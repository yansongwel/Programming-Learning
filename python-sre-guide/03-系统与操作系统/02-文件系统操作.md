# os/shutil/pathlib 文件系统操作 — 用 Python 替代 find/cp/mv/rm/chmod 实现文件管理自动化

## 1. 是什么 & 为什么用它

### 三大核心库

| 库 | 定位 | Shell 等效 |
|---|---|---|
| `os` | 操作系统底层接口（文件/目录/环境变量） | 基础命令 ls, mkdir, chmod, chown |
| `shutil` | 高级文件操作（复制、移动、压缩） | cp, mv, rm -rf, tar |
| `pathlib` | 面向对象的路径操作（Python 3.4+，推荐） | 路径拼接、通配符匹配 |

### 为什么 SRE 需要用 Python 而非 Shell 处理文件

```
Shell 方式的痛点:
  find /var/log -name "*.log" -mtime +30 -exec rm {} \;
  - 文件名含空格会出错
  - 错误处理困难（rm 失败怎么办？）
  - 跨平台不兼容
  - 复杂逻辑难以表达（按大小+时间+类型组合筛选）

Python 方式的优势:
  - 路径处理安全（pathlib 自动处理分隔符和特殊字符）
  - 完善的异常处理（try/except 精确捕获每类错误）
  - 跨平台（Windows/Linux/macOS）
  - 可组合复杂逻辑（条件筛选、正则匹配、数据库记录）
```

## 3. 核心概念与原理

### pathlib 路径对象模型

```
                    PurePath
                   /        \
          PurePosixPath   PureWindowsPath
                |              |
              Path           Path
             /    \            |
      PosixPath  WindowsPath  WindowsPath
     (Linux/Mac)  (Windows)

路径对象的核心属性:
  path = Path('/var/log/nginx/access.log')

  path.parent    → /var/log/nginx      # 父目录
  path.name      → access.log          # 文件名
  path.stem      → access              # 不含后缀的文件名
  path.suffix    → .log                # 后缀
  path.parts     → ('/', 'var', 'log', 'nginx', 'access.log')
```

### 三库协作关系

```
+--------------------------------------------------+
|                  你的 SRE 脚本                     |
+--------------------------------------------------+
|  pathlib (推荐入口)                                |
|  - 路径构建与解析                                   |
|  - 文件读写 (read_text/write_text)                 |
|  - 通配符匹配 (glob/rglob)                        |
|  - 文件属性查询 (stat/exists/is_file)              |
+--------------------------------------------------+
|  shutil (批量操作)        |  os (底层操作)          |
|  - 复制文件/目录树        |  - 权限管理             |
|  - 移动/重命名            |  - 符号链接             |
|  - 删除目录树             |  - 环境变量             |
|  - 压缩/解压              |  - 进程相关             |
+--------------------------------------------------+
|              操作系统文件系统 API                    |
+--------------------------------------------------+
```

## 4. 基础用法

### 4.1 路径操作（对比 Shell 路径处理）

```python
#!/usr/bin/env python3
"""路径操作基础 —— pathlib vs Shell"""

from pathlib import Path

# ============================================================
# Shell 对比: DIR=$(dirname "$FILE"); BASE=$(basename "$FILE")
# Python 不需要处理引号和空格问题
# ============================================================

# --- 创建路径对象 ---
p = Path('/var/log/nginx/access.log')

# 路径分解（对比 dirname/basename）
print(f"完整路径: {p}")
print(f"父目录:   {p.parent}")        # 对比: dirname
print(f"文件名:   {p.name}")           # 对比: basename
print(f"主干名:   {p.stem}")           # 无 Shell 等效，需要 ${FILE%.*}
print(f"后缀:     {p.suffix}")         # 对比: ${FILE##*.}
print(f"所有后缀: {p.suffixes}")       # 处理 .tar.gz 这种双后缀

# --- 路径拼接（对比 Shell 的字符串拼接）---
# Shell: LOG_DIR="/var/log"; FILE="$LOG_DIR/app/error.log"
# 问题: 如果 LOG_DIR 末尾有 / 就会变成 //
log_dir = Path('/var/log')
error_log = log_dir / 'app' / 'error.log'  # 使用 / 运算符拼接
print(f"拼接路径: {error_log}")

# --- 路径解析 ---
# 对比: readlink -f 或 realpath
relative = Path('./logs/../data/test.txt')
print(f"规范化:   {relative.resolve()}")  # 解析 .. 和符号链接

# 当前目录和家目录
print(f"当前目录: {Path.cwd()}")           # 对比: pwd
print(f"家目录:   {Path.home()}")          # 对比: $HOME

# --- 路径判断 ---
p = Path('/etc/hosts')
print(f"\n文件是否存在: {p.exists()}")      # 对比: test -e
print(f"是否为文件:   {p.is_file()}")       # 对比: test -f
print(f"是否为目录:   {p.is_dir()}")        # 对比: test -d
print(f"是否为链接:   {p.is_symlink()}")    # 对比: test -L
```

### 4.2 目录遍历（对比 find）

```python
#!/usr/bin/env python3
"""目录遍历 —— 对比 find 命令"""

from pathlib import Path
import os

# ============================================================
# Shell 对比: find /var/log -name "*.log" -type f
# ============================================================

log_dir = Path('/var/log')

# --- 方法 1: glob（非递归）---
# 对比: ls /var/log/*.log
print("=== /var/log 下的 .log 文件 ===")
for f in log_dir.glob('*.log'):
    print(f"  {f}")

# --- 方法 2: rglob（递归）---
# 对比: find /var/log -name "*.log" -type f
print("\n=== 递归查找所有 .log 文件 ===")
for f in log_dir.rglob('*.log'):
    if f.is_file():  # 确保是文件而非目录
        print(f"  {f} ({f.stat().st_size} 字节)")

# --- 方法 3: os.walk（更精细的控制）---
# 对比: find /var/log -maxdepth 2 -name "*.log"
print("\n=== os.walk 遍历（限制深度）===")
base_depth = str(log_dir).count(os.sep)
for root, dirs, files in os.walk(log_dir):
    current_depth = root.count(os.sep) - base_depth
    if current_depth >= 2:  # 限制深度为 2
        dirs.clear()  # 不再深入子目录
        continue
    for name in files:
        if name.endswith('.log'):
            filepath = Path(root) / name
            print(f"  深度{current_depth}: {filepath}")

# --- 方法 4: 复杂条件筛选（find 难以实现）---
# 找出所有大于 1MB 且 30 天未修改的 .log 文件
import time
print("\n=== 复杂条件筛选 ===")
now = time.time()
thirty_days_ago = now - (30 * 24 * 3600)

for f in log_dir.rglob('*.log'):
    try:
        if f.is_file():
            stat = f.stat()
            size_mb = stat.st_size / (1024 * 1024)
            mtime = stat.st_mtime
            if size_mb > 1 and mtime < thirty_days_ago:
                age_days = (now - mtime) / (24 * 3600)
                print(f"  {f}: {size_mb:.1f}MB, {age_days:.0f}天前")
    except PermissionError:
        pass  # 跳过无权限的文件
```

### 4.3 文件复制、移动、删除（对比 cp/mv/rm）

```python
#!/usr/bin/env python3
"""文件操作 —— 对比 cp/mv/rm 命令"""

import shutil
from pathlib import Path

# 创建测试目录
test_dir = Path('/tmp/sre_file_test')
test_dir.mkdir(exist_ok=True)

# --- 创建文件（对比 echo > 或 touch）---
# Shell: echo "hello" > /tmp/test.txt
(test_dir / 'source.txt').write_text('这是测试内容\n第二行\n', encoding='utf-8')

# Shell: touch /tmp/empty.txt
(test_dir / 'empty.txt').touch()

# --- 复制文件（对比 cp）---
# Shell: cp source.txt dest.txt
src = test_dir / 'source.txt'
dst = test_dir / 'dest.txt'
shutil.copy2(src, dst)  # copy2 保留元数据（时间戳等），对比 cp -p
print(f"复制文件: {src} -> {dst}")

# --- 复制目录树（对比 cp -r）---
# Shell: cp -r /tmp/sre_file_test /tmp/sre_file_backup
src_dir = test_dir
dst_dir = Path('/tmp/sre_file_backup')
if dst_dir.exists():
    shutil.rmtree(dst_dir)  # 先删除已存在的目标
shutil.copytree(src_dir, dst_dir)
print(f"复制目录: {src_dir} -> {dst_dir}")

# --- 移动/重命名（对比 mv）---
# Shell: mv dest.txt renamed.txt
old_path = test_dir / 'dest.txt'
new_path = test_dir / 'renamed.txt'
shutil.move(str(old_path), str(new_path))
print(f"移动文件: {old_path} -> {new_path}")

# pathlib 方式重命名（同目录内）
# Shell: mv renamed.txt final.txt
new_path.rename(test_dir / 'final.txt')
print(f"重命名: renamed.txt -> final.txt")

# --- 删除文件（对比 rm）---
# Shell: rm /tmp/sre_file_test/empty.txt
(test_dir / 'empty.txt').unlink()
print(f"删除文件: empty.txt")

# 安全删除（missing_ok=True 不报错，对比 rm -f）
(test_dir / 'nonexistent.txt').unlink(missing_ok=True)

# --- 删除目录树（对比 rm -rf）---
# Shell: rm -rf /tmp/sre_file_backup
shutil.rmtree(dst_dir)
print(f"删除目录树: {dst_dir}")

# --- 创建目录（对比 mkdir -p）---
# Shell: mkdir -p /tmp/sre_test/a/b/c
deep_dir = Path('/tmp/sre_test/a/b/c')
deep_dir.mkdir(parents=True, exist_ok=True)
print(f"创建多级目录: {deep_dir}")

# 清理
shutil.rmtree(Path('/tmp/sre_test'), ignore_errors=True)
shutil.rmtree(test_dir, ignore_errors=True)
print("清理完成")
```

### 4.4 权限管理（对比 chmod/chown）

```python
#!/usr/bin/env python3
"""文件权限管理 —— 对比 chmod/chown/stat 命令"""

import os
import stat
from pathlib import Path

# 创建测试文件
test_file = Path('/tmp/perm_test.txt')
test_file.write_text('权限测试文件\n')

# --- 查看权限（对比 ls -la 或 stat）---
file_stat = test_file.stat()
mode = file_stat.st_mode

# 转换为八进制格式（对比 stat -c %a）
octal_perm = oct(stat.S_IMODE(mode))
print(f"文件: {test_file}")
print(f"权限（八进制）: {octal_perm}")  # 如 0o644
print(f"所有者 UID: {file_stat.st_uid}")
print(f"所属组 GID: {file_stat.st_gid}")
print(f"大小: {file_stat.st_size} 字节")

# 人类可读的权限字符串（对比 ls -la 的 -rw-r--r--）
def perm_to_str(mode):
    """将权限模式转换为 rwx 格式"""
    perms = ''
    for who in ('USR', 'GRP', 'OTH'):
        for what in ('R', 'W', 'X'):
            flag = getattr(stat, f'S_I{what}{who}')
            perms += 'rwx'[['R','W','X'].index(what)] if mode & flag else '-'
    return perms

print(f"权限（rwx）: {perm_to_str(mode)}")

# --- 修改权限（对比 chmod）---
# Shell: chmod 755 /tmp/perm_test.txt
os.chmod(test_file, 0o755)
print(f"\nchmod 755 后: {oct(stat.S_IMODE(test_file.stat().st_mode))}")

# Shell: chmod u+x /tmp/perm_test.txt
# Python 需要位运算
current_mode = test_file.stat().st_mode
os.chmod(test_file, current_mode | stat.S_IXUSR)  # 添加用户执行权限
print(f"添加 u+x 后: {perm_to_str(test_file.stat().st_mode)}")

# --- 修改所有者（对比 chown，需要 root 权限）---
# Shell: chown user:group /tmp/perm_test.txt
# os.chown(test_file, uid, gid)  # 需要 root 权限
# 获取用户信息
import pwd
import grp
try:
    current_user = pwd.getpwuid(os.getuid())
    print(f"\n当前用户: {current_user.pw_name} (UID={current_user.pw_uid})")
    current_group = grp.getgrgid(os.getgid())
    print(f"当前组:   {current_group.gr_name} (GID={current_group.gr_gid})")
except KeyError:
    print("无法获取用户信息")

# 清理
test_file.unlink(missing_ok=True)
```

### 4.5 磁盘空间检查

```python
#!/usr/bin/env python3
"""磁盘空间检查 —— 对比 df -h 和 du -sh"""

import shutil
import os
from pathlib import Path

# --- df -h 等效 ---
def check_disk_usage(path='/'):
    """检查磁盘使用情况（对比 df -h）"""
    total, used, free = shutil.disk_usage(path)

    def human(n):
        for u in ['B','KB','MB','GB','TB']:
            if n < 1024: return f"{n:.1f}{u}"
            n /= 1024

    print(f"路径: {path}")
    print(f"  总量: {human(total)}")
    print(f"  已用: {human(used)} ({used/total*100:.1f}%)")
    print(f"  空闲: {human(free)} ({free/total*100:.1f}%)")
    return used / total * 100

print("=== 磁盘使用（对比 df -h）===")
percent = check_disk_usage('/')
if percent > 80:
    print("  [警告] 磁盘使用超过 80%!")

# --- du -sh 等效 ---
def get_dir_size(path):
    """计算目录大小（对比 du -sh）"""
    total = 0
    path = Path(path)
    for f in path.rglob('*'):
        try:
            if f.is_file() and not f.is_symlink():
                total += f.stat().st_size
        except (PermissionError, OSError):
            pass
    return total

def human_size(n):
    for u in ['B','KB','MB','GB','TB']:
        if n < 1024: return f"{n:.1f}{u}"
        n /= 1024
    return f"{n:.1f}PB"

print("\n=== 目录大小（对比 du -sh）===")
dirs_to_check = ['/var/log', '/tmp']
for d in dirs_to_check:
    if Path(d).exists():
        size = get_dir_size(d)
        print(f"  {d}: {human_size(size)}")
```

### 5.1 文件压缩与解压（对比 tar/zip）

```python
#!/usr/bin/env python3
"""文件压缩解压 —— 对比 tar/zip 命令"""

import shutil
import tarfile
import zipfile
from pathlib import Path

# 创建测试数据
test_dir = Path('/tmp/compress_test')
test_dir.mkdir(exist_ok=True)
(test_dir / 'file1.txt').write_text('文件1内容\n' * 100)
(test_dir / 'file2.log').write_text('日志内容\n' * 200)
sub_dir = test_dir / 'subdir'
sub_dir.mkdir(exist_ok=True)
(sub_dir / 'file3.txt').write_text('子目录文件\n' * 50)

# --- tar.gz 压缩（对比 tar czf）---
# Shell: tar czf /tmp/backup.tar.gz -C /tmp compress_test
output = shutil.make_archive(
    '/tmp/backup',        # 输出文件名（不含后缀）
    'gztar',              # 格式：gztar, bztar, xztar, tar, zip
    root_dir='/tmp',      # 根目录
    base_dir='compress_test'  # 要打包的子目录
)
print(f"tar.gz 压缩完成: {output}")

# --- tar.gz 解压（对比 tar xzf）---
# Shell: tar xzf /tmp/backup.tar.gz -C /tmp/extract
extract_dir = Path('/tmp/extract')
extract_dir.mkdir(exist_ok=True)
shutil.unpack_archive('/tmp/backup.tar.gz', extract_dir)
print(f"解压到: {extract_dir}")

# --- 精细控制 tar 操作 ---
# 选择性打包（只打包 .txt 文件）
with tarfile.open('/tmp/selective.tar.gz', 'w:gz') as tar:
    for f in test_dir.rglob('*.txt'):
        # arcname 控制压缩包内的路径
        tar.add(f, arcname=f.relative_to(test_dir))
        print(f"  添加: {f.name}")
print("选择性打包完成")

# 列出 tar 内容（对比 tar tzf）
with tarfile.open('/tmp/selective.tar.gz', 'r:gz') as tar:
    print("\ntar 包内容:")
    for member in tar.getmembers():
        print(f"  {member.name} ({member.size} 字节)")

# --- ZIP 操作（对比 zip/unzip）---
# Shell: zip -r /tmp/backup.zip /tmp/compress_test
with zipfile.ZipFile('/tmp/backup.zip', 'w', zipfile.ZIP_DEFLATED) as zf:
    for f in test_dir.rglob('*'):
        if f.is_file():
            zf.write(f, f.relative_to(test_dir))
print(f"\nZIP 压缩完成: /tmp/backup.zip")

# 清理
shutil.rmtree(test_dir, ignore_errors=True)
shutil.rmtree(extract_dir, ignore_errors=True)
for f in ['/tmp/backup.tar.gz', '/tmp/backup.zip', '/tmp/selective.tar.gz']:
    Path(f).unlink(missing_ok=True)
```

### 5.2 watchdog 文件监控

```python
#!/usr/bin/env python3
"""
文件系统实时监控 —— watchdog
对比 Shell 的 inotifywait 命令
Shell: inotifywait -m -r /var/log -e create,modify,delete
"""

import time
from pathlib import Path
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler

class SREFileHandler(FileSystemEventHandler):
    """SRE 文件事件处理器"""

    def __init__(self, watch_patterns=None, ignore_patterns=None):
        """
        :param watch_patterns: 要监控的文件模式（如 ['*.log', '*.conf']）
        :param ignore_patterns: 要忽略的文件模式（如 ['*.tmp', '*.swp']）
        """
        self.watch_patterns = watch_patterns or ['*']
        self.ignore_patterns = ignore_patterns or ['*.swp', '*.tmp', '*~']

    def _should_process(self, path):
        """判断文件是否应该处理"""
        name = Path(path).name
        # 检查忽略模式
        for pattern in self.ignore_patterns:
            if Path(name).match(pattern):
                return False
        # 检查监控模式
        if self.watch_patterns == ['*']:
            return True
        for pattern in self.watch_patterns:
            if Path(name).match(pattern):
                return True
        return False

    def on_created(self, event):
        if not event.is_directory and self._should_process(event.src_path):
            print(f"[创建] {event.src_path}")

    def on_modified(self, event):
        if not event.is_directory and self._should_process(event.src_path):
            size = Path(event.src_path).stat().st_size if Path(event.src_path).exists() else 0
            print(f"[修改] {event.src_path} (大小: {size} 字节)")

    def on_deleted(self, event):
        if not event.is_directory and self._should_process(event.src_path):
            print(f"[删除] {event.src_path}")

    def on_moved(self, event):
        if not event.is_directory:
            print(f"[移动] {event.src_path} -> {event.dest_path}")


def watch_directory(path, patterns=None, recursive=True):
    """
    监控目录变化
    :param path: 要监控的目录
    :param patterns: 文件模式列表
    :param recursive: 是否递归监控子目录
    """
    handler = SREFileHandler(watch_patterns=patterns)
    observer = Observer()
    observer.schedule(handler, str(path), recursive=recursive)
    observer.start()

    print(f"开始监控: {path} (Ctrl+C 退出)")
    print(f"监控模式: {patterns or ['*']}")

    try:
        while True:
            time.sleep(1)
    except KeyboardInterrupt:
        observer.stop()
    observer.join()
    print("监控已停止")


if __name__ == '__main__':
    # 监控 /tmp 目录下的 .log 和 .txt 文件变化
    watch_directory('/tmp', patterns=['*.log', '*.txt'])
```

## 6. SRE 实战案例：自动清理过期日志文件

```python
#!/usr/bin/env python3
"""
SRE 实战：自动清理过期日志文件
功能：
  - 扫描指定目录的日志文件
  - 按年龄、大小、磁盘空间条件筛选
  - 支持压缩归档再删除
  - 详细的清理报告
  - dry-run 模式（安全预览）

对比 Shell 方式:
  find /var/log -name "*.log" -mtime +30 -delete
  - Shell 版本无法: 按磁盘空间动态决策、生成报告、压缩归档、错误恢复
"""

import os
import time
import shutil
import tarfile
import logging
from pathlib import Path
from datetime import datetime, timedelta
from dataclasses import dataclass, field
from typing import List

# 设置日志
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s [%(levelname)s] %(message)s',
    handlers=[
        logging.StreamHandler(),
        logging.FileHandler('/tmp/log_cleaner.log'),
    ]
)
logger = logging.getLogger('LogCleaner')


@dataclass
class CleanRule:
    """清理规则"""
    pattern: str = '*.log'        # 文件匹配模式
    max_age_days: int = 30        # 最大保留天数
    max_size_mb: float = 0        # 最大文件大小（MB），0 表示不限
    keep_latest: int = 5          # 至少保留最新的 N 个文件
    archive_before_delete: bool = True  # 删除前是否归档


@dataclass
class CleanResult:
    """清理结果"""
    scanned: int = 0
    deleted: int = 0
    archived: int = 0
    freed_bytes: int = 0
    errors: List[str] = field(default_factory=list)
    details: List[dict] = field(default_factory=list)


def human_size(n):
    """字节转人类可读"""
    for u in ['B', 'KB', 'MB', 'GB', 'TB']:
        if n < 1024:
            return f"{n:.1f}{u}"
        n /= 1024
    return f"{n:.1f}PB"


class LogCleaner:
    """日志文件清理器"""

    def __init__(self, directories, rules=None, archive_dir=None):
        """
        :param directories: 要扫描的目录列表
        :param rules: 清理规则列表
        :param archive_dir: 归档目录（默认 /tmp/log_archives）
        """
        self.directories = [Path(d) for d in directories]
        self.rules = rules or [CleanRule()]
        self.archive_dir = Path(archive_dir or '/tmp/log_archives')

    def scan(self):
        """扫描并返回需要清理的文件列表"""
        candidates = []
        now = time.time()

        for directory in self.directories:
            if not directory.exists():
                logger.warning(f"目录不存在: {directory}")
                continue

            for rule in self.rules:
                # 收集匹配的文件
                matched_files = []
                for f in directory.rglob(rule.pattern):
                    try:
                        if f.is_file() and not f.is_symlink():
                            stat = f.stat()
                            matched_files.append({
                                'path': f,
                                'size': stat.st_size,
                                'mtime': stat.st_mtime,
                                'age_days': (now - stat.st_mtime) / 86400,
                                'rule': rule,
                            })
                    except (PermissionError, OSError) as e:
                        logger.warning(f"无法访问: {f}: {e}")

                # 按修改时间排序（最新的在前）
                matched_files.sort(key=lambda x: x['mtime'], reverse=True)

                # 跳过最新的 N 个文件
                protected = matched_files[:rule.keep_latest]
                checkable = matched_files[rule.keep_latest:]

                for item in checkable:
                    should_clean = False
                    reason = []

                    # 条件1: 超过最大年龄
                    if rule.max_age_days > 0 and item['age_days'] > rule.max_age_days:
                        should_clean = True
                        reason.append(f"超过{rule.max_age_days}天")

                    # 条件2: 超过最大大小
                    if rule.max_size_mb > 0:
                        size_mb = item['size'] / (1024 * 1024)
                        if size_mb > rule.max_size_mb:
                            should_clean = True
                            reason.append(f"大小{size_mb:.1f}MB超过{rule.max_size_mb}MB")

                    if should_clean:
                        item['reason'] = ', '.join(reason)
                        candidates.append(item)

        return candidates

    def _archive_file(self, filepath):
        """归档单个文件到压缩包"""
        self.archive_dir.mkdir(parents=True, exist_ok=True)
        date_str = datetime.now().strftime('%Y%m%d')
        archive_name = self.archive_dir / f"archived_{date_str}.tar.gz"

        mode = 'a:gz' if archive_name.exists() else 'w:gz'
        # tar 的 append 模式不支持 gzip，如果文件已存在就直接追加到新包
        if archive_name.exists():
            # 重命名避免冲突
            counter = 1
            while archive_name.exists():
                archive_name = self.archive_dir / f"archived_{date_str}_{counter}.tar.gz"
                counter += 1

        with tarfile.open(archive_name, 'w:gz') as tar:
            tar.add(filepath, arcname=filepath.name)

        return archive_name

    def clean(self, dry_run=True):
        """
        执行清理
        :param dry_run: 如果为 True，只显示将要做什么，不实际操作
        """
        candidates = self.scan()
        result = CleanResult(scanned=len(candidates))

        if not candidates:
            logger.info("没有需要清理的文件")
            return result

        mode = "[DRY RUN] " if dry_run else ""
        logger.info(f"{mode}找到 {len(candidates)} 个待清理文件")

        for item in candidates:
            filepath = item['path']
            size = item['size']

            try:
                logger.info(
                    f"{mode}清理: {filepath} "
                    f"(大小: {human_size(size)}, "
                    f"年龄: {item['age_days']:.0f}天, "
                    f"原因: {item['reason']})"
                )

                if not dry_run:
                    # 归档
                    if item['rule'].archive_before_delete:
                        archive_path = self._archive_file(filepath)
                        logger.info(f"  已归档到: {archive_path}")
                        result.archived += 1

                    # 删除
                    filepath.unlink()
                    logger.info(f"  已删除: {filepath}")

                result.deleted += 1
                result.freed_bytes += size
                result.details.append({
                    'path': str(filepath),
                    'size': size,
                    'age_days': round(item['age_days'], 1),
                    'reason': item['reason'],
                })

            except Exception as e:
                error_msg = f"清理失败 {filepath}: {e}"
                logger.error(error_msg)
                result.errors.append(error_msg)

        return result

    def generate_report(self, result):
        """生成清理报告"""
        report = []
        report.append("=" * 60)
        report.append("  日志清理报告")
        report.append(f"  时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        report.append("=" * 60)
        report.append(f"  扫描目录: {', '.join(str(d) for d in self.directories)}")
        report.append(f"  待清理文件数: {result.scanned}")
        report.append(f"  已删除文件数: {result.deleted}")
        report.append(f"  已归档文件数: {result.archived}")
        report.append(f"  释放空间: {human_size(result.freed_bytes)}")

        if result.errors:
            report.append(f"\n  错误 ({len(result.errors)}):")
            for err in result.errors:
                report.append(f"    - {err}")

        if result.details:
            report.append(f"\n  清理详情:")
            for d in result.details:
                report.append(
                    f"    {d['path']} | {human_size(d['size'])} | "
                    f"{d['age_days']}天 | {d['reason']}"
                )

        report.append("=" * 60)
        return '\n'.join(report)


if __name__ == '__main__':
```

## 7. 常见坑点与最佳实践

### Do's (推荐做法)

```python
# 1. 优先使用 pathlib 而非 os.path
# 好：
from pathlib import Path
path = Path('/var/log') / 'nginx' / 'access.log'

# 差：
import os
path = os.path.join('/var/log', 'nginx', 'access.log')

# 2. 删除前检查（防止误删）
target = Path('/important/data')
if target.exists() and target.is_dir():
    # 二次确认：检查路径是否合理
    if str(target).startswith(('/tmp', '/var/log')):
        shutil.rmtree(target)

# 3. 使用 exist_ok 和 missing_ok 避免异常
Path('/tmp/mydir').mkdir(parents=True, exist_ok=True)  # 不会因为已存在报错
Path('/tmp/nonexist.txt').unlink(missing_ok=True)       # 不会因为不存在报错

# 4. 大文件使用流式读取
def count_lines(filepath):
    """统计大文件行数（不会把整个文件加载到内存）"""
    count = 0
    with open(filepath, 'rb') as f:
        for _ in f:
            count += 1
    return count

# 5. 原子写入（防止写到一半崩溃产生损坏文件）
import tempfile
def safe_write(filepath, content):
    """安全写入：先写临时文件再重命名"""
    filepath = Path(filepath)
    fd, tmp_path = tempfile.mkstemp(dir=filepath.parent)
    try:
        with os.fdopen(fd, 'w') as f:
            f.write(content)
        os.replace(tmp_path, filepath)  # 原子操作
    except:
        os.unlink(tmp_path)
        raise
```

### Don'ts (避免做法)

```python
# 1. 不要用字符串拼接路径
# 坏：
path = '/var/log' + '/' + 'app' + '/' + 'error.log'  # Windows 上不兼容

# 2. 不要在未检查的情况下 rmtree
# 坏（极度危险！）：
shutil.rmtree(some_variable)  # some_variable 可能是 '/' ！

# 3. 不要忽略编码
# 坏：
content = open('file.txt').read()  # 默认编码可能不对
# 好：
content = Path('file.txt').read_text(encoding='utf-8')

# 4. 不要一次读取巨大文件
# 坏（可能占几 GB 内存）：
data = Path('/var/log/huge.log').read_text()
# 好：
with open('/var/log/huge.log') as f:
    for line in f:
        process(line)
```

```
# ==================== 文件操作速查表 ====================

# --- pathlib 路径操作 ---
Path('/a/b/c.txt').parent          → /a/b           # dirname
Path('/a/b/c.txt').name            → c.txt          # basename
Path('/a/b/c.txt').stem            → c              # ${f%.*}
Path('/a/b/c.txt').suffix          → .txt           # ${f##*.}
Path('/a') / 'b' / 'c'            → /a/b/c         # 路径拼接
Path('.').resolve()                → /abs/path      # realpath
Path.cwd()                        → 当前目录        # pwd
Path.home()                       → 家目录          # $HOME

# --- 文件判断 ---
path.exists()                      # test -e
path.is_file()                     # test -f
path.is_dir()                      # test -d
path.is_symlink()                  # test -L

path.mkdir(parents=True, exist_ok=True)    # mkdir -p
```
