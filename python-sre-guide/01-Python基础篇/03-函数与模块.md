## 1. 是什么 & 为什么用它

### Shell 函数的局限

```bash
# Shell 函数：简单但受限
check_disk() {
    local threshold=$1                    # 只能通过位置参数
    df -h | awk -v t="$threshold" '$5+0 > t {print $0}'
    return $?                              # 只能返回 0-255 的整数
}

# 想返回复杂数据？只能 echo + 管道
get_server_info() {
    echo "web-01|192.168.1.1|running"     # 只能返回字符串
}
result=$(get_server_info)                  # 用子 shell 捕获输出
```

### Python 函数的优势

```python
# Python 函数：功能强大
def check_disk(threshold=80, mount_point="/"):
    """检查磁盘使用率（支持默认参数、类型丰富的返回值）"""
    import shutil
    usage = shutil.disk_usage(mount_point)
    percent = usage.used / usage.total * 100
    return {                               # 可以返回任何类型的数据
        "mount": mount_point,
        "percent": round(percent, 1),
        "alert": percent > threshold
    }
```

| 对比维度     | Shell function                | Python def                    |
|-------------|-------------------------------|-------------------------------|
| 参数传递     | $1, $2 位置参数               | 位置、默认、关键字、*args等   |
| 返回值       | return 0-255 或 echo 字符串   | return 任何对象               |
| 作用域       | local 关键字，容易出错        | 函数内自动局部作用域          |
| 文档         | 注释                          | docstring（可自动生成文档）   |
| 模块化       | source 其他脚本               | import 模块系统               |

## 3. 参数详解

### 3.1 位置参数与默认参数

```python
#!/usr/bin/env python3
"""
函数参数的各种写法
"""

# ==========================================
# 位置参数（和 Shell $1 $2 类似）
# ==========================================
def ping主机(主机, 次数):
    """位置参数：调用时必须按顺序传入"""
    print(f"ping {主机} -c {次数}")

ping主机("192.168.1.1", 3)       # 正确
# ping主机(3, "192.168.1.1")     # 语法上不报错，但语义错了


# ==========================================
# 默认参数（Shell 中用 ${1:-默认值} 模拟）
# ==========================================
# Shell: check_port() { local port=${1:-80}; ... }
def 检查端口(主机, 端口=80, 超时=5):
    """默认参数：有默认值的参数可以不传"""
    print(f"检查 {主机}:{端口}，超时 {超时}s")

检查端口("10.0.1.1")              # 使用默认端口 80，超时 5s
检查端口("10.0.1.1", 443)         # 指定端口 443
检查端口("10.0.1.1", 8080, 10)    # 指定端口和超时


# ==========================================
# 关键字参数（Shell 完全没有的功能）
# ==========================================
# 可以不按顺序传参，用参数名指定
检查端口("10.0.1.1", 超时=10, 端口=443)   # 通过参数名指定
检查端口(主机="10.0.1.1", 超时=10)         # 全部用关键字参数


# ==========================================
# 强制关键字参数（* 后面的参数必须用关键字传递）
# ==========================================
def 部署服务(服务名, *, 版本, 环境="production"):
    """星号后面的参数必须以关键字形式传入"""
    print(f"部署 {服务名} v{版本} 到 {环境}")

部署服务("nginx", 版本="1.24.0")                    # 正确
部署服务("nginx", 版本="1.24.0", 环境="staging")    # 正确
# 部署服务("nginx", "1.24.0")                       # 错误！版本必须用关键字
```

### 3.2 *args 和 **kwargs

```python
#!/usr/bin/env python3
"""
可变参数：*args（收集多余的位置参数）和 **kwargs（收集多余的关键字参数）
"""

# ==========================================
# *args：接受任意数量的位置参数
# ==========================================
# 对比 Shell: 使用 $@ 获取所有参数
# process_servers() { for server in "$@"; do echo $server; done; }

def 批量检查(*主机列表):
    """接受任意数量的主机名"""
    print(f"共 {len(主机列表)} 台主机需要检查")
    for 主机 in 主机列表:
        print(f"  检查: {主机}")

批量检查("web-01", "web-02", "web-03")
批量检查("db-01")                    # 只传一个也行


# ==========================================
# **kwargs：接受任意数量的关键字参数
# ==========================================
def 创建告警(**告警信息):
    """接受任意数量的关键字参数，构建告警"""
    print("=== 告警信息 ===")
    for 键, 值 in 告警信息.items():
        print(f"  {键}: {值}")

创建告警(
    级别="WARNING",
    主机="web-01",
    指标="CPU",
    当前值="95%",
    阈值="80%"
)


# ==========================================
# 混合使用
# ==========================================
def 运维命令(操作, *目标, **选项):
    """
    混合参数示例

    参数:
        操作: 要执行的操作（必需）
        *目标: 操作目标列表（可变长度）
        **选项: 额外选项（关键字参数）
    """
    print(f"操作: {操作}")
    print(f"目标: {目标}")
    print(f"选项: {选项}")

运维命令("restart", "nginx", "php-fpm", timeout=30, force=True)
# 输出:
# 操作: restart
# 目标: ('nginx', 'php-fpm')
# 选项: {'timeout': 30, 'force': True}


# ==========================================
# 参数解包（反向操作）
# ==========================================
参数列表 = ["web-01", "web-02", "web-03"]
批量检查(*参数列表)                  # 等同于 批量检查("web-01", "web-02", "web-03")

配置 = {"级别": "ERROR", "主机": "db-01", "指标": "磁盘"}
创建告警(**配置)                     # 等同于 创建告警(级别="ERROR", 主机="db-01", 指标="磁盘")
```

## 4. 返回值（对比 Shell return/echo）

```python
#!/usr/bin/env python3
"""
Python 返回值 vs Shell 返回值
"""

# ==========================================
# Shell 的返回值局限：
# 1. return 只能返回 0-255 的整数（退出码）
# 2. 返回字符串只能用 echo + $() 捕获
# 3. 无法直接返回数组或复杂数据
#
# get_info() {
#     echo "hostname|ip|status"    # 只能返回字符串
#     return 0
# }
# result=$(get_info)
# IFS='|' read -r host ip status <<< "$result"
# ==========================================

# Python 可以返回任何类型的数据
def 获取磁盘信息():
    """返回字典：包含多个字段"""
    import shutil
    使用情况 = shutil.disk_usage("/")
    return {
        "总量GB": round(使用情况.total / (1024**3), 1),
        "已用GB": round(使用情况.used / (1024**3), 1),
        "可用GB": round(使用情况.free / (1024**3), 1),
        "使用率": round(使用情况.used / 使用情况.total * 100, 1),
    }

信息 = 获取磁盘信息()
print(f"磁盘使用率: {信息['使用率']}%")
print(f"可用空间: {信息['可用GB']}GB")


# ==========================================
# 返回多个值（实际上是返回元组）
# ==========================================
def 分析日志行(行):
    """解析日志行，返回多个值"""
    部分 = 行.split()
    日期 = 部分[0]
    时间 = 部分[1]
    级别 = 部分[2]
    消息 = " ".join(部分[3:])
    return 日期, 时间, 级别, 消息          # 返回元组

# 接收多个返回值
日期, 时间, 级别, 消息 = 分析日志行("2024-01-15 10:30:45 ERROR nginx crashed")
print(f"日期={日期}, 时间={时间}, 级别={级别}")

# 只关心部分返回值，用 _ 忽略其他
_, _, 级别, _ = 分析日志行("2024-01-15 10:30:45 WARNING high cpu")
print(f"级别: {级别}")


# ==========================================
# 返回布尔值（常用于检查类函数）
# ==========================================
def 端口是否开放(主机, 端口):
    """检查端口是否开放"""
    import socket
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(3)
        结果 = sock.connect_ex((主机, 端口))
        sock.close()
        return 结果 == 0              # 返回 True 或 False
    except Exception:
        return False

# 直接用在 if 中
# if 端口是否开放("localhost", 80):
#     print("80 端口已开放")


# ==========================================
# 没有 return 语句时，默认返回 None
# ==========================================
def 打印信息(消息):
    print(消息)
    # 没有 return，默认返回 None

结果 = 打印信息("hello")
print(f"返回值: {结果}")            # 返回值: None
```

## 5. lambda 匿名函数

```python
#!/usr/bin/env python3
"""
lambda：一行定义的小函数，常用于排序和过滤
"""

# ==========================================
# 基础 lambda
# ==========================================
# 普通函数
def 平方(x):
    return x * x

# 等价的 lambda
平方 = lambda x: x * x
print(平方(5))    # 25


# ==========================================
# lambda 在排序中的应用（SRE 常用场景）
# ==========================================
服务器列表 = [
    {"名称": "web-01", "CPU": 85, "内存": 60},
    {"名称": "db-01",  "CPU": 45, "内存": 90},
    {"名称": "web-02", "CPU": 92, "内存": 75},
    {"名称": "api-01", "CPU": 30, "内存": 40},
]

# 按 CPU 使用率排序
按CPU排序 = sorted(服务器列表, key=lambda s: s["CPU"], reverse=True)
print("按 CPU 使用率降序:")
for s in 按CPU排序:
    print(f"  {s['名称']}: CPU={s['CPU']}%")

# 按内存使用率排序
按内存排序 = sorted(服务器列表, key=lambda s: s["内存"], reverse=True)
print("\n按内存使用率降序:")
for s in 按内存排序:
    print(f"  {s['名称']}: 内存={s['内存']}%")


# ==========================================
# lambda 在过滤中的应用
# ==========================================
# 找出 CPU > 80% 的服务器
高负载 = list(filter(lambda s: s["CPU"] > 80, 服务器列表))
print(f"\nCPU > 80% 的服务器: {[s['名称'] for s in 高负载]}")

# 等价的列表推导式（更推荐）
高负载 = [s for s in 服务器列表 if s["CPU"] > 80]
```

```python
import socket
```

```python
        "标准输出": 结果.stdout.strip(),

    }


# 模块级常量
```

## 8. if \_\_name\_\_ == "\_\_main\_\_"

```python
#!/usr/bin/env python3
"""
if __name__ == "__main__" 详解

这是 Python 中最重要的惯用写法之一：
- 当脚本被直接运行时，__name__ 的值是 "__main__"
- 当脚本被其他模块导入时，__name__ 的值是模块名

对比 Shell：
- Shell 没有直接对应的机制
- Shell 中 source 和直接执行没有区别
"""


def 检查磁盘(路径="/"):
    """检查磁盘使用率"""
    import shutil
    使用情况 = shutil.disk_usage(路径)
    使用率 = 使用情况.used / 使用情况.total * 100
    return round(使用率, 1)


def 检查内存():
    """检查内存使用率（模拟）"""
    return 65.3


def main():
    """主函数：脚本直接运行时执行"""
    print("=== 系统检查 ===")
    print(f"磁盘使用率: {检查磁盘()}%")
    print(f"内存使用率: {检查内存()}%")


# 关键：这个判断让脚本同时支持"直接运行"和"被导入"
if __name__ == "__main__":
    # 直接运行：python3 this_file.py -> 执行 main()
    main()

# 被导入时（from this_file import 检查磁盘）：
```

## 9. SRE 实战案例：模块化运维工具箱

```python
#!/usr/bin/env python3
"""
SRE 实战：模块化运维工具箱
功能：展示如何用函数和模块组织一个完整的运维检查工具
可直接运行：python3 sre_toolkit.py [check|report|alert]
"""

import os
import sys
import json
import socket
import shutil
from datetime import datetime


# ==========================================
# 配置区域
# ==========================================
配置 = {
    "磁盘阈值": 80,
    "检查路径": ["/", "/tmp", "/var"],
    "检查端口": [
        {"主机": "localhost", "端口": 22, "服务": "SSH"},
    ],
    "输出格式": "text",    # text 或 json
}


# ==========================================
# 检查函数
# ==========================================
def 获取系统信息():
    """获取基本系统信息"""
    return {
        "主机名": socket.gethostname(),
        "平台": sys.platform,
        "Python版本": sys.version.split()[0],
        "CPU核心": os.cpu_count(),
        "当前时间": datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
    }


def 检查磁盘(路径="/", 阈值=80):
    """检查指定路径的磁盘使用率"""
    try:
        使用情况 = shutil.disk_usage(路径)
        总量GB = round(使用情况.total / (1024**3), 1)
        已用GB = round(使用情况.used / (1024**3), 1)
        可用GB = round(使用情况.free / (1024**3), 1)
        使用率 = round(使用情况.used / 使用情况.total * 100, 1)

        return {
            "路径": 路径,
            "总量GB": 总量GB,
            "已用GB": 已用GB,
            "可用GB": 可用GB,
            "使用率": 使用率,
            "告警": 使用率 > 阈值,
            "状态": "告警" if 使用率 > 阈值 else "正常",
        }
    except OSError as e:
        return {
            "路径": 路径,
            "错误": str(e),
            "状态": "检查失败",
        }


def 检查端口(主机, 端口, 超时=3):
    """检查端口是否可达"""
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(超时)
        结果 = sock.connect_ex((主机, 端口))
        sock.close()
        开放 = (结果 == 0)
        return {
            "主机": 主机,
            "端口": 端口,
            "开放": 开放,
            "状态": "开放" if 开放 else "关闭",
        }
    except Exception as e:
        return {
            "主机": 主机,
            "端口": 端口,
            "错误": str(e),
            "状态": "检查失败",
        }


def 批量检查磁盘(路径列表, 阈值=80):
    """批量检查多个路径的磁盘使用率"""
    return [检查磁盘(路径, 阈值) for 路径 in 路径列表]


def 批量检查端口(端口列表):
    """批量检查多个端口"""
    结果 = []
    for 项 in 端口列表:
        检查结果 = 检查端口(项["主机"], 项["端口"])
        检查结果["服务"] = 项.get("服务", "未知")
        结果.append(检查结果)
    return 结果


# ==========================================
# 输出函数
# ==========================================
def 文本报告(系统信息, 磁盘结果, 端口结果):
    """生成文本格式报告"""
    分隔线 = "=" * 60
    细线 = "-" * 60

    报告 = []
    报告.append(分隔线)
    报告.append(f"{'系统健康检查报告':^50}")
    报告.append(分隔线)

    # 系统信息
    报告.append("\n【系统信息】")
    for 键, 值 in 系统信息.items():
        报告.append(f"  {键}: {值}")

    # 磁盘检查
    报告.append(f"\n【磁盘检查】（阈值: {配置['磁盘阈值']}%）")
    报告.append(f"  {'路径':<15} {'总量':>8} {'已用':>8} {'可用':>8} {'使用率':>8} {'状态':>6}")
    报告.append(f"  {细线[:56]}")
    for 项 in 磁盘结果:
        if "错误" in 项:
            报告.append(f"  {项['路径']:<15} {'检查失败':>40}")
        else:
            标记 = "!!!" if 项["告警"] else ""
            报告.append(
                f"  {项['路径']:<15} {项['总量GB']:>7}G {项['已用GB']:>7}G "
                f"{项['可用GB']:>7}G {项['使用率']:>7}% {项['状态']:>5} {标记}"
            )

    # 端口检查
    报告.append(f"\n【端口检查】")
    报告.append(f"  {'服务':<10} {'主机':<15} {'端口':>6} {'状态':>8}")
    报告.append(f"  {细线[:42]}")
    for 项 in 端口结果:
        报告.append(
            f"  {项['服务']:<10} {项['主机']:<15} {项['端口']:>6} {项['状态']:>8}"
        )

    # 汇总
    磁盘告警数 = sum(1 for d in 磁盘结果 if d.get("告警", False))
    端口关闭数 = sum(1 for p in 端口结果 if p.get("状态") != "开放")

    报告.append(f"\n{'汇总':}")
    报告.append(分隔线)
    报告.append(f"  磁盘告警: {磁盘告警数}/{len(磁盘结果)}")
    报告.append(f"  端口异常: {端口关闭数}/{len(端口结果)}")

    整体状态 = "正常" if (磁盘告警数 == 0 and 端口关闭数 == 0) else "异常"
    报告.append(f"  整体状态: {整体状态}")
    报告.append(分隔线)

    return "\n".join(报告)


def JSON报告(系统信息, 磁盘结果, 端口结果):
    """生成 JSON 格式报告"""
    数据 = {
        "检查时间": datetime.now().isoformat(),
        "系统信息": 系统信息,
        "磁盘检查": 磁盘结果,
        "端口检查": 端口结果,
    }
    return json.dumps(数据, ensure_ascii=False, indent=2)


# ==========================================
# 主入口
# ==========================================
def main():
    """主函数"""
    # 获取命令行参数
    命令 = sys.argv[1] if len(sys.argv) > 1 else "report"

    if 命令 == "check":
        # 快速检查模式
        print("执行快速检查...")
        磁盘 = 检查磁盘("/")
        print(f"磁盘 /: {磁盘['使用率']}% - {磁盘['状态']}")

    elif 命令 == "report":
        # 完整报告模式
        系统信息 = 获取系统信息()
        磁盘结果 = 批量检查磁盘(配置["检查路径"], 配置["磁盘阈值"])
        端口结果 = 批量检查端口(配置["检查端口"])

        if 配置["输出格式"] == "json":
            print(JSON报告(系统信息, 磁盘结果, 端口结果))
        else:
            print(文本报告(系统信息, 磁盘结果, 端口结果))

    elif 命令 == "json":
        # JSON 输出模式
        系统信息 = 获取系统信息()
        磁盘结果 = 批量检查磁盘(配置["检查路径"])
        端口结果 = 批量检查端口(配置["检查端口"])
        print(JSON报告(系统信息, 磁盘结果, 端口结果))

    else:
        print(f"用法: {sys.argv[0]} [check|report|json]")
        print("  check  - 快速检查")
        print("  report - 完整文本报告")
        print("  json   - JSON 格式报告")
        sys.exit(1)


if __name__ == "__main__":
    main()
```

## 10. 常见坑点与最佳实践

### 坑点 1：可变默认参数

```python
# 错误！列表是可变对象，会在所有调用之间共享
def 添加日志(消息, 日志列表=[]):
    日志列表.append(消息)
    return 日志列表

print(添加日志("error1"))    # ['error1']
print(添加日志("error2"))    # ['error1', 'error2'] — 不是预期的 ['error2']！

# 正确写法：用 None 作为默认值
def 添加日志(消息, 日志列表=None):
    if 日志列表 is None:
        日志列表 = []
    日志列表.append(消息)
    return 日志列表
```

### 坑点 2：循环导入

```python
# a.py
from b import 函数B    # a 导入 b

# b.py
from a import 函数A    # b 也导入 a -> 循环导入！

# 解决方案：
# 1. 重新组织代码，把公共部分提取到第三个模块
# 2. 把 import 放到函数内部（延迟导入）
def 某函数():
    from a import 函数A   # 延迟导入，在函数被调用时才导入
    函数A()
```

### 坑点 3：全局变量

```python
# 函数内部可以读取全局变量，但不能直接修改
计数器 = 0

def 增加():
    # 计数器 += 1    # 报错！UnboundLocalError
    global 计数器     # 需要声明 global（但不推荐使用全局变量）
    计数器 += 1

# 最佳实践：避免全局变量，用参数和返回值传递数据
def 增加(计数器):
    return 计数器 + 1
```

### 最佳实践总结

```python
# 1. 函数单一职责：每个函数只做一件事
# 2. 写 docstring：说明函数的功能、参数、返回值
# 3. 使用类型提示（Python 3.5+）
def 检查端口(主机: str, 端口: int, 超时: int = 3) -> bool:
    """检查端口是否开放"""
    ...

# 4. 每个脚本都写 if __name__ == "__main__"
# 5. 用 requirements.txt 管理第三方依赖
# 6. 避免 from xxx import *
# 7. 导入顺序：标准库 -> 第三方库 -> 本地模块
```
