## 1. 是什么 & 为什么用它

### Shell 与 Python 语法的根本区别

```bash
# Shell：一切皆字符串，命令驱动
name="nginx"
port=80                    # 其实是字符串 "80"
echo "服务: ${name}, 端口: ${port}"

# Shell 中的 "变量" 本质上都是字符串
result=$((port + 1))       # 需要特殊语法做数学运算
```

```python
# Python：强类型，对象驱动
name = "nginx"             # 字符串类型 str
port = 80                  # 整数类型 int
print(f"服务: {name}, 端口: {port}")

# Python 中变量有明确的类型
result = port + 1          # 直接做数学运算，自然且安全
```

| 对比维度         | Shell                        | Python                        |
|------------------|------------------------------|-------------------------------|
| 变量类型         | 全是字符串                   | 强类型（str/int/float/bool）  |
| 数组             | 数组 / 关联数组              | list / dict（功能更强大）     |
| 字符串处理       | sed / awk / grep / cut       | 内置方法 / 正则模块           |
| 条件判断         | if [ ]; then / fi            | if / elif / else（更直观）    |
| 循环             | for / while / do / done      | for / while（简洁）           |
| 数学运算         | $(( )) / expr / bc           | 原生支持 +  -  *  /          |

## 3. 字符串操作（对比 sed/awk/grep）

### 3.1 基础字符串方法

```python
#!/usr/bin/env python3
"""
字符串操作：Python 内置方法 vs Shell 命令
"""

日志行 = "2024-01-15 10:30:45 ERROR [nginx] 502 Bad Gateway - /api/health"

# ==========================================
# 分割字符串（对比 awk/cut）
# ==========================================
# Shell: echo "$line" | awk '{print $1}'
# Shell: echo "$line" | cut -d' ' -f1
日期 = 日志行.split()[0]          # split() 按空格分割，取第一个元素
print(f"日期: {日期}")            # 2024-01-15

# 按指定分隔符分割
路径 = "/var/log/nginx/access.log"
部分 = 路径.split("/")            # ['', 'var', 'log', 'nginx', 'access.log']
文件名 = 路径.split("/")[-1]     # access.log（取最后一个元素）

# ==========================================
# 查找与替换（对比 grep/sed）
# ==========================================
# Shell: echo "$line" | grep -q "ERROR" && echo "found"
if "ERROR" in 日志行:
    print("找到 ERROR")

# Shell: echo "$line" | sed 's/ERROR/WARNING/'
新行 = 日志行.replace("ERROR", "WARNING")
print(新行)

# 查找位置
位置 = 日志行.find("ERROR")      # 返回索引位置，找不到返回 -1
print(f"ERROR 在位置: {位置}")

# ==========================================
# 判断开头和结尾（对比 grep ^... 和 grep ...$）
# ==========================================
# Shell: echo "$line" | grep "^2024"
if 日志行.startswith("2024"):
    print("是2024年的日志")

# Shell: echo "$file" | grep "\.log$"
if 路径.endswith(".log"):
    print("是日志文件")

# ==========================================
# 大小写转换
# ==========================================
# Shell: echo "$str" | tr '[:lower:]' '[:upper:]'
print("hello".upper())           # HELLO
print("HELLO".lower())           # hello
print("hello world".title())     # Hello World

# ==========================================
# 去除空白（对比 Shell 的各种 trim 技巧）
# ==========================================
# Shell: echo "$str" | sed 's/^[ \t]*//;s/[ \t]*$//'
脏数据 = "  hello world  \n"
print(脏数据.strip())             # "hello world"（去除两端空白）
print(脏数据.lstrip())            # "hello world  \n"（只去左边）
print(脏数据.rstrip())            # "  hello world"（只去右边）

# ==========================================
# 填充与对齐（对比 printf）
# ==========================================
# Shell: printf "%-20s %5d\n" "$name" "$count"
print("nginx".ljust(20), str(80).rjust(5))
print("nginx".center(20, "-"))    # -------nginx--------
print("5".zfill(3))               # 005（前面补零）
```

### 3.2 f-string 格式化（对比 echo 变量替换）

```python
#!/usr/bin/env python3
"""
f-string：Python 3.6+ 的字符串格式化利器
"""

服务名 = "nginx"
端口 = 80
CPU = 85.732
状态 = "running"

# ==========================================
# Shell 的变量替换
# ==========================================
# echo "服务: ${name}, 端口: ${port}"
# echo "CPU 使用率: $(printf '%.1f' $cpu)%"
# printf "%-10s %5d %s\n" "$name" "$port" "$status"

# ==========================================
# Python f-string（推荐！）
# ==========================================
# 基础用法：f"..." 中 {} 内放表达式
print(f"服务: {服务名}, 端口: {端口}")

# 格式化数字
print(f"CPU 使用率: {CPU:.1f}%")         # 85.7%（保留1位小数）
print(f"CPU 使用率: {CPU:.2f}%")         # 85.73%（保留2位小数）

# 对齐和填充（类似 printf）
print(f"{'服务名':<10} {'端口':>5} {'状态':^10}")
print(f"{服务名:<10} {端口:>5} {状态:^10}")

# 千分位分隔符
连接数 = 1234567
print(f"当前连接数: {连接数:,}")          # 1,234,567

# 在 {} 内可以写表达式
print(f"端口+1: {端口 + 1}")
print(f"大写: {服务名.upper()}")
print(f"状态: {'正常' if 状态 == 'running' else '异常'}")

# 多行 f-string
报告 = f"""
========== 服务状态报告 ==========
服务名称: {服务名}
监听端口: {端口}
CPU 使用率: {CPU:.1f}%
运行状态: {状态}
==================================
"""
print(报告)
```

## 4. 列表 list（对比 Shell 数组）

```python
#!/usr/bin/env python3
"""
Python 列表 vs Shell 数组
"""

# ==========================================
# Shell 数组：
# servers=("web-01" "web-02" "web-03")
# echo ${servers[0]}           # 取第一个元素
# echo ${servers[@]}           # 取所有元素
# echo ${#servers[@]}          # 取长度
# ==========================================

# Python 列表：更强大、更灵活
服务器列表 = ["web-01", "web-02", "web-03", "db-01", "db-02"]

# 基础操作
print(f"第一台: {服务器列表[0]}")        # web-01（索引从0开始）
print(f"最后一台: {服务器列表[-1]}")     # db-02（负索引：从末尾开始）
print(f"服务器数量: {len(服务器列表)}")  # 5

# 切片操作（Shell 没有的强大功能）
print(f"前3台: {服务器列表[:3]}")        # ['web-01', 'web-02', 'web-03']
print(f"后2台: {服务器列表[-2:]}")       # ['db-01', 'db-02']
print(f"第2到4: {服务器列表[1:4]}")      # ['web-02', 'web-03', 'db-01']

# 添加元素
# Shell: servers+=("cache-01")
服务器列表.append("cache-01")            # 末尾添加
服务器列表.insert(0, "lb-01")            # 指定位置插入

# 删除元素
服务器列表.remove("db-02")               # 按值删除
被删除的 = 服务器列表.pop()              # 弹出最后一个
del 服务器列表[0]                        # 按索引删除

# 查找
if "web-01" in 服务器列表:
    print("web-01 在列表中")
    位置 = 服务器列表.index("web-01")
    print(f"位置: {位置}")

# 排序
端口列表 = [8080, 443, 80, 3306, 6379]
端口列表.sort()                           # 原地排序
print(f"排序后: {端口列表}")             # [80, 443, 3306, 6379, 8080]

# 反转
端口列表.reverse()
print(f"反转后: {端口列表}")

# 列表可以包含不同类型（但不推荐）
混合列表 = ["nginx", 80, True, 3.14]

# 列表合并
web服务器 = ["web-01", "web-02"]
db服务器 = ["db-01", "db-02"]
所有服务器 = web服务器 + db服务器         # 合并两个列表
print(f"所有服务器: {所有服务器}")
```

## 5. 字典 dict（对比关联数组）

```python
#!/usr/bin/env python3
"""
Python 字典 vs Shell 关联数组
"""

# ==========================================
# Shell 关联数组（Bash 4.0+）：
# declare -A server
# server[name]="web-01"
# server[ip]="192.168.1.10"
# server[port]=80
# echo ${server[name]}
# ==========================================

# Python 字典：键值对存储，功能强大
服务器 = {
    "名称": "web-01",
    "IP": "192.168.1.10",
    "端口": 80,
    "状态": "running",
    "标签": ["production", "web"],     # 值可以是列表
}

# 访问值
print(f"服务器名: {服务器['名称']}")
print(f"IP: {服务器['IP']}")

# 安全访问（键不存在时不报错）
# Shell: ${server[region]:-"未知"}
区域 = 服务器.get("区域", "未知")       # 不存在返回默认值 "未知"
print(f"区域: {区域}")

# 添加/修改
服务器["区域"] = "华东"                  # 添加新键
服务器["状态"] = "stopped"               # 修改已有键

# 删除
del 服务器["标签"]                       # 删除指定键
弹出值 = 服务器.pop("区域", None)       # 弹出并返回值

# 遍历字典
print("\n--- 服务器信息 ---")
for 键, 值 in 服务器.items():
    print(f"  {键}: {值}")

# 只遍历键
for 键 in 服务器.keys():
    print(键)

# 只遍历值
for 值 in 服务器.values():
    print(值)

# 检查键是否存在
# Shell: if [[ -v server[name] ]]; then ...
if "名称" in 服务器:
    print("名称 存在")

# 嵌套字典（用于复杂配置）
集群配置 = {
    "web-01": {"ip": "10.0.1.1", "cpu": 4, "memory": 8},
    "web-02": {"ip": "10.0.1.2", "cpu": 4, "memory": 8},
    "db-01":  {"ip": "10.0.2.1", "cpu": 8, "memory": 32},
}

# 访问嵌套值
print(f"web-01 的 IP: {集群配置['web-01']['ip']}")
print(f"db-01 的内存: {集群配置['db-01']['memory']}GB")

# 遍历嵌套字典
for 主机名, 配置 in 集群配置.items():
    print(f"{主机名}: IP={配置['ip']}, CPU={配置['cpu']}核, 内存={配置['memory']}GB")
```

## 6. 条件判断（对比 Shell if）

```python
#!/usr/bin/env python3
"""
Python 条件判断 vs Shell if
"""

# ==========================================
# Shell 条件判断：
# if [ "$status" = "running" ]; then
#     echo "服务正常"
# elif [ "$status" = "stopped" ]; then
#     echo "服务已停止"
# else
#     echo "状态未知"
# fi
# ==========================================

# Python 条件判断：更直观
状态 = "running"

if 状态 == "running":
    print("服务正常")
elif 状态 == "stopped":
    print("服务已停止")
else:
    print("状态未知")

# ==========================================
# 数值比较
# ==========================================
# Shell: if [ "$cpu" -gt 80 ]; then
CPU使用率 = 85

if CPU使用率 > 90:
    print("严重告警：CPU > 90%")
elif CPU使用率 > 80:
    print("警告：CPU > 80%")
elif CPU使用率 > 60:
    print("注意：CPU > 60%")
else:
    print("正常")

# ==========================================
# 逻辑运算符
# ==========================================
# Shell: if [ "$cpu" -gt 80 ] && [ "$mem" -gt 80 ]; then
CPU = 85
内存 = 75
磁盘 = 92

# and / or / not（对比 Shell 的 && / || / !）
if CPU > 80 and 内存 > 80:
    print("CPU 和内存都高")

if CPU > 90 or 磁盘 > 90:
    print("CPU 或磁盘使用率过高")

if not 磁盘 > 95:
    print("磁盘还没到极限")

# ==========================================
# 字符串判断
# ==========================================
# Shell: if [ -z "$str" ]; then echo "空字符串"
# Shell: if [ -n "$str" ]; then echo "非空"
文本 = ""
if not 文本:                 # 空字符串为 False
    print("文本为空")

文本 = "hello"
if 文本:                     # 非空字符串为 True
    print("文本非空")

# ==========================================
# in 运算符（Shell 没有直接对应的写法）
# ==========================================
允许的状态 = ["running", "starting", "reloading"]
当前状态 = "running"

if 当前状态 in 允许的状态:
    print(f"{当前状态} 是允许的状态")

# ==========================================
# 三元表达式（对比 Shell: result=$([ cond ] && echo "a" || echo "b")）
# ==========================================
状态 = "running"
状态文本 = "正常" if 状态 == "running" else "异常"
print(f"状态: {状态文本}")
```

## 7. 循环（对比 Shell for/while）

```python
#!/usr/bin/env python3
"""
Python 循环 vs Shell 循环
"""

# ==========================================
# for 循环
# ==========================================

# Shell:
# for server in web-01 web-02 web-03; do
#     echo "检查 $server"
# done

# Python:
服务器列表 = ["web-01", "web-02", "web-03"]
for 服务器 in 服务器列表:
    print(f"检查 {服务器}")

# ==========================================
# 带索引的循环（enumerate）
# ==========================================
# Shell:
# i=0
# for server in "${servers[@]}"; do
#     echo "$i: $server"
#     ((i++))
# done

# Python（使用 enumerate，更优雅）:
for 索引, 服务器 in enumerate(服务器列表):
    print(f"{索引}: {服务器}")

# 从 1 开始编号
for 序号, 服务器 in enumerate(服务器列表, start=1):
    print(f"第{序号}台: {服务器}")

# ==========================================
# 数值范围循环
# ==========================================
# Shell: for i in $(seq 1 5); do echo $i; done
# Shell: for ((i=1; i<=5; i++)); do echo $i; done

# Python:
for i in range(1, 6):        # range(start, stop)，不包含 stop
    print(f"第 {i} 次检查")

# range 的其他用法
for i in range(5):            # 0, 1, 2, 3, 4
    pass

for i in range(0, 100, 10):  # 0, 10, 20, ..., 90（步长10）
    print(f"进度: {i}%")

# ==========================================
# while 循环
# ==========================================
# Shell:
# count=0
# while [ $count -lt 5 ]; do
#     echo "尝试 $count"
#     ((count++))
# done

# Python:
尝试次数 = 0
最大重试 = 5
while 尝试次数 < 最大重试:
    print(f"尝试第 {尝试次数 + 1} 次")
    尝试次数 += 1

# ==========================================
# break 和 continue
# ==========================================
# 和 Shell 中一样，break 退出循环，continue 跳过本次

端口列表 = [80, 443, 8080, 3306, 6379]
for 端口 in 端口列表:
    if 端口 == 3306:
        print(f"跳过数据库端口 {端口}")
        continue
    if 端口 > 8000:
        print("不再检查更大的端口")
        break
    print(f"检查端口: {端口}")

# ==========================================
# 遍历字典
# ==========================================
# Shell 关联数组遍历：
# for key in "${!server[@]}"; do
#     echo "$key: ${server[$key]}"
# done

服务器状态 = {"web-01": "running", "web-02": "stopped", "db-01": "running"}

for 主机, 状态 in 服务器状态.items():
    print(f"{主机}: {状态}")

# ==========================================
# 同时遍历多个列表（zip）
# ==========================================
主机名 = ["web-01", "web-02", "web-03"]
IP列表 = ["10.0.1.1", "10.0.1.2", "10.0.1.3"]

for 主机, ip in zip(主机名, IP列表):
    print(f"{主机} -> {ip}")
```

## 8. 列表推导式

```python
#!/usr/bin/env python3
"""
列表推导式：Python 的一大特色，Shell 中没有直接对应的语法
"""

# ==========================================
# 基础列表推导式
# ==========================================

# 传统写法
端口列表 = []
for i in range(8080, 8086):
    端口列表.append(i)

# 列表推导式（一行搞定）
端口列表 = [i for i in range(8080, 8086)]
print(f"端口列表: {端口列表}")   # [8080, 8081, 8082, 8083, 8084, 8085]

# ==========================================
# 带条件过滤
# ==========================================
# Shell: echo "${servers[@]}" | tr ' ' '\n' | grep "^web"

服务器列表 = ["web-01", "web-02", "db-01", "cache-01", "web-03"]

# 只保留 web 服务器
web服务器 = [s for s in 服务器列表 if s.startswith("web")]
print(f"Web 服务器: {web服务器}")   # ['web-01', 'web-02', 'web-03']

# ==========================================
# 带转换
# ==========================================
# 把所有主机名转大写
大写列表 = [s.upper() for s in 服务器列表]
print(f"大写: {大写列表}")

# 生成 IP 列表
IP列表 = [f"192.168.1.{i}" for i in range(1, 11)]
print(f"IP 列表: {IP列表}")

# ==========================================
# 字典推导式
# ==========================================
# 从两个列表构建字典
主机名列表 = ["web-01", "web-02", "web-03"]
IP列表 = ["10.0.1.1", "10.0.1.2", "10.0.1.3"]

主机IP映射 = {主机: ip for 主机, ip in zip(主机名列表, IP列表)}
print(f"映射: {主机IP映射}")

# 过滤字典
服务器状态 = {"web-01": "running", "web-02": "stopped", "db-01": "running"}
运行中 = {k: v for k, v in 服务器状态.items() if v == "running"}
print(f"运行中的服务器: {运行中}")

# ==========================================
# 集合推导式
# ==========================================
日志级别 = ["ERROR", "WARNING", "ERROR", "INFO", "ERROR", "WARNING"]
唯一级别 = {级别 for 级别 in 日志级别}
print(f"唯一的日志级别: {唯一级别}")  # {'ERROR', 'WARNING', 'INFO'}
```

## 9. SRE 实战案例：服务器批量检查脚本

```python
#!/usr/bin/env python3
"""
SRE 实战：服务器状态批量检查
功能：检查多台服务器的服务状态，生成状态报告
展示：综合运用变量、列表、字典、循环、条件判断、f-string
"""

import random  # 模拟数据用

# ==========================================
# 服务器配置（实际中可从配置文件读取）
# ==========================================
服务器清单 = [
    {"主机名": "web-01",   "IP": "10.0.1.1", "角色": "web",   "端口": 80},
    {"主机名": "web-02",   "IP": "10.0.1.2", "角色": "web",   "端口": 80},
    {"主机名": "web-03",   "IP": "10.0.1.3", "角色": "web",   "端口": 80},
    {"主机名": "api-01",   "IP": "10.0.2.1", "角色": "api",   "端口": 8080},
    {"主机名": "api-02",   "IP": "10.0.2.2", "角色": "api",   "端口": 8080},
    {"主机名": "db-01",    "IP": "10.0.3.1", "角色": "db",    "端口": 3306},
    {"主机名": "db-02",    "IP": "10.0.3.2", "角色": "db",    "端口": 3306},
    {"主机名": "cache-01", "IP": "10.0.4.1", "角色": "cache", "端口": 6379},
]


def 模拟检查(服务器):
    """模拟服务器检查（实际中应该用 SSH 或 API 调用）"""
    return {
        "状态": random.choice(["running", "running", "running", "stopped"]),
        "CPU": round(random.uniform(10, 95), 1),
        "内存": round(random.uniform(20, 90), 1),
        "磁盘": round(random.uniform(30, 95), 1),
    }


def 生成报告(检查结果):
    """根据检查结果生成状态报告"""

    # 分隔线
    分隔线 = "=" * 80
    细线 = "-" * 80

    print(f"\n{分隔线}")
    print(f"{'服务器状态检查报告':^70}")
    print(f"{分隔线}")

    # 表头
    print(f"{'主机名':<12} {'IP':<15} {'角色':<8} {'状态':<10} "
          f"{'CPU%':>6} {'内存%':>6} {'磁盘%':>6} {'告警':>6}")
    print(细线)

    # 统计变量
    总数 = len(检查结果)
    正常数 = 0
    告警列表 = []

    # 遍历每台服务器的检查结果
    for 服务器, 指标 in 检查结果.items():
        # 判断是否需要告警
        告警标记 = []
        if 指标["状态"] != "running":
            告警标记.append("宕机")
        if 指标["CPU"] > 80:
            告警标记.append("CPU高")
        if 指标["内存"] > 85:
            告警标记.append("内存高")
        if 指标["磁盘"] > 90:
            告警标记.append("磁盘满")

        告警文本 = ",".join(告警标记) if 告警标记 else "正常"

        if not 告警标记:
            正常数 += 1
        else:
            告警列表.append(f"  - {服务器}: {告警文本}")

        # 获取服务器基本信息
        基本信息 = next(s for s in 服务器清单 if s["主机名"] == 服务器)

        # 输出一行
        print(f"{服务器:<12} {基本信息['IP']:<15} {基本信息['角色']:<8} "
              f"{指标['状态']:<10} {指标['CPU']:>6.1f} {指标['内存']:>6.1f} "
              f"{指标['磁盘']:>6.1f} {告警文本:>6}")

    # 汇总信息
    print(细线)
    print(f"总计: {总数} 台服务器, {正常数} 台正常, {总数 - 正常数} 台告警")

    if 告警列表:
        print(f"\n{'告警详情':}")
        for 告警 in 告警列表:
            print(告警)

    # 按角色统计
    角色统计 = {}
    for 服务器 in 服务器清单:
        角色 = 服务器["角色"]
        角色统计[角色] = 角色统计.get(角色, 0) + 1

    print(f"\n按角色分布:")
    for 角色, 数量 in 角色统计.items():
        print(f"  {角色}: {数量} 台")

    print(分隔线)


def main():
    """主函数"""
    print("开始检查服务器状态...")

    # 执行检查
    检查结果 = {}
    for 服务器 in 服务器清单:
        主机名 = 服务器["主机名"]
        检查结果[主机名] = 模拟检查(服务器)

    # 生成报告
    生成报告(检查结果)


if __name__ == "__main__":
    main()
```

## 10. 常见坑点与最佳实践

### 坑点 1：缩进

```python
# Python 用缩进表示代码块（Shell 用 fi/done/esac）
# 缩进错误会直接报 IndentationError

# 错误：混用空格和 Tab
if True:
    print("用4个空格")   # 正确
	print("用Tab")       # 错误！不要混用

# 最佳实践：始终使用 4 个空格（在编辑器中配置 Tab = 4空格）
```

### 坑点 2：== 与 is

```python
# == 比较值，is 比较对象身份（内存地址）
a = [1, 2, 3]
b = [1, 2, 3]
print(a == b)    # True（值相同）
print(a is b)    # False（不是同一个对象）

# 只有和 None 比较时使用 is
if 值 is None:
    print("值为 None")
```

### 坑点 3：可变对象作为默认参数

```python
# 错误！列表是可变对象，会在调用之间共享
def 添加服务器(名称, 列表=[]):
    列表.append(名称)
    return 列表

# 正确写法
def 添加服务器(名称, 列表=None):
    if 列表 is None:
        列表 = []
    列表.append(名称)
    return 列表
```

## 11. 速查表

```python
# =====================================================
# Python 基础语法速查表（SRE 运维视角）
# =====================================================

# --- 变量与类型 ---
s = "hello"              # 字符串
n = 42                   # 整数
f = 3.14                 # 浮点数
b = True                 # 布尔值
none = None              # 空值
type(s)                  # 查看类型 -> <class 'str'>
int("42")                # 字符串 -> 整数
str(42)                  # 整数 -> 字符串

# --- 字符串 ---
s.split("/")             # 分割     (对比 cut -d'/')
s.replace("a", "b")      # 替换     (对比 sed 's/a/b/')
s.startswith("http")     # 判断开头 (对比 grep ^http)
s.endswith(".log")       # 判断结尾 (对比 grep '\.log$')
s.strip()                # 去空白   (对比 sed trim)
s.upper() / s.lower()   # 大小写   (对比 tr)
s.find("sub")            # 查找     (对比 grep -o)
f"值={变量}"              # 格式化   (对比 echo "$var")
"x" in s                 # 包含     (对比 grep -q)

# --- 列表 ---
lst = [1, 2, 3]          # 创建     (对比 arr=(1 2 3))
lst[0]                   # 取值     (对比 ${arr[0]})
lst[-1]                  # 最后一个
lst[1:3]                 # 切片
lst.append(4)            # 追加     (对比 arr+=(4))
lst.remove(2)            # 删除
len(lst)                 # 长度     (对比 ${#arr[@]})
sorted(lst)              # 排序     (对比 sort)

# --- 字典 ---
d = {"key": "value"}     # 创建     (对比 declare -A)
d["key"]                 # 取值     (对比 ${d[key]})
d.get("key", "默认")     # 安全取值 (对比 ${d[key]:-默认})
d["new"] = "val"         # 添加
d.items()                # 遍历键值对
"key" in d               # 判断键存在

# --- 条件 ---
if x > 0:               # (对比 if [ $x -gt 0 ]; then)
elif x == 0:             # (对比 elif [ $x -eq 0 ]; then)
else:                    # (对比 else)
# 无需 fi！用缩进控制范围

# --- 循环 ---
for x in 列表:           # (对比 for x in ${arr[@]}; do)
for i in range(10):      # (对比 for i in $(seq 0 9); do)
for i, x in enumerate(列表):  # 带索引遍历
while 条件:              # (对比 while [ 条件 ]; do)
break / continue         # 和 Shell 一样

# --- 列表推导式 ---
[x for x in 列表]                    # 基础
[x for x in 列表 if 条件]            # 带过滤
[f(x) for x in 列表]                 # 带转换
{k: v for k, v in 字典.items()}      # 字典推导式
```
