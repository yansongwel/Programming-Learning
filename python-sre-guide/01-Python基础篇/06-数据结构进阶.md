## 1. 是什么 & 为什么用它

### Shell 数据处理的局限

```bash
# Shell 处理复杂数据非常痛苦
# 统计 Nginx 日志中各状态码出现次数：
awk '{print $9}' access.log | sort | uniq -c | sort -rn
# 问题：
# 1. 中间结果全是字符串，需要反复 sort/uniq/awk
# 2. 无法方便地做进一步分析（如按时间窗口统计）
# 3. 数据结构单一，只有字符串和数组
```

### Python 的优势

```python
# Python 一行搞定，且可以进一步分析
from collections import Counter
with open("access.log") as f:
    状态码统计 = Counter(line.split()[8] for line in f if len(line.split()) > 8)
print(状态码统计.most_common(10))  # 获取 Top 10 状态码
```

| 场景             | Shell 方式                      | Python 方式                    |
|------------------|---------------------------------|-------------------------------|
| 统计频次         | sort \| uniq -c                 | Counter                       |
| 去重             | sort -u                         | set                           |
| 键值映射         | declare -A (Bash 4+)           | dict / defaultdict            |
| 有序数据         | 无原生支持                      | OrderedDict / 排序            |
| 队列/栈          | 无原生支持                      | deque                         |
| 结构化记录       | 无                              | namedtuple / dataclass        |
| 集合运算         | comm / diff                     | set 交集/并集/差集            |

```python
#!/usr/bin/env python3
"""
列表切片：比 Shell 数组操作强大得多

日志行列表 = [
    "2024-01-15 10:00:01 INFO 请求处理成功",
    "2024-01-15 10:00:02 ERROR 数据库连接超时",
    "2024-01-15 10:00:03 INFO 健康检查通过",
    "2024-01-15 10:00:04 WARNING 内存使用率 85%",
    "2024-01-15 10:00:05 INFO 请求处理成功",
    "2024-01-15 10:00:06 ERROR SSL 证书即将过期",
    "2024-01-15 10:00:07 INFO 配置重载完成",
    "2024-01-15 10:00:08 WARNING 磁盘 IO 偏高",
    "2024-01-15 10:00:09 INFO 连接池正常",
    "2024-01-15 10:00:10 CRITICAL 主数据库宕机",
]

# 基础切片 [start:stop:step]
print("前3条:", 日志行列表[:3])              # 前3个元素
print("后3条:", 日志行列表[-3:])             # 最后3个元素
print("第3到7条:", 日志行列表[2:7])          # 索引2到6
print("偶数位置:", 日志行列表[::2])          # 每隔一个取一个
print("反转列表:", 日志行列表[::-1])         # 完全反转

# 切片赋值（批量替换）
副本 = 日志行列表[:]                         # 浅拷贝整个列表
副本[0:2] = ["替换行1", "替换行2"]          # 替换前两个元素

# 用切片实现分页
def 分页(数据, 页码, 每页数量=3):
    """模拟分页查询"""
    开始 = (页码 - 1) * 每页数量
    结束 = 开始 + 每页数量
    return 数据[开始:结束]

print(f"\n第1页: {分页(日志行列表, 1)}")
print(f"第2页: {分页(日志行列表, 2)}")
```

### 2.2 排序

```python
#!/usr/bin/env python3
"""
列表排序：灵活的多条件排序
"""

# 服务器列表
服务器列表 = [
    {"名称": "web-01",   "CPU": 85, "内存": 60, "角色": "web"},
    {"名称": "db-01",    "CPU": 45, "内存": 90, "角色": "db"},
    {"名称": "web-02",   "CPU": 92, "内存": 75, "角色": "web"},
    {"名称": "api-01",   "CPU": 30, "内存": 40, "角色": "api"},
    {"名称": "web-03",   "CPU": 85, "内存": 88, "角色": "web"},
    {"名称": "cache-01", "CPU": 55, "内存": 65, "角色": "cache"},
]

# ==========================================
# 单条件排序
# ==========================================
# 对比 Shell: sort -t'|' -k2 -nr
按CPU排序 = sorted(服务器列表, key=lambda s: s["CPU"], reverse=True)
print("按 CPU 降序:")
for s in 按CPU排序:
    print(f"  {s['名称']:<10} CPU: {s['CPU']}%")

# ==========================================
# 多条件排序（先按 CPU 降序，CPU 相同时按内存降序）
# ==========================================
# 对比 Shell: sort -t'|' -k2 -nr -k3 -nr
多条件排序 = sorted(
    服务器列表,
    key=lambda s: (-s["CPU"], -s["内存"])   # 负号表示降序
)
print("\n按 CPU 降序 -> 内存降序:")
for s in 多条件排序:
    print(f"  {s['名称']:<10} CPU: {s['CPU']}%, 内存: {s['内存']}%")

# ==========================================
# 按字符串排序
# ==========================================
名称列表 = ["web-10", "web-2", "web-1", "web-20", "web-3"]

# 字典序排序（不符合预期）
print(f"\n字典序: {sorted(名称列表)}")
# ['web-1', 'web-10', 'web-2', 'web-20', 'web-3']

# 自然排序（按数字大小）
import re
def 自然排序键(s):
    """提取字符串中的数字部分用于排序"""
    return [int(x) if x.isdigit() else x for x in re.split(r'(\d+)', s)]

print(f"自然序: {sorted(名称列表, key=自然排序键)}")
# ['web-1', 'web-2', 'web-3', 'web-10', 'web-20']
```

### 2.3 filter 和 map

```python
#!/usr/bin/env python3
"""
filter 和 map：函数式数据处理
"""

# 端口列表
端口列表 = [22, 80, 443, 3306, 5432, 6379, 8080, 8443, 9090, 27017]

# ==========================================
# filter：过滤数据（对比 grep）
# ==========================================
# Shell: echo "${ports[@]}" | tr ' ' '\n' | awk '$1 > 1000'

# 方式1：filter 函数
大端口 = list(filter(lambda p: p > 1000, 端口列表))
print(f"大于1000的端口: {大端口}")

# 方式2：列表推导式（更推荐！）
大端口 = [p for p in 端口列表 if p > 1000]
print(f"大于1000的端口: {大端口}")

# ==========================================
# map：转换数据（对比 awk '{print ...}'）
# ==========================================
# Shell: for port in "${ports[@]}"; do echo "端口: $port"; done

# 方式1：map 函数
端口描述 = list(map(lambda p: f":{p}", 端口列表))
print(f"端口格式化: {端口描述}")

# 方式2：列表推导式（更推荐！）
端口描述 = [f":{p}" for p in 端口列表]

# ==========================================
# 组合使用：过滤 + 转换
# ==========================================
服务器数据 = [
    {"名称": "web-01", "CPU": 85, "状态": "running"},
    {"名称": "web-02", "CPU": 92, "状态": "running"},
    {"名称": "db-01",  "CPU": 45, "状态": "stopped"},
    {"名称": "web-03", "CPU": 78, "状态": "running"},
]

# 找出所有运行中且 CPU > 80% 的服务器名称
告警服务器 = [
    s["名称"]
    for s in 服务器数据
    if s["状态"] == "running" and s["CPU"] > 80
]
print(f"需要告警的服务器: {告警服务器}")

# ==========================================
# any() 和 all()：批量判断
# ==========================================
# 是否有任何服务器 CPU > 90%（对比 grep -q）
有高负载 = any(s["CPU"] > 90 for s in 服务器数据)
print(f"存在高负载服务器: {有高负载}")

# 是否所有服务器都在运行
全部运行 = all(s["状态"] == "running" for s in 服务器数据)
print(f"全部运行中: {全部运行}")
```

```python
#!/usr/bin/env python3
"""
字典高级用法：SRE 数据处理的核心
"""

# ==========================================
# 字典合并
# ==========================================
默认配置 = {"端口": 80, "workers": 4, "日志级别": "INFO"}
自定义配置 = {"端口": 8080, "日志级别": "DEBUG", "超时": 30}

# Python 3.9+ 合并运算符
# 合并配置 = 默认配置 | 自定义配置

# Python 3.5+ 解包方式（兼容性好）
合并配置 = {**默认配置, **自定义配置}    # 后面的覆盖前面的
print(f"合并配置: {合并配置}")
# {'端口': 8080, 'workers': 4, '日志级别': 'DEBUG', '超时': 30}

# ==========================================
# 字典推导式
# ==========================================
# 从列表构建字典
主机列表 = ["web-01", "web-02", "web-03"]
主机状态 = {主机: "unknown" for 主机 in 主机列表}
print(f"初始状态: {主机状态}")

# 过滤字典
所有指标 = {"CPU": 85, "内存": 60, "磁盘": 92, "网络": 30, "IO": 75}
告警指标 = {k: v for k, v in 所有指标.items() if v > 80}
print(f"告警指标: {告警指标}")

# 反转字典（键值互换）
端口服务 = {80: "http", 443: "https", 22: "ssh", 3306: "mysql"}
服务端口 = {v: k for k, v in 端口服务.items()}
print(f"服务到端口: {服务端口}")

# ==========================================
# setdefault：获取或设置默认值
# ==========================================
访问统计 = {}
日志行 = ["web-01", "web-02", "web-01", "web-03", "web-01", "web-02"]

for 主机 in 日志行:
    访问统计.setdefault(主机, 0)
    访问统计[主机] += 1

print(f"访问统计: {访问统计}")

# ==========================================
# 字典排序
# ==========================================
# 按值排序（获取 CPU 最高的服务器）
CPU数据 = {"web-01": 85, "web-02": 92, "db-01": 45, "api-01": 78}

按CPU排序 = dict(sorted(CPU数据.items(), key=lambda x: x[1], reverse=True))
print(f"CPU 排序: {按CPU排序}")

# 获取最大/最小值
最高CPU主机 = max(CPU数据, key=CPU数据.get)
最低CPU主机 = min(CPU数据, key=CPU数据.get)
print(f"CPU 最高: {最高CPU主机} ({CPU数据[最高CPU主机]}%)")
print(f"CPU 最低: {最低CPU主机} ({CPU数据[最低CPU主机]}%)")

# ==========================================
# 嵌套字典的安全访问
# ==========================================
配置 = {
    "server": {
        "nginx": {
            "port": 80,
            "workers": 4
        }
    }
}

# 不安全的方式（KeyError 风险）
# port = 配置["server"]["apache"]["port"]    # KeyError!

# 安全的逐级获取
def 安全获取(字典, *键, 默认值=None):
    """安全地从嵌套字典中获取值"""
    当前 = 字典
    for 键名 in 键:
        if isinstance(当前, dict):
            当前 = 当前.get(键名, 默认值)
        else:
            return 默认值
    return 当前

print(f"nginx 端口: {安全获取(配置, 'server', 'nginx', 'port')}")          # 80
print(f"apache 端口: {安全获取(配置, 'server', 'apache', 'port', 默认值=8080)}")  # 8080
```

## 4. set 集合运算

```python
#!/usr/bin/env python3
"""
集合操作：高效处理去重、对比、查找差异
对比 Shell 的 comm、diff、sort -u
"""

# ==========================================
# 基础集合操作
# ==========================================
# Shell: sort file | uniq   -> Python: set(列表)
IP列表 = ["10.0.1.1", "10.0.1.2", "10.0.1.1", "10.0.1.3", "10.0.1.2"]
唯一IP = set(IP列表)
print(f"去重后: {唯一IP}")
print(f"原始 {len(IP列表)} 个, 去重后 {len(唯一IP)} 个")

# ==========================================
# 集合运算（SRE 超实用！）
# ==========================================
# 场景：对比两批服务器列表

线上服务器 = {"web-01", "web-02", "web-03", "db-01", "db-02", "cache-01"}
CMDB记录 = {"web-01", "web-02", "web-04", "db-01", "db-03", "cache-01"}

# 交集：两边都有的（正常匹配的服务器）
# 对比 Shell: comm -12 <(sort file1) <(sort file2)
匹配的 = 线上服务器 & CMDB记录                # 或 线上服务器.intersection(CMDB记录)
print(f"\n两边都有: {匹配的}")

# 差集：线上有但 CMDB 没有的（未登记的服务器）
# 对比 Shell: comm -23 <(sort file1) <(sort file2)
未登记 = 线上服务器 - CMDB记录                 # 或 线上服务器.difference(CMDB记录)
print(f"未登记:   {未登记}")

# 差集：CMDB 有但线上没有的（已下线的服务器）
已下线 = CMDB记录 - 线上服务器
print(f"已下线:   {已下线}")

# 并集：所有出现过的服务器
# 对比 Shell: sort -u file1 file2
所有的 = 线上服务器 | CMDB记录                 # 或 线上服务器.union(CMDB记录)
print(f"全部:     {所有的}")

# 对称差集：只在一边出现的（有差异的服务器）
有差异的 = 线上服务器 ^ CMDB记录               # 或 线上服务器.symmetric_difference(CMDB记录)
print(f"有差异:   {有差异的}")

# ==========================================
# 子集和超集判断
# ==========================================
核心服务器 = {"web-01", "db-01"}
print(f"\n核心服务器是线上的子集: {核心服务器 <= 线上服务器}")       # True
print(f"线上服务器包含所有核心: {线上服务器 >= 核心服务器}")        # True

# ==========================================
# 集合的性能优势（in 操作 O(1) vs 列表的 O(n)）
# ==========================================
# 当需要频繁查找元素是否存在时，用 set 比 list 快得多
import time

大列表 = list(range(1000000))
大集合 = set(大列表)

# 在列表中查找（慢）
开始 = time.time()
for _ in range(1000):
    999999 in 大列表
列表耗时 = time.time() - 开始

# 在集合中查找（快）
开始 = time.time()
for _ in range(1000):
    999999 in 大集合
集合耗时 = time.time() - 开始

print(f"\n列表查找耗时: {列表耗时:.4f}s")
print(f"集合查找耗时: {集合耗时:.4f}s")
print(f"集合快了 {列表耗时/集合耗时:.0f} 倍")
```

## 6. collections 模块

### 6.1 defaultdict

```python
#!/usr/bin/env python3
"""
defaultdict：带默认值的字典，省去 if key not in dict 的判断
"""

from collections import defaultdict

# ==========================================
# 普通字典的问题
# ==========================================
统计 = {}
日志级别 = ["INFO", "ERROR", "INFO", "WARNING", "ERROR", "INFO", "ERROR"]

# 普通写法：需要检查键是否存在
for 级别 in 日志级别:
    if 级别 not in 统计:
        统计[级别] = 0
    统计[级别] += 1

# ==========================================
# defaultdict 写法：简洁！
# ==========================================
统计 = defaultdict(int)          # 默认值为 int()，即 0
for 级别 in 日志级别:
    统计[级别] += 1              # 键不存在时自动创建，值为 0
print(f"日志统计: {dict(统计)}")

# ==========================================
# defaultdict(list)：自动创建空列表
# ==========================================
# 场景：按角色分组服务器
服务器列表 = [
    ("web",   "web-01"),
    ("web",   "web-02"),
    ("db",    "db-01"),
    ("web",   "web-03"),
    ("cache", "cache-01"),
    ("db",    "db-02"),
]

按角色分组 = defaultdict(list)
for 角色, 名称 in 服务器列表:
    按角色分组[角色].append(名称)

print("\n按角色分组:")
for 角色, 服务器 in 按角色分组.items():
    print(f"  {角色}: {服务器}")

# ==========================================
# defaultdict(set)：自动创建空集合
# ==========================================
# 场景：记录每个 IP 访问过的 URL（自动去重）
访问记录 = defaultdict(set)
日志数据 = [
    ("10.0.1.1", "/api/health"),
    ("10.0.1.2", "/api/login"),
    ("10.0.1.1", "/api/health"),     # 重复
    ("10.0.1.1", "/api/data"),
    ("10.0.1.2", "/api/login"),      # 重复
]

for IP, URL in 日志数据:
    访问记录[IP].add(URL)

print("\n每个 IP 访问的唯一 URL:")
for IP, URLs in 访问记录.items():
    print(f"  {IP}: {URLs}")
```

### 6.2 Counter

```python
#!/usr/bin/env python3
"""
Counter：计数器，专门用于统计频次
对比 Shell: sort | uniq -c | sort -rn
"""

from collections import Counter

# ==========================================
# 基础用法
# ==========================================
# 对比 Shell: echo "200 200 404 200 500 404 200" | tr ' ' '\n' | sort | uniq -c
状态码列表 = [200, 200, 404, 200, 500, 404, 200, 301, 200, 500, 404, 200]

统计 = Counter(状态码列表)
print(f"状态码统计: {统计}")
# Counter({200: 6, 404: 3, 500: 2, 301: 1})

# 获取最常见的 N 个
print(f"Top 3: {统计.most_common(3)}")
# [(200, 6), (404, 3), (500, 2)]

# ==========================================
# 从字符串统计字符频次
# ==========================================
文本 = "abracadabra"
字符统计 = Counter(文本)
print(f"\n字符统计: {字符统计}")
# Counter({'a': 5, 'b': 2, 'r': 2, 'c': 1, 'd': 1})

# ==========================================
# Counter 的运算
# ==========================================
上午日志 = Counter({"INFO": 100, "WARNING": 20, "ERROR": 5})
下午日志 = Counter({"INFO": 150, "WARNING": 30, "ERROR": 15})

# 相加
全天日志 = 上午日志 + 下午日志
print(f"\n全天统计: {全天日志}")
# Counter({'INFO': 250, 'WARNING': 50, 'ERROR': 20})

# 相减
差异 = 下午日志 - 上午日志
print(f"下午比上午多: {差异}")

# ==========================================
# 从可迭代对象构建 Counter
# ==========================================
# 统计单词频次
命令历史 = "ls cd ls pwd ls cd cd git ls git git status push ls"
命令统计 = Counter(命令历史.split())
print(f"\n最常用的命令: {命令统计.most_common(5)}")

# ==========================================
# 更新计数
# ==========================================
计数器 = Counter()
计数器.update(["INFO", "INFO", "ERROR"])     # 批量更新
计数器.update(["INFO", "WARNING"])           # 继续累加
计数器["ERROR"] += 1                          # 单个更新
```

### 6.3 OrderedDict

```python
#!/usr/bin/env python3
"""
OrderedDict：有序字典
注意：Python 3.7+ 的普通 dict 已经保持插入顺序
OrderedDict 的额外价值在于它支持 move_to_end() 等操作
"""

from collections import OrderedDict

# ==========================================
# 用 OrderedDict 实现 LRU 缓存（简化版）
# ==========================================
class LRU缓存:
    """最近最少使用缓存"""

    def __init__(self, 容量=100):
        self.容量 = 容量
        self.缓存 = OrderedDict()

    def 获取(self, 键):
        if 键 in self.缓存:
            self.缓存.move_to_end(键)    # 移到末尾（标记为最近使用）
            return self.缓存[键]
        return None

    def 设置(self, 键, 值):
        if 键 in self.缓存:
            self.缓存.move_to_end(键)
        self.缓存[键] = 值
        if len(self.缓存) > self.容量:
            self.缓存.popitem(last=False)  # 删除最老的（最前面的）

    def __repr__(self):
        return f"LRU缓存({dict(self.缓存)})"


# 使用示例
DNS缓存 = LRU缓存(容量=3)
DNS缓存.设置("google.com", "142.250.80.46")
DNS缓存.设置("github.com", "140.82.121.4")
DNS缓存.设置("python.org", "151.101.128.223")
print(f"缓存: {DNS缓存}")

# 再添加一个，最老的会被淘汰
DNS缓存.设置("reddit.com", "151.101.1.140")
print(f"添加后: {DNS缓存}")  # google.com 被淘汰
```

### 6.5 deque

```python
#!/usr/bin/env python3
"""
deque：双端队列，高效的两端操作
"""

from collections import deque
import time

# ==========================================
# deque vs list 性能对比
# ==========================================
# list 在头部插入是 O(n)，deque 是 O(1)

# list 头部插入（慢）
大列表 = []
开始 = time.time()
for i in range(100000):
    大列表.insert(0, i)           # 每次都要移动所有元素
列表耗时 = time.time() - 开始

# deque 头部插入（快）
大队列 = deque()
开始 = time.time()
for i in range(100000):
    大队列.appendleft(i)          # O(1) 操作
队列耗时 = time.time() - 开始

print(f"list 头部插入 10万次: {列表耗时:.3f}s")
print(f"deque 头部插入 10万次: {队列耗时:.3f}s")
print(f"deque 快了 {列表耗时/队列耗时:.0f} 倍")

# ==========================================
# deque 的基本操作
# ==========================================
队列 = deque()
队列.append("任务1")              # 右端添加
队列.append("任务2")
队列.appendleft("紧急任务")       # 左端添加
print(f"\n队列: {队列}")

左边 = 队列.popleft()             # 从左端取出
print(f"取出: {左边}, 剩余: {队列}")

# ==========================================
# 固定大小的 deque（自动淘汰旧数据）
# ==========================================
# 场景：只保留最近 5 条日志
最近日志 = deque(maxlen=5)

for i in range(10):
    最近日志.append(f"日志 #{i+1}")

print(f"\n最近5条日志: {list(最近日志)}")
# 只保留了最后5条，前5条自动被淘汰

# ==========================================
# 实用场景：滑动窗口（计算移动平均值）
# ==========================================
def 移动平均(数据, 窗口大小=5):
    """计算移动平均值（用于平滑监控指标）"""
    窗口 = deque(maxlen=窗口大小)
    结果 = []
    for 值 in 数据:
        窗口.append(值)
        平均值 = sum(窗口) / len(窗口)
        结果.append(round(平均值, 1))
    return 结果

# CPU 监控数据（每分钟一个点）
CPU数据 = [60, 65, 80, 92, 85, 70, 65, 88, 95, 78, 62, 58]
平滑数据 = 移动平均(CPU数据, 窗口大小=3)
```

## 7. SRE 实战案例：用 Counter 分析 Nginx 日志

```python
#!/usr/bin/env python3
"""
SRE 实战：Nginx 日志分析工具
功能：分析 Nginx access.log，统计状态码、Top IP、Top URL、请求速率等
用法：python3 nginx_analyzer.py [日志文件路径]

模拟了一份 Nginx 日志数据用于演示
"""

import sys
import re
from collections import Counter, defaultdict
from datetime import datetime


# ==========================================
# 日志解析
# ==========================================
# Nginx 默认日志格式 (combined):
# 10.0.1.1 - - [15/Jan/2024:10:30:45 +0800] "GET /api/health HTTP/1.1" 200 15 "-" "curl/7.68"

NGINX日志正则 = re.compile(
    r'(?P<ip>\S+)\s+'              # IP 地址
    r'\S+\s+'                       # ident
    r'\S+\s+'                       # user
    r'\[(?P<time>[^\]]+)\]\s+'     # 时间
    r'"(?P<method>\S+)\s+'          # HTTP 方法
    r'(?P<url>\S+)\s+'              # URL
    r'\S+"\s+'                      # HTTP 版本
    r'(?P<status>\d+)\s+'          # 状态码
    r'(?P<size>\d+)'               # 响应大小
)


def 解析日志行(行):
    """解析单行 Nginx 日志"""
    匹配 = NGINX日志正则.match(行)
    if 匹配:
        return 匹配.groupdict()
    return None


def 生成模拟日志():
    """生成模拟 Nginx 日志数据"""
    import random

    IP列表 = [
        "10.0.1.100", "10.0.1.101", "10.0.1.102",
        "192.168.1.50", "192.168.1.51",
        "172.16.0.10", "172.16.0.11", "172.16.0.12",
        "10.0.2.200", "10.0.2.201",
    ]

    URL列表 = [
        "/api/health", "/api/users", "/api/login", "/api/data",
        "/api/orders", "/api/products", "/static/js/app.js",
        "/static/css/style.css", "/index.html", "/favicon.ico",
        "/api/search", "/api/upload", "/api/config",
    ]

    方法列表 = ["GET", "GET", "GET", "GET", "POST", "POST", "PUT", "DELETE"]
    状态码列表 = [200, 200, 200, 200, 200, 200, 200, 301, 302, 304,
                404, 404, 500, 502, 503]

    日志行列表 = []
    基准时间 = datetime(2024, 1, 15, 10, 0, 0)

    for i in range(500):
        ip = random.choice(IP列表)
        # 让某些 IP 出现更多（模拟真实场景）
        if random.random() < 0.3:
            ip = "10.0.1.100"

        url = random.choice(URL列表)
        method = random.choice(方法列表)
        status = random.choice(状态码列表)
        size = random.randint(100, 50000)

        秒偏移 = i * 2 + random.randint(0, 3)
        时间对象 = datetime(2024, 1, 15, 10, 秒偏移 // 60, 秒偏移 % 60)
        时间字符串 = 时间对象.strftime("%d/%b/%Y:%H:%M:%S +0800")

        行 = (f'{ip} - - [{时间字符串}] '
              f'"{method} {url} HTTP/1.1" {status} {size} '
              f'"-" "Mozilla/5.0"')
        日志行列表.append(行)

    return 日志行列表


def 分析日志(日志行列表):
    """分析 Nginx 日志，生成统计报告"""

    # 统计器
    状态码统计 = Counter()
    IP统计 = Counter()
    URL统计 = Counter()
    方法统计 = Counter()
    错误URL统计 = Counter()
    IP_URL映射 = defaultdict(Counter)
    每分钟请求 = Counter()
    总流量 = 0
    总请求数 = 0
    解析失败 = 0

    # 逐行解析和统计
    for 行 in 日志行列表:
        条目 = 解析日志行(行)
        if 条目 is None:
            解析失败 += 1
            continue

        总请求数 += 1
        总流量 += int(条目["size"])

        状态码统计[条目["status"]] += 1
        IP统计[条目["ip"]] += 1
        URL统计[条目["url"]] += 1
        方法统计[条目["method"]] += 1
        IP_URL映射[条目["ip"]][条目["url"]] += 1

        # 统计错误请求的 URL
        if 条目["status"].startswith(("4", "5")):
            错误URL统计[f'{条目["status"]} {条目["url"]}'] += 1

        # 按分钟统计（用于计算 QPS）
        try:
            时间部分 = 条目["time"].split()[0]
            分钟 = ":".join(时间部分.split(":")[:2])
            每分钟请求[分钟] += 1
        except (IndexError, ValueError):
            pass

    return {
        "总请求数": 总请求数,
        "解析失败": 解析失败,
        "总流量": 总流量,
        "状态码统计": 状态码统计,
        "IP统计": IP统计,
        "URL统计": URL统计,
        "方法统计": 方法统计,
        "错误URL统计": 错误URL统计,
        "IP_URL映射": IP_URL映射,
        "每分钟请求": 每分钟请求,
    }


def 生成报告(统计数据):
    """生成分析报告"""
    分隔线 = "=" * 65
    细线 = "-" * 65

    print(f"\n{分隔线}")
    print(f"{'Nginx 日志分析报告':^55}")
    print(f"{分隔线}")

    # 概览
    总流量MB = 统计数据["总流量"] / (1024 * 1024)
    print(f"\n【概览】")
    print(f"  总请求数: {统计数据['总请求数']:,}")
    print(f"  解析失败: {统计数据['解析失败']}")
    print(f"  总流量:   {总流量MB:.2f} MB")

    # 状态码分布
    print(f"\n【状态码分布】")
    print(f"  {'状态码':<8} {'次数':>8} {'占比':>8} {'条形图'}")
    print(f"  {细线[:50]}")

    总数 = 统计数据["总请求数"]
    for 状态码, 次数 in 统计数据["状态码统计"].most_common():
        占比 = 次数 / 总数 * 100
        条形 = "█" * int(占比 / 2)
        print(f"  {状态码:<8} {次数:>8} {占比:>7.1f}% {条形}")

    # 按状态码类别汇总
    类别统计 = Counter()
    for 码, 数 in 统计数据["状态码统计"].items():
        类别 = f"{码[0]}xx"
        类别统计[类别] += 数

    print(f"\n  按类别: ", end="")
    for 类别 in sorted(类别统计):
        print(f"{类别}={类别统计[类别]}", end="  ")
    print()

    # Top 10 IP
    print(f"\n【Top 10 IP 地址】")
    print(f"  {'排名':<4} {'IP 地址':<20} {'请求数':>8} {'占比':>8}")
    print(f"  {细线[:45]}")
    for 排名, (ip, 次数) in enumerate(统计数据["IP统计"].most_common(10), 1):
        占比 = 次数 / 总数 * 100
        print(f"  {排名:<4} {ip:<20} {次数:>8} {占比:>7.1f}%")

    # Top 10 URL
    print(f"\n【Top 10 URL】")
    print(f"  {'排名':<4} {'URL':<30} {'请求数':>8} {'占比':>8}")
    print(f"  {细线[:55]}")
    for 排名, (url, 次数) in enumerate(统计数据["URL统计"].most_common(10), 1):
        占比 = 次数 / 总数 * 100
        print(f"  {排名:<4} {url:<30} {次数:>8} {占比:>7.1f}%")

    # HTTP 方法分布
    print(f"\n【HTTP 方法分布】")
    for 方法, 次数 in 统计数据["方法统计"].most_common():
        占比 = 次数 / 总数 * 100
        print(f"  {方法:<8} {次数:>8} ({占比:.1f}%)")

    # 错误请求 Top 5
    if 统计数据["错误URL统计"]:
        print(f"\n【Top 错误请求】")
        for 项, 次数 in 统计数据["错误URL统计"].most_common(5):
            print(f"  {次数:>5}次  {项}")

    # QPS 分析
    if 统计数据["每分钟请求"]:
        每分钟列表 = list(统计数据["每分钟请求"].values())
        平均QPS = sum(每分钟列表) / len(每分钟列表) / 60
        最大QPS = max(每分钟列表) / 60
        print(f"\n【QPS 分析】")
        print(f"  平均 QPS: {平均QPS:.1f}")
        print(f"  峰值 QPS: {最大QPS:.1f}")

    # 可疑 IP（请求量异常高的）
    if 统计数据["IP统计"]:
        平均请求 = 总数 / len(统计数据["IP统计"])
        可疑IP = [
            (ip, 次数) for ip, 次数 in 统计数据["IP统计"].items()
            if 次数 > 平均请求 * 3
        ]
        if 可疑IP:
            print(f"\n【可疑 IP（请求量 > 平均值 3 倍）】")
            for ip, 次数 in sorted(可疑IP, key=lambda x: x[1], reverse=True):
                print(f"  {ip:<20} {次数} 次请求")

    print(f"\n{分隔线}")


def main():
    """主函数"""
    if len(sys.argv) > 1 and sys.argv[1] != "--demo":
```

## 8. 常见坑点与最佳实践

### 坑点 1：dict.keys() 在迭代时修改字典

```python
# 错误：在遍历字典时删除元素
数据 = {"a": 1, "b": 2, "c": 3}
# for k in 数据:
#     if 数据[k] < 3:
#         del 数据[k]       # RuntimeError: dictionary changed size during iteration

要删除 = [k for k, v in 数据.items() if v < 3]
for k in 要删除:
    del 数据[k]

# 或者直接用字典推导式创建新字典
数据 = {k: v for k, v in 数据.items() if v >= 3}
```

## 9. 速查表

```python
# =====================================================
# 数据结构进阶速查表
# =====================================================

# --- list 高级 ---

max(d, key=d.get)                 # 获取最大值的键

x in s                            # 成员检查 O(1)

from collections import Counter, defaultdict, deque, namedtuple, OrderedDict
```
