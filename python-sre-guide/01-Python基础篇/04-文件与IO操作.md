## 1. 是什么 & 为什么用它

### Shell 文件操作的局限

```bash
# Shell 文件操作：简单直接但容易出错
cat /var/log/nginx/access.log                      # 读取文件
echo "hello" > /tmp/test.txt                       # 写入文件
echo "world" >> /tmp/test.txt                      # 追加内容
cp -r /src /dst                                    # 复制
mv /old /new                                       # 移动/重命名
rm -rf /path                                       # 删除（危险！）

# 问题：
# 1. 没有错误处理，rm -rf / 直接删根目录
# 2. 编码处理困难
# 3. 处理复杂文本（如 JSON、YAML）很痛苦
# 4. 大文件处理容易撑爆内存
```

### Python 的优势

| 对比维度       | Shell                       | Python                          |
|---------------|-----------------------------|---------------------------------|
| 读文件        | cat / head / tail           | open() / pathlib.read_text()    |
| 写文件        | echo > / tee                | open('w') / pathlib.write_text()|
| 路径操作      | dirname / basename          | pathlib.Path / os.path          |
| 复制/移动     | cp / mv                     | shutil.copy / shutil.move       |
| 删除          | rm -rf                      | shutil.rmtree（更安全）         |
| 临时文件      | mktemp                      | tempfile 模块                   |
| 编码处理      | iconv / 手动处理            | encoding 参数自动处理           |
| 错误处理      | set -e / || 手动处理        | try/except 结构化处理           |

## 2. open() 读写文件（对比 cat/echo/tee）

### 2.1 读取文件

```python
#!/usr/bin/env python3
"""
文件读取的各种方式
"""

# ==========================================
# 方式1：读取整个文件（对比 cat）
# ==========================================
# Shell: content=$(cat /etc/hostname)
with open("/etc/hostname", "r") as f:
    内容 = f.read()                     # 读取全部内容为字符串
    print(f"主机名: {内容.strip()}")

# ==========================================
# 方式2：逐行读取（对比 while read line）
# ==========================================
# Shell:
# while IFS= read -r line; do
#     echo "$line"
# done < /etc/passwd

with open("/etc/passwd", "r") as f:
    for 行 in f:                         # 逐行读取，内存友好
        字段 = 行.strip().split(":")
        用户名 = 字段[0]
        UID = 字段[2]
        if int(UID) >= 1000:             # 只显示普通用户
            print(f"用户: {用户名}, UID: {UID}")

# ==========================================
# 方式3：读取所有行到列表（对比 readarray）
# ==========================================
# Shell: readarray -t lines < /etc/hosts
with open("/etc/hosts", "r") as f:
    所有行 = f.readlines()               # 返回列表，每行一个元素
    print(f"hosts 文件共 {len(所有行)} 行")

# ==========================================
# 方式4：读取前 N 行（对比 head -n 5）
# ==========================================
with open("/etc/passwd", "r") as f:
    for i, 行 in enumerate(f):
        if i >= 5:
            break
        print(行.strip())

# ==========================================
# 方式5：读取最后 N 行（对比 tail -n 5）
# ==========================================
from collections import deque

def 读取末尾(文件路径, 行数=5):
    """读取文件最后 N 行"""
    with open(文件路径, "r") as f:
        最后几行 = deque(f, maxlen=行数)
    return list(最后几行)
```

### 2.2 写入文件

```python
#!/usr/bin/env python3
"""
文件写入的各种方式
"""

# ==========================================
# 写入文件（覆盖模式，对比 echo > file）
# ==========================================
# Shell: echo "hello world" > /tmp/test.txt
with open("/tmp/test_py.txt", "w") as f:
    f.write("第一行内容\n")
    f.write("第二行内容\n")

# ==========================================
# 追加文件（对比 echo >> file）
# ==========================================
# Shell: echo "new line" >> /tmp/test.txt
with open("/tmp/test_py.txt", "a") as f:
    f.write("追加的内容\n")

# ==========================================
# 写入多行（对比 cat << EOF > file）
# ==========================================
# Shell:
# cat << 'EOF' > /tmp/config.conf
# server=nginx
# port=80
# EOF

配置内容 = """[server]
name = nginx
port = 80
workers = 4
log_level = info
"""

with open("/tmp/config.conf", "w") as f:
    f.write(配置内容)

# ==========================================
# 写入列表（对比 printf）
# ==========================================
服务器列表 = ["web-01", "web-02", "web-03", "db-01"]

with open("/tmp/servers.txt", "w") as f:
    for 服务器 in 服务器列表:
        f.write(f"{服务器}\n")

# 更简洁的写法
with open("/tmp/servers2.txt", "w") as f:
    f.writelines(f"{s}\n" for s in 服务器列表)

print("文件写入完成")
```

## 3. with 上下文管理器

```python
#!/usr/bin/env python3
"""
with 语句：自动管理资源（文件关闭、锁释放等）
这是 Python 中非常重要的模式，Shell 中没有对应概念
"""

# ==========================================
# 不使用 with（不推荐）
# ==========================================
f = open("/tmp/test_py.txt", "r")
内容 = f.read()
f.close()                     # 容易忘记关闭！如果中间出错，文件句柄泄漏

# ==========================================
# 使用 with（推荐！）
# ==========================================
with open("/tmp/test_py.txt", "r") as f:
    内容 = f.read()
# 离开 with 块后，文件自动关闭，即使出错也会关闭

# ==========================================
# 同时打开多个文件（对比 Shell 的重定向）
# ==========================================
# Shell: paste file1 file2 > file3
with open("/tmp/test_py.txt", "r") as 输入文件, \
     open("/tmp/output.txt", "w") as 输出文件:
    for 行 in 输入文件:
        输出文件.write(行.upper())      # 读入转大写后写出

# ==========================================
# with 的工作原理
# ==========================================
# with 语句调用对象的 __enter__() 和 __exit__() 方法
# __enter__() 在进入 with 块时执行（打开文件）
# __exit__()  在离开 with 块时执行（关闭文件），即使发生异常也会执行
# 类似于 Shell 中的 trap 'cleanup' EXIT，但更优雅
```

## 5. pathlib 路径操作（对比 dirname/basename）

```python
#!/usr/bin/env python3
"""
pathlib：现代 Python 路径操作（Python 3.4+，推荐！）
对比 Shell 的 dirname, basename, readlink 等
"""

from pathlib import Path

# ==========================================
# 创建 Path 对象
# ==========================================
路径 = Path("/var/log/nginx/access.log")

# ==========================================
# 路径分解（对比 dirname / basename）
# ==========================================
# Shell: dirname /var/log/nginx/access.log   -> /var/log/nginx
# Shell: basename /var/log/nginx/access.log  -> access.log
print(f"父目录: {路径.parent}")             # /var/log/nginx
print(f"文件名: {路径.name}")               # access.log
print(f"主名称: {路径.stem}")               # access（不含后缀）
print(f"后缀:   {路径.suffix}")             # .log
print(f"所有后缀: {路径.suffixes}")         # ['.log']

# 多级后缀
压缩文件 = Path("/backup/data.tar.gz")
print(f"后缀列表: {压缩文件.suffixes}")     # ['.tar', '.gz']
print(f"主名称: {压缩文件.stem}")           # data.tar

# ==========================================
# 路径拼接（对比 Shell 的字符串拼接）
# ==========================================
# Shell: log_dir="/var/log"; file="${log_dir}/nginx/error.log"
日志目录 = Path("/var/log")
完整路径 = 日志目录 / "nginx" / "error.log"   # 用 / 运算符拼接！
print(f"完整路径: {完整路径}")

# ==========================================
# 路径检查
# ==========================================
# Shell: [ -f /etc/hosts ] && echo "exists"
# Shell: [ -d /tmp ] && echo "is dir"
测试路径 = Path("/etc/hosts")
print(f"存在: {测试路径.exists()}")          # True
print(f"是文件: {测试路径.is_file()}")       # True
print(f"是目录: {测试路径.is_dir()}")        # False

# ==========================================
# 获取文件信息
# ==========================================
if 测试路径.exists():
    信息 = 测试路径.stat()
    print(f"文件大小: {信息.st_size} 字节")
    print(f"修改时间: {信息.st_mtime}")

# ==========================================
# 遍历目录（对比 ls / find）
# ==========================================
# Shell: ls /tmp/*.txt
# Shell: find /tmp -name "*.txt"
临时目录 = Path("/tmp")

# 列出当前目录下的 .txt 文件（非递归）
for 文件 in 临时目录.glob("*.txt"):
    print(f"文件: {文件}")

# 递归查找所有 .log 文件（对比 find /var/log -name "*.log"）
# for 文件 in Path("/var/log").rglob("*.log"):
#     print(f"日志: {文件}")

# ==========================================
# 文件读写快捷方式
# ==========================================
# 写入
Path("/tmp/pathlib_test.txt").write_text("Hello from pathlib!\n", encoding="utf-8")

# 读取
内容 = Path("/tmp/pathlib_test.txt").read_text(encoding="utf-8")
print(f"读取: {内容}")

# ==========================================
# 路径修改
# ==========================================
原始路径 = Path("/var/log/nginx/access.log")

# 修改后缀
新路径 = 原始路径.with_suffix(".bak")
print(f"备份路径: {新路径}")               # /var/log/nginx/access.bak

# 修改文件名
新路径 = 原始路径.with_name("error.log")
print(f"错误日志: {新路径}")               # /var/log/nginx/error.log

# ==========================================
# 当前目录和家目录
# ==========================================
print(f"当前目录: {Path.cwd()}")           # 对比 pwd
print(f"家目录:   {Path.home()}")          # 对比 echo $HOME

# ==========================================
# 创建目录
# ==========================================
# Shell: mkdir -p /tmp/test/sub/dir
新目录 = Path("/tmp/test_pathlib/sub/dir")
新目录.mkdir(parents=True, exist_ok=True)  # parents=True 等同于 -p
print(f"目录已创建: {新目录}")
```

## 7. shutil 文件操作（对比 cp/mv/rm）

```python
#!/usr/bin/env python3
"""
shutil：高级文件操作
对比 Shell 的 cp、mv、rm 命令
"""

import shutil
import os
from pathlib import Path

# 准备测试环境
os.makedirs("/tmp/shutil_test/src", exist_ok=True)
os.makedirs("/tmp/shutil_test/dst", exist_ok=True)
Path("/tmp/shutil_test/src/test.txt").write_text("hello world\n")

# ==========================================
# 复制文件（对比 cp）
# ==========================================
# Shell: cp /tmp/src/test.txt /tmp/dst/test.txt
shutil.copy("/tmp/shutil_test/src/test.txt", "/tmp/shutil_test/dst/test.txt")
print("文件复制完成")

# copy2 保留元数据（修改时间等），类似 cp -p
shutil.copy2("/tmp/shutil_test/src/test.txt", "/tmp/shutil_test/dst/test2.txt")
print("文件复制完成（保留元数据）")

# ==========================================
# 复制目录（对比 cp -r）
# ==========================================
# Shell: cp -r /tmp/src /tmp/src_backup
if os.path.exists("/tmp/shutil_test/src_backup"):
    shutil.rmtree("/tmp/shutil_test/src_backup")
shutil.copytree("/tmp/shutil_test/src", "/tmp/shutil_test/src_backup")
print("目录复制完成")

# ==========================================
# 移动/重命名（对比 mv）
# ==========================================
# Shell: mv /tmp/dst/test2.txt /tmp/dst/renamed.txt
shutil.move(
    "/tmp/shutil_test/dst/test2.txt",
    "/tmp/shutil_test/dst/renamed.txt"
)
print("文件移动/重命名完成")

# ==========================================
# 删除目录（对比 rm -rf）
# ==========================================
# Shell: rm -rf /tmp/src_backup
# 注意：shutil.rmtree 是不可逆的，使用前要确认路径！
shutil.rmtree("/tmp/shutil_test/src_backup")
print("目录删除完成")

# ==========================================
# 磁盘使用率（对比 df）
# ==========================================
# Shell: df -h /
使用情况 = shutil.disk_usage("/")
print(f"\n磁盘使用情况 (/):")
print(f"  总量: {使用情况.total / (1024**3):.1f} GB")
print(f"  已用: {使用情况.used / (1024**3):.1f} GB")
print(f"  可用: {使用情况.free / (1024**3):.1f} GB")
print(f"  使用率: {使用情况.used / 使用情况.total * 100:.1f}%")

# ==========================================
# 查找命令路径（对比 which）
# ==========================================
# Shell: which python3
python路径 = shutil.which("python3")
print(f"\npython3 路径: {python路径}")

# 清理测试目录
shutil.rmtree("/tmp/shutil_test")
print("\n测试目录已清理")
```

## 8. tempfile 临时文件（对比 mktemp）

```python
#!/usr/bin/env python3
"""
tempfile：安全创建临时文件和目录
对比 Shell 的 mktemp 命令
"""

import tempfile
import os

# ==========================================
# 创建临时文件（对比 mktemp）
# ==========================================
# Shell: tmpfile=$(mktemp /tmp/myapp.XXXXXX)

# 方式1：自动管理（推荐）
with tempfile.NamedTemporaryFile(mode="w", suffix=".log", prefix="sre_",
                                  delete=False) as tmp:
    tmp.write("临时日志内容\n")
    tmp.write("第二行\n")
    临时文件路径 = tmp.name
    print(f"临时文件: {临时文件路径}")

# 读取临时文件
with open(临时文件路径, "r") as f:
    print(f"内容: {f.read()}")

# 手动清理
os.unlink(临时文件路径)

# 方式2：with 块结束后自动删除
with tempfile.NamedTemporaryFile(mode="w", suffix=".txt", delete=True) as tmp:
    tmp.write("这个文件会自动删除\n")
    tmp.flush()                        # 刷新缓冲区
    print(f"临时文件存在: {os.path.exists(tmp.name)}")  # True

# with 块结束后文件已删除

# ==========================================
# 创建临时目录（对比 mktemp -d）
# ==========================================
# Shell: tmpdir=$(mktemp -d /tmp/myapp.XXXXXX)
with tempfile.TemporaryDirectory(prefix="sre_") as 临时目录:
    print(f"临时目录: {临时目录}")

    # 在临时目录中创建文件
    临时文件 = os.path.join(临时目录, "test.txt")
    with open(临时文件, "w") as f:
        f.write("临时内容\n")

    print(f"临时文件存在: {os.path.exists(临时文件)}")  # True

# with 块结束后，整个目录自动删除
# print(f"临时目录存在: {os.path.exists(临时目录)}")  # False

# ==========================================
# 获取临时目录路径
# ==========================================
# Shell: echo $TMPDIR 或 /tmp
print(f"系统临时目录: {tempfile.gettempdir()}")
```

```python
#!/usr/bin/env python3
"""
编码问题：SRE 经常遇到的坑
"""

# ==========================================
# Python 3 字符串默认是 Unicode（UTF-8）
# ==========================================
中文文本 = "你好，SRE！"
print(f"类型: {type(中文文本)}")         # <class 'str'>
print(f"长度: {len(中文文本)}")          # 7（按字符计算）

# 编码为字节
字节数据 = 中文文本.encode("utf-8")
print(f"字节类型: {type(字节数据)}")     # <class 'bytes'>
print(f"字节长度: {len(字节数据)}")      # 17（中文 UTF-8 每字符3字节）

# 解码回字符串
恢复文本 = 字节数据.decode("utf-8")
print(f"恢复: {恢复文本}")

# ==========================================
# 读写文件时指定编码
# ==========================================
# 写入 UTF-8（推荐）
with open("/tmp/utf8.txt", "w", encoding="utf-8") as f:
    f.write("UTF-8 编码的中文内容\n")

# 读取时指定编码
with open("/tmp/utf8.txt", "r", encoding="utf-8") as f:
    print(f.read())

# ==========================================
# 处理未知编码的文件
# ==========================================
def 安全读取(文件路径, 编码列表=None):
    """尝试多种编码读取文件"""
    if 编码列表 is None:
        编码列表 = ["utf-8", "gbk", "gb2312", "latin-1"]

    for 编码 in 编码列表:
        try:
            with open(文件路径, "r", encoding=编码) as f:
                内容 = f.read()
            print(f"成功使用 {编码} 编码读取")
            return 内容
        except UnicodeDecodeError:
            continue

    # 最后的备选方案：忽略错误
    with open(文件路径, "r", encoding="utf-8", errors="ignore") as f:
        return f.read()

# 使用示例
# 内容 = 安全读取("/path/to/unknown_encoding.txt")

# 清理
os.unlink("/tmp/utf8.txt")
print("编码示例完成")
```

```python
import time

# ANSI 颜色代码
class 颜色:

    黄色 = "\033[93m"
    绿色 = "\033[92m"
    蓝色 = "\033[94m"
    重置 = "\033[0m"
    粗体 = "\033[1m"


def 高亮关键字(行, 关键字映射=None):
    """对关键字进行颜色高亮"""
    if 关键字映射 is None:
        关键字映射 = {
            "ERROR":   颜色.红色,
            "WARN":    颜色.黄色,
            "WARNING": 颜色.黄色,
            "INFO":    颜色.绿色,
            "DEBUG":   颜色.蓝色,
        }

    结果 = 行
    for 关键字, 色彩 in 关键字映射.items():
        if 关键字 in 结果:
            结果 = 结果.replace(关键字, f"{色彩}{颜色.粗体}{关键字}{颜色.重置}")
    return 结果


def tail_f(文件路径, 过滤关键字=None, 初始行数=10):
    """
    实现 tail -f 功能

    参数:
        文件路径: 要监控的文件路径
        过滤关键字: 只显示包含该关键字的行（类似 grep）
        初始行数: 初始显示最后几行
    """
    路径 = Path(文件路径)

    # 检查文件是否存在
    if not 路径.exists():
        print(f"错误: 文件不存在 - {文件路径}")
        sys.exit(1)

    if not 路径.is_file():
        print(f"错误: 不是文件 - {文件路径}")
        sys.exit(1)

    print(f"{'=' * 60}")
    print(f"监控文件: {文件路径}")
    if 过滤关键字:
        print(f"过滤关键字: {过滤关键字}")
    print(f"按 Ctrl+C 退出")
    print(f"{'=' * 60}\n")

    with open(文件路径, "r", encoding="utf-8", errors="replace") as f:
        # 先定位到文件末尾附近，显示最后几行
        # 读取最后部分内容
        f.seek(0, 2)                  # 移到文件末尾
        文件大小 = f.tell()
        读取位置 = max(0, 文件大小 - 4096)  # 从末尾前 4KB 开始
        f.seek(读取位置)

        if 读取位置 > 0:
            f.readline()              # 跳过可能不完整的第一行

        # 读取并显示最后几行
        所有行 = f.readlines()
        最后几行 = 所有行[-初始行数:] if len(所有行) > 初始行数 else 所有行

        for 行 in 最后几行:
            行 = 行.rstrip("\n")
            if 过滤关键字 and 过滤关键字 not in 行:
                continue
            print(高亮关键字(行))

        # 持续监控新内容
        try:
            while True:
                行 = f.readline()
                if 行:
                    行 = 行.rstrip("\n")
                    if 过滤关键字 and 过滤关键字 not in 行:
                        continue
                    print(高亮关键字(行))
                else:
                    # 检查文件是否被轮转（inode 改变）
                    try:
                        当前inode = os.stat(文件路径).st_ino
                        打开inode = os.fstat(f.fileno()).st_ino
                        if 当前inode != 打开inode:
                            print(f"\n{颜色.黄色}[文件已轮转，重新打开]{颜色.重置}")
                            break  # 跳出循环，重新打开文件
                    except OSError:
                        pass
                    time.sleep(0.1)         # 没有新内容，短暂等待

        except KeyboardInterrupt:
            print(f"\n\n{颜色.绿色}监控已停止{颜色.重置}")


def 生成测试日志(文件路径="/tmp/test_tail.log"):
    """生成测试日志文件，用于演示"""
    import random
    from datetime import datetime

    级别列表 = ["INFO", "INFO", "INFO", "WARNING", "ERROR", "DEBUG"]
    消息列表 = [
        "请求处理完成",
        "数据库连接正常",
        "磁盘使用率超过阈值",
        "服务启动成功",
        "连接超时，正在重试",

        "配置文件加载完成",
        "健康检查通过",
        "SSL 证书即将过期",
        "API 响应时间过长",
    ]

    with open(文件路径, "w") as f:
        for i in range(20):
            时间戳 = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            级别 = random.choice(级别列表)

            f.write(f"{时间戳} {级别} [{f'worker-{i%4}'}] {消息}\n")

    print(f"测试日志已生成: {文件路径}")
    return 文件路径


def main():
    """主函数"""
    if len(sys.argv) < 2:
        print(f"用法: {sys.argv[0]} <文件路径> [过滤关键字]")
        print(f"示例: {sys.argv[0]} /var/log/syslog ERROR")

if __name__ == "__main__":
    main()
```

## 11. 常见坑点与最佳实践

### 坑点 1：忘记关闭文件

```python
# 错误：不关闭文件会导致资源泄漏
f = open("file.txt", "r")
内容 = f.read()
# 忘记 f.close()

# 正确：始终使用 with 语句
with open("file.txt", "r") as f:
    内容 = f.read()
# 自动关闭
```

### 坑点 2：大文件一次性读入内存

```python
# 错误：大文件直接 read() 会撑爆内存
# with open("huge_file.log", "r") as f:
#     所有内容 = f.read()             # 10GB 文件直接卡死

# 正确：逐行读取
# with open("huge_file.log", "r") as f:
#     for 行 in f:                    # 逐行迭代，内存友好
#         处理(行)

# 正确：分块读取
# with open("huge_binary.dat", "rb") as f:
#     while True:
#         块 = f.read(8192)           # 每次读 8KB
#         if not 块:
#             break
#         处理(块)
```

### 坑点 3：路径拼接用字符串相加

```python
# 错误：手动拼接路径容易出错
# path = "/var/log" + "/" + "nginx" + "/" + "access.log"
# path = "/var/log/" + "/nginx/"  # 双斜杠！

# 正确：使用 pathlib 或 os.path.join
from pathlib import Path
path = Path("/var/log") / "nginx" / "access.log"

# 或
import os
path = os.path.join("/var/log", "nginx", "access.log")
```

### 坑点 4：不处理文件不存在的情况

```python
# 错误：直接打开可能不存在的文件
# with open("/var/log/maybe_not_exist.log") as f:
#     内容 = f.read()    # FileNotFoundError!

# 正确：先检查或用 try/except
from pathlib import Path

文件 = Path("/var/log/maybe_not_exist.log")
if 文件.exists():
    内容 = 文件.read_text()
else:
    print("文件不存在")

# 或者用 try/except
try:
    with open("/var/log/maybe_not_exist.log") as f:
        内容 = f.read()
except FileNotFoundError:
    print("文件不存在")
except PermissionError:
    print("没有读取权限")
```
