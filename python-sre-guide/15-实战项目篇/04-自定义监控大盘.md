# 实战项目四：自定义监控大盘

## 1. 项目概述

### 1.1 解决什么问题

运维团队需要实时掌握基础设施和应用的健康状态。商业监控方案成本高昂，而开源方案配置复杂。
本项目实现了一个**轻量级自定义监控大盘**，集成本机指标采集、Prometheus 指标暴露、
Kubernetes 集群状态、AWS 云资源监控，并提供终端可视化大盘和 REST API，
适合中小团队快速搭建监控体系。

### 4.2 本机系统指标采集 `collectors/system.py`

```python
"""
本机系统指标采集模块
使用 psutil 采集 CPU、内存、磁盘、网络、进程等指标
"""

import time
import logging
from typing import Dict, List
from dataclasses import dataclass, field

import psutil

logger = logging.getLogger(__name__)


@dataclass
class SystemMetrics:
    """
    系统指标数据类
    包含本机各项性能指标
    """
    timestamp: float = 0.0

    # CPU 指标
    cpu_usage_percent: float = 0.0
    cpu_usage_per_core: list = field(default_factory=list)
    cpu_count_logical: int = 0
    cpu_count_physical: int = 0
    cpu_freq_current: float = 0.0
    load_1min: float = 0.0
    load_5min: float = 0.0
    load_15min: float = 0.0
    cpu_ctx_switches: int = 0
    cpu_interrupts: int = 0

    # 内存指标
    mem_total_bytes: int = 0
    mem_used_bytes: int = 0
    mem_available_bytes: int = 0
    mem_usage_percent: float = 0.0
    mem_buffers_bytes: int = 0
    mem_cached_bytes: int = 0
    swap_total_bytes: int = 0
    swap_used_bytes: int = 0
    swap_usage_percent: float = 0.0

    # 磁盘指标
    disk_partitions: list = field(default_factory=list)
    disk_io_read_bytes: int = 0
    disk_io_write_bytes: int = 0
    disk_io_read_count: int = 0
    disk_io_write_count: int = 0

    # 网络指标
    net_bytes_sent: int = 0
    net_bytes_recv: int = 0
    net_packets_sent: int = 0
    net_packets_recv: int = 0
    net_connections: int = 0
    net_connections_by_status: dict = field(default_factory=dict)

    # 进程指标
    process_total: int = 0
    process_running: int = 0
    process_sleeping: int = 0
    process_zombie: int = 0
    top_cpu_processes: list = field(default_factory=list)
    top_mem_processes: list = field(default_factory=list)

    def to_dict(self) -> dict:
        """转换为字典"""
        from dataclasses import asdict
        return asdict(self)


class SystemCollector:
    """
    系统指标采集器
    定期采集本机各项系统性能指标
    """

    def __init__(self, config: dict = None):
        """
        初始化系统采集器

        参数:
            config: 采集配置
        """
        self.config = config or {}
        self._prev_net_io = None     # 上次网络 IO（用于计算速率）
        self._prev_disk_io = None    # 上次磁盘 IO
        self._prev_time = None       # 上次采集时间

    def collect(self) -> SystemMetrics:
        """
        采集所有系统指标

        返回:
            SystemMetrics: 系统指标数据对象
        """
        metrics = SystemMetrics(timestamp=time.time())

        self._collect_cpu(metrics)
        self._collect_memory(metrics)
        self._collect_disk(metrics)
        self._collect_network(metrics)
        self._collect_processes(metrics)

        self._prev_time = time.time()
        return metrics

    def _collect_cpu(self, m: SystemMetrics):
        """
        采集 CPU 指标

        包括：总使用率、每核使用率、核心数、频率、负载、上下文切换
        """
        try:
            # CPU 使用率
            m.cpu_usage_percent = psutil.cpu_percent(interval=1)
            m.cpu_usage_per_core = psutil.cpu_percent(interval=0, percpu=True)

            # CPU 核心数
            m.cpu_count_logical = psutil.cpu_count(logical=True) or 0
            m.cpu_count_physical = psutil.cpu_count(logical=False) or 0

            # CPU 频率
            freq = psutil.cpu_freq()
            if freq:
                m.cpu_freq_current = freq.current

            # 系统负载
            load = psutil.getloadavg()
            m.load_1min = load[0]
            m.load_5min = load[1]
            m.load_15min = load[2]

            # CPU 统计
            cpu_stats = psutil.cpu_stats()
            m.cpu_ctx_switches = cpu_stats.ctx_switches
            m.cpu_interrupts = cpu_stats.interrupts

        except Exception as e:
            logger.error(f"CPU 指标采集失败: {e}")

    def _collect_memory(self, m: SystemMetrics):
        """
        采集内存指标

        包括：总内存、已用、可用、使用率、缓冲区、缓存、Swap
        """
        try:
            # 物理内存
            mem = psutil.virtual_memory()
            m.mem_total_bytes = mem.total
            m.mem_used_bytes = mem.used
            m.mem_available_bytes = mem.available
            m.mem_usage_percent = mem.percent

            # Linux 特有字段
            if hasattr(mem, "buffers"):
                m.mem_buffers_bytes = mem.buffers
            if hasattr(mem, "cached"):
                m.mem_cached_bytes = mem.cached

            # Swap
            swap = psutil.swap_memory()
            m.swap_total_bytes = swap.total
            m.swap_used_bytes = swap.used
            m.swap_usage_percent = swap.percent

        except Exception as e:
            logger.error(f"内存指标采集失败: {e}")

    def _collect_disk(self, m: SystemMetrics):
        """
        采集磁盘指标

        包括：各分区使用率、磁盘 IO 计数器
        """
        try:
            # 磁盘分区
            partitions = []
            for part in psutil.disk_partitions():
                try:
                    usage = psutil.disk_usage(part.mountpoint)
                    partitions.append({
                        "device": part.device,
                        "mountpoint": part.mountpoint,
                        "fstype": part.fstype,
                        "total_bytes": usage.total,
                        "used_bytes": usage.used,
                        "free_bytes": usage.free,
                        "usage_percent": usage.percent,
                    })
                except (PermissionError, OSError):
                    continue
            m.disk_partitions = partitions

            # 磁盘 IO
            disk_io = psutil.disk_io_counters()
            if disk_io:
                m.disk_io_read_bytes = disk_io.read_bytes
                m.disk_io_write_bytes = disk_io.write_bytes
                m.disk_io_read_count = disk_io.read_count
                m.disk_io_write_count = disk_io.write_count

                self._prev_disk_io = disk_io

        except Exception as e:
            logger.error(f"磁盘指标采集失败: {e}")

    def _collect_network(self, m: SystemMetrics):
        """
        采集网络指标

        包括：流量统计、包统计、连接数、连接状态分布
        """
        try:
            # 网络 IO
            net_io = psutil.net_io_counters()
            m.net_bytes_sent = net_io.bytes_sent
            m.net_bytes_recv = net_io.bytes_recv
            m.net_packets_sent = net_io.packets_sent
            m.net_packets_recv = net_io.packets_recv

            self._prev_net_io = net_io

            # 网络连接
            try:
                connections = psutil.net_connections(kind="inet")
                m.net_connections = len(connections)

                # 按状态统计连接数
                status_counts = {}
                for conn in connections:
                    status = conn.status
                    status_counts[status] = status_counts.get(status, 0) + 1
                m.net_connections_by_status = status_counts

            except (psutil.AccessDenied, PermissionError):
                # 非 root 用户可能无权限
                m.net_connections = 0

        except Exception as e:
            logger.error(f"网络指标采集失败: {e}")

    def _collect_processes(self, m: SystemMetrics):
        """
        采集进程指标

        包括：进程总数、按状态分类、CPU/内存占用 Top5
        """
        try:
            procs = []
            status_counts = {"running": 0, "sleeping": 0, "zombie": 0}

            for proc in psutil.process_iter(
                ["pid", "name", "cpu_percent", "memory_percent", "status"]
            ):
                try:
                    info = proc.info
                    procs.append(info)

                    status = info.get("status", "")
                    if status == psutil.STATUS_RUNNING:
                        status_counts["running"] += 1
                    elif status == psutil.STATUS_SLEEPING:
                        status_counts["sleeping"] += 1
                    elif status == psutil.STATUS_ZOMBIE:
                        status_counts["zombie"] += 1

                except (psutil.NoSuchProcess, psutil.AccessDenied):
                    continue

            m.process_total = len(procs)
            m.process_running = status_counts["running"]
            m.process_sleeping = status_counts["sleeping"]
            m.process_zombie = status_counts["zombie"]

            # Top 5 CPU 占用进程
            procs_sorted_cpu = sorted(
                procs,
                key=lambda p: p.get("cpu_percent", 0) or 0,
                reverse=True
            )
            m.top_cpu_processes = [
                {
                    "pid": p["pid"],
                    "name": p["name"],
                    "cpu_percent": round(p.get("cpu_percent", 0) or 0, 1),
                }
                for p in procs_sorted_cpu[:5]
            ]

            # Top 5 内存占用进程
            procs_sorted_mem = sorted(
                procs,
                key=lambda p: p.get("memory_percent", 0) or 0,
                reverse=True
            )
            m.top_mem_processes = [
                {
                    "pid": p["pid"],
                    "name": p["name"],
                    "memory_percent": round(p.get("memory_percent", 0) or 0, 1),
                }
                for p in procs_sorted_mem[:5]
            ]

        except Exception as e:
            logger.error(f"进程指标采集失败: {e}")
```

### 4.3 Kubernetes 集群指标采集 `collectors/kubernetes_col.py`

```python
"""
Kubernetes 集群指标采集模块
使用 kubernetes client 获取集群和工作负载状态
"""

import logging
from typing import Dict, List, Optional
from dataclasses import dataclass, field

logger = logging.getLogger(__name__)


@dataclass
class K8sMetrics:
    """
    Kubernetes 集群指标数据类
    """
    timestamp: float = 0.0

    # 节点信息
    node_count: int = 0
    node_ready_count: int = 0
    nodes: list = field(default_factory=list)

    # Pod 信息
    pod_total: int = 0
    pod_running: int = 0
    pod_pending: int = 0
    pod_failed: int = 0
    pod_succeeded: int = 0

    # Deployment 信息
    deployments: list = field(default_factory=list)
    deployment_count: int = 0
    deployment_available: int = 0

    # 资源使用
    cluster_cpu_requests: float = 0.0
    cluster_cpu_limits: float = 0.0
    cluster_mem_requests_bytes: int = 0
    cluster_mem_limits_bytes: int = 0

    # 异常 Pod
    unhealthy_pods: list = field(default_factory=list)

    def to_dict(self) -> dict:
        from dataclasses import asdict
        return asdict(self)


class KubernetesCollector:
    """
    Kubernetes 集群采集器
    采集节点、Pod、Deployment 等资源状态
    """

    def __init__(self, config: dict):
        """
        初始化 K8s 采集器

        参数:
            config: K8s 配置字典
        """
        self.namespaces = config.get("namespaces", [])
        self._client_initialized = False
        self._core_v1 = None
        self._apps_v1 = None

        # 初始化 K8s 客户端
        self._init_client(config.get("kubeconfig"))

    def _init_client(self, kubeconfig: str = None):
        """
        初始化 Kubernetes Python 客户端

        参数:
            kubeconfig: kubeconfig 文件路径
        """
        try:
            from kubernetes import client, config as k8s_config

            # 优先使用 kubeconfig 文件
            if kubeconfig:
                k8s_config.load_kube_config(config_file=kubeconfig)
            else:
                try:
                    # 尝试集群内配置（Pod 内运行时）
                    k8s_config.load_incluster_config()
                except k8s_config.ConfigException:
                    # 回退到默认 kubeconfig
                    k8s_config.load_kube_config()

            self._core_v1 = client.CoreV1Api()
            self._apps_v1 = client.AppsV1Api()
            self._client_initialized = True
            logger.info("Kubernetes 客户端初始化成功")

        except ImportError:
            logger.warning("kubernetes 库未安装，跳过 K8s 采集")
        except Exception as e:
            logger.error(f"Kubernetes 客户端初始化失败: {e}")

    def collect(self) -> Optional[K8sMetrics]:
        """
        采集 Kubernetes 集群指标

        返回:
            Optional[K8sMetrics]: 集群指标，初始化失败返回 None
        """
        if not self._client_initialized:
            return None

        import time
        metrics = K8sMetrics(timestamp=time.time())

        self._collect_nodes(metrics)
        self._collect_pods(metrics)
        self._collect_deployments(metrics)

        return metrics

    def _collect_nodes(self, m: K8sMetrics):
        """
        采集节点信息

        包括：节点数量、就绪状态、资源容量
        """
        try:
            nodes = self._core_v1.list_node()
            m.node_count = len(nodes.items)

            for node in nodes.items:
                node_info = {
                    "name": node.metadata.name,
                    "status": "Unknown",
                    "cpu_capacity": node.status.capacity.get("cpu", "0"),
                    "mem_capacity": node.status.capacity.get("memory", "0"),
                    "pods_capacity": node.status.capacity.get("pods", "0"),
                }

                # 检查节点就绪状态
                for condition in node.status.conditions or []:
                    if condition.type == "Ready":
                        node_info["status"] = (
                            "Ready" if condition.status == "True" else "NotReady"
                        )
                        if condition.status == "True":
                            m.node_ready_count += 1

                m.nodes.append(node_info)

        except Exception as e:
            logger.error(f"节点信息采集失败: {e}")

    def _collect_pods(self, m: K8sMetrics):
        """
        采集 Pod 信息

        包括：各状态 Pod 数量、异常 Pod 列表
        """
        try:
            # 按命名空间采集
            if self.namespaces:
                all_pods = []
                for ns in self.namespaces:
                    pods = self._core_v1.list_namespaced_pod(namespace=ns)
                    all_pods.extend(pods.items)
            else:
                pods = self._core_v1.list_pod_for_all_namespaces()
                all_pods = pods.items

            m.pod_total = len(all_pods)

            for pod in all_pods:
                phase = pod.status.phase

                if phase == "Running":
                    m.pod_running += 1
                elif phase == "Pending":
                    m.pod_pending += 1
                    # 记录异常 Pod
                    m.unhealthy_pods.append({
                        "name": pod.metadata.name,
                        "namespace": pod.metadata.namespace,
                        "status": "Pending",
                        "reason": self._get_pod_reason(pod),
                    })
                elif phase == "Failed":
                    m.pod_failed += 1
                    m.unhealthy_pods.append({
                        "name": pod.metadata.name,
                        "namespace": pod.metadata.namespace,
                        "status": "Failed",
                        "reason": self._get_pod_reason(pod),
                    })
                elif phase == "Succeeded":
                    m.pod_succeeded += 1

                # 检查 CrashLoopBackOff
                if pod.status.container_statuses:
                    for cs in pod.status.container_statuses:
                        if cs.state and cs.state.waiting:
                            reason = cs.state.waiting.reason or ""
                            if "CrashLoopBackOff" in reason:
                                m.unhealthy_pods.append({
                                    "name": pod.metadata.name,
                                    "namespace": pod.metadata.namespace,
                                    "status": "CrashLoopBackOff",
                                    "reason": reason,
                                    "restart_count": cs.restart_count,
                                })

        except Exception as e:
            logger.error(f"Pod 信息采集失败: {e}")

    def _collect_deployments(self, m: K8sMetrics):
        """
        采集 Deployment 信息

        包括：部署数量、就绪状态、副本数
        """
        try:
            if self.namespaces:
                all_deployments = []
                for ns in self.namespaces:
                    deps = self._apps_v1.list_namespaced_deployment(namespace=ns)
                    all_deployments.extend(deps.items)
            else:
                deps = self._apps_v1.list_deployment_for_all_namespaces()
                all_deployments = deps.items

            m.deployment_count = len(all_deployments)

            for dep in all_deployments:
                desired = dep.spec.replicas or 0
                available = dep.status.available_replicas or 0
                ready = dep.status.ready_replicas or 0

                dep_info = {
                    "name": dep.metadata.name,
                    "namespace": dep.metadata.namespace,
                    "desired_replicas": desired,
                    "available_replicas": available,
                    "ready_replicas": ready,
                    "up_to_date": dep.status.updated_replicas or 0,
                    "healthy": available == desired and desired > 0,
                }
                m.deployments.append(dep_info)

                if available == desired and desired > 0:
                    m.deployment_available += 1

        except Exception as e:
            logger.error(f"Deployment 信息采集失败: {e}")

    @staticmethod
    def _get_pod_reason(pod) -> str:
        """获取 Pod 异常原因"""
        if pod.status.conditions:
            for condition in pod.status.conditions:
                if condition.status == "False" and condition.message:
                    return condition.message[:100]
        return ""
```

### 4.4 AWS 资源指标采集 `collectors/aws_col.py`

```python
"""
AWS 资源指标采集模块
使用 boto3 获取 EC2、RDS、ELB 等服务的状态
"""

import logging
from typing import Optional, List
from dataclasses import dataclass, field

logger = logging.getLogger(__name__)


@dataclass
class AWSMetrics:
    """
    AWS 资源指标数据类
    """
    timestamp: float = 0.0

    # EC2 实例
    ec2_total: int = 0
    ec2_running: int = 0
    ec2_stopped: int = 0
    ec2_instances: list = field(default_factory=list)

    # RDS 数据库
    rds_total: int = 0
    rds_available: int = 0
    rds_instances: list = field(default_factory=list)

    # ELB 负载均衡
    elb_total: int = 0
    elb_healthy: int = 0
    elb_list: list = field(default_factory=list)

    def to_dict(self) -> dict:
        from dataclasses import asdict
        return asdict(self)


class AWSCollector:
    """
    AWS 资源采集器
    采集 EC2、RDS、ELB 等核心服务状态
    """

    def __init__(self, config: dict):
        """
        初始化 AWS 采集器

        参数:
            config: AWS 配置字典
        """
        self.config = config
        self.region = config.get("region", "us-east-1")
        self.services = config.get("services", {})
        self._clients = {}
        self._initialized = False

        self._init_clients()

    def _init_clients(self):
        """初始化 boto3 客户端"""
        try:
            import boto3

            # 构建会话参数
            session_kwargs = {"region_name": self.region}

            access_key = self.config.get("access_key_id")
            secret_key = self.config.get("secret_access_key")

            if access_key and secret_key:
                session_kwargs["aws_access_key_id"] = access_key
                session_kwargs["aws_secret_access_key"] = secret_key

            session = boto3.Session(**session_kwargs)

            # 按需创建客户端
            if self.services.get("ec2"):
                self._clients["ec2"] = session.client("ec2")
            if self.services.get("rds"):
                self._clients["rds"] = session.client("rds")
            if self.services.get("elb"):
                self._clients["elbv2"] = session.client("elbv2")

            self._initialized = True
            logger.info(f"AWS 客户端初始化成功 (region={self.region})")

        except ImportError:
            logger.warning("boto3 未安装，跳过 AWS 采集")
        except Exception as e:
            logger.error(f"AWS 客户端初始化失败: {e}")

    def collect(self) -> Optional[AWSMetrics]:
        """
        采集所有 AWS 资源指标

        返回:
            Optional[AWSMetrics]: AWS 指标，初始化失败返回 None
        """
        if not self._initialized:
            return None

        import time
        metrics = AWSMetrics(timestamp=time.time())

        if "ec2" in self._clients:
            self._collect_ec2(metrics)
        if "rds" in self._clients:
            self._collect_rds(metrics)
        if "elbv2" in self._clients:
            self._collect_elb(metrics)

        return metrics

    def _collect_ec2(self, m: AWSMetrics):
        """
        采集 EC2 实例状态

        包括：实例数量、运行状态、实例类型
        """
        try:
            ec2 = self._clients["ec2"]
            response = ec2.describe_instances()

            for reservation in response["Reservations"]:
                for instance in reservation["Instances"]:
                    state = instance["State"]["Name"]
                    m.ec2_total += 1

                    if state == "running":
                        m.ec2_running += 1
                    elif state == "stopped":
                        m.ec2_stopped += 1

                    # 获取实例名称
                    name = ""
                    for tag in instance.get("Tags", []):
                        if tag["Key"] == "Name":
                            name = tag["Value"]
                            break

                    m.ec2_instances.append({
                        "id": instance["InstanceId"],
                        "name": name,
                        "type": instance["InstanceType"],
                        "state": state,
                        "az": instance["Placement"]["AvailabilityZone"],
                        "private_ip": instance.get("PrivateIpAddress", ""),
                        "public_ip": instance.get("PublicIpAddress", ""),
                    })

            logger.info(
                f"EC2: 总计 {m.ec2_total}, "
                f"运行 {m.ec2_running}, 停止 {m.ec2_stopped}"
            )

        except Exception as e:
            logger.error(f"EC2 信息采集失败: {e}")

    def _collect_rds(self, m: AWSMetrics):
        """
        采集 RDS 数据库实例状态

        包括：实例数量、可用状态、引擎类型、存储空间
        """
        try:
            rds = self._clients["rds"]
            response = rds.describe_db_instances()

            for db in response["DBInstances"]:
                m.rds_total += 1
                status = db["DBInstanceStatus"]

                if status == "available":
                    m.rds_available += 1

                m.rds_instances.append({
                    "id": db["DBInstanceIdentifier"],
                    "engine": f"{db['Engine']} {db.get('EngineVersion', '')}",
                    "class": db["DBInstanceClass"],
                    "status": status,
                    "storage_gb": db.get("AllocatedStorage", 0),
                    "multi_az": db.get("MultiAZ", False),
                    "endpoint": db.get("Endpoint", {}).get("Address", ""),
                })

            logger.info(
                f"RDS: 总计 {m.rds_total}, 可用 {m.rds_available}"
            )

        except Exception as e:
            logger.error(f"RDS 信息采集失败: {e}")

    def _collect_elb(self, m: AWSMetrics):
        """
        采集 ELB (ALB/NLB) 状态

        包括：负载均衡器数量、状态、目标组健康状况
        """
        try:
            elbv2 = self._clients["elbv2"]
            response = elbv2.describe_load_balancers()

            for lb in response["LoadBalancers"]:
                m.elb_total += 1
                state = lb["State"]["Code"]

                if state == "active":
                    m.elb_healthy += 1

                m.elb_list.append({
                    "name": lb["LoadBalancerName"],
                    "dns": lb["DNSName"],
                    "type": lb["Type"],      # application / network
                    "scheme": lb["Scheme"],  # internet-facing / internal
                    "state": state,
                    "az_count": len(lb.get("AvailabilityZones", [])),
                })

            logger.info(
                f"ELB: 总计 {m.elb_total}, 活跃 {m.elb_healthy}"
            )

        except Exception as e:
            logger.error(f"ELB 信息采集失败: {e}")
```

### 4.6 Redis 指标缓存 `cache/redis_cache.py`

```python
"""
Redis 指标缓存模块
缓存最近的指标数据，支持历史查询和趋势分析
"""

import json
import time
import logging
from typing import Optional, List

import redis

logger = logging.getLogger(__name__)


class MetricsCache:
    """
    指标数据缓存
    使用 Redis 存储最近的指标采样数据
    """

    KEY_PREFIX = "metrics:"
    MAX_SAMPLES = 720  # 最多保存 720 个采样点（如10秒一次=2小时）

    def __init__(self, config: dict):
        """
        初始化 Redis 缓存

        参数:
            config: Redis 配置字典
        """
        self.redis = redis.Redis(
            host=config.get("host", "127.0.0.1"),
            port=config.get("port", 6379),
            db=config.get("db", 2),
            password=config.get("password") or None,
            decode_responses=True,
        )
        self.redis.ping()
        logger.info("Redis 缓存连接成功")

    def store_system_metrics(self, metrics_dict: dict):
        """
        缓存系统指标数据点

        参数:
            metrics_dict: 指标数据字典
        """
        key = f"{self.KEY_PREFIX}system:latest"
        # 存储最新值
        self.redis.set(key, json.dumps(metrics_dict), ex=120)

        # 存储历史序列（用于趋势图）
        ts_key = f"{self.KEY_PREFIX}system:timeseries"
        sample = {
            "ts": time.time(),
            "cpu": metrics_dict.get("cpu_usage_percent", 0),
            "mem": metrics_dict.get("mem_usage_percent", 0),
            "load": metrics_dict.get("load_1min", 0),
            "net_recv": metrics_dict.get("net_bytes_recv", 0),
            "net_sent": metrics_dict.get("net_bytes_sent", 0),
            "connections": metrics_dict.get("net_connections", 0),
        }
        self.redis.lpush(ts_key, json.dumps(sample))
        self.redis.ltrim(ts_key, 0, self.MAX_SAMPLES - 1)

    def store_k8s_metrics(self, metrics_dict: dict):
        """缓存 K8s 指标"""
        key = f"{self.KEY_PREFIX}k8s:latest"
        self.redis.set(key, json.dumps(metrics_dict), ex=120)

    def store_aws_metrics(self, metrics_dict: dict):
        """缓存 AWS 指标"""
        key = f"{self.KEY_PREFIX}aws:latest"
        self.redis.set(key, json.dumps(metrics_dict), ex=300)

    def get_latest_system(self) -> Optional[dict]:
        """获取最新的系统指标"""
        data = self.redis.get(f"{self.KEY_PREFIX}system:latest")
        return json.loads(data) if data else None

    def get_latest_k8s(self) -> Optional[dict]:
        """获取最新的 K8s 指标"""
        data = self.redis.get(f"{self.KEY_PREFIX}k8s:latest")
        return json.loads(data) if data else None

    def get_latest_aws(self) -> Optional[dict]:
        """获取最新的 AWS 指标"""
        data = self.redis.get(f"{self.KEY_PREFIX}aws:latest")
        return json.loads(data) if data else None

    def get_timeseries(self, count: int = 60) -> List[dict]:
        """
        获取历史时间序列数据（用于趋势图）

        参数:
            count: 获取数据点数量

        返回:
            List[dict]: 时间序列数据列表（从旧到新）
        """
        key = f"{self.KEY_PREFIX}system:timeseries"
        data = self.redis.lrange(key, 0, count - 1)
        # 反转为时间正序
        return [json.loads(d) for d in reversed(data)]

    def store_alert(self, alert_type: str, message: str, value: float):
        """
        存储告警记录

        参数:
            alert_type: 告警类型
            message: 告警消息
            value: 触发告警的指标值
        """
        key = f"{self.KEY_PREFIX}alerts"
        alert = {
            "type": alert_type,
            "message": message,
            "value": value,
            "time": time.time(),
        }
        self.redis.lpush(key, json.dumps(alert))
        self.redis.ltrim(key, 0, 99)  # 保留最近 100 条告警

    def get_recent_alerts(self, count: int = 20) -> List[dict]:
        """获取最近的告警记录"""
        key = f"{self.KEY_PREFIX}alerts"
        data = self.redis.lrange(key, 0, count - 1)
        return [json.loads(d) for d in data]
```

### 4.7 定时任务调度 `scheduler/jobs.py`

```python
"""
定时任务调度模块
使用 APScheduler 管理各采集器的定时执行
"""

import logging

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.triggers.interval import IntervalTrigger

from collectors.system import SystemCollector
from collectors.kubernetes_col import KubernetesCollector
from collectors.aws_col import AWSCollector
from exporters.prometheus import PrometheusExporter
from cache.redis_cache import MetricsCache

logger = logging.getLogger(__name__)


class MetricsScheduler:
    """
    指标采集调度器
    管理各采集器的定时执行和数据流转
    """

    def __init__(self, config: dict, cache: MetricsCache, exporter: PrometheusExporter):
        """
        初始化调度器

        参数:
            config: 全局配置
            cache: Redis 缓存实例
            exporter: Prometheus 导出器实例
        """
        self.config = config
        self.cache = cache
        self.exporter = exporter
        self.alert_thresholds = config.get("alerts", {})

        # 创建采集器
        self.system_collector = SystemCollector(config.get("system", {}))

        self.k8s_collector = None
        if config.get("kubernetes", {}).get("enabled"):
            self.k8s_collector = KubernetesCollector(config["kubernetes"])

        self.aws_collector = None
        if config.get("aws", {}).get("enabled"):
            self.aws_collector = AWSCollector(config["aws"])

        # 创建调度器
        self.scheduler = BackgroundScheduler(
            timezone="UTC",
            job_defaults={
                "coalesce": True,       # 合并错过的执行
                "max_instances": 1,     # 同一任务最多一个实例
                "misfire_grace_time": 30,
            }
        )

    def start(self):
        """
        启动调度器，注册所有定时任务
        """
        intervals = self.config.get("intervals", {})

        # 注册系统指标采集任务
        system_interval = intervals.get("system", 10)
        self.scheduler.add_job(
            self._collect_system,
            trigger=IntervalTrigger(seconds=system_interval),
            id="system_collector",
            name="系统指标采集",
        )

        # 注册 K8s 采集任务
        if self.k8s_collector:
            k8s_interval = intervals.get("kubernetes", 30)
            self.scheduler.add_job(
                self._collect_k8s,
                trigger=IntervalTrigger(seconds=k8s_interval),
                id="k8s_collector",
                name="K8s集群采集",
            )

        # 注册 AWS 采集任务
        if self.aws_collector:
            aws_interval = intervals.get("aws", 60)
            self.scheduler.add_job(
                self._collect_aws,
                trigger=IntervalTrigger(seconds=aws_interval),
                id="aws_collector",
                name="AWS资源采集",
            )

        self.scheduler.start()
        logger.info("指标采集调度器已启动")

        # 立即执行一次采集
        self._collect_system()
        if self.k8s_collector:
            self._collect_k8s()
        if self.aws_collector:
            self._collect_aws()

    def stop(self):
        """停止调度器"""
        self.scheduler.shutdown()
        logger.info("指标采集调度器已停止")

    def _collect_system(self):
        """系统指标采集任务"""
        try:
            metrics = self.system_collector.collect()
            metrics_dict = metrics.to_dict()

            # 更新 Prometheus 指标
            self.exporter.update_system_metrics(metrics)

            # 存入 Redis 缓存
            self.cache.store_system_metrics(metrics_dict)

            # 检查告警
            self._check_alerts(metrics)

            logger.debug(
                f"系统采集完成: CPU={metrics.cpu_usage_percent:.1f}%, "
                f"MEM={metrics.mem_usage_percent:.1f}%"
            )

        except Exception as e:
            logger.error(f"系统指标采集失败: {e}")

    def _collect_k8s(self):
        """K8s 集群采集任务"""
        try:
            metrics = self.k8s_collector.collect()
            if metrics:
                self.exporter.update_k8s_metrics(metrics)
                self.cache.store_k8s_metrics(metrics.to_dict())
                logger.debug(
                    f"K8s 采集完成: Nodes={metrics.node_count}, "
                    f"Pods={metrics.pod_total}"
                )
        except Exception as e:
            logger.error(f"K8s 指标采集失败: {e}")

    def _collect_aws(self):
        """AWS 资源采集任务"""
        try:
            metrics = self.aws_collector.collect()
            if metrics:
                self.cache.store_aws_metrics(metrics.to_dict())
                logger.debug(
                    f"AWS 采集完成: EC2={metrics.ec2_total}, "
                    f"RDS={metrics.rds_total}"
                )
        except Exception as e:
            logger.error(f"AWS 指标采集失败: {e}")

    def _check_alerts(self, metrics):
        """
        检查指标是否超过告警阈值

        参数:
            metrics: 系统指标对象
        """
        # CPU 告警
        cpu_threshold = self.alert_thresholds.get("cpu_usage_percent", 85)
        if metrics.cpu_usage_percent > cpu_threshold:
            self.cache.store_alert(
                "cpu_high",
                f"CPU 使用率 {metrics.cpu_usage_percent:.1f}% 超过阈值 {cpu_threshold}%",
                metrics.cpu_usage_percent
            )

        # 内存告警
        mem_threshold = self.alert_thresholds.get("memory_usage_percent", 90)
        if metrics.mem_usage_percent > mem_threshold:
            self.cache.store_alert(
                "memory_high",
                f"内存使用率 {metrics.mem_usage_percent:.1f}% 超过阈值 {mem_threshold}%",
                metrics.mem_usage_percent
            )

        # 磁盘告警
        disk_threshold = self.alert_thresholds.get("disk_usage_percent", 90)
        for part in metrics.disk_partitions:
            if part["usage_percent"] > disk_threshold:
                self.cache.store_alert(
                    "disk_high",
                    f"磁盘 {part['mountpoint']} 使用率 {part['usage_percent']:.1f}% 超过阈值",
                    part["usage_percent"]
                )

        # 僵尸进程告警
        if metrics.process_zombie > 0:
            self.cache.store_alert(
                "zombie_process",
                f"检测到 {metrics.process_zombie} 个僵尸进程",
                metrics.process_zombie
            )
```

```python
def _bytes_to_human(n: int) -> str:
    """
    将字节数转换为人类可读格式

    参数:
        n: 字节数

    返回:
        str: 如 "1.5 GB"
    """
    for unit in ("B", "KB", "MB", "GB", "TB"):
        if abs(n) < 1024:
            return f"{n:.1f} {unit}"
        n /= 1024
    return f"{n:.1f} PB"


def _percent_bar(percent: float, width: int = 20) -> Text:
    """
    生成百分比进度条

    参数:
        percent: 百分比值 (0-100)
        width: 进度条宽度

    返回:
        Text: rich 格式的进度条
    """
    filled = int(width * percent / 100)
    empty = width - filled

    # 根据百分比选择颜色
    if percent < 60:
        color = "green"
    elif percent < 80:
        color = "yellow"
    else:
        color = "red"

    bar = Text()
    bar.append("█" * filled, style=color)
    bar.append("░" * empty, style="dim")
    bar.append(f" {percent:.1f}%", style=f"bold {color}")
    return bar


def _create_cpu_panel(data: dict) -> Panel:

def _create_network_panel(data: dict) -> Panel:
    """创建网络信息面板"""

    content.append(f"发送: {_bytes_to_human(data.get('net_bytes_sent', 0))}\n")
    content.append(f"接收: {_bytes_to_human(data.get('net_bytes_recv', 0))}\n")
    content.append(f"连接数: {data.get('net_connections', 0)}\n")

    # 连接状态分布

def _create_alert_panel(cache_instance: MetricsCache) -> Panel:
    """创建告警面板"""

    content = Text()

    if not alerts:
        content.append("无告警", style="green")
    else:
        for alert in alerts:
            import datetime
            ts = datetime.datetime.fromtimestamp(
                alert.get("time", 0)
            ).strftime("%H:%M:%S")
            content.append(f"[{ts}] ", style="dim")
            content.append(f"{alert.get('message', '')}\n", style="red")

    return Panel(content, title="[bold red]告警[/]", border_style="red")


def run_dashboard(cache_instance: MetricsCache, refresh_interval: int = 5):
    """
    运行终端监控大盘

    参数:
        cache_instance: Redis 缓存实例
        refresh_interval: 刷新间隔（秒）
    """
    console.print("[bold]启动终端监控大盘...[/bold]\n")
    console.print("按 Ctrl+C 退出\n")

    def generate_layout():
        """生成大盘布局"""
        data = cache_instance.get_latest_system() or {}

        layout = Layout()
        layout.split_column(
            Layout(name="header", size=3),
            Layout(name="main", ratio=3),
            Layout(name="bottom", ratio=2),
        )

        # 头部
        header_text = Text(
            f"  监控大盘 | {time.strftime('%Y-%m-%d %H:%M:%S')} | "
            f"刷新间隔: {refresh_interval}s",
            style="bold white on blue"
        )
        layout["header"].update(Panel(header_text, style="blue"))

        # 主区域（上下分栏）
        layout["main"].split_row(
            Layout(_create_cpu_panel(data), name="cpu"),
            Layout(_create_memory_panel(data), name="memory"),
            Layout(_create_network_panel(data), name="network"),
        )

        layout["bottom"].split_row(
            Layout(_create_disk_panel(data), name="disk"),
            Layout(_create_process_panel(data), name="process"),
            Layout(_create_alert_panel(cache_instance), name="alerts"),
        )

        return layout

    try:
        with Live(
            generate_layout(),
            console=console,
            refresh_per_second=1,
            screen=True,
        ) as live:
            while True:
                time.sleep(refresh_interval)
                live.update(generate_layout())

    except KeyboardInterrupt:
        console.print("\n[bold]大盘已退出[/bold]")
```

```python
import logging
import sys
from typing import Optional

import typer

    help="自定义监控大盘 - 采集、展示、告警"

def load_config(path: str) -> dict:
    """加载配置"""
    with open(path, "r", encoding="utf-8") as f:
        return yaml.safe_load(f)


def setup_logging(level: str = "INFO"):
    """配置日志"""
    logging.basicConfig(
        level=getattr(logging, level.upper(), logging.INFO),
        format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
        handlers=[logging.StreamHandler(sys.stdout)],
    )


@app.command()
def start(
    config_file: str = typer.Option("config/config.yaml", "--config", "-c"),
    mode: str = typer.Option(
        "all", "--mode", "-m",
        help="运行模式: all(全部) / api(仅API) / dashboard(仅终端大盘) / collect(仅采集)"
    ),
    log_level: str = typer.Option("INFO", "--log-level"),
):
    """
    启动监控大盘

    模式:
    - all: 启动采集器 + API + Prometheus
    - api: 仅启动 API 服务
    - dashboard: 仅启动终端大盘
    - collect: 仅启动采集器（不启动 API）
    """
    setup_logging(log_level)
    config = load_config(config_file)

    from cache.redis_cache import MetricsCache
    from exporters.prometheus import PrometheusExporter
    from scheduler.jobs import MetricsScheduler

    # 初始化组件
    cache_instance = MetricsCache(config["redis"])

    prom_config = config.get("prometheus", {})
    exporter_instance = PrometheusExporter(
        prefix=prom_config.get("prefix", "sre_monitor"),
        port=prom_config.get("port", 9090),
    )

    if mode in ("all", "collect"):
        # 启动采集调度器
        scheduler = MetricsScheduler(config, cache_instance, exporter_instance)
        scheduler.start()
        typer.echo("采集调度器已启动")

    if mode == "dashboard":
        # 先启动采集
        scheduler = MetricsScheduler(config, cache_instance, exporter_instance)
        scheduler.start()

        # 启动终端大盘
        from dashboard.terminal import run_dashboard
        dash_config = config.get("dashboard", {})
        run_dashboard(
            cache_instance,
            refresh_interval=dash_config.get("refresh_interval", 5),
        )
        return

    if mode in ("all", "api"):
        # 启动 API 服务
        from api import server as api_module
        api_module.cache = cache_instance
        api_module.exporter = exporter_instance

        api_config = config.get("api", {})
        typer.echo(
            f"API 服务: http://{api_config.get('host', '0.0.0.0')}"
            f":{api_config.get('port', 8080)}"
        )
        typer.echo(
            f"Prometheus: http://0.0.0.0:{prom_config.get('port', 9090)}/metrics"
        )

        uvicorn.run(
            api_module.api_app,
            host=api_config.get("host", "0.0.0.0"),
            port=api_config.get("port", 8080),
        )

    if mode == "collect":
        # 采集模式下保持运行
        typer.echo("采集模式运行中，按 Ctrl+C 停止")
        try:
            import time
            while True:
                time.sleep(1)
        except KeyboardInterrupt:
            typer.echo("\n已停止")
```

```python
    assert d["cpu_usage_percent"] == 50.0
    assert d["mem_usage_percent"] == 70.0
```

## 7. 扩展思路

1. **Grafana 集成**：直接作为 Prometheus 数据源接入 Grafana 展示
2. **告警通知**：集成钉钉/企业微信/邮件/PagerDuty 告警通道
3. **自定义指标**：支持用户自定义采集脚本和指标定义
4. **多节点监控**：Agent 模式，中心服务器聚合多节点数据
5. **SLO/SLI 计算**：基于采集数据自动计算服务等级目标
6. **容量预测**：基于历史数据的线性回归预测资源用尽时间
7. **自动扩容**：与 K8s HPA 或 AWS AutoScaling 集成
8. **历史数据持久化**：将 Redis 中的时间序列定期归档到 InfluxDB/TimescaleDB
