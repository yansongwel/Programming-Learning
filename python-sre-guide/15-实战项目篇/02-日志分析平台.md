# 实战项目二：日志分析平台

## 1. 项目概述

### 1.1 解决什么问题

在生产环境中，Nginx 和各类应用每天产生海量日志。运维工程师需要快速定位问题、分析访问趋势、识别异常流量。
本项目实现了一个**全功能日志分析平台**，支持多种日志格式解析、数据聚合分析、Elasticsearch 存储查询、
REST API 接口以及 HTML 报告生成，同时支持实时日志流监控模式。

### 4.2 解析器基类 `parser/base.py`

```python
"""
日志解析器基类
定义日志解析的统一接口和通用逻辑
"""

import re
import logging
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from datetime import datetime
from typing import Optional, List, Iterator, Dict
from pathlib import Path

logger = logging.getLogger(__name__)


@dataclass
class LogEntry:
    """
    解析后的日志条目数据类
    所有解析器的输出统一为此格式
    """
    timestamp: Optional[datetime] = None     # 日志时间
    ip_address: str = ""                     # 客户端 IP
    method: str = ""                         # HTTP 方法
    url: str = ""                            # 请求 URL
    status_code: int = 0                     # HTTP 状态码
    response_size: int = 0                   # 响应大小（字节）
    response_time: float = 0.0              # 响应时间（秒）
    user_agent: str = ""                     # User-Agent
    referer: str = ""                        # Referer
    log_level: str = ""                      # 日志级别（应用日志用）
    message: str = ""                        # 日志消息
    source_file: str = ""                    # 来源文件
    raw_line: str = ""                       # 原始日志行
    extra: dict = field(default_factory=dict)  # 扩展字段

    def to_dict(self) -> dict:
        """转换为字典，方便存入 ES"""
        result = {
            "timestamp": self.timestamp.isoformat() if self.timestamp else None,
            "ip_address": self.ip_address,
            "method": self.method,
            "url": self.url,
            "status_code": self.status_code,
            "response_size": self.response_size,
            "response_time": self.response_time,
            "user_agent": self.user_agent,
            "referer": self.referer,
            "log_level": self.log_level,
            "message": self.message,
            "source_file": self.source_file,
        }
        result.update(self.extra)
        return result


class BaseLogParser(ABC):
    """
    日志解析器抽象基类
    定义所有解析器必须实现的接口
    """

    def __init__(self, name: str = "base"):
        """
        初始化解析器

        参数:
            name: 解析器名称标识
        """
        self.name = name
        self._compiled_patterns: Dict[str, re.Pattern] = {}
        self._parse_count = 0      # 已解析行数
        self._error_count = 0      # 解析失败行数

    @abstractmethod
    def parse_line(self, line: str) -> Optional[LogEntry]:
        """
        解析单行日志

        参数:
            line: 日志原始行文本

        返回:
            Optional[LogEntry]: 解析成功返回 LogEntry，失败返回 None
        """
        pass

    def parse_file(self, file_path: str) -> Iterator[LogEntry]:
        """
        解析整个日志文件（生成器方式，节省内存）

        参数:
            file_path: 日志文件路径

        生成:
            LogEntry: 逐行解析的日志条目
        """
        path = Path(file_path)
        if not path.exists():
            logger.error(f"日志文件不存在: {file_path}")
            return

        logger.info(f"开始解析日志文件: {file_path}")

        with open(path, "r", encoding="utf-8", errors="replace") as f:
            for line_num, line in enumerate(f, 1):
                line = line.strip()
                if not line:
                    continue

                self._parse_count += 1
                entry = self.parse_line(line)

                if entry:
                    entry.source_file = str(file_path)
                    entry.raw_line = line
                    yield entry
                else:
                    self._error_count += 1
                    if self._error_count <= 10:
                        logger.warning(
                            f"第 {line_num} 行解析失败: {line[:100]}"
                        )

        logger.info(
            f"文件解析完成: {file_path}, "
            f"总行数={self._parse_count}, "
            f"失败={self._error_count}"
        )

    def parse_files(self, file_paths: List[str]) -> Iterator[LogEntry]:
        """
        解析多个日志文件

        参数:
            file_paths: 日志文件路径列表

        生成:
            LogEntry: 所有文件中的日志条目
        """
        for fp in file_paths:
            yield from self.parse_file(fp)

    def _compile_pattern(self, name: str, pattern: str) -> re.Pattern:
        """
        编译并缓存正则表达式（避免重复编译）

        参数:
            name: 模式名称
            pattern: 正则表达式字符串

        返回:
            re.Pattern: 编译后的正则对象
        """
        if name not in self._compiled_patterns:
            self._compiled_patterns[name] = re.compile(pattern)
        return self._compiled_patterns[name]

    @property
    def stats(self) -> dict:
        """返回解析统计"""
        return {
            "parser": self.name,
            "total_lines": self._parse_count,
            "failed_lines": self._error_count,
            "success_rate": (
                f"{(1 - self._error_count / self._parse_count) * 100:.1f}%"
                if self._parse_count > 0 else "N/A"
            ),
        }
```

### 4.3 Nginx 日志解析器 `parser/nginx.py`

```python
"""
Nginx 日志解析器
支持 combined 格式和自定义格式
"""

import re
import logging
from datetime import datetime
from typing import Optional

from parser.base import BaseLogParser, LogEntry

logger = logging.getLogger(__name__)

# Nginx combined 日志格式正则
# 示例：192.168.1.1 - - [25/Mar/2026:10:15:32 +0800] "GET /api/users HTTP/1.1" 200 1234 "https://example.com" "Mozilla/5.0"
NGINX_COMBINED_PATTERN = (
    r'(?P<ip>\d+\.\d+\.\d+\.\d+)\s+'          # 客户端 IP
    r'(?P<ident>\S+)\s+'                         # 身份标识
    r'(?P<user>\S+)\s+'                          # 用户名
    r'\[(?P<timestamp>[^\]]+)\]\s+'              # 时间戳
    r'"(?P<method>\S+)\s+'                       # HTTP 方法
    r'(?P<url>\S+)\s+'                           # 请求 URL
    r'(?P<protocol>[^"]+)"\s+'                   # 协议版本
    r'(?P<status>\d{3})\s+'                      # 状态码
    r'(?P<size>\d+|-)\s+'                        # 响应大小
    r'"(?P<referer>[^"]*)"\s+'                   # Referer
    r'"(?P<user_agent>[^"]*)"'                   # User-Agent
)

# 带响应时间的扩展格式
NGINX_EXTENDED_PATTERN = (
    NGINX_COMBINED_PATTERN +
    r'(?:\s+(?P<response_time>[\d.]+))?'         # 响应时间（可选）
)

# Nginx 日志时间格式
NGINX_TIME_FORMAT = "%d/%b/%Y:%H:%M:%S %z"


class NginxLogParser(BaseLogParser):
    """
    Nginx 访问日志解析器
    支持标准 combined 格式和带响应时间的扩展格式
    """

    def __init__(self, custom_pattern: str = None):
        """
        初始化 Nginx 日志解析器

        参数:
            custom_pattern: 自定义正则表达式（可选）
        """
        super().__init__(name="nginx")

        # 使用自定义或默认正则
        if custom_pattern:
            self._pattern = self._compile_pattern("custom", custom_pattern)
        else:
            self._pattern = self._compile_pattern(
                "extended", NGINX_EXTENDED_PATTERN
            )
            # 备用：标准 combined 格式
            self._fallback_pattern = self._compile_pattern(
                "combined", NGINX_COMBINED_PATTERN
            )

    def parse_line(self, line: str) -> Optional[LogEntry]:
        """
        解析单行 Nginx 日志

        参数:
            line: Nginx 日志原始行

        返回:
            Optional[LogEntry]: 解析成功返回 LogEntry，失败返回 None
        """
        # 尝试匹配扩展格式
        match = self._pattern.match(line)

        # 回退到标准 combined 格式
        if not match and hasattr(self, "_fallback_pattern"):
            match = self._fallback_pattern.match(line)

        if not match:
            return None

        groups = match.groupdict()

        # 解析时间戳
        timestamp = None
        try:
            timestamp = datetime.strptime(
                groups["timestamp"], NGINX_TIME_FORMAT
            )
        except (ValueError, KeyError):
            # 时间解析失败不影响其他字段
            pass

        # 解析响应大小（可能为 "-"）
        try:
            size = int(groups.get("size", 0))
        except (ValueError, TypeError):
            size = 0

        # 解析响应时间
        response_time = 0.0
        if groups.get("response_time"):
            try:
                response_time = float(groups["response_time"])
            except (ValueError, TypeError):
                pass

        return LogEntry(
            timestamp=timestamp,
            ip_address=groups.get("ip", ""),
            method=groups.get("method", ""),
            url=groups.get("url", ""),
            status_code=int(groups.get("status", 0)),
            response_size=size,
            response_time=response_time,
            referer=groups.get("referer", "-"),
            user_agent=groups.get("user_agent", ""),
        )


class NginxErrorLogParser(BaseLogParser):
    """
    Nginx 错误日志解析器
    """

    # 错误日志正则
    # 示例：2026/03/25 10:15:32 [error] 1234#0: *56789 connect() failed
    ERROR_PATTERN = (
        r'(?P<timestamp>\d{4}/\d{2}/\d{2}\s+\d{2}:\d{2}:\d{2})\s+'
        r'\[(?P<level>\w+)\]\s+'
        r'(?P<pid>\d+)#(?P<tid>\d+):\s+'
        r'(?:\*(?P<conn_id>\d+)\s+)?'
        r'(?P<message>.*)'
    )

    def __init__(self):
        super().__init__(name="nginx-error")
        self._pattern = self._compile_pattern("error", self.ERROR_PATTERN)

    def parse_line(self, line: str) -> Optional[LogEntry]:
        """
        解析单行 Nginx 错误日志

        参数:
            line: Nginx 错误日志原始行

        返回:
            Optional[LogEntry]: 解析结果
        """
        match = self._pattern.match(line)
        if not match:
            return None

        groups = match.groupdict()

        timestamp = None
        try:
            timestamp = datetime.strptime(
                groups["timestamp"], "%Y/%m/%d %H:%M:%S"
            )
        except ValueError:
            pass

        return LogEntry(
            timestamp=timestamp,
            log_level=groups.get("level", ""),
            message=groups.get("message", ""),
            extra={
                "pid": groups.get("pid"),
                "connection_id": groups.get("conn_id"),
            }
        )
```

### 4.4 应用日志解析器 `parser/app_log.py`

```python
"""
应用日志解析器
支持标准文本格式和 JSON 格式的应用日志
"""

import json
import logging
from datetime import datetime
from typing import Optional

from parser.base import BaseLogParser, LogEntry

logger = logging.getLogger(__name__)

# 标准应用日志格式正则
# 示例：2026-03-25 10:15:32.123 [INFO] module_name - 这是一条日志消息
APP_LOG_PATTERN = (
    r'(?P<timestamp>\d{4}-\d{2}-\d{2}\s+\d{2}:\d{2}:\d{2}(?:\.\d+)?)\s+'
    r'\[(?P<level>\w+)\]\s+'
    r'(?P<module>\S+)\s*-\s*'
    r'(?P<message>.*)'
)


class AppLogParser(BaseLogParser):
    """
    应用日志解析器
    支持标准文本格式的应用日志
    """

    def __init__(self):
        super().__init__(name="app-standard")
        self._pattern = self._compile_pattern("app", APP_LOG_PATTERN)

    def parse_line(self, line: str) -> Optional[LogEntry]:
        """
        解析单行应用日志

        参数:
            line: 日志原始行

        返回:
            Optional[LogEntry]: 解析结果
        """
        match = self._pattern.match(line)
        if not match:
            return None

        groups = match.groupdict()

        timestamp = None
        try:
            ts_str = groups["timestamp"]
            # 尝试带毫秒的格式
            if "." in ts_str:
                timestamp = datetime.strptime(ts_str, "%Y-%m-%d %H:%M:%S.%f")
            else:
                timestamp = datetime.strptime(ts_str, "%Y-%m-%d %H:%M:%S")
        except ValueError:
            pass

        return LogEntry(
            timestamp=timestamp,
            log_level=groups.get("level", "").upper(),
            message=groups.get("message", ""),
            extra={"module": groups.get("module", "")},
        )


class JsonLogParser(BaseLogParser):
    """
    JSON 格式日志解析器
    每行一个 JSON 对象
    """

    # JSON 日志字段映射（JSON 字段名 -> LogEntry 字段名）
    FIELD_MAPPING = {
        "timestamp": "timestamp",
        "time": "timestamp",
        "@timestamp": "timestamp",
        "level": "log_level",
        "severity": "log_level",
        "msg": "message",
        "message": "message",
        "ip": "ip_address",
        "remote_addr": "ip_address",
        "method": "method",
        "request_method": "method",
        "url": "url",
        "path": "url",
        "request_uri": "url",
        "status": "status_code",
        "status_code": "status_code",
        "response_size": "response_size",
        "body_bytes_sent": "response_size",
        "response_time": "response_time",
        "request_time": "response_time",
        "user_agent": "user_agent",
        "http_user_agent": "user_agent",
    }

    def __init__(self):
        super().__init__(name="json")

    def parse_line(self, line: str) -> Optional[LogEntry]:
        """
        解析单行 JSON 日志

        参数:
            line: JSON 格式的日志行

        返回:
            Optional[LogEntry]: 解析结果
        """
        try:
            data = json.loads(line)
        except json.JSONDecodeError:
            return None

        if not isinstance(data, dict):
            return None

        entry = LogEntry()
        extra_fields = {}

        for json_key, value in data.items():
            if json_key in self.FIELD_MAPPING:
                attr_name = self.FIELD_MAPPING[json_key]

                # 时间戳特殊处理
                if attr_name == "timestamp" and isinstance(value, str):
                    try:
                        value = datetime.fromisoformat(
                            value.replace("Z", "+00:00")
                        )
                    except ValueError:
                        value = None

                # 状态码和大小转为整数
                if attr_name in ("status_code", "response_size"):
                    try:
                        value = int(value)
                    except (ValueError, TypeError):
                        value = 0

                # 响应时间转为浮点数
                if attr_name == "response_time":
                    try:
                        value = float(value)
                    except (ValueError, TypeError):
                        value = 0.0

                setattr(entry, attr_name, value)
            else:
                # 未映射的字段放入 extra
                extra_fields[json_key] = value

        entry.extra = extra_fields
        return entry
```

### 4.5 统计分析模块 `analyzer/stats.py`

```python
"""
统计分析模块
使用 pandas 和 Counter 对日志数据进行多维度统计分析
"""

import logging
from collections import Counter
from datetime import datetime, timedelta
from typing import List, Dict, Any
from dataclasses import dataclass, field

import pandas as pd

from parser.base import LogEntry

logger = logging.getLogger(__name__)


@dataclass
class AnalysisResult:
    """
    分析结果数据类
    汇总所有维度的统计数据
    """
    # 基本统计
    total_requests: int = 0                      # 请求总数
    time_range_start: datetime = None             # 日志时间范围起始
    time_range_end: datetime = None               # 日志时间范围结束

    # 状态码统计
    status_code_counts: Dict[int, int] = field(default_factory=dict)
    error_rate: float = 0.0                       # 错误率（4xx+5xx）

    # IP 统计
    top_ips: List[tuple] = field(default_factory=list)  # Top IP

    # URL 统计
    top_urls: List[tuple] = field(default_factory=list)  # Top URL

    # 响应时间统计
    avg_response_time: float = 0.0
    p50_response_time: float = 0.0               # P50 响应时间
    p90_response_time: float = 0.0               # P90 响应时间
    p99_response_time: float = 0.0               # P99 响应时间
    slow_requests: List[dict] = field(default_factory=list)  # 慢请求列表

    # 流量统计
    total_bytes: int = 0                          # 总流量（字节）
    requests_per_minute: Dict[str, int] = field(default_factory=dict)  # 每分钟请求数

    # HTTP 方法统计
    method_counts: Dict[str, int] = field(default_factory=dict)

    # User-Agent 统计
    top_user_agents: List[tuple] = field(default_factory=list)

    def to_dict(self) -> dict:
        """转换为字典，方便 JSON 序列化"""
        return {
            "total_requests": self.total_requests,
            "time_range": {
                "start": str(self.time_range_start),
                "end": str(self.time_range_end),
            },
            "status_codes": self.status_code_counts,
            "error_rate": f"{self.error_rate:.2f}%",
            "top_ips": [{"ip": ip, "count": c} for ip, c in self.top_ips],
            "top_urls": [{"url": u, "count": c} for u, c in self.top_urls],
            "response_time": {
                "avg": round(self.avg_response_time, 3),
                "p50": round(self.p50_response_time, 3),
                "p90": round(self.p90_response_time, 3),
                "p99": round(self.p99_response_time, 3),
            },
            "total_bytes": self.total_bytes,
            "method_counts": self.method_counts,
        }


class LogAnalyzer:
    """
    日志分析器
    对解析后的日志条目进行多维度统计分析
    """

    def __init__(self, top_n: int = 20, slow_threshold: float = 3.0):
        """
        初始化分析器

        参数:
            top_n: Top N 统计数量
            slow_threshold: 慢请求阈值（秒）
        """
        self.top_n = top_n
        self.slow_threshold = slow_threshold

        # 内部计数器
        self._ip_counter = Counter()
        self._url_counter = Counter()
        self._status_counter = Counter()
        self._method_counter = Counter()
        self._ua_counter = Counter()
        self._entries: List[dict] = []

    def add_entry(self, entry: LogEntry):
        """
        添加一条日志条目进行分析

        参数:
            entry: 解析后的日志条目
        """
        # 更新各维度计数器
        if entry.ip_address:
            self._ip_counter[entry.ip_address] += 1
        if entry.url:
            # URL 去除查询参数后统计
            clean_url = entry.url.split("?")[0]
            self._url_counter[clean_url] += 1
        if entry.status_code:
            self._status_counter[entry.status_code] += 1
        if entry.method:
            self._method_counter[entry.method] += 1
        if entry.user_agent:
            # User-Agent 截取前 80 字符
            short_ua = entry.user_agent[:80]
            self._ua_counter[short_ua] += 1

        # 存储条目数据（用于 pandas 分析）
        self._entries.append(entry.to_dict())

    def add_entries(self, entries):
        """
        批量添加日志条目

        参数:
            entries: 日志条目迭代器
        """
        count = 0
        for entry in entries:
            self.add_entry(entry)
            count += 1
            if count % 100000 == 0:
                logger.info(f"已处理 {count} 条日志...")

        logger.info(f"共处理 {count} 条日志")

    def analyze(self) -> AnalysisResult:
        """
        执行全量分析，返回分析结果

        返回:
            AnalysisResult: 包含所有维度统计数据的结果对象
        """
        result = AnalysisResult()

        if not self._entries:
            logger.warning("无日志数据可分析")
            return result

        # 转换为 pandas DataFrame
        df = pd.DataFrame(self._entries)
        result.total_requests = len(df)

        logger.info(f"开始分析 {result.total_requests} 条日志...")

        # 时间范围
        self._analyze_time_range(df, result)

        # 状态码统计
        self._analyze_status_codes(df, result)

        # IP 统计
        result.top_ips = self._ip_counter.most_common(self.top_n)

        # URL 统计
        result.top_urls = self._url_counter.most_common(self.top_n)

        # HTTP 方法统计
        result.method_counts = dict(self._method_counter)

        # User-Agent 统计
        result.top_user_agents = self._ua_counter.most_common(10)

        # 响应时间分析
        self._analyze_response_time(df, result)

        # 流量统计
        self._analyze_traffic(df, result)

        # 每分钟请求数
        self._analyze_requests_per_minute(df, result)

        logger.info("日志分析完成")
        return result

    def _analyze_time_range(self, df: pd.DataFrame, result: AnalysisResult):
        """
        分析日志时间范围

        参数:
            df: 日志数据 DataFrame
            result: 分析结果对象
        """
        if "timestamp" in df.columns:
            timestamps = pd.to_datetime(
                df["timestamp"], errors="coerce"
            ).dropna()
            if not timestamps.empty:
                result.time_range_start = timestamps.min().to_pydatetime()
                result.time_range_end = timestamps.max().to_pydatetime()

    def _analyze_status_codes(self, df: pd.DataFrame, result: AnalysisResult):
        """
        分析 HTTP 状态码分布

        参数:
            df: 日志数据 DataFrame
            result: 分析结果对象
        """
        result.status_code_counts = dict(self._status_counter)

        # 计算错误率（4xx + 5xx）
        total = sum(self._status_counter.values())
        if total > 0:
            error_count = sum(
                count for code, count in self._status_counter.items()
                if code >= 400
            )
            result.error_rate = (error_count / total) * 100

    def _analyze_response_time(self, df: pd.DataFrame, result: AnalysisResult):
        """
        分析响应时间分布，计算 P50/P90/P99 等百分位

        参数:
            df: 日志数据 DataFrame
            result: 分析结果对象
        """
        if "response_time" not in df.columns:
            return

        # 过滤有效的响应时间
        rt = pd.to_numeric(df["response_time"], errors="coerce").dropna()
        rt = rt[rt > 0]

        if rt.empty:
            return

        result.avg_response_time = rt.mean()
        result.p50_response_time = rt.quantile(0.50)
        result.p90_response_time = rt.quantile(0.90)
        result.p99_response_time = rt.quantile(0.99)

        # 慢请求列表
        slow_mask = rt > self.slow_threshold
        if slow_mask.any():
            slow_df = df.loc[slow_mask.index[slow_mask]].head(50)
            result.slow_requests = [
                {
                    "url": row.get("url", ""),
                    "method": row.get("method", ""),
                    "response_time": row.get("response_time", 0),
                    "status_code": row.get("status_code", 0),
                    "timestamp": str(row.get("timestamp", "")),
                }
                for _, row in slow_df.iterrows()
            ]

    def _analyze_traffic(self, df: pd.DataFrame, result: AnalysisResult):
        """
        分析流量数据

        参数:
            df: 日志数据 DataFrame
            result: 分析结果对象
        """
        if "response_size" in df.columns:
            sizes = pd.to_numeric(
                df["response_size"], errors="coerce"
            ).fillna(0)
            result.total_bytes = int(sizes.sum())

    def _analyze_requests_per_minute(
        self, df: pd.DataFrame, result: AnalysisResult
    ):
        """
        统计每分钟请求数（用于绘制趋势图）

        参数:
            df: 日志数据 DataFrame
            result: 分析结果对象
        """
        if "timestamp" not in df.columns:
            return

        ts = pd.to_datetime(df["timestamp"], errors="coerce").dropna()
        if ts.empty:
            return

        # 按分钟聚合
        minute_counts = ts.dt.floor("min").value_counts().sort_index()
        result.requests_per_minute = {
            str(k): int(v) for k, v in minute_counts.items()
        }
```

```python
"""
实时日志分析模块
支持 tail -f 模式的实时日志流监控
"""

import os
import time
import logging
import threading
from typing import Callable, Optional
from pathlib import Path

from parser.base import BaseLogParser, LogEntry

logger = logging.getLogger(__name__)


class LogTailer:
    """
    日志文件实时跟踪器
    类似 tail -f 功能，持续监控日志文件的新增内容
    """

    def __init__(
        self,
        file_path: str,
        parser: BaseLogParser,
        callback: Callable[[LogEntry], None],
        poll_interval: float = 0.5
    ):
        """
        初始化日志跟踪器

        参数:
            file_path: 日志文件路径
            parser: 日志解析器实例
            callback: 新日志回调函数
            poll_interval: 轮询间隔（秒）
        """
        self.file_path = file_path
        self.parser = parser
        self.callback = callback
        self.poll_interval = poll_interval
        self._running = False
        self._thread: Optional[threading.Thread] = None
        self._position = 0     # 文件读取位置
        self._inode = None     # 文件 inode（用于检测文件轮转）

    def start(self):
        """
        启动实时跟踪（非阻塞，在后台线程运行）
        """
        if self._running:
            logger.warning("日志跟踪器已在运行")
            return

        self._running = True

        # 移动到文件末尾
        path = Path(self.file_path)
        if path.exists():
            self._position = path.stat().st_size
            self._inode = path.stat().st_ino

        self._thread = threading.Thread(
            target=self._tail_loop,
            daemon=True,
            name=f"tailer-{self.file_path}"
        )
        self._thread.start()
        logger.info(f"开始实时跟踪: {self.file_path}")

    def stop(self):
        """停止实时跟踪"""
        self._running = False
        if self._thread:
            self._thread.join(timeout=5)
        logger.info(f"停止实时跟踪: {self.file_path}")

    def _tail_loop(self):
        """
        后台轮询循环
        持续检查文件是否有新内容
        """
        while self._running:
            try:
                path = Path(self.file_path)

                if not path.exists():
                    time.sleep(self.poll_interval)
                    continue

                current_stat = path.stat()

                # 检测文件轮转（inode 变化表示文件被重新创建）
                if self._inode and current_stat.st_ino != self._inode:
                    logger.info("检测到日志文件轮转，重新打开")
                    self._position = 0
                    self._inode = current_stat.st_ino

                # 检查是否有新内容
                if current_stat.st_size > self._position:
                    self._read_new_lines()
                elif current_stat.st_size < self._position:
                    # 文件被截断
                    logger.info("检测到文件截断，从头开始读取")
                    self._position = 0

            except Exception as e:
                logger.error(f"实时跟踪出错: {e}")

            time.sleep(self.poll_interval)

    def _read_new_lines(self):
        """
        读取文件中的新增行并解析
        """
        try:
            with open(self.file_path, "r", encoding="utf-8", errors="replace") as f:
                f.seek(self._position)

                for line in f:
                    line = line.strip()
                    if not line:
                        continue

                    # 解析日志行
                    entry = self.parser.parse_line(line)
                    if entry:
                        entry.source_file = self.file_path
                        entry.raw_line = line
                        # 调用回调函数处理
                        try:
                            self.callback(entry)
                        except Exception as e:
                            logger.error(f"回调函数执行失败: {e}")

                # 更新读取位置
                self._position = f.tell()

        except Exception as e:
            logger.error(f"读取新日志失败: {e}")


class RealtimeAnalyzer:
    """
    实时日志分析器
    聚合实时日志流的统计数据，支持滑动窗口
    """

    def __init__(self, window_seconds: int = 60):
        """
        初始化实时分析器

        参数:
            window_seconds: 滑动窗口大小（秒）
        """
        self.window_seconds = window_seconds
        self._lock = threading.Lock()  # 线程安全锁
        self._entries: list = []       # 窗口内的日志条目
        self._total_count = 0          # 总计数

        # 实时统计
        self.current_rps = 0.0         # 当前 QPS
        self.current_error_rate = 0.0  # 当前错误率
        self.current_avg_rt = 0.0      # 当前平均响应时间

    def on_new_entry(self, entry: LogEntry):
        """
        处理新的日志条目（作为 LogTailer 的回调函数）

        参数:
            entry: 新的日志条目
        """
        with self._lock:
            now = time.time()
            self._entries.append({
                "time": now,
                "status_code": entry.status_code,
                "response_time": entry.response_time,
                "url": entry.url,
            })
            self._total_count += 1

            # 清理过期条目
            cutoff = now - self.window_seconds
            self._entries = [
                e for e in self._entries if e["time"] > cutoff
            ]

            # 更新实时统计
            self._update_stats()

    def _update_stats(self):
        """更新实时统计指标"""
        count = len(self._entries)
        if count == 0:
            return

        # QPS = 窗口内请求数 / 窗口时间
        self.current_rps = count / self.window_seconds

        # 错误率
        error_count = sum(
            1 for e in self._entries if e["status_code"] >= 400
        )
        self.current_error_rate = (error_count / count) * 100

        # 平均响应时间
        response_times = [
            e["response_time"] for e in self._entries
            if e["response_time"] > 0
        ]
        if response_times:
            self.current_avg_rt = sum(response_times) / len(response_times)

    def get_stats(self) -> dict:
        """
        获取当前实时统计数据

        返回:
            dict: 统计数据字典
        """
        with self._lock:
            return {
                "window_seconds": self.window_seconds,
                "requests_in_window": len(self._entries),
                "total_requests": self._total_count,
                "current_rps": round(self.current_rps, 2),
                "current_error_rate": round(self.current_error_rate, 2),
                "current_avg_response_time": round(self.current_avg_rt, 4),
            }
```

### 4.7 Elasticsearch 存储 `storage/es_client.py`

```python
"""
Elasticsearch 存储模块
封装日志数据的索引、查询和聚合操作
"""

import logging
from datetime import datetime, date
from typing import List, Dict, Optional

from elasticsearch import Elasticsearch, helpers

from parser.base import LogEntry

logger = logging.getLogger(__name__)


class ESStorage:
    """
    Elasticsearch 存储类
    管理日志数据的索引和查询
    """

    def __init__(self, config: dict):
        """
        初始化 ES 连接

        参数:
            config: ES 配置字典，包含 hosts/username/password 等
        """
        es_kwargs = {
            "hosts": config.get("hosts", ["http://localhost:9200"]),
            "timeout": config.get("timeout", 30),
        }

        # 可选认证
        username = config.get("username")
        password = config.get("password")
        if username and password:
            es_kwargs["basic_auth"] = (username, password)

        self.es = Elasticsearch(**es_kwargs)
        self.index_prefix = config.get("index_prefix", "log-analysis")

        # 验证连接
        if self.es.ping():
            logger.info("Elasticsearch 连接成功")
        else:
            logger.error("Elasticsearch 连接失败")

    def _get_index_name(self, dt: date = None) -> str:
        """
        获取索引名称（按日期分索引）

        参数:
            dt: 日期，默认为今天

        返回:
            str: 索引名称，如 "log-analysis-2026.03.25"
        """
        if dt is None:
            dt = date.today()
        return f"{self.index_prefix}-{dt.strftime('%Y.%m.%d')}"

    def create_index_template(self):
        """
        创建索引模板，定义字段映射
        """
        template_body = {
            "index_patterns": [f"{self.index_prefix}-*"],
            "template": {
                "settings": {
                    "number_of_shards": 1,
                    "number_of_replicas": 0,
                    "refresh_interval": "5s",
                },
                "mappings": {
                    "properties": {
                        "timestamp": {"type": "date"},
                        "ip_address": {"type": "ip"},
                        "method": {"type": "keyword"},
                        "url": {"type": "keyword"},
                        "status_code": {"type": "integer"},
                        "response_size": {"type": "long"},
                        "response_time": {"type": "float"},
                        "user_agent": {"type": "text"},
                        "referer": {"type": "keyword"},
                        "log_level": {"type": "keyword"},
                        "message": {"type": "text"},
                        "source_file": {"type": "keyword"},
                    }
                }
            }
        }

        self.es.indices.put_index_template(
            name=f"{self.index_prefix}-template",
            body=template_body
        )
        logger.info("索引模板创建成功")

    def bulk_index(self, entries: List[LogEntry], batch_size: int = 1000):
        """
        批量索引日志条目

        参数:
            entries: 日志条目列表
            batch_size: 每批提交数量
        """
        actions = []
        indexed = 0
        errors = 0

        for entry in entries:
            doc = entry.to_dict()

            # 确定目标索引
            dt = entry.timestamp.date() if entry.timestamp else date.today()
            index_name = self._get_index_name(dt)

            action = {
                "_index": index_name,
                "_source": doc,
            }
            actions.append(action)

            # 达到批次大小时提交
            if len(actions) >= batch_size:
                success, err = self._do_bulk(actions)
                indexed += success
                errors += err
                actions = []

        # 提交剩余数据
        if actions:
            success, err = self._do_bulk(actions)
            indexed += success
            errors += err

        logger.info(f"ES 索引完成: 成功={indexed}, 失败={errors}")

    def _do_bulk(self, actions: list) -> tuple:
        """
        执行批量操作

        参数:
            actions: ES bulk 操作列表

        返回:
            tuple: (成功数, 失败数)
        """
        try:
            success, errors = helpers.bulk(
                self.es, actions,
                raise_on_error=False,
                stats_only=True
            )
            return success, errors
        except Exception as e:
            logger.error(f"ES bulk 操作失败: {e}")
            return 0, len(actions)

    def search_logs(
        self,
        keyword: Optional[str] = None,
        status_code: Optional[int] = None,
        ip_address: Optional[str] = None,
        start_time: Optional[str] = None,
        end_time: Optional[str] = None,
        size: int = 100
    ) -> List[dict]:
        """
        搜索日志

        参数:
            keyword: 关键词搜索（匹配 URL 和 message）
            status_code: 状态码筛选
            ip_address: IP 地址筛选
            start_time: 开始时间（ISO 格式）
            end_time: 结束时间（ISO 格式）
            size: 返回数量

        返回:
            List[dict]: 匹配的日志列表
        """
        must_clauses = []

        if keyword:
            must_clauses.append({
                "multi_match": {
                    "query": keyword,
                    "fields": ["url", "message"],
                }
            })

        if status_code:
            must_clauses.append({"term": {"status_code": status_code}})

        if ip_address:
            must_clauses.append({"term": {"ip_address": ip_address}})

        if start_time or end_time:
            time_range = {}
            if start_time:
                time_range["gte"] = start_time
            if end_time:
                time_range["lte"] = end_time
            must_clauses.append({"range": {"timestamp": time_range}})

        query = {
            "bool": {"must": must_clauses} if must_clauses else {"match_all": {}}
        }

        result = self.es.search(
            index=f"{self.index_prefix}-*",
            query=query,
            size=size,
            sort=[{"timestamp": {"order": "desc"}}],
        )

        return [hit["_source"] for hit in result["hits"]["hits"]]

    def get_aggregations(self, field: str, size: int = 20) -> List[dict]:
        """
        对指定字段进行聚合统计

        参数:
            field: 聚合字段名
            size: 返回数量

        返回:
            List[dict]: 聚合结果
        """
        result = self.es.search(
            index=f"{self.index_prefix}-*",
            size=0,
            aggs={
                "field_agg": {
                    "terms": {
                        "field": field,
                        "size": size,
                    }
                }
            }
        )

        buckets = result["aggregations"]["field_agg"]["buckets"]
        return [
            {"value": b["key"], "count": b["doc_count"]}
            for b in buckets
        ]
```

```python
    """
    报告生成器
    将分析结果渲染为 HTML 报告
    """

    def __init__(self, output_dir: str = "./reports"):
        """
        初始化报告生成器

        参数:
            output_dir: 报告输出目录
        """
        self.output_dir = Path(output_dir)
        self.output_dir.mkdir(parents=True, exist_ok=True)

        # 初始化 Jinja2 环境
        self.env = Environment(
            autoescape=select_autoescape(["html"]),
        )

    def generate_html(
        self,
        result: AnalysisResult,
        slow_threshold: float = 3.0,
        filename: str = None
    ) -> str:
        """
        生成 HTML 分析报告

        参数:
            result: 分析结果对象
            slow_threshold: 慢请求阈值
            filename: 输出文件名（可选）

        返回:
            str: 报告文件路径
        """
        # 渲染模板
        template = self.env.from_string(REPORT_TEMPLATE)
        html_content = template.render(
            result=result,
            report_time=datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
            slow_threshold=slow_threshold,
        )

        # 写入文件
        if not filename:
            filename = f"report_{datetime.now().strftime('%Y%m%d_%H%M%S')}.html"

        output_path = self.output_dir / filename
        output_path.write_text(html_content, encoding="utf-8")

        logger.info(f"HTML 报告已生成: {output_path}")
        return str(output_path)
```

### 4.10 CLI 入口 `main.py`

```python
"""
日志分析平台 - CLI 入口
"""

import logging
import sys
import glob as glob_module
from pathlib import Path
from typing import Optional

import typer
import yaml
import uvicorn

from parser.nginx import NginxLogParser, NginxErrorLogParser
from parser.app_log import AppLogParser, JsonLogParser
from analyzer.stats import LogAnalyzer
from analyzer.realtime import LogTailer, RealtimeAnalyzer
from storage.es_client import ESStorage
from report.generator import ReportGenerator

app = typer.Typer(
    name="log-analyzer",
    help="日志分析平台 - 解析、分析、查询、监控"
)


def load_config(path: str) -> dict:
    """加载配置文件"""
    with open(path, "r", encoding="utf-8") as f:
        return yaml.safe_load(f)


def get_parser(log_type: str, config: dict = None):
    """
    根据类型获取对应的日志解析器

    参数:
        log_type: 日志类型（nginx/nginx-error/app/json）
        config: 解析器配置

    返回:
        BaseLogParser: 解析器实例
    """
    parsers = {
        "nginx": NginxLogParser,
        "nginx-error": NginxErrorLogParser,
        "app": AppLogParser,
        "json": JsonLogParser,
    }

    parser_cls = parsers.get(log_type)
    if not parser_cls:
        typer.echo(f"不支持的日志类型: {log_type}")
        raise typer.Exit(code=1)

    return parser_cls()


def resolve_paths(patterns: list) -> list:
    """将 glob 模式展开为实际文件路径列表"""
    paths = []
    for pattern in patterns:
        matched = glob_module.glob(pattern)
        paths.extend(sorted(matched))
    return paths


@app.command()
def analyze(
    log_files: list[str] = typer.Argument(..., help="日志文件路径（支持多个）"),
    log_type: str = typer.Option("nginx", "--type", "-t", help="日志类型"),
    top_n: int = typer.Option(20, "--top", help="Top N 数量"),
    slow: float = typer.Option(3.0, "--slow", help="慢请求阈值(秒)"),
    output: Optional[str] = typer.Option(None, "--output", "-o", help="报告输出目录"),
    to_es: bool = typer.Option(False, "--to-es", help="同时写入 ES"),
    config_file: str = typer.Option("config/config.yaml", "--config", "-c"),
):
    """
    分析日志文件并生成报告

    示例:
        python main.py analyze /var/log/nginx/access.log -t nginx
        python main.py analyze *.log -t json --to-es
    """
    config = load_config(config_file)
    parser = get_parser(log_type, config)
    analyzer = LogAnalyzer(top_n=top_n, slow_threshold=slow)

    typer.echo(f"日志类型: {log_type}")
    typer.echo(f"待分析文件: {len(log_files)} 个\n")

    # 解析并分析
    all_entries = []
    for fp in log_files:
        typer.echo(f"解析中: {fp}")
        for entry in parser.parse_file(fp):
            analyzer.add_entry(entry)
            if to_es:
                all_entries.append(entry)

    # 执行分析
    result = analyzer.analyze()

    # 输出汇总
    typer.echo(f"\n{'='*50}")
    typer.echo(f"总请求数: {result.total_requests:,}")
    typer.echo(f"错误率: {result.error_rate:.2f}%")
    typer.echo(f"平均响应时间: {result.avg_response_time:.3f}s")
    typer.echo(f"P99 响应时间: {result.p99_response_time:.3f}s")
    typer.echo(f"总流量: {result.total_bytes / 1048576:.1f} MB")

    # 生成报告
    output_dir = output or config.get("report", {}).get("output_dir", "./reports")
    generator = ReportGenerator(output_dir=output_dir)
    report_path = generator.generate_html(result, slow_threshold=slow)
    typer.echo(f"\n报告已生成: {report_path}")

    # 写入 ES
    if to_es and all_entries:
        typer.echo(f"\n写入 ES ({len(all_entries)} 条)...")
        es = ESStorage(config["elasticsearch"])
        es.create_index_template()
        es.bulk_index(all_entries)
        typer.echo("ES 写入完成")


@app.command()
def tail(
    log_file: str = typer.Argument(..., help="日志文件路径"),
    log_type: str = typer.Option("nginx", "--type", "-t"),
    config_file: str = typer.Option("config/config.yaml", "--config", "-c"),
):
    """
    实时跟踪日志文件（类似 tail -f）

    实时解析新增日志并显示统计数据
    """
    parser = get_parser(log_type)
    rt_analyzer = RealtimeAnalyzer(window_seconds=60)

    def on_entry(entry):
        """新日志回调：打印并更新统计"""
        rt_analyzer.on_new_entry(entry)
        stats = rt_analyzer.get_stats()
        status = f"[RPS: {stats['current_rps']:.1f} | " \
                 f"错误率: {stats['current_error_rate']:.1f}% | " \
                 f"平均RT: {stats['current_avg_response_time']:.3f}s]"
        typer.echo(f"{status} {entry.method} {entry.url} {entry.status_code}")

    typer.echo(f"开始实时跟踪: {log_file}")
    typer.echo("按 Ctrl+C 停止\n")

    tailer = LogTailer(log_file, parser, callback=on_entry)
    tailer.start()

    try:
        import time
        while True:
            time.sleep(1)
    except KeyboardInterrupt:
        tailer.stop()
        typer.echo("\n已停止")


@app.command()
def serve(
    config_file: str = typer.Option("config/config.yaml", "--config", "-c"),
    host: str = typer.Option("0.0.0.0", "--host"),
    port: int = typer.Option(8080, "--port", "-p"),
):
    """启动 API 服务"""
    config = load_config(config_file)

    # 初始化 ES 存储并注入到 API 模块
    from api import server as api_module
    api_module.es_storage = ESStorage(config["elasticsearch"])

    typer.echo(f"启动 API 服务: http://{host}:{port}")
    uvicorn.run(api_module.api_app, host=host, port=port)


if __name__ == "__main__":
    app()
```
