# 实战项目一：服务器资产采集工具

## 1. 项目概述

### 1.1 解决什么问题

在大规模服务器环境中，运维团队需要准确掌握每台服务器的硬件配置、资源使用情况和网络信息。
手动登录每台服务器查看信息效率极低且容易出错。本项目实现了一个**自动化服务器资产采集工具**，
能够批量连接服务器、并发采集资产信息、存储到数据库，并在终端以美观的方式展示结果。

```python
"""
数据库模型定义
使用 SQLAlchemy ORM 定义服务器资产数据表结构
"""

from datetime import datetime
from dataclasses import dataclass, field
from typing import List, Optional

from sqlalchemy import (
    create_engine, Column, Integer, String,

from sqlalchemy.ext.declarative import declarative_base

# 创建 ORM 基类
Base = declarative_base()


class ServerAsset(Base):

    服务器资产信息表

    os_version = Column(String(100), comment="操作系统版本")

    # CPU 信息
    cpu_model = Column(String(255), comment="CPU型号")
    cpu_physical_count = Column(Integer, comment="物理CPU数")
    cpu_logical_count = Column(Integer, comment="逻辑CPU数")
    cpu_usage_percent = Column(Float, comment="CPU使用率(%)")
    cpu_load_1min = Column(Float, comment="1分钟负载")
    cpu_load_5min = Column(Float, comment="5分钟负载")
    cpu_load_15min = Column(Float, comment="15分钟负载")

    # 内存信息
    memory_total_gb = Column(Float, comment="总内存(GB)")
    memory_used_gb = Column(Float, comment="已用内存(GB)")
    memory_available_gb = Column(Float, comment="可用内存(GB)")
    memory_usage_percent = Column(Float, comment="内存使用率(%)")
    swap_total_gb = Column(Float, comment="Swap总量(GB)")
    swap_used_gb = Column(Float, comment="Swap已用(GB)")

    # 磁盘信息（以 JSON 格式存储多个分区）

    disk_total_gb = Column(Float, comment="磁盘总容量(GB)")
    disk_used_gb = Column(Float, comment="磁盘已用(GB)")
    disk_usage_percent = Column(Float, comment="磁盘使用率(%)")

    # 网络信息
    network_interfaces = Column(JSON, comment="网络接口详情(JSON)")

            f"ip={self.ip_address}, "
            f"cpu_usage={self.cpu_usage_percent}%)>"

class CollectTask(Base):

    采集任务记录表

    finished_at = Column(DateTime, comment="结束时间")
    error_details = Column(JSON, comment="错误详情(JSON)")

    def __repr__(self):
        return (
            f"<CollectTask(task_id={self.task_id}, "
            f"status={self.status})>"

@dataclass
class ServerInfo:
    """
    服务器采集结果的数据类
    在采集过程中用于临时存储采集到的信息，最终写入数据库
    """
    hostname: str = ""
    ip_address: str = ""
    server_name: str = ""

    # 操作系统
    os_type: str = ""
    os_version: str = ""
    kernel_version: str = ""
    architecture: str = ""
    uptime_seconds: int = 0

    # CPU
    cpu_model: str = ""
    cpu_physical_count: int = 0
    cpu_logical_count: int = 0
    cpu_usage_percent: float = 0.0
    cpu_load_1min: float = 0.0
    cpu_load_5min: float = 0.0
    cpu_load_15min: float = 0.0

    # 内存
    memory_total_gb: float = 0.0
    memory_used_gb: float = 0.0
    memory_available_gb: float = 0.0
    memory_usage_percent: float = 0.0
    swap_total_gb: float = 0.0
    swap_used_gb: float = 0.0

    # 磁盘
    disk_partitions: list = field(default_factory=list)
    disk_total_gb: float = 0.0
    disk_used_gb: float = 0.0
    disk_usage_percent: float = 0.0

    # 网络
    network_interfaces: list = field(default_factory=list)
    network_connections: int = 0

    # 进程
    process_count: int = 0
    top_processes: list = field(default_factory=list)

    # 标签
    tags: list = field(default_factory=list)

    def to_dict(self) -> dict:
        """转换为字典，方便写入数据库"""
        from dataclasses import asdict
        return asdict(self)


def init_database(db_url: str) -> sessionmaker:
    """
    初始化数据库连接并创建表结构

    参数:
        db_url: 数据库连接字符串，如 mysql+pymysql://user:pass@host/db

    返回:
        sessionmaker: SQLAlchemy 会话工厂
    """
    engine = create_engine(
        db_url,
        pool_size=10,           # 连接池大小
        max_overflow=20,        # 最大溢出连接数
        pool_recycle=3600,      # 连接回收时间（秒）
        echo=False              # 关闭 SQL 日志
    )

    # 自动创建不存在的表
    Base.metadata.create_all(engine)

    # 返回会话工厂
    Session = sessionmaker(bind=engine)
    return Session
```

### 4.3 SSH 连接管理 `collector/ssh_client.py`

```python
"""
SSH 连接管理模块
封装 paramiko 实现安全的 SSH 连接与命令执行
"""

import logging
from typing import Optional, Tuple
from pathlib import Path

import paramiko

logger = logging.getLogger(__name__)


class SSHClient:
    """
    SSH 客户端封装类
    支持密码认证和密钥认证，提供命令执行和文件传输功能
    """

    def __init__(
        self,
        host: str,
        port: int = 22,
        username: str = "root",
        password: Optional[str] = None,
        key_file: Optional[str] = None,
        timeout: int = 10
    ):
        """
        初始化 SSH 客户端

        参数:
            host: 目标服务器 IP 或域名
            port: SSH 端口号
            username: 登录用户名
            password: 登录密码（与 key_file 二选一）
            key_file: SSH 私钥文件路径
            timeout: 连接超时时间（秒）
        """
        self.host = host
        self.port = port
        self.username = username
        self.password = password
        self.key_file = key_file
        self.timeout = timeout
        self._client: Optional[paramiko.SSHClient] = None

    def connect(self) -> bool:
        """
        建立 SSH 连接

        返回:
            bool: 连接是否成功
        """
        try:
            self._client = paramiko.SSHClient()
            # 自动接受未知主机密钥（生产环境建议使用 known_hosts）
            self._client.set_missing_host_key_policy(
                paramiko.AutoAddPolicy()
            )

            # 构建连接参数
            connect_kwargs = {
                "hostname": self.host,
                "port": self.port,
                "username": self.username,
                "timeout": self.timeout,
            }

            # 优先使用密钥认证
            if self.key_file and Path(self.key_file).exists():
                private_key = paramiko.RSAKey.from_private_key_file(
                    self.key_file
                )
                connect_kwargs["pkey"] = private_key
                logger.info(f"[{self.host}] 使用密钥认证连接")
            elif self.password:
                connect_kwargs["password"] = self.password
                logger.info(f"[{self.host}] 使用密码认证连接")
            else:
                logger.error(f"[{self.host}] 未提供认证方式")
                return False

            self._client.connect(**connect_kwargs)
            logger.info(f"[{self.host}] SSH 连接成功")
            return True

        except paramiko.AuthenticationException:
            logger.error(f"[{self.host}] SSH 认证失败：用户名或密码错误")
            return False
        except paramiko.SSHException as e:
            logger.error(f"[{self.host}] SSH 连接异常：{e}")
            return False
        except Exception as e:
            logger.error(f"[{self.host}] 连接失败：{e}")
            return False

    def execute(self, command: str) -> Tuple[str, str, int]:
        """
        执行远程命令

        参数:
            command: 要执行的 Shell 命令

        返回:
            Tuple[str, str, int]: (标准输出, 标准错误, 退出码)
        """
        if not self._client:
            raise RuntimeError("SSH 未连接，请先调用 connect()")

        try:
            stdin, stdout, stderr = self._client.exec_command(
                command, timeout=30
            )
            exit_code = stdout.channel.recv_exit_status()
            stdout_text = stdout.read().decode("utf-8", errors="replace").strip()
            stderr_text = stderr.read().decode("utf-8", errors="replace").strip()

            if exit_code != 0:
                logger.warning(
                    f"[{self.host}] 命令执行返回非零退出码: "
                    f"cmd='{command}', exit_code={exit_code}, "
                    f"stderr='{stderr_text[:200]}'"
                )

            return stdout_text, stderr_text, exit_code

        except Exception as e:
            logger.error(f"[{self.host}] 命令执行失败: {e}")
            return "", str(e), -1

    def close(self):
        """关闭 SSH 连接"""
        if self._client:
            self._client.close()
            logger.info(f"[{self.host}] SSH 连接已关闭")
            self._client = None

    def __enter__(self):
        """上下文管理器入口"""
        self.connect()
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        """上下文管理器退出，自动关闭连接"""
        self.close()
        return False
```

### 4.4 系统信息采集 `collector/system_info.py`

```python
"""
系统信息采集模块
通过 SSH 远程执行命令采集服务器的各项系统信息
也支持使用 psutil 采集本机信息
"""

import json
import logging
from typing import Optional

import psutil

from collector.ssh_client import SSHClient
from models.asset import ServerInfo

logger = logging.getLogger(__name__)


class SystemInfoCollector:
    """
    系统信息采集器
    支持通过 SSH 远程采集，也支持本机直接采集
    """

    def __init__(self, ssh_client: Optional[SSHClient] = None):
        """
        初始化采集器

        参数:
            ssh_client: SSH 客户端实例，为 None 时采集本机信息
        """
        self.ssh = ssh_client
        self.is_remote = ssh_client is not None

    def _run_command(self, command: str) -> str:
        """
        执行命令（远程或本地）

        参数:
            command: Shell 命令

        返回:
            str: 命令输出
        """
        if self.is_remote:
            stdout, stderr, code = self.ssh.execute(command)
            return stdout
        else:
            import subprocess
            result = subprocess.run(
                command, shell=True,
                capture_output=True, text=True
            )
            return result.stdout.strip()

    def collect_all(self, server_name: str = "", tags: list = None) -> ServerInfo:
        """
        采集所有系统信息

        参数:
            server_name: 服务器别名
            tags: 标签列表

        返回:
            ServerInfo: 采集结果数据对象
        """
        info = ServerInfo(
            server_name=server_name,
            tags=tags or [],
            ip_address=self.ssh.host if self.is_remote else "127.0.0.1"
        )

        logger.info(f"[{info.ip_address}] 开始采集系统信息...")

        # 逐项采集，单项失败不影响其他项
        self._collect_os_info(info)
        self._collect_cpu_info(info)
        self._collect_memory_info(info)
        self._collect_disk_info(info)
        self._collect_network_info(info)
        self._collect_process_info(info)

        logger.info(f"[{info.ip_address}] 系统信息采集完成")
        return info

    def _collect_os_info(self, info: ServerInfo):
        """
        采集操作系统信息

        包括：主机名、操作系统类型、版本、内核版本、系统架构、运行时间
        """
        try:
            # 获取主机名
            info.hostname = self._run_command("hostname")

            # 获取操作系统信息
            info.os_type = self._run_command(
                "cat /etc/os-release | grep ^ID= | cut -d= -f2 | tr -d '\"'"
            )
            info.os_version = self._run_command(
                "cat /etc/os-release | grep VERSION_ID | cut -d= -f2 | tr -d '\"'"
            )

            # 内核版本
            info.kernel_version = self._run_command("uname -r")

            # 系统架构
            info.architecture = self._run_command("uname -m")

            # 系统运行时间（秒）
            uptime_str = self._run_command(
                "cat /proc/uptime | awk '{print $1}'"
            )
            info.uptime_seconds = int(float(uptime_str)) if uptime_str else 0

            logger.info(f"[{info.ip_address}] 操作系统信息采集完成")

        except Exception as e:
            logger.error(f"[{info.ip_address}] 操作系统信息采集失败: {e}")

    def _collect_cpu_info(self, info: ServerInfo):
        """
        采集 CPU 信息

        包括：CPU 型号、物理/逻辑核心数、使用率、系统负载
        """
        try:
            # CPU 型号
            info.cpu_model = self._run_command(
                "cat /proc/cpuinfo | grep 'model name' | head -1 "
                "| cut -d: -f2 | xargs"
            )

            # 物理 CPU 数量
            physical = self._run_command(
                "cat /proc/cpuinfo | grep 'physical id' | sort -u | wc -l"
            )
            info.cpu_physical_count = int(physical) if physical else 1

            # 逻辑 CPU 数量
            logical = self._run_command("nproc")
            info.cpu_logical_count = int(logical) if logical else 1

            # CPU 使用率（采样 1 秒）
            cpu_usage = self._run_command(
                "top -bn1 | grep 'Cpu(s)' | awk '{print $2}'"
            )
            info.cpu_usage_percent = float(cpu_usage) if cpu_usage else 0.0

            # 系统负载
            loadavg = self._run_command("cat /proc/loadavg")
            if loadavg:
                parts = loadavg.split()
                info.cpu_load_1min = float(parts[0])
                info.cpu_load_5min = float(parts[1])
                info.cpu_load_15min = float(parts[2])

            logger.info(f"[{info.ip_address}] CPU 信息采集完成")

        except Exception as e:
            logger.error(f"[{info.ip_address}] CPU 信息采集失败: {e}")

    def _collect_memory_info(self, info: ServerInfo):
        """
        采集内存信息

        包括：总内存、已用内存、可用内存、使用率、Swap 信息
        """
        try:
            # 使用 free 命令获取内存信息（单位 MB）
            mem_output = self._run_command("free -m | grep Mem")
            if mem_output:
                parts = mem_output.split()
                total_mb = int(parts[1])
                used_mb = int(parts[2])
                available_mb = int(parts[6]) if len(parts) > 6 else total_mb - used_mb

                info.memory_total_gb = round(total_mb / 1024, 2)
                info.memory_used_gb = round(used_mb / 1024, 2)
                info.memory_available_gb = round(available_mb / 1024, 2)
                info.memory_usage_percent = round(
                    (used_mb / total_mb) * 100, 1
                ) if total_mb > 0 else 0.0

            # Swap 信息
            swap_output = self._run_command("free -m | grep Swap")
            if swap_output:
                parts = swap_output.split()
                info.swap_total_gb = round(int(parts[1]) / 1024, 2)
                info.swap_used_gb = round(int(parts[2]) / 1024, 2)

            logger.info(f"[{info.ip_address}] 内存信息采集完成")

        except Exception as e:
            logger.error(f"[{info.ip_address}] 内存信息采集失败: {e}")

    def _collect_disk_info(self, info: ServerInfo):
        """
        采集磁盘信息

        包括：各分区容量、使用率、挂载点
        """
        try:
            # 获取磁盘分区信息
            disk_output = self._run_command(
                "df -h --output=source,fstype,size,used,avail,pcent,target "
                "| grep -v tmpfs | grep -v devtmpfs | tail -n +2"
            )

            partitions = []
            total_size = 0.0
            total_used = 0.0

            if disk_output:
                for line in disk_output.strip().split("\n"):
                    parts = line.split()
                    if len(parts) >= 7:
                        partition = {
                            "device": parts[0],        # 设备名
                            "fstype": parts[1],        # 文件系统类型
                            "size": parts[2],          # 总大小
                            "used": parts[3],          # 已用
                            "available": parts[4],     # 可用
                            "usage_percent": parts[5], # 使用率
                            "mount_point": parts[6],   # 挂载点
                        }
                        partitions.append(partition)

                        # 累计总量（将各种单位转为 GB）
                        size_gb = self._parse_size_to_gb(parts[2])
                        used_gb = self._parse_size_to_gb(parts[3])
                        total_size += size_gb
                        total_used += used_gb

            info.disk_partitions = partitions
            info.disk_total_gb = round(total_size, 2)
            info.disk_used_gb = round(total_used, 2)
            info.disk_usage_percent = round(
                (total_used / total_size) * 100, 1
            ) if total_size > 0 else 0.0

            logger.info(f"[{info.ip_address}] 磁盘信息采集完成")

        except Exception as e:
            logger.error(f"[{info.ip_address}] 磁盘信息采集失败: {e}")

    def _collect_network_info(self, info: ServerInfo):
        """
        采集网络信息

        包括：网络接口、IP 地址、MAC 地址、当前连接数
        """
        try:
            # 获取网络接口信息
            net_output = self._run_command(
                "ip -j addr show 2>/dev/null || ip addr show"
            )

            interfaces = []
            # 尝试用 ip 命令的简化输出
            iface_output = self._run_command(
                "ip -o addr show | grep -v '127.0.0.1' | grep -v '::1'"
            )

            if iface_output:
                for line in iface_output.strip().split("\n"):
                    parts = line.split()
                    if len(parts) >= 4:
                        iface = {
                            "name": parts[1],
                            "ip": parts[3].split("/")[0] if "/" in parts[3] else parts[3],
                        }
                        interfaces.append(iface)

            info.network_interfaces = interfaces

            # 当前网络连接数
            conn_count = self._run_command(
                "ss -tun | tail -n +2 | wc -l"
            )
            info.network_connections = int(conn_count) if conn_count else 0

            logger.info(f"[{info.ip_address}] 网络信息采集完成")

        except Exception as e:
            logger.error(f"[{info.ip_address}] 网络信息采集失败: {e}")

    def _collect_process_info(self, info: ServerInfo):
        """
        采集进程信息

        包括：进程总数、CPU 占用 Top10 进程
        """
        try:
            # 进程总数
            proc_count = self._run_command(
                "ps aux --no-headers | wc -l"
            )
            info.process_count = int(proc_count) if proc_count else 0

            # CPU 占用 Top10 进程
            top_output = self._run_command(
                "ps aux --sort=-%cpu --no-headers | head -10"
            )

            top_processes = []
            if top_output:
                for line in top_output.strip().split("\n"):
                    parts = line.split(None, 10)
                    if len(parts) >= 11:
                        proc = {
                            "user": parts[0],
                            "pid": int(parts[1]),
                            "cpu_percent": float(parts[2]),
                            "mem_percent": float(parts[3]),
                            "command": parts[10][:100],  # 截断过长的命令
                        }
                        top_processes.append(proc)

            info.top_processes = top_processes

            logger.info(f"[{info.ip_address}] 进程信息采集完成")

        except Exception as e:
            logger.error(f"[{info.ip_address}] 进程信息采集失败: {e}")

    @staticmethod
    def _parse_size_to_gb(size_str: str) -> float:
        """
        将带单位的大小字符串解析为 GB 数值

        参数:
            size_str: 如 "100G", "500M", "1.5T"

        返回:
            float: GB 数值
        """
        size_str = size_str.strip().upper()
        try:
            if size_str.endswith("T"):
                return float(size_str[:-1]) * 1024
            elif size_str.endswith("G"):
                return float(size_str[:-1])
            elif size_str.endswith("M"):
                return float(size_str[:-1]) / 1024
            elif size_str.endswith("K"):
                return float(size_str[:-1]) / (1024 * 1024)
            else:
                return float(size_str) / (1024 ** 3)  # 假设为字节
        except ValueError:
            return 0.0


def collect_local_info() -> ServerInfo:
    """
    使用 psutil 直接采集本机信息（无需 SSH）

    返回:
        ServerInfo: 本机系统信息
    """
    import platform
    import socket

    info = ServerInfo()
    info.hostname = socket.gethostname()
    info.ip_address = "127.0.0.1"
    info.server_name = "localhost"

    # 操作系统信息
    info.os_type = platform.system().lower()
    info.os_version = platform.version()
    info.kernel_version = platform.release()
    info.architecture = platform.machine()
    info.uptime_seconds = int(
        psutil.time.time() - psutil.boot_time()
    )

    # CPU 信息
    info.cpu_physical_count = psutil.cpu_count(logical=False) or 1
    info.cpu_logical_count = psutil.cpu_count(logical=True) or 1
    info.cpu_usage_percent = psutil.cpu_percent(interval=1)
    load = psutil.getloadavg()
    info.cpu_load_1min = load[0]
    info.cpu_load_5min = load[1]
    info.cpu_load_15min = load[2]

    # 内存信息
    mem = psutil.virtual_memory()
    info.memory_total_gb = round(mem.total / (1024 ** 3), 2)
    info.memory_used_gb = round(mem.used / (1024 ** 3), 2)
    info.memory_available_gb = round(mem.available / (1024 ** 3), 2)
    info.memory_usage_percent = mem.percent

    swap = psutil.swap_memory()
    info.swap_total_gb = round(swap.total / (1024 ** 3), 2)
    info.swap_used_gb = round(swap.used / (1024 ** 3), 2)

    # 磁盘信息
    partitions = []
    total_disk = 0.0
    used_disk = 0.0
    for part in psutil.disk_partitions():
        try:
            usage = psutil.disk_usage(part.mountpoint)
            partitions.append({
                "device": part.device,
                "fstype": part.fstype,
                "mount_point": part.mountpoint,
                "total_gb": round(usage.total / (1024 ** 3), 2),
                "used_gb": round(usage.used / (1024 ** 3), 2),
                "usage_percent": usage.percent,
            })
            total_disk += usage.total / (1024 ** 3)
            used_disk += usage.used / (1024 ** 3)
        except PermissionError:
            continue

    info.disk_partitions = partitions
    info.disk_total_gb = round(total_disk, 2)
    info.disk_used_gb = round(used_disk, 2)
    info.disk_usage_percent = round(
        (used_disk / total_disk) * 100, 1
    ) if total_disk > 0 else 0.0

    # 网络信息
    net_if = psutil.net_if_addrs()
    interfaces = []
    for name, addrs in net_if.items():
        for addr in addrs:
            if addr.family.name == "AF_INET" and addr.address != "127.0.0.1":
                interfaces.append({
                    "name": name,
                    "ip": addr.address,
                })
    info.network_interfaces = interfaces
    info.network_connections = len(psutil.net_connections())

    # 进程信息
    info.process_count = len(psutil.pids())

    return info
```

### 4.5 并发采集引擎 `collector/engine.py`

```python
"""
并发采集引擎
使用 ThreadPoolExecutor 实现多服务器并发采集
"""

import uuid
import logging
from datetime import datetime
from concurrent.futures import ThreadPoolExecutor, as_completed
from typing import List, Dict, Tuple

from collector.ssh_client import SSHClient
from collector.system_info import SystemInfoCollector
from models.asset import ServerInfo

logger = logging.getLogger(__name__)


class CollectEngine:
    """
    并发采集引擎
    管理多个服务器的并发采集任务
    """

    def __init__(self, max_workers: int = 10, ssh_config: dict = None):
        """
        初始化采集引擎

        参数:
            max_workers: 最大并发线程数
            ssh_config: SSH 默认配置
        """
        self.max_workers = max_workers
        self.ssh_config = ssh_config or {}
        self.task_id = str(uuid.uuid4())

    def collect_single(self, server: dict) -> Tuple[bool, ServerInfo, str]:
        """
        采集单台服务器信息

        参数:
            server: 服务器配置字典，包含 host/port/password 等

        返回:
            Tuple[bool, ServerInfo, str]: (是否成功, 采集结果, 错误信息)
        """
        host = server["host"]
        try:
            # 合并默认配置和服务器专属配置
            ssh_params = {
                "host": host,
                "port": server.get("port", self.ssh_config.get("port", 22)),
                "username": server.get(
                    "username", self.ssh_config.get("username", "root")
                ),
                "password": server.get(
                    "password", self.ssh_config.get("password")
                ),
                "key_file": server.get(
                    "key_file", self.ssh_config.get("key_file")
                ),
                "timeout": self.ssh_config.get("timeout", 10),
            }

            # 使用上下文管理器确保连接关闭
            with SSHClient(**ssh_params) as ssh:
                collector = SystemInfoCollector(ssh_client=ssh)
                info = collector.collect_all(
                    server_name=server.get("name", host),
                    tags=server.get("tags", [])
                )
                return True, info, ""

        except Exception as e:
            logger.error(f"[{host}] 采集失败: {e}")
            return False, ServerInfo(ip_address=host), str(e)

    def collect_batch(
        self,
        servers: List[dict],
        progress_callback=None
    ) -> Dict:
        """
        批量并发采集多台服务器

        参数:
            servers: 服务器列表
            progress_callback: 进度回调函数，签名 callback(completed, total, server_info)

        返回:
            Dict: 采集结果汇总
        """
        results = {
            "task_id": self.task_id,
            "total": len(servers),
            "success": [],
            "failed": [],
            "started_at": datetime.now(),
            "finished_at": None,
        }

        logger.info(
            f"开始批量采集任务 {self.task_id}，"
            f"共 {len(servers)} 台服务器，"
            f"并发数 {self.max_workers}"
        )

        # 使用线程池并发采集
        with ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            # 提交所有采集任务
            future_to_server = {
                executor.submit(self.collect_single, server): server
                for server in servers
            }

            completed = 0
            # 按完成顺序处理结果
            for future in as_completed(future_to_server):
                server = future_to_server[future]
                completed += 1

                try:
                    success, info, error = future.result()

                    if success:
                        results["success"].append(info)
                        logger.info(
                            f"[{server['host']}] 采集成功 "
                            f"({completed}/{len(servers)})"
                        )
                    else:
                        results["failed"].append({
                            "host": server["host"],
                            "name": server.get("name", ""),
                            "error": error,
                        })
                        logger.warning(
                            f"[{server['host']}] 采集失败 "
                            f"({completed}/{len(servers)}): {error}"
                        )

                    # 调用进度回调
                    if progress_callback:
                        progress_callback(completed, len(servers), info)

                except Exception as e:
                    results["failed"].append({
                        "host": server["host"],
                        "name": server.get("name", ""),
                        "error": str(e),
                    })
                    logger.error(f"[{server['host']}] 任务异常: {e}")

        results["finished_at"] = datetime.now()
        duration = (
            results["finished_at"] - results["started_at"]
        ).total_seconds()

        logger.info(
            f"批量采集完成: 成功 {len(results['success'])}，"
            f"失败 {len(results['failed'])}，"
            f"耗时 {duration:.1f} 秒"
        )

        return results
```

```python
                network_connections=info.network_connections,
                process_count=info.process_count,
                top_processes=info.top_processes,
                tags=info.tags,
                collected_at=datetime.now(),
            )
            session.add(asset)
            session.commit()

            logger.info(

                self.save_server_info(info)

            except Exception as e:

    def query_assets(

        返回:
            List[ServerAsset]: 资产记录列表

        try:
            query = session.query(ServerAsset)

            if ip_address:

                    ServerAsset.ip_address == ip_address
                )

            if tag:
                # JSON 字段查询

                    ServerAsset.tags.contains(tag)
                )

            ).limit(limit).all()

        finally:
            session.close()

    def save_task(self, results: dict):

                finished_at=results["finished_at"],
                error_details=results["failed"] if results["failed"] else None,
            )
            session.add(task)
            session.commit()
            logger.info(f"采集任务 {task.task_id} 已记录")

        except Exception as e:
            session.rollback()
            logger.error(f"保存任务记录失败: {e}")
        finally:
            session.close()
```

```python
            show_header=True,
            header_style="bold red"
        )
        fail_table.add_column("服务器", width=20)
        fail_table.add_column("IP", width=15)
        fail_table.add_column("错误原因", width=40)

        for fail in results["failed"]:
            fail_table.add_row(

def create_progress() -> Progress:
    """
    创建 rich 进度条

    返回:
        Progress: 进度条对象
    """
    return Progress(
        SpinnerColumn(),
        TextColumn("[bold blue]{task.description}"),
        BarColumn(),
        TextColumn("[progress.percentage]{task.percentage:>3.0f}%"),
        TextColumn("({task.completed}/{task.total})"),
        console=console
    )


def _get_status_style(percent: float) -> str:
    """根据百分比返回样式颜色"""
    if percent < 60:
        return "green"
    elif percent < 80:
        return "yellow"
    else:
        return "red"


def _get_status_text(percent: float, style: str) -> str:
    """根据百分比返回状态文本"""
    if percent < 60:
        return f"[{style}]正常[/]"
    elif percent < 80:
        return f"[{style}]警告[/]"
    else:
        return f"[{style}]危险[/]"
```

```python
def setup_logging(level: str = "INFO", log_file: str = None):
    """
    配置日志

    参数:
        level: 日志级别
        log_file: 日志文件路径
    """
    handlers = [logging.StreamHandler(sys.stdout)]

    if log_file:
        log_path = Path(log_file)
        log_path.parent.mkdir(parents=True, exist_ok=True)
        handlers.append(logging.FileHandler(log_file, encoding="utf-8"))

    logging.basicConfig(
        level=getattr(logging, level.upper(), logging.INFO),
        format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
        handlers=handlers,
    )


def load_config(config_path: str) -> dict:
    """
    加载 YAML 配置文件

    参数:
        config_path: 配置文件路径

    返回:
        dict: 配置字典
    """
    path = Path(config_path)
    if not path.exists():
        console.print(f"[red]配置文件不存在: {config_path}[/red]")
        raise typer.Exit(code=1)

    with open(path, "r", encoding="utf-8") as f:
        config = yaml.safe_load(f)

    return config


@app.command()
def collect(
    config_file: str = typer.Option(
        "config/config.yaml",
        "--config", "-c",
        help="配置文件路径"
    ),
    local: bool = typer.Option(
        False, "--local", "-l",
        help="仅采集本机信息"
    ),
    host: Optional[str] = typer.Option(
        None, "--host", "-H",
        help="采集指定服务器（覆盖配置文件）"
    ),
    workers: Optional[int] = typer.Option(
        None, "--workers", "-w",
        help="并发线程数"
    ),
    no_db: bool = typer.Option(
        False, "--no-db",
        help="不保存到数据库"
    ),
):
    """
    执行服务器资产采集

    支持三种模式：
    1. 批量模式：从配置文件读取服务器列表并发采集
    2. 本机模式：使用 --local 仅采集本机信息
    3. 单机模式：使用 --host 指定单台服务器
    """
    console.print("[bold blue]服务器资产采集工具 v1.0[/bold blue]\n")

    # 加载配置
    config = load_config(config_file)

    # 配置日志
    log_config = config.get("logging", {})
    setup_logging(
        level=log_config.get("level", "INFO"),
        log_file=log_config.get("file")
    )

    # 本机采集模式
    if local:
        console.print("[cyan]模式: 本机采集[/cyan]\n")
        info = collect_local_info()
        display_server_info(info)

        if not no_db:
            _save_to_db(config, [info])
        return

    # 确定采集目标
    if host:
        # 单机模式
        servers = [{
            "host": host,
            "name": host,
            "password": config.get("ssh", {}).get("password", ""),
        }]
    else:
        # 批量模式
        servers = config.get("servers", [])

    if not servers:
        console.print("[red]未找到采集目标服务器[/red]")
        raise typer.Exit(code=1)

    console.print(f"[cyan]采集目标: {len(servers)} 台服务器[/cyan]\n")

    # 创建采集引擎
    max_workers = workers or config.get("concurrency", {}).get("max_workers", 10)
    engine = CollectEngine(
        max_workers=max_workers,
        ssh_config=config.get("ssh", {})
    )

    # 使用进度条执行采集
    progress = create_progress()
    with progress:
        task = progress.add_task("采集中...", total=len(servers))

        def on_progress(completed, total, info):
            progress.update(task, completed=completed)

        results = engine.collect_batch(servers, progress_callback=on_progress)

    # 展示结果
    console.print()
    display_summary(results)

    # 展示每台服务器详情
    for info in results["success"]:
        console.print()
        display_server_info(info)

    # 保存到数据库
    if not no_db:
        _save_to_db(config, results["success"], results)


@app.command()
def query(
    config_file: str = typer.Option(

    ),
):

    查询已采集的资产信息
    """

    if not assets:

def _save_to_db(config: dict, info_list: list, results: dict = None):
    """
    保存采集结果到数据库

    参数:
        config: 配置字典
        info_list: 服务器信息列表
        results: 完整的采集结果（可选）
    """
    try:
        storage = AssetStorage(config["database"])
        saved = storage.save_batch(info_list)
        console.print(f"\n[green]已保存 {saved} 条资产记录到数据库[/green]")

        if results:
            storage.save_task(results)

    except Exception as e:
        console.print(f"\n[red]保存到数据库失败: {e}[/red]")
        console.print("[yellow]提示: 请检查数据库配置是否正确[/yellow]")
```
