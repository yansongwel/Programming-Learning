# 实战项目三：自动化部署机器人

## 1. 项目概述

### 1.1 解决什么问题

在微服务架构下，频繁的发布部署是常态。手动部署容易出错、难以追踪状态、回滚困难。
本项目实现了一个**自动化部署机器人**，支持从代码拉取、Docker 镜像构建、容器部署到健康检查的完整流程，
提供 WebSocket 实时进度推送、Redis 状态存储和 Sentry 异常监控，并内置回滚能力。

### 4.2 Git 操作 `deployer/git_manager.py`

```python
"""
Git 仓库管理模块
使用 gitpython 实现代码拉取、版本切换等操作
"""

import logging
import shutil
from pathlib import Path
from typing import Optional

import git
from git import Repo, GitCommandError

logger = logging.getLogger(__name__)


class GitManager:
    """
    Git 仓库管理器
    封装代码克隆、拉取、分支切换、版本信息获取等操作
    """

    def __init__(self, repo_url: str, work_dir: str, branch: str = "main"):
        """
        初始化 Git 管理器

        参数:
            repo_url: Git 仓库地址
            work_dir: 本地工作目录
            branch: 目标分支
        """
        self.repo_url = repo_url
        self.work_dir = Path(work_dir)
        self.branch = branch
        self._repo: Optional[Repo] = None

    def prepare(self) -> str:
        """
        准备代码仓库（克隆或更新）

        返回:
            str: 当前 commit hash
        """
        if self.work_dir.exists() and (self.work_dir / ".git").exists():
            # 仓库已存在，执行更新
            return self._pull()
        else:
            # 仓库不存在，执行克隆
            return self._clone()

    def _clone(self) -> str:
        """
        克隆代码仓库

        返回:
            str: 最新 commit hash
        """
        logger.info(f"克隆仓库: {self.repo_url} -> {self.work_dir}")

        # 确保父目录存在
        self.work_dir.parent.mkdir(parents=True, exist_ok=True)

        # 如果目录存在但不是 git 仓库，先清理
        if self.work_dir.exists():
            shutil.rmtree(self.work_dir)

        self._repo = Repo.clone_from(
            self.repo_url,
            str(self.work_dir),
            branch=self.branch,
            depth=1,             # 浅克隆，只获取最新一次提交
        )

        commit_hash = self._repo.head.commit.hexsha[:8]
        logger.info(f"克隆完成，当前版本: {commit_hash}")
        return commit_hash

    def _pull(self) -> str:
        """
        更新已有仓库到最新版本

        返回:
            str: 最新 commit hash
        """
        logger.info(f"更新仓库: {self.work_dir}")

        self._repo = Repo(str(self.work_dir))

        # 放弃本地修改
        self._repo.git.checkout("--", ".")
        self._repo.git.clean("-fd")

        # 切换到目标分支
        try:
            self._repo.git.checkout(self.branch)
        except GitCommandError:
            # 如果分支不存在本地，从远程创建
            self._repo.git.checkout("-b", self.branch, f"origin/{self.branch}")

        # 拉取最新代码
        origin = self._repo.remotes.origin
        origin.pull(self.branch)

        commit_hash = self._repo.head.commit.hexsha[:8]
        logger.info(f"更新完成，当前版本: {commit_hash}")
        return commit_hash

    def checkout_version(self, commit_hash: str):
        """
        切换到指定版本

        参数:
            commit_hash: Git commit hash
        """
        if not self._repo:
            self._repo = Repo(str(self.work_dir))

        logger.info(f"切换版本: {commit_hash}")
        self._repo.git.checkout(commit_hash)

    def get_commit_info(self) -> dict:
        """
        获取当前版本的详细信息

        返回:
            dict: 包含 hash、作者、时间、消息等信息
        """
        if not self._repo:
            self._repo = Repo(str(self.work_dir))

        commit = self._repo.head.commit
        return {
            "hash": commit.hexsha[:8],
            "full_hash": commit.hexsha,
            "author": str(commit.author),
            "email": commit.author.email,
            "date": commit.committed_datetime.isoformat(),
            "message": commit.message.strip(),
            "branch": self.branch,
        }

    def get_recent_commits(self, count: int = 10) -> list:
        """
        获取最近的 commit 列表

        参数:
            count: 获取数量

        返回:
            list: commit 信息列表
        """
        if not self._repo:
            self._repo = Repo(str(self.work_dir))

        commits = []
        for commit in self._repo.iter_commits(max_count=count):
            commits.append({
                "hash": commit.hexsha[:8],
                "author": str(commit.author),
                "date": commit.committed_datetime.isoformat(),
                "message": commit.message.strip()[:80],
            })
        return commits

    def get_diff_files(self, from_hash: str, to_hash: str = "HEAD") -> list:
        """
        获取两个版本之间的变更文件列表

        参数:
            from_hash: 起始版本
            to_hash: 结束版本

        返回:
            list: 变更文件列表
        """
        if not self._repo:
            self._repo = Repo(str(self.work_dir))

        diff = self._repo.git.diff("--name-status", from_hash, to_hash)
        files = []
        for line in diff.strip().split("\n"):
            if line:
                parts = line.split("\t")
                files.append({
                    "status": parts[0],   # A=新增, M=修改, D=删除
                    "file": parts[1] if len(parts) > 1 else "",
                })
        return files
```

### 4.3 Docker 构建 `deployer/docker_builder.py`

```python
"""
Docker 构建与管理模块
使用 docker-py 实现镜像构建、推送和容器管理
"""

import logging
import time
from typing import Optional, Generator

import docker
from docker.errors import BuildError, APIError, ImageNotFound

logger = logging.getLogger(__name__)


class DockerBuilder:
    """
    Docker 镜像构建与容器管理器
    """

    def __init__(self, registry_config: dict = None):
        """
        初始化 Docker 客户端

        参数:
            registry_config: Docker Registry 配置
        """
        self.client = docker.from_env()
        self.registry_config = registry_config or {}

        # 登录 Registry
        if self.registry_config.get("url"):
            self._login()

    def _login(self):
        """登录 Docker Registry"""
        try:
            self.client.login(
                username=self.registry_config.get("username", ""),
                password=self.registry_config.get("password", ""),
                registry=self.registry_config.get("url", ""),
            )
            logger.info(
                f"Docker Registry 登录成功: "
                f"{self.registry_config['url']}"
            )
        except APIError as e:
            logger.error(f"Docker Registry 登录失败: {e}")
            raise

    def build_image(
        self,
        path: str,
        image_name: str,
        tag: str = "latest",
        dockerfile: str = "Dockerfile",
        build_args: dict = None,
        progress_callback=None,
    ) -> str:
        """
        构建 Docker 镜像

        参数:
            path: 构建上下文路径（Dockerfile 所在目录）
            image_name: 镜像名称
            tag: 镜像标签
            dockerfile: Dockerfile 文件名
            build_args: 构建参数
            progress_callback: 构建进度回调函数

        返回:
            str: 镜像完整标签 (image_name:tag)
        """
        full_tag = f"{image_name}:{tag}"
        logger.info(f"开始构建镜像: {full_tag}")

        try:
            # 使用低层 API 获取构建日志流
            build_logs = self.client.api.build(
                path=path,
                tag=full_tag,
                dockerfile=dockerfile,
                buildargs=build_args or {},
                rm=True,            # 构建完成后删除中间容器
                forcerm=True,       # 构建失败也删除中间容器
                decode=True,        # 解码 JSON 流
            )

            # 处理构建日志
            for log_entry in build_logs:
                if "stream" in log_entry:
                    line = log_entry["stream"].strip()
                    if line:
                        logger.debug(f"[构建] {line}")
                        if progress_callback:
                            progress_callback("building", line)

                if "error" in log_entry:
                    error_msg = log_entry["error"]
                    logger.error(f"构建错误: {error_msg}")
                    raise BuildError(error_msg, build_logs)

            logger.info(f"镜像构建成功: {full_tag}")
            return full_tag

        except BuildError as e:
            logger.error(f"镜像构建失败: {e}")
            raise
        except APIError as e:
            logger.error(f"Docker API 错误: {e}")
            raise

    def push_image(
        self,
        image_name: str,
        tag: str = "latest",
        progress_callback=None,
    ) -> bool:
        """
        推送镜像到 Registry

        参数:
            image_name: 镜像名称
            tag: 镜像标签
            progress_callback: 推送进度回调

        返回:
            bool: 是否推送成功
        """
        full_tag = f"{image_name}:{tag}"
        logger.info(f"推送镜像: {full_tag}")

        try:
            push_logs = self.client.api.push(
                image_name, tag=tag, stream=True, decode=True
            )

            for log_entry in push_logs:
                if "status" in log_entry:
                    status = log_entry["status"]
                    if progress_callback:
                        progress_callback("pushing", status)

                if "error" in log_entry:
                    logger.error(f"推送错误: {log_entry['error']}")
                    return False

            logger.info(f"镜像推送成功: {full_tag}")
            return True

        except APIError as e:
            logger.error(f"镜像推送失败: {e}")
            return False

    def deploy_container(
        self,
        image_name: str,
        tag: str = "latest",
        container_name: str = None,
        ports: dict = None,
        environment: dict = None,
        volumes: dict = None,
        network: str = None,
        restart_policy: dict = None,
    ) -> str:
        """
        部署容器（停止旧容器 + 启动新容器）

        参数:
            image_name: 镜像名称
            tag: 镜像标签
            container_name: 容器名称
            ports: 端口映射，如 {"8080/tcp": 8080}
            environment: 环境变量
            volumes: 数据卷映射
            network: Docker 网络
            restart_policy: 重启策略

        返回:
            str: 新容器 ID
        """
        full_tag = f"{image_name}:{tag}"

        # 停止并移除旧容器
        if container_name:
            self._remove_old_container(container_name)

        # 拉取最新镜像
        logger.info(f"拉取镜像: {full_tag}")
        try:
            self.client.images.pull(image_name, tag=tag)
        except APIError as e:
            logger.warning(f"拉取镜像失败（使用本地镜像）: {e}")

        # 启动新容器
        logger.info(f"启动容器: {container_name}")
        container = self.client.containers.run(
            full_tag,
            name=container_name,
            ports=ports or {},
            environment=environment or {},
            volumes=volumes or {},
            network=network,
            restart_policy=restart_policy or {"Name": "unless-stopped"},
            detach=True,          # 后台运行
        )

        logger.info(
            f"容器启动成功: {container_name} (ID: {container.short_id})"
        )
        return container.short_id

    def _remove_old_container(self, container_name: str):
        """
        停止并移除指定名称的旧容器

        参数:
            container_name: 容器名称
        """
        try:
            old_container = self.client.containers.get(container_name)
            logger.info(f"停止旧容器: {container_name}")
            old_container.stop(timeout=30)
            old_container.remove()
            logger.info(f"旧容器已移除: {container_name}")
        except docker.errors.NotFound:
            logger.info(f"未发现旧容器: {container_name}")
        except Exception as e:
            logger.warning(f"移除旧容器异常: {e}")

    def get_container_status(self, container_name: str) -> dict:
        """
        获取容器运行状态

        参数:
            container_name: 容器名称

        返回:
            dict: 容器状态信息
        """
        try:
            container = self.client.containers.get(container_name)
            return {
                "id": container.short_id,
                "name": container.name,
                "status": container.status,
                "image": container.image.tags[0] if container.image.tags else "",
                "created": container.attrs["Created"],
                "ports": container.ports,
            }
        except docker.errors.NotFound:
            return {"name": container_name, "status": "not_found"}
        except Exception as e:
            return {"name": container_name, "status": "error", "error": str(e)}

    def cleanup_old_images(self, image_name: str, keep_count: int = 5):
        """
        清理旧版本镜像，保留最新的 N 个

        参数:
            image_name: 镜像名称
            keep_count: 保留数量
        """
        try:
            images = self.client.images.list(name=image_name)
            if len(images) <= keep_count:
                return

            # 按创建时间排序，删除最旧的
            images.sort(
                key=lambda i: i.attrs.get("Created", ""),
                reverse=True
            )

            for img in images[keep_count:]:
                try:
                    self.client.images.remove(img.id, force=True)
                    logger.info(f"已清理旧镜像: {img.short_id}")
                except Exception as e:
                    logger.warning(f"清理镜像失败: {e}")

        except Exception as e:
            logger.error(f"镜像清理异常: {e}")
```

### 4.4 远程执行 `deployer/remote_executor.py`

```python
"""
远程命令执行模块
使用 Fabric 在远程服务器上执行部署命令
"""

import logging
from typing import List, Optional

from fabric import Connection, Config
from invoke import Responder
from paramiko import RSAKey

logger = logging.getLogger(__name__)


class RemoteExecutor:
    """
    远程命令执行器
    封装 Fabric 实现远程服务器操作
    """

    def __init__(
        self,
        host: str,
        user: str = "deploy",
        password: Optional[str] = None,
        key_file: Optional[str] = None,
        port: int = 22,
    ):
        """
        初始化远程执行器

        参数:
            host: 目标服务器
            user: SSH 用户名
            password: SSH 密码
            key_file: SSH 私钥路径
            port: SSH 端口
        """
        self.host = host
        self.user = user

        # 构建连接参数
        connect_kwargs = {}
        if key_file:
            connect_kwargs["key_filename"] = key_file
        elif password:
            connect_kwargs["password"] = password

        self.conn = Connection(
            host=host,
            user=user,
            port=port,
            connect_kwargs=connect_kwargs,
            connect_timeout=15,
        )

    def execute(self, command: str, sudo: bool = False) -> dict:
        """
        在远程服务器执行命令

        参数:
            command: 要执行的命令
            sudo: 是否使用 sudo

        返回:
            dict: 包含 stdout, stderr, exit_code
        """
        try:
            logger.info(f"[{self.host}] 执行: {command}")

            if sudo:
                result = self.conn.sudo(command, hide=True, warn=True)
            else:
                result = self.conn.run(command, hide=True, warn=True)

            return {
                "stdout": result.stdout.strip(),
                "stderr": result.stderr.strip(),
                "exit_code": result.return_code,
                "success": result.return_code == 0,
            }

        except Exception as e:
            logger.error(f"[{self.host}] 命令执行失败: {e}")
            return {
                "stdout": "",
                "stderr": str(e),
                "exit_code": -1,
                "success": False,
            }

    def upload_file(self, local_path: str, remote_path: str):
        """
        上传文件到远程服务器

        参数:
            local_path: 本地文件路径
            remote_path: 远程目标路径
        """
        logger.info(f"[{self.host}] 上传: {local_path} -> {remote_path}")
        self.conn.put(local_path, remote_path)

    def deploy_with_docker(
        self,
        image_name: str,
        tag: str,
        container_name: str,
        port_mapping: str,
        env_vars: dict = None,
    ) -> bool:
        """
        在远程服务器上用 Docker 部署

        参数:
            image_name: 镜像名称
            tag: 镜像标签
            container_name: 容器名称
            port_mapping: 端口映射，如 "8080:8080"
            env_vars: 环境变量字典

        返回:
            bool: 部署是否成功
        """
        full_image = f"{image_name}:{tag}"

        # 构建环境变量参数
        env_args = ""
        if env_vars:
            env_args = " ".join(
                f'-e {k}="{v}"' for k, v in env_vars.items()
            )

        commands = [
            # 拉取最新镜像
            f"docker pull {full_image}",
            # 停止旧容器
            f"docker stop {container_name} 2>/dev/null || true",
            # 移除旧容器
            f"docker rm {container_name} 2>/dev/null || true",
            # 启动新容器
            (
                f"docker run -d --name {container_name} "
                f"-p {port_mapping} {env_args} "
                f"--restart unless-stopped {full_image}"
            ),
        ]

        for cmd in commands:
            result = self.execute(cmd, sudo=True)
            if not result["success"] and "pull" in cmd:
                # 拉取失败是致命错误
                logger.error(f"[{self.host}] 镜像拉取失败")
                return False

        # 验证容器是否运行
        check = self.execute(
            f"docker inspect --format='{{{{.State.Running}}}}' {container_name}"
        )
        if check["stdout"] == "true":
            logger.info(f"[{self.host}] 容器部署成功")
            return True
        else:
            logger.error(f"[{self.host}] 容器未正常运行")
            return False

    def rollback_container(
        self,
        container_name: str,
        old_image: str,
        port_mapping: str,
    ) -> bool:
        """
        回滚到旧版本容器

        参数:
            container_name: 容器名称
            old_image: 旧版本镜像标签
            port_mapping: 端口映射

        返回:
            bool: 回滚是否成功
        """
        logger.info(f"[{self.host}] 开始回滚到: {old_image}")

        commands = [
            f"docker stop {container_name} 2>/dev/null || true",
            f"docker rm {container_name} 2>/dev/null || true",
            (
                f"docker run -d --name {container_name} "
                f"-p {port_mapping} "
                f"--restart unless-stopped {old_image}"
            ),
        ]

        for cmd in commands:
            self.execute(cmd, sudo=True)

        # 验证
        check = self.execute(
            f"docker inspect --format='{{{{.State.Running}}}}' {container_name}"
        )
        success = check["stdout"] == "true"

        if success:
            logger.info(f"[{self.host}] 回滚成功")
        else:
            logger.error(f"[{self.host}] 回滚失败")

        return success

    def close(self):
        """关闭连接"""
        self.conn.close()
```

### 4.5 健康检查 `deployer/health_checker.py`

```python
"""
健康检查模块
对已部署的服务进行多种方式的健康检查
"""

import time
import logging
from typing import Optional

import requests

logger = logging.getLogger(__name__)


class HealthChecker:
    """
    服务健康检查器
    支持 HTTP 健康检查和端口探测
    """

    def __init__(
        self,
        timeout: int = 60,
        interval: int = 5,
        expected_status: int = 200,
    ):
        """
        初始化健康检查器

        参数:
            timeout: 总超时时间（秒）
            interval: 检查间隔（秒）
            expected_status: 期望的 HTTP 状态码
        """
        self.timeout = timeout
        self.interval = interval
        self.expected_status = expected_status

    def check_http(
        self,
        url: str,
        progress_callback=None
    ) -> bool:
        """
        HTTP 健康检查（带重试）

        参数:
            url: 健康检查 URL
            progress_callback: 进度回调函数

        返回:
            bool: 健康检查是否通过
        """
        logger.info(f"开始健康检查: {url} (超时: {self.timeout}s)")

        start_time = time.time()
        attempt = 0

        while time.time() - start_time < self.timeout:
            attempt += 1
            elapsed = time.time() - start_time

            try:
                response = requests.get(url, timeout=5)

                if response.status_code == self.expected_status:
                    logger.info(
                        f"健康检查通过 (第 {attempt} 次, "
                        f"耗时 {elapsed:.1f}s)"
                    )
                    if progress_callback:
                        progress_callback("healthy", f"检查通过，耗时 {elapsed:.1f}s")
                    return True
                else:
                    logger.warning(
                        f"健康检查返回非预期状态码: {response.status_code} "
                        f"(期望 {self.expected_status})"
                    )

            except requests.ConnectionError:
                logger.debug(f"第 {attempt} 次检查: 连接失败，等待重试...")
            except requests.Timeout:
                logger.debug(f"第 {attempt} 次检查: 请求超时")
            except Exception as e:
                logger.debug(f"第 {attempt} 次检查异常: {e}")

            if progress_callback:
                progress_callback(
                    "checking",
                    f"第 {attempt} 次检查未通过，{self.interval}s 后重试"
                )

            time.sleep(self.interval)

        logger.error(f"健康检查超时 ({self.timeout}s)")
        if progress_callback:
            progress_callback("unhealthy", "健康检查超时")
        return False

    def check_container_health(
        self,
        container_name: str,
        docker_client=None
    ) -> bool:
        """
        检查 Docker 容器健康状态

        参数:
            container_name: 容器名称
            docker_client: Docker 客户端

        返回:
            bool: 容器是否健康
        """
        if not docker_client:
            import docker
            docker_client = docker.from_env()

        try:
            container = docker_client.containers.get(container_name)

            # 检查容器状态
            if container.status != "running":
                logger.error(f"容器未运行: {container.status}")
                return False

            # 检查容器健康状态（如果定义了 HEALTHCHECK）
            health = container.attrs.get("State", {}).get("Health")
            if health:
                status = health.get("Status", "none")
                if status == "healthy":
                    logger.info("容器健康检查通过")
                    return True
                elif status == "unhealthy":
                    logger.error("容器健康状态: unhealthy")
                    return False

            # 没有 HEALTHCHECK，只检查是否在运行
            logger.info("容器正在运行（无 HEALTHCHECK 定义）")
            return True

        except Exception as e:
            logger.error(f"容器健康检查失败: {e}")
            return False
```

### 4.6 Redis 状态存储 `state/redis_store.py`

```python
"""
Redis 状态存储模块
管理部署任务的状态、历史记录和版本信息
"""

import json
import logging
from datetime import datetime
from typing import Optional, List

import redis

logger = logging.getLogger(__name__)


class DeployStateStore:
    """
    部署状态存储
    使用 Redis 存储部署任务的完整生命周期状态
    """

    # Redis Key 前缀
    KEY_PREFIX = "deploy:"

    def __init__(self, config: dict):
        """
        初始化 Redis 连接

        参数:
            config: Redis 配置字典
        """
        self.redis = redis.Redis(
            host=config.get("host", "127.0.0.1"),
            port=config.get("port", 6379),
            db=config.get("db", 1),
            password=config.get("password") or None,
            decode_responses=True,   # 自动将字节解码为字符串
        )

        # 验证连接
        self.redis.ping()
        logger.info("Redis 连接成功")

    def create_deployment(
        self,
        deploy_id: str,
        app_name: str,
        version: str,
        operator: str = "system"
    ):
        """
        创建部署记录

        参数:
            deploy_id: 部署唯一 ID
            app_name: 应用名称
            version: 部署版本（git hash）
            operator: 操作人
        """
        key = f"{self.KEY_PREFIX}task:{deploy_id}"
        data = {
            "deploy_id": deploy_id,
            "app_name": app_name,
            "version": version,
            "operator": operator,
            "status": "created",
            "current_step": "",
            "progress": 0,
            "steps": json.dumps([]),
            "started_at": datetime.now().isoformat(),
            "finished_at": "",
            "error": "",
        }
        self.redis.hset(key, mapping=data)
        self.redis.expire(key, 86400 * 7)  # 7 天过期

        # 添加到应用的部署历史列表
        history_key = f"{self.KEY_PREFIX}history:{app_name}"
        self.redis.lpush(history_key, deploy_id)
        self.redis.ltrim(history_key, 0, 99)  # 只保留最近 100 条

        logger.info(f"创建部署记录: {deploy_id}")

    def update_step(
        self,
        deploy_id: str,
        step_name: str,
        status: str,
        message: str = "",
        progress: int = 0
    ):
        """
        更新部署步骤状态

        参数:
            deploy_id: 部署 ID
            step_name: 步骤名称
            status: 步骤状态 (running/success/failed)
            message: 状态消息
            progress: 整体进度百分比 (0-100)
        """
        key = f"{self.KEY_PREFIX}task:{deploy_id}"

        # 更新当前步骤
        self.redis.hset(key, "current_step", step_name)
        self.redis.hset(key, "progress", progress)

        if status == "failed":
            self.redis.hset(key, "status", "failed")
            self.redis.hset(key, "error", message)
        elif status == "running":
            self.redis.hset(key, "status", "deploying")

        # 追加步骤日志
        step_log = {
            "step": step_name,
            "status": status,
            "message": message,
            "time": datetime.now().isoformat(),
        }
        steps_key = f"{self.KEY_PREFIX}steps:{deploy_id}"
        self.redis.rpush(steps_key, json.dumps(step_log))
        self.redis.expire(steps_key, 86400 * 7)

    def complete_deployment(self, deploy_id: str, success: bool):
        """
        标记部署完成

        参数:
            deploy_id: 部署 ID
            success: 是否成功
        """
        key = f"{self.KEY_PREFIX}task:{deploy_id}"
        self.redis.hset(key, "status", "success" if success else "failed")
        self.redis.hset(key, "finished_at", datetime.now().isoformat())
        self.redis.hset(key, "progress", 100 if success else -1)

    def get_deployment(self, deploy_id: str) -> Optional[dict]:
        """
        获取部署记录

        参数:
            deploy_id: 部署 ID

        返回:
            Optional[dict]: 部署信息，不存在返回 None
        """
        key = f"{self.KEY_PREFIX}task:{deploy_id}"
        data = self.redis.hgetall(key)

        if not data:
            return None

        # 获取步骤日志
        steps_key = f"{self.KEY_PREFIX}steps:{deploy_id}"
        steps = self.redis.lrange(steps_key, 0, -1)
        data["steps"] = [json.loads(s) for s in steps]

        return data

    def get_deploy_history(
        self, app_name: str, limit: int = 20
    ) -> List[dict]:
        """
        获取应用的部署历史

        参数:
            app_name: 应用名称
            limit: 返回数量

        返回:
            List[dict]: 部署记录列表
        """
        history_key = f"{self.KEY_PREFIX}history:{app_name}"
        deploy_ids = self.redis.lrange(history_key, 0, limit - 1)

        records = []
        for did in deploy_ids:
            record = self.get_deployment(did)
            if record:
                records.append(record)

        return records

    def save_version_info(
        self, app_name: str, version: str, image_tag: str
    ):
        """
        保存版本信息（用于回滚）

        参数:
            app_name: 应用名称
            version: Git commit hash
            image_tag: Docker 镜像标签
        """
        key = f"{self.KEY_PREFIX}versions:{app_name}"
        version_data = json.dumps({
            "version": version,
            "image_tag": image_tag,
            "deployed_at": datetime.now().isoformat(),
        })
        self.redis.lpush(key, version_data)
        self.redis.ltrim(key, 0, 19)  # 保留最近 20 个版本

    def get_previous_version(
        self, app_name: str, offset: int = 1
    ) -> Optional[dict]:
        """
        获取上一个版本信息（用于回滚）

        参数:
            app_name: 应用名称
            offset: 偏移量（1=上一个版本，2=上上个版本）

        返回:
            Optional[dict]: 版本信息
        """
        key = f"{self.KEY_PREFIX}versions:{app_name}"
        data = self.redis.lindex(key, offset)

        if data:
            return json.loads(data)
        return None
```

### 4.7 部署流水线 `deployer/pipeline.py`

```python
"""
部署流水线模块
编排完整的部署流程：拉代码 -> 构建镜像 -> 推送 -> 部署 -> 健康检查
"""

import uuid
import logging
from datetime import datetime
from typing import Optional, Callable

from deployer.git_manager import GitManager
from deployer.docker_builder import DockerBuilder
from deployer.remote_executor import RemoteExecutor
from deployer.health_checker import HealthChecker
from state.redis_store import DeployStateStore

logger = logging.getLogger(__name__)


class DeployPipeline:
    """
    部署流水线
    按顺序执行部署各阶段，支持失败回滚
    """

    # 部署阶段定义
    STAGES = [
        ("git_pull", "拉取代码", 15),
        ("docker_build", "构建镜像", 35),
        ("docker_push", "推送镜像", 50),
        ("deploy", "部署容器", 75),
        ("health_check", "健康检查", 90),
        ("cleanup", "清理旧版本", 100),
    ]

    def __init__(
        self,
        app_config: dict,
        registry_config: dict,
        state_store: DeployStateStore,
        progress_callback: Optional[Callable] = None,
    ):
        """
        初始化部署流水线

        参数:
            app_config: 应用配置
            registry_config: Docker Registry 配置
            state_store: Redis 状态存储
            progress_callback: 进度回调 callback(deploy_id, stage, status, message)
        """
        self.app_config = app_config
        self.registry_config = registry_config
        self.state = state_store
        self.progress_callback = progress_callback
        self.deploy_id = str(uuid.uuid4())[:8]

    def _notify(self, stage: str, status: str, message: str, progress: int = 0):
        """
        发送进度通知

        参数:
            stage: 当前阶段
            status: 状态
            message: 消息
            progress: 进度百分比
        """
        # 更新 Redis 状态
        self.state.update_step(
            self.deploy_id, stage, status, message, progress
        )

        # 调用回调
        if self.progress_callback:
            self.progress_callback(
                self.deploy_id, stage, status, message
            )

        logger.info(f"[{self.deploy_id}] [{stage}] {status}: {message}")

    def run(self, operator: str = "system") -> bool:
        """
        执行完整部署流水线

        参数:
            operator: 操作人

        返回:
            bool: 部署是否成功
        """
        app_name = self.app_config["name"]
        logger.info(f"开始部署: {app_name} (ID: {self.deploy_id})")

        # 创建部署记录
        self.state.create_deployment(
            self.deploy_id, app_name, "", operator
        )

        try:
            # 阶段1: 拉取代码
            self._notify("git_pull", "running", "正在拉取代码...", 5)
            git_mgr = GitManager(
                repo_url=self.app_config["git_repo"],
                work_dir=self.app_config["work_dir"],
                branch=self.app_config.get("branch", "main"),
            )
            commit_hash = git_mgr.prepare()
            commit_info = git_mgr.get_commit_info()
            self._notify(
                "git_pull", "success",
                f"代码更新完成: {commit_hash} - {commit_info['message'][:50]}",
                15
            )

            # 阶段2: 构建 Docker 镜像
            image_tag = f"{commit_hash}-{datetime.now().strftime('%Y%m%d%H%M')}"
            image_name = self.app_config["docker_image"]
            full_tag = f"{image_name}:{image_tag}"

            self._notify("docker_build", "running", f"构建镜像: {full_tag}", 20)
            builder = DockerBuilder(self.registry_config)
            builder.build_image(
                path=str(self.app_config["work_dir"]),
                image_name=image_name,
                tag=image_tag,
                dockerfile=self.app_config.get("dockerfile", "Dockerfile"),
            )
            self._notify("docker_build", "success", f"镜像构建完成: {full_tag}", 35)

            # 阶段3: 推送镜像
            self._notify("docker_push", "running", "推送镜像到 Registry...", 40)
            push_ok = builder.push_image(image_name, tag=image_tag)
            if not push_ok:
                raise RuntimeError("镜像推送失败")
            self._notify("docker_push", "success", "镜像推送完成", 50)

            # 阶段4: 部署到目标服务器
            self._notify("deploy", "running", "开始部署到目标服务器...", 55)
            deploy_targets = self.app_config.get("deploy_targets", [])
            container_name = self.app_config["container_name"]
            container_port = self.app_config["container_port"]

            deploy_success = True
            for i, target in enumerate(deploy_targets):
                self._notify(
                    "deploy", "running",
                    f"部署到 {target['host']} ({i+1}/{len(deploy_targets)})",
                    55 + (20 * (i + 1) // len(deploy_targets))
                )

                executor = RemoteExecutor(
                    host=target["host"],
                    user=target.get("user", "deploy"),
                    key_file=target.get("key_file"),
                    password=target.get("password"),
                )

                ok = executor.deploy_with_docker(
                    image_name=image_name,
                    tag=image_tag,
                    container_name=container_name,
                    port_mapping=f"{container_port}:{container_port}",
                )
                executor.close()

                if not ok:
                    deploy_success = False
                    self._notify(
                        "deploy", "failed",
                        f"部署到 {target['host']} 失败",
                        75
                    )
                    break

            if not deploy_success:
                raise RuntimeError("容器部署失败")

            self._notify("deploy", "success", "所有服务器部署完成", 75)

            # 阶段5: 健康检查
            health_url = self.app_config.get("health_check_url")
            if health_url:
                self._notify("health_check", "running", "执行健康检查...", 80)
                checker = HealthChecker(timeout=60, interval=5)
                healthy = checker.check_http(health_url)

                if not healthy:
                    self._notify(
                        "health_check", "failed", "健康检查未通过", 90
                    )
                    # 触发回滚
                    self._rollback(image_name, container_name, container_port)
                    raise RuntimeError("健康检查失败，已回滚")

                self._notify("health_check", "success", "健康检查通过", 90)

            # 阶段6: 清理
            self._notify("cleanup", "running", "清理旧版本...", 95)
            builder.cleanup_old_images(image_name, keep_count=5)

            # 保存版本信息用于回滚
            self.state.save_version_info(
                app_name, commit_hash, full_tag
            )

            self._notify("cleanup", "success", "部署完成", 100)

            # 标记部署成功
            self.state.complete_deployment(self.deploy_id, success=True)
            logger.info(f"部署成功: {app_name} -> {image_tag}")
            return True

        except Exception as e:
            logger.error(f"部署失败: {e}")
            self.state.complete_deployment(self.deploy_id, success=False)

            # 报告 Sentry
            try:
                import sentry_sdk
                sentry_sdk.capture_exception(e)
            except ImportError:
                pass

            return False

    def _rollback(
        self, image_name: str, container_name: str, port: int
    ):
        """
        执行回滚操作

        参数:
            image_name: 镜像名称
            container_name: 容器名称
            port: 容器端口
        """
        app_name = self.app_config["name"]
        prev = self.state.get_previous_version(app_name, offset=0)

        if not prev:
            logger.error("无可回滚版本")
            return

        old_image = prev["image_tag"]
        logger.info(f"回滚到: {old_image}")
        self._notify("rollback", "running", f"回滚到: {old_image}", 85)

        for target in self.app_config.get("deploy_targets", []):
            executor = RemoteExecutor(
                host=target["host"],
                user=target.get("user", "deploy"),
                key_file=target.get("key_file"),
            )
            executor.rollback_container(
                container_name, old_image, f"{port}:{port}"
            )
            executor.close()

    def rollback_to(self, version_offset: int = 1) -> bool:
        """
        手动回滚到指定版本

        参数:
            version_offset: 版本偏移 (1=上一版本)

        返回:
            bool: 回滚是否成功
        """
        app_name = self.app_config["name"]
        prev = self.state.get_previous_version(app_name, offset=version_offset)

        if not prev:
            logger.error(f"找不到偏移 {version_offset} 的历史版本")
            return False

        container_name = self.app_config["container_name"]
        port = self.app_config["container_port"]

        logger.info(f"手动回滚 {app_name} 到: {prev['image_tag']}")

        all_ok = True
        for target in self.app_config.get("deploy_targets", []):
            executor = RemoteExecutor(
                host=target["host"],
                user=target.get("user", "deploy"),
                key_file=target.get("key_file"),
            )
            ok = executor.rollback_container(
                container_name, prev["image_tag"],
                f"{port}:{port}"
            )
            executor.close()
            if not ok:
                all_ok = False

        return all_ok
```

### 4.8 Sentry 集成 `notify/sentry.py`

```python
"""
Sentry 异常监控集成模块
记录部署失败、异常等信息到 Sentry
"""

import logging
from typing import Optional

logger = logging.getLogger(__name__)


def init_sentry(config: dict):
    """
    初始化 Sentry SDK

    参数:
        config: Sentry 配置字典，包含 dsn 和 environment
    """
    dsn = config.get("dsn")
    if not dsn:
        logger.info("未配置 Sentry DSN，跳过初始化")
        return

    try:
        import sentry_sdk

        sentry_sdk.init(
            dsn=dsn,
            environment=config.get("environment", "production"),
            traces_sample_rate=0.1,     # 性能追踪采样率
            send_default_pii=False,     # 不发送用户敏感信息
        )
        logger.info(f"Sentry 初始化成功: {config.get('environment')}")

    except ImportError:
        logger.warning("sentry-sdk 未安装，跳过 Sentry 初始化")
    except Exception as e:
        logger.error(f"Sentry 初始化失败: {e}")


def report_deploy_failure(
    app_name: str,
    deploy_id: str,
    error: str,
    extra: dict = None,
):
    """
    向 Sentry 报告部署失败事件

    参数:
        app_name: 应用名称
        deploy_id: 部署 ID
        error: 错误信息
        extra: 附加信息
    """
    try:
        import sentry_sdk

        with sentry_sdk.push_scope() as scope:
            scope.set_tag("app_name", app_name)
            scope.set_tag("deploy_id", deploy_id)
            scope.set_context("deploy_info", {
                "app_name": app_name,
                "deploy_id": deploy_id,
                "error": error,
                **(extra or {}),
            })

            sentry_sdk.capture_message(
                f"部署失败: {app_name} ({deploy_id}): {error}",
                level="error"
            )

        logger.info(f"已向 Sentry 报告部署失败: {deploy_id}")

    except ImportError:
        pass
    except Exception as e:
        logger.error(f"Sentry 报告失败: {e}")
```

### 4.9 FastAPI 接口 `api/server.py`

```python
"""
FastAPI 控制面板
提供部署触发、进度查询、WebSocket 实时推送、回滚等接口
"""

import json
import asyncio
import logging
from typing import Optional, Dict, Set

from fastapi import FastAPI, WebSocket, WebSocketDisconnect, HTTPException, Query
from pydantic import BaseModel

from deployer.pipeline import DeployPipeline
from state.redis_store import DeployStateStore

logger = logging.getLogger(__name__)

# FastAPI 应用
api_app = FastAPI(
    title="自动化部署机器人 API",
    description="支持部署触发、进度追踪、回滚等操作",
    version="1.0.0"
)

# 全局变量（在 main.py 中初始化）
deploy_config: dict = {}
state_store: Optional[DeployStateStore] = None

# WebSocket 连接管理
ws_connections: Dict[str, Set[WebSocket]] = {}  # deploy_id -> set of websockets


class DeployRequest(BaseModel):
    """部署请求体"""
    app_name: str             # 应用名称
    branch: Optional[str] = None  # 指定分支（覆盖配置）
    operator: str = "api"     # 操作人


class RollbackRequest(BaseModel):
    """回滚请求体"""
    app_name: str
    version_offset: int = 1   # 回滚到第几个历史版本


@api_app.post("/api/deploy")
async def trigger_deploy(req: DeployRequest):
    """
    触发部署

    接收应用名称，启动后台部署任务
    """
    app_config = deploy_config.get("applications", {}).get(req.app_name)
    if not app_config:
        raise HTTPException(
            status_code=404,
            detail=f"应用不存在: {req.app_name}"
        )

    # 如果指定了分支，覆盖配置
    if req.branch:
        app_config = {**app_config, "branch": req.branch}

    # 创建流水线
    pipeline = DeployPipeline(
        app_config=app_config,
        registry_config=deploy_config.get("docker_registry", {}),
        state_store=state_store,
        progress_callback=_on_progress,
    )

    # 在后台线程执行部署
    import threading
    thread = threading.Thread(
        target=pipeline.run,
        kwargs={"operator": req.operator},
        daemon=True
    )
    thread.start()

    return {
        "deploy_id": pipeline.deploy_id,
        "app_name": req.app_name,
        "status": "started",
        "message": "部署已启动，通过 WebSocket 或轮询接口查看进度",
        "ws_url": f"/ws/deploy/{pipeline.deploy_id}",
    }


@api_app.post("/api/rollback")
async def trigger_rollback(req: RollbackRequest):
    """
    触发回滚操作

    回滚到指定的历史版本
    """
    app_config = deploy_config.get("applications", {}).get(req.app_name)
    if not app_config:
        raise HTTPException(status_code=404, detail=f"应用不存在: {req.app_name}")

    pipeline = DeployPipeline(
        app_config=app_config,
        registry_config=deploy_config.get("docker_registry", {}),
        state_store=state_store,
    )

    success = pipeline.rollback_to(req.version_offset)
    return {
        "app_name": req.app_name,
        "success": success,
        "message": "回滚成功" if success else "回滚失败",
    }


@api_app.get("/api/deploy/{deploy_id}")
async def get_deploy_status(deploy_id: str):
    """获取部署状态"""
    record = state_store.get_deployment(deploy_id)
    if not record:
        raise HTTPException(status_code=404, detail="部署记录不存在")
    return record


@api_app.get("/api/deploy/history/{app_name}")
async def get_deploy_history(
    app_name: str,
    limit: int = Query(20, ge=1, le=100),
):
    """获取部署历史"""
    history = state_store.get_deploy_history(app_name, limit=limit)
    return {"app_name": app_name, "total": len(history), "records": history}


@api_app.websocket("/ws/deploy/{deploy_id}")
async def websocket_deploy_progress(websocket: WebSocket, deploy_id: str):
    """
    WebSocket 实时部署进度推送

    客户端连接后，实时接收部署各阶段的状态更新
    """
    await websocket.accept()
    logger.info(f"WebSocket 已连接: {deploy_id}")

    # 注册连接
    if deploy_id not in ws_connections:
        ws_connections[deploy_id] = set()
    ws_connections[deploy_id].add(websocket)

    try:
        # 先发送当前状态
        current = state_store.get_deployment(deploy_id)
        if current:
            await websocket.send_json(current)

        # 保持连接，等待消息或断开
        while True:
            try:
                # 等待客户端消息（心跳或命令）
                data = await asyncio.wait_for(
                    websocket.receive_text(), timeout=30
                )
                if data == "ping":
                    await websocket.send_text("pong")
            except asyncio.TimeoutError:
                # 超时发送心跳
                await websocket.send_text("ping")

    except WebSocketDisconnect:
        logger.info(f"WebSocket 已断开: {deploy_id}")
    finally:
        ws_connections.get(deploy_id, set()).discard(websocket)


def _on_progress(deploy_id: str, stage: str, status: str, message: str):
    """
    部署进度回调函数
    将进度推送到 WebSocket 客户端

    参数:
        deploy_id: 部署 ID
        stage: 当前阶段
        status: 状态
        message: 消息
    """
    ws_set = ws_connections.get(deploy_id, set())
    if not ws_set:
        return

    data = {
        "deploy_id": deploy_id,
        "stage": stage,
        "status": status,
        "message": message,
    }

    # 异步发送到所有连接的客户端
    import asyncio
    loop = asyncio.new_event_loop()

    async def send_all():
        disconnected = set()
        for ws in ws_set:
            try:
                await ws.send_json(data)
            except Exception:
                disconnected.add(ws)
        for ws in disconnected:
            ws_set.discard(ws)

    loop.run_until_complete(send_all())
    loop.close()


@api_app.get("/api/health")
async def health_check():
    """健康检查"""
    return {"status": "ok"}
```

### 4.10 CLI 入口 `main.py`

```python
"""
自动化部署机器人 - CLI 入口
"""

import logging
import sys
from typing import Optional

import typer
import yaml
import uvicorn

from deployer.pipeline import DeployPipeline
from state.redis_store import DeployStateStore
from notify.sentry import init_sentry

app = typer.Typer(
    name="deploy-bot",
    help="自动化部署机器人 - 支持部署、回滚、状态查询"
)


def load_config(path: str) -> dict:
    """加载配置"""
    with open(path, "r", encoding="utf-8") as f:
        return yaml.safe_load(f)


@app.command()
def deploy(
    app_name: str = typer.Argument(..., help="应用名称"),
    config_file: str = typer.Option("config/config.yaml", "--config", "-c"),
    branch: Optional[str] = typer.Option(None, "--branch", "-b", help="指定分支"),
    operator: str = typer.Option("cli", "--operator"),
):
    """
    执行部署

    示例: python main.py deploy web-app --branch feature/new
    """
    config = load_config(config_file)
    init_sentry(config.get("sentry", {}))

    app_config = config.get("applications", {}).get(app_name)
    if not app_config:
        typer.echo(f"应用不存在: {app_name}")
        typer.echo(f"可用应用: {list(config.get('applications', {}).keys())}")
        raise typer.Exit(code=1)

    if branch:
        app_config["branch"] = branch

    state = DeployStateStore(config["redis"])

    def on_progress(deploy_id, stage, status, message):
        icon = {"running": "...", "success": "OK", "failed": "FAIL"}.get(status, "")
        typer.echo(f"  [{icon}] {stage}: {message}")

    pipeline = DeployPipeline(
        app_config=app_config,
        registry_config=config.get("docker_registry", {}),
        state_store=state,
        progress_callback=on_progress,
    )

    typer.echo(f"开始部署: {app_name}")
    typer.echo(f"部署ID: {pipeline.deploy_id}\n")

    success = pipeline.run(operator=operator)

    if success:
        typer.echo("\n部署成功!")
    else:
        typer.echo("\n部署失败!")
        raise typer.Exit(code=1)


@app.command()
def rollback(
    app_name: str = typer.Argument(..., help="应用名称"),
    version_offset: int = typer.Option(1, "--offset", "-n", help="回滚版本偏移"),
    config_file: str = typer.Option("config/config.yaml", "--config", "-c"),
):
    """
    回滚到历史版本

    示例: python main.py rollback web-app -n 1
    """
    config = load_config(config_file)
    state = DeployStateStore(config["redis"])
    app_config = config.get("applications", {}).get(app_name)

    if not app_config:
        typer.echo(f"应用不存在: {app_name}")
        raise typer.Exit(code=1)

    # 显示可回滚版本
    prev = state.get_previous_version(app_name, offset=version_offset)
    if not prev:
        typer.echo("无可回滚版本")
        raise typer.Exit(code=1)

    typer.echo(f"回滚目标: {prev['image_tag']}")
    typer.echo(f"部署时间: {prev['deployed_at']}")

    if not typer.confirm("确认回滚?"):
        raise typer.Abort()

    pipeline = DeployPipeline(
        app_config=app_config,
        registry_config=config.get("docker_registry", {}),
        state_store=state,
    )

    success = pipeline.rollback_to(version_offset)
    typer.echo("回滚成功!" if success else "回滚失败!")


@app.command()
def status(
    deploy_id: str = typer.Argument(..., help="部署ID"),
    config_file: str = typer.Option("config/config.yaml", "--config", "-c"),
):
    """查看部署状态"""
    config = load_config(config_file)
    state = DeployStateStore(config["redis"])

    record = state.get_deployment(deploy_id)
    if not record:
        typer.echo(f"部署记录不存在: {deploy_id}")
        raise typer.Exit(code=1)

    typer.echo(f"部署ID:   {record['deploy_id']}")
    typer.echo(f"应用:     {record['app_name']}")
    typer.echo(f"状态:     {record['status']}")
    typer.echo(f"进度:     {record['progress']}%")
    typer.echo(f"开始时间: {record['started_at']}")
    typer.echo(f"结束时间: {record.get('finished_at', '-')}")

    if record.get("steps"):
        typer.echo("\n执行步骤:")
        for step in record["steps"]:
            icon = {"success": "+", "failed": "x", "running": "~"}.get(
                step["status"], " "
            )
            typer.echo(f"  [{icon}] {step['step']}: {step['message']}")


@app.command()
def history(
    app_name: str = typer.Argument(..., help="应用名称"),
    limit: int = typer.Option(10, "--limit", "-n"),
    config_file: str = typer.Option("config/config.yaml", "--config", "-c"),

    config = load_config(config_file)
    state = DeployStateStore(config["redis"])

    records = state.get_deploy_history(app_name, limit=limit)
    if not records:
        typer.echo(f"暂无部署记录: {app_name}")
        return

    typer.echo(f"应用 {app_name} 的部署历史 (最近 {len(records)} 条):\n")
    for r in records:
        status_icon = {"success": "+", "failed": "x"}.get(r["status"], "?")
        typer.echo(
            f"  [{status_icon}] {r['deploy_id']} | "
            f"{r['status']:10s} | {r['started_at']}"
        )

def serve(

    from api import server as api_module

    typer.echo(f"启动部署服务: http://{host}:{port}")
```
