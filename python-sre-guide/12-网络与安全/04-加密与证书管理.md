# 加密与证书管理 — 用 cryptography 库实现 SSL 证书巡检

## 1. 是什么 & 为什么用它

`cryptography` 是 Python 最主流的加密库，提供了对称加密、非对称加密、哈希、证书管理等全套功能。
对 SRE 来说，加密和证书管理是安全运维的核心：

- **SSL 证书到期检查**：避免证书过期导致服务中断
- **密码/密钥管理**：安全地处理敏感信息
- **X.509 证书解析**：自动化证书信息提取
- **数据加密传输**：保护配置中的敏感字段

证书过期是 SRE 最常见的生产事故之一（Google、Microsoft 都出过此类故障）。

## 3. 核心概念与原理

### 3.1 加密类型

```
对称加密（AES）
  同一个密钥加密和解密
  快速，适合大数据
  密钥分发是难点

非对称加密（RSA/ECC）
  公钥加密 → 私钥解密
  私钥签名 → 公钥验证
  慢，适合小数据和密钥交换

哈希（SHA256/MD5）
  单向不可逆
  验证数据完整性
  密码存储（加盐）

数字证书（X.509）
  公钥 + 身份信息 + CA 签名
  用于 HTTPS/TLS
  信任链：Root CA → Intermediate CA → End Entity
```

### 3.2 SSL/TLS 证书关键字段

| 字段 | 说明 |
|------|------|
| Subject | 证书持有者（域名等） |
| Issuer | 签发者（CA） |
| Not Before | 有效期开始 |
| Not After | 有效期结束 |
| Serial Number | 序列号 |
| SAN | 主题备用名称（多域名） |
| Public Key | 公钥 |
| Signature Algorithm | 签名算法 |

## 4. 基础用法

### 4.1 哈希与摘要

```python
#!/usr/bin/env python3
"""哈希与摘要计算"""

import hashlib
import hmac
import secrets

# === MD5（不推荐用于安全场景） ===
data = "Hello, SRE World!".encode('utf-8')
md5_hash = hashlib.md5(data).hexdigest()
print(f"MD5: {md5_hash}")

# === SHA256（推荐） ===
sha256_hash = hashlib.sha256(data).hexdigest()
print(f"SHA256: {sha256_hash}")

# === SHA512 ===
sha512_hash = hashlib.sha512(data).hexdigest()
print(f"SHA512: {sha512_hash}")

# === 文件哈希（校验完整性） ===
def file_hash(filepath: str, algorithm: str = 'sha256') -> str:
    """计算文件哈希"""
    h = hashlib.new(algorithm)
    with open(filepath, 'rb') as f:
        while chunk := f.read(8192):
            h.update(chunk)
    return h.hexdigest()

# 写入测试文件
from pathlib import Path
test_file = Path("/tmp/test_hash.txt")
test_file.write_text("这是一个测试文件\n")
print(f"\n文件 SHA256: {file_hash(str(test_file))}")

# === HMAC（带密钥的哈希，用于消息认证） ===
secret_key = b"my-secret-key"
message = b"important data"
mac = hmac.new(secret_key, message, hashlib.sha256).hexdigest()
print(f"\nHMAC-SHA256: {mac}")

# 验证 HMAC
expected_mac = mac
is_valid = hmac.compare_digest(
    hmac.new(secret_key, message, hashlib.sha256).hexdigest(),
    expected_mac
)
print(f"HMAC 验证: {'通过' if is_valid else '失败'}")

# === 密码哈希（加盐） ===
def hash_password(password: str) -> tuple:
    """密码哈希（加盐）"""
    salt = secrets.token_hex(16)
    hashed = hashlib.pbkdf2_hmac('sha256', password.encode(), salt.encode(), 100000)
    return salt, hashed.hex()

def verify_password(password: str, salt: str, expected_hash: str) -> bool:
    """验证密码"""
    hashed = hashlib.pbkdf2_hmac('sha256', password.encode(), salt.encode(), 100000)
    return hmac.compare_digest(hashed.hex(), expected_hash)

salt, pwd_hash = hash_password("my_secure_password")
print(f"\n密码哈希: salt={salt}, hash={pwd_hash[:32]}...")
print(f"密码验证: {verify_password('my_secure_password', salt, pwd_hash)}")
print(f"错误密码: {verify_password('wrong_password', salt, pwd_hash)}")
```

### 4.2 对称加密（AES）

```python
#!/usr/bin/env python3
"""对称加密：AES"""

from cryptography.fernet import Fernet
from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes
from cryptography.hazmat.primitives import padding
import os
import base64

# === Fernet（推荐，简单安全） ===
# 生成密钥
key = Fernet.generate_key()
print(f"Fernet 密钥: {key.decode()}")

# 加密
f = Fernet(key)
plaintext = "数据库密码: P@ssw0rd123!".encode('utf-8')
ciphertext = f.encrypt(plaintext)
print(f"密文: {ciphertext[:60]}...")

# 解密
decrypted = f.decrypt(ciphertext)
print(f"明文: {decrypted.decode('utf-8')}")

# === AES-CBC 底层实现 ===
def aes_encrypt(plaintext: bytes, key: bytes) -> tuple:
    """AES-CBC 加密"""
    iv = os.urandom(16)  # 初始化向量
    # 填充到 AES 块大小（16 字节）
    padder = padding.PKCS7(128).padder()
    padded_data = padder.update(plaintext) + padder.finalize()
    # 加密
    cipher = Cipher(algorithms.AES(key), modes.CBC(iv))
    encryptor = cipher.encryptor()
    ciphertext = encryptor.update(padded_data) + encryptor.finalize()
    return iv, ciphertext

def aes_decrypt(iv: bytes, ciphertext: bytes, key: bytes) -> bytes:
    """AES-CBC 解密"""
    cipher = Cipher(algorithms.AES(key), modes.CBC(iv))
    decryptor = cipher.decryptor()
    padded_data = decryptor.update(ciphertext) + decryptor.finalize()
    # 去除填充
    unpadder = padding.PKCS7(128).unpadder()
    return unpadder.update(padded_data) + unpadder.finalize()

# 使用 AES-256
aes_key = os.urandom(32)  # 256 位密钥
iv, encrypted = aes_encrypt(b"Secret SRE data", aes_key)
decrypted = aes_decrypt(iv, encrypted, aes_key)
print(f"\nAES-256-CBC 解密: {decrypted}")
```

### 4.3 非对称加密（RSA）

```python
#!/usr/bin/env python3
"""非对称加密：RSA"""

from cryptography.hazmat.primitives.asymmetric import rsa, padding as asym_padding
from cryptography.hazmat.primitives import hashes, serialization

# === 生成 RSA 密钥对 ===
private_key = rsa.generate_private_key(
    public_exponent=65537,
    key_size=2048,
)
public_key = private_key.public_key()

# 序列化密钥（PEM 格式）
private_pem = private_key.private_bytes(
    encoding=serialization.Encoding.PEM,
    format=serialization.PrivateFormat.PKCS8,
    encryption_algorithm=serialization.NoEncryption()
)
public_pem = public_key.public_bytes(
    encoding=serialization.Encoding.PEM,
    format=serialization.PublicFormat.SubjectPublicKeyInfo
)

print("RSA 私钥 (前两行):")
for line in private_pem.decode().split('\n')[:2]:
    print(f"  {line}")

print("\nRSA 公钥:")
print(public_pem.decode())

# === 加密 / 解密 ===
message = "敏感配置数据: db_password=secret123".encode('utf-8')

# 公钥加密
ciphertext = public_key.encrypt(
    message,
    asym_padding.OAEP(
        mgf=asym_padding.MGF1(algorithm=hashes.SHA256()),
        algorithm=hashes.SHA256(),
        label=None
    )
)
print(f"RSA 密文长度: {len(ciphertext)} bytes")

# 私钥解密
plaintext = private_key.decrypt(
    ciphertext,
    asym_padding.OAEP(
        mgf=asym_padding.MGF1(algorithm=hashes.SHA256()),
        algorithm=hashes.SHA256(),
        label=None
    )
)
print(f"RSA 解密: {plaintext.decode('utf-8')}")

# === 数字签名 ===
data = b"Release v2.0.0 checksum: abc123def456"

# 私钥签名
signature = private_key.sign(
    data,
    asym_padding.PSS(
        mgf=asym_padding.MGF1(hashes.SHA256()),
        salt_length=asym_padding.PSS.MAX_LENGTH
    ),
    hashes.SHA256()
)
print(f"\n签名长度: {len(signature)} bytes")

# 公钥验证
try:
    public_key.verify(
        signature,
        data,
        asym_padding.PSS(
            mgf=asym_padding.MGF1(hashes.SHA256()),
            salt_length=asym_padding.PSS.MAX_LENGTH
        ),
        hashes.SHA256()
    )
    print("签名验证: 通过")
except Exception:
    print("签名验证: 失败")
```

### 4.4 X.509 证书解析

```python
#!/usr/bin/env python3
"""X.509 证书解析"""

import ssl
import socket
from cryptography import x509
from cryptography.hazmat.primitives import hashes
from datetime import datetime, timezone

def get_ssl_cert(hostname: str, port: int = 443, timeout: float = 10) -> dict:
    """获取远程服务器的 SSL 证书信息"""
    context = ssl.create_default_context()

    try:
        with socket.create_connection((hostname, port), timeout=timeout) as sock:
            with context.wrap_socket(sock, server_hostname=hostname) as ssock:
                # 获取 DER 格式证书
                cert_der = ssock.getpeercert(binary_form=True)
                cert = x509.load_der_x509_certificate(cert_der)

                # 提取信息
                now = datetime.now(timezone.utc)
                not_after = cert.not_valid_after_utc if hasattr(cert, 'not_valid_after_utc') else cert.not_valid_after.replace(tzinfo=timezone.utc)
                not_before = cert.not_valid_before_utc if hasattr(cert, 'not_valid_before_utc') else cert.not_valid_before.replace(tzinfo=timezone.utc)
                days_remaining = (not_after - now).days

                # 提取 SAN（主题备用名称）
                san = []
                try:
                    ext = cert.extensions.get_extension_for_class(x509.SubjectAlternativeName)
                    san = ext.value.get_values_for_type(x509.DNSName)
                except x509.ExtensionNotFound:
                    pass

                # 提取 Subject
                subject = {}
                for attr in cert.subject:
                    subject[attr.oid._name] = attr.value

                # 提取 Issuer
                issuer = {}
                for attr in cert.issuer:
                    issuer[attr.oid._name] = attr.value

                return {
                    'hostname': hostname,
                    'subject': subject,
                    'issuer': issuer,
                    'serial_number': cert.serial_number,
                    'not_before': not_before.isoformat(),
                    'not_after': not_after.isoformat(),
                    'days_remaining': days_remaining,
                    'san': san,
                    'signature_algorithm': cert.signature_algorithm_oid._name,
                    'key_size': cert.public_key().key_size,
                    'fingerprint_sha256': cert.fingerprint(hashes.SHA256()).hex(),
                    'version': cert.version.name,
                    'is_expired': days_remaining < 0,
                    'is_warning': 0 < days_remaining <= 30,
                }

    except Exception as e:
        return {
            'hostname': hostname,
            'error': str(e),
            'is_expired': True,
        }

# 查询证书
print("=== SSL 证书信息 ===")
cert_info = get_ssl_cert("example.com")
if 'error' not in cert_info:
    print(f"域名: {cert_info['hostname']}")
    print(f"Subject: {cert_info['subject']}")
    print(f"Issuer: {cert_info['issuer']}")
    print(f"有效期: {cert_info['not_before']} ~ {cert_info['not_after']}")
    print(f"剩余天数: {cert_info['days_remaining']}")
    print(f"SAN: {cert_info['san']}")
    print(f"签名算法: {cert_info['signature_algorithm']}")
    print(f"密钥长度: {cert_info['key_size']} bits")
    print(f"指纹: {cert_info['fingerprint_sha256'][:32]}...")
else:
    print(f"获取失败: {cert_info['error']}")
```

## 5. 进阶用法

### 5.1 自签名证书生成

```python
#!/usr/bin/env python3
"""生成自签名 SSL 证书"""

from cryptography import x509
from cryptography.x509.oid import NameOID
from cryptography.hazmat.primitives import hashes, serialization
from cryptography.hazmat.primitives.asymmetric import rsa
from datetime import datetime, timedelta, timezone
import ipaddress

def generate_self_signed_cert(
    common_name: str,
    san_domains: list = None,
    san_ips: list = None,
    days: int = 365,
    key_size: int = 2048,
) -> tuple:
    """生成自签名证书"""
    # 生成私钥
    key = rsa.generate_private_key(
        public_exponent=65537,
        key_size=key_size,
    )

    # 构造 Subject
    subject = issuer = x509.Name([
        x509.NameAttribute(NameOID.COUNTRY_NAME, "CN"),
        x509.NameAttribute(NameOID.STATE_OR_PROVINCE_NAME, "Beijing"),
        x509.NameAttribute(NameOID.LOCALITY_NAME, "Beijing"),
        x509.NameAttribute(NameOID.ORGANIZATION_NAME, "SRE Team"),
        x509.NameAttribute(NameOID.COMMON_NAME, common_name),
    ])

    # SAN 列表
    san_list = []
    for domain in (san_domains or []):
        san_list.append(x509.DNSName(domain))
    for ip in (san_ips or []):
        san_list.append(x509.IPAddress(ipaddress.ip_address(ip)))

    # 构造证书
    now = datetime.now(timezone.utc)
    cert = (
        x509.CertificateBuilder()
        .subject_name(subject)
        .issuer_name(issuer)
        .public_key(key.public_key())
        .serial_number(x509.random_serial_number())
        .not_valid_before(now)
        .not_valid_after(now + timedelta(days=days))
        .add_extension(
            x509.SubjectAlternativeName(san_list),
            critical=False,
        )
        .add_extension(
            x509.BasicConstraints(ca=False, path_length=None),
            critical=True,
        )
        .sign(key, hashes.SHA256())
    )

    # 序列化
    cert_pem = cert.public_bytes(serialization.Encoding.PEM)
    key_pem = key.private_bytes(
        encoding=serialization.Encoding.PEM,
        format=serialization.PrivateFormat.TraditionalOpenSSL,
        encryption_algorithm=serialization.NoEncryption()
    )

    return cert_pem, key_pem

# 生成证书
cert_pem, key_pem = generate_self_signed_cert(
    common_name="*.sre.internal",
    san_domains=["sre.internal", "*.sre.internal", "monitoring.sre.internal"],
    san_ips=["10.0.0.1", "192.168.1.1"],
    days=365,
)

from pathlib import Path
Path("/tmp/sre-cert.pem").write_bytes(cert_pem)
Path("/tmp/sre-key.pem").write_bytes(key_pem)
print("证书已生成: /tmp/sre-cert.pem")
print("私钥已生成: /tmp/sre-key.pem")
print(f"\n证书内容:\n{cert_pem.decode()[:200]}...")
```

## 6. SRE 实战案例：SSL 证书到期自动巡检告警

```python
#!/usr/bin/env python3
"""
SRE 实战：SSL 证书到期自动巡检告警
功能：
1. 批量检查多个域名的 SSL 证书
2. 检测证书到期时间
3. 检测证书链完整性
4. 检测弱加密算法
5. 生成巡检报告
6. 告警通知（模拟）
"""

import ssl
import socket
import json
import time
from datetime import datetime, timezone, timedelta
from pathlib import Path
from typing import Dict, List, Optional
from dataclasses import dataclass, field, asdict
from concurrent.futures import ThreadPoolExecutor, as_completed

# 尝试导入 cryptography
try:
    from cryptography import x509
    from cryptography.hazmat.primitives import hashes
    HAS_CRYPTOGRAPHY = True
except ImportError:
    HAS_CRYPTOGRAPHY = False

# ============================================================
# 数据模型
# ============================================================
@dataclass
class CertCheckResult:
    """证书检查结果"""
    hostname: str
    port: int = 443
    status: str = ""          # ok / warning / critical / error
    subject_cn: str = ""
    issuer_cn: str = ""
    not_before: str = ""
    not_after: str = ""
    days_remaining: int = 0
    san_domains: List[str] = field(default_factory=list)
    signature_algorithm: str = ""
    key_size: int = 0
    fingerprint: str = ""
    protocol_version: str = ""
    cipher_name: str = ""
    cipher_bits: int = 0
    chain_length: int = 0
    issues: List[str] = field(default_factory=list)
    latency_ms: float = 0
    check_time: str = ""
    error: str = ""

    def __post_init__(self):
        if not self.check_time:
            self.check_time = datetime.now().isoformat()

# ============================================================
# SSL 证书检查器
# ============================================================
class SSLCertChecker:
    """SSL 证书检查器"""

    # 弱加密算法
    WEAK_ALGORITHMS = ['md5', 'sha1', 'md2']
    # 最小密钥长度
    MIN_RSA_KEY_SIZE = 2048
    MIN_ECC_KEY_SIZE = 256

    def __init__(self, timeout: float = 10.0):
        self.timeout = timeout

    def check(self, hostname: str, port: int = 443) -> CertCheckResult:
        """检查单个域名的 SSL 证书"""
        result = CertCheckResult(hostname=hostname, port=port)
        start_time = time.time()

        try:
            # 创建 SSL 上下文
            context = ssl.create_default_context()

            with socket.create_connection((hostname, port), timeout=self.timeout) as sock:
                with context.wrap_socket(sock, server_hostname=hostname) as ssock:
                    result.latency_ms = round((time.time() - start_time) * 1000, 2)

                    # 获取连接信息
                    result.protocol_version = ssock.version()
                    cipher = ssock.cipher()
                    if cipher:
                        result.cipher_name = cipher[0]
                        result.cipher_bits = cipher[2]

                    # 获取证书链长度
                    cert_chain = ssock.getpeercert(binary_form=False)
                    peer_cert_der = ssock.getpeercert(binary_form=True)

                    # 使用 cryptography 解析详细信息
                    if HAS_CRYPTOGRAPHY and peer_cert_der:
                        cert = x509.load_der_x509_certificate(peer_cert_der)
                        self._parse_cert_details(cert, result)
                    elif cert_chain:
                        self._parse_cert_basic(cert_chain, result)

                    # 计算剩余天数
                    if result.not_after:
                        not_after = datetime.fromisoformat(result.not_after)
                        if not_after.tzinfo is None:
                            not_after = not_after.replace(tzinfo=timezone.utc)
                        now = datetime.now(timezone.utc)
                        result.days_remaining = (not_after - now).days

                    # 安全检查
                    self._security_checks(result)

                    # 设置状态
                    if result.days_remaining < 0:
                        result.status = 'critical'
                        result.issues.append(f"证书已过期 {abs(result.days_remaining)} 天")
                    elif result.days_remaining <= 7:
                        result.status = 'critical'
                        result.issues.append(f"证书将在 {result.days_remaining} 天内过期")
                    elif result.days_remaining <= 30:
                        result.status = 'warning'
                        result.issues.append(f"证书将在 {result.days_remaining} 天内过期")
                    elif result.issues:
                        result.status = 'warning'
                    else:
                        result.status = 'ok'

        except ssl.SSLCertVerificationError as e:
            result.status = 'critical'
            result.error = f"证书验证失败: {e}"
            result.issues.append(str(e))
        except socket.timeout:
            result.status = 'critical'
            result.error = "连接超时"
            result.issues.append("SSL 连接超时")
        except ConnectionRefusedError:
            result.status = 'critical'
            result.error = "连接被拒绝"
            result.issues.append(f"端口 {port} 连接被拒绝")
        except Exception as e:
            result.status = 'error'
            result.error = str(e)
            result.issues.append(f"检查异常: {e}")

        result.latency_ms = round((time.time() - start_time) * 1000, 2)
        return result

    def _parse_cert_details(self, cert, result: CertCheckResult):
        """使用 cryptography 解析证书详情"""
        # Subject
        for attr in cert.subject:
            if attr.oid._name == 'commonName':
                result.subject_cn = attr.value

        # Issuer
        for attr in cert.issuer:
            if attr.oid._name == 'commonName':
                result.issuer_cn = attr.value

        # 有效期
        not_before = cert.not_valid_before_utc if hasattr(cert, 'not_valid_before_utc') else cert.not_valid_before.replace(tzinfo=timezone.utc)
        not_after = cert.not_valid_after_utc if hasattr(cert, 'not_valid_after_utc') else cert.not_valid_after.replace(tzinfo=timezone.utc)
        result.not_before = not_before.isoformat()
        result.not_after = not_after.isoformat()

        # SAN
        try:
            ext = cert.extensions.get_extension_for_class(x509.SubjectAlternativeName)
            result.san_domains = ext.value.get_values_for_type(x509.DNSName)
        except x509.ExtensionNotFound:
            pass

        # 签名算法
        result.signature_algorithm = cert.signature_algorithm_oid._name

        # 密钥信息
        result.key_size = cert.public_key().key_size

        # 指纹
        result.fingerprint = cert.fingerprint(hashes.SHA256()).hex()

    def _parse_cert_basic(self, cert_dict: dict, result: CertCheckResult):
        """使用标准库解析基本证书信息"""
        subject = dict(x[0] for x in cert_dict.get('subject', ()))
        result.subject_cn = subject.get('commonName', '')

        issuer = dict(x[0] for x in cert_dict.get('issuer', ()))
        result.issuer_cn = issuer.get('commonName', '')

        result.not_before = cert_dict.get('notBefore', '')
        result.not_after = cert_dict.get('notAfter', '')

        san = cert_dict.get('subjectAltName', ())
        result.san_domains = [v for t, v in san if t == 'DNS']

    def _security_checks(self, result: CertCheckResult):
        """安全性检查"""
        # 检查弱签名算法
        if result.signature_algorithm:
            algo_lower = result.signature_algorithm.lower()
            for weak in self.WEAK_ALGORITHMS:
                if weak in algo_lower:
                    result.issues.append(f"使用弱签名算法: {result.signature_algorithm}")

        # 检查密钥长度
        if result.key_size > 0 and result.key_size < self.MIN_RSA_KEY_SIZE:
            result.issues.append(f"RSA 密钥过短: {result.key_size} bits (建议 >= {self.MIN_RSA_KEY_SIZE})")

        # 检查 TLS 版本
        if result.protocol_version and result.protocol_version in ('TLSv1', 'TLSv1.1', 'SSLv3'):
            result.issues.append(f"使用不安全的协议: {result.protocol_version}")

# 证书巡检系统
# ============================================================
class SSLCertInspector:
    """SSL 证书巡检系统"""

    def __init__(self, config: dict):
        self.config = config
        self.checker = SSLCertChecker(timeout=config.get('timeout', 10))
        self.results: List[CertCheckResult] = []

        # 告警阈值
        self.warning_days = config.get('warning_days', 30)
        self.critical_days = config.get('critical_days', 7)

    def inspect(self, domains: List[dict], max_workers: int = 5):
        """批量巡检"""
        print(f"\n{'='*60}")
        print(f"SSL 证书巡检")
        print(f"时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        print(f"域名数: {len(domains)}")
        print(f"{'='*60}\n")

        with ThreadPoolExecutor(max_workers=max_workers) as executor:
            futures = {}
            for domain_conf in domains:
                hostname = domain_conf['hostname']
                port = domain_conf.get('port', 443)
                future = executor.submit(self.checker.check, hostname, port)
                futures[future] = domain_conf

            for future in as_completed(futures):
                result = future.result()
                self.results.append(result)

                # 实时输出
                icon = {'ok': '+', 'warning': '!', 'critical': 'X', 'error': '?'}.get(result.status, '?')
                print(f"  [{icon}] {result.hostname}:{result.port}  "
                      f"剩余 {result.days_remaining} 天  "
                      f"[{result.status.upper()}]  "
                      f"{result.latency_ms:.0f}ms")
                if result.issues:
                    for issue in result.issues:
                        print(f"      └─ {issue}")

    def get_alerts(self) -> List[Dict]:
        """获取需要告警的结果"""
        alerts = []
        for r in self.results:
            if r.status in ('warning', 'critical', 'error'):
                alerts.append({
                    'hostname': r.hostname,
                    'port': r.port,
                    'status': r.status,
                    'days_remaining': r.days_remaining,
                    'issues': r.issues,
                    'error': r.error,
                })
        return alerts

    def generate_report(self) -> str:
        """生成巡检报告"""
        lines = [
            f"\n{'#'*60}",
            f"#  SSL 证书巡检报告",
            f"#  时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}",
            f"{'#'*60}",
        ]

        # 汇总
        total = len(self.results)
        ok_count = sum(1 for r in self.results if r.status == 'ok')
        warn_count = sum(1 for r in self.results if r.status == 'warning')
        crit_count = sum(1 for r in self.results if r.status == 'critical')
        err_count = sum(1 for r in self.results if r.status == 'error')

        lines.append(f"\n--- 汇总 ---")
        lines.append(f"  总数: {total}")
        lines.append(f"  正常: {ok_count}")
        lines.append(f"  警告: {warn_count}")
        lines.append(f"  严重: {crit_count}")
        lines.append(f"  错误: {err_count}")

        # 详细结果
        lines.append(f"\n--- 详细结果 ---")
        lines.append(f"{'域名':<30s} {'端口':<6s} {'状态':<10s} {'剩余天数':<10s} {'签发者':<20s} {'到期时间'}")
        lines.append("-" * 110)

        # 按状态排序：critical > warning > error > ok
        status_order = {'critical': 0, 'warning': 1, 'error': 2, 'ok': 3}
        sorted_results = sorted(self.results, key=lambda r: status_order.get(r.status, 4))

        for r in sorted_results:
            lines.append(
                f"{r.hostname:<30s} {r.port:<6d} {r.status:<10s} "
                f"{r.days_remaining:<10d} {r.issuer_cn[:20]:<20s} {r.not_after}"
            )
            if r.issues:
                for issue in r.issues:
                    lines.append(f"  └─ {issue}")

        # 即将到期列表
        expiring = [r for r in self.results if 0 < r.days_remaining <= 90 and r.status != 'error']
        if expiring:
            lines.append(f"\n--- 90 天内到期证书 ---")
            for r in sorted(expiring, key=lambda x: x.days_remaining):
                lines.append(f"  {r.hostname}: {r.days_remaining} 天后到期 ({r.not_after})")

        return '\n'.join(lines)

def main():
    # 巡检配置
    config = {
        'timeout': 10,
        'warning_days': 30,
        'critical_days': 7,
    }

    # 要巡检的域名列表
```

## 7. 常见坑点与最佳实践

### 坑点

1. **证书链不完整**：服务器未配置中间证书，某些客户端验证失败
2. **SNI 问题**：同 IP 多域名时需要正确设置 `server_hostname`
3. **自签名证书**：默认会验证失败，需要配置信任
4. **时区问题**：证书时间是 UTC，本地时间需转换
5. **密钥安全**：私钥不能硬编码，使用环境变量或密钥管理服务

### 最佳实践

```python
# 1. 证书检查加入 CI/CD 流水线
# 2. 证书到期前 60/30/7 天分级告警
# 3. 使用 Let's Encrypt 自动续期
# 4. 密钥使用 Vault/KMS 管理，不存储在代码仓库
# 5. 定期扫描弱加密算法和协议版本
```
