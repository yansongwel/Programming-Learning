# IP 地址与子网计算 — 用 ipaddress/netaddr 管理网络地址

## 1. 是什么 & 为什么用它

IP 地址和子网管理是 SRE 的基础技能。Python 提供了强大的 IP 地址处理能力：

- **ipaddress（标准库）**：IP 地址/网络/接口对象，CIDR 计算
- **netaddr（第三方）**：更丰富的网络地址操作，IP 集合运算

日常使用场景：
- IP 白名单/黑名单管理
- 子网划分与容量规划
- CIDR 聚合与拆分
- 防火墙规则校验
- 网络地址合法性检查

对比命令行工具：
- `ipcalc`：只能做单次计算
- `sipcalc`：功能有限
- `Python ipaddress`：可编程、可批量处理、可集成

## 3. 核心概念与原理

### 3.1 IP 地址基础

```
IPv4 地址：32 位，点分十进制
  192.168.1.100 = 11000000.10101000.00000001.01100100

IPv6 地址：128 位，冒号十六进制
  2001:0db8:85a3:0000:0000:8a2e:0370:7334

CIDR（无类别域间路由）：
  192.168.1.0/24 → 网络地址/前缀长度
  /24 → 子网掩码 255.255.255.0 → 256 个地址 → 254 个可用主机

子网掩码对照：
  /8  → 255.0.0.0       → 16,777,216 个地址
  /16 → 255.255.0.0     → 65,536 个地址
  /24 → 255.255.255.0   → 256 个地址
  /32 → 255.255.255.255 → 1 个地址（主机路由）
```

### 3.2 ipaddress 三大对象

| 对象 | 说明 | 示例 |
|------|------|------|
| `ip_address` | 单个 IP 地址 | `192.168.1.100` |
| `ip_network` | 网络/子网 | `192.168.1.0/24` |
| `ip_interface` | IP + 网络 | `192.168.1.100/24` |

## 4. 基础用法

### 4.1 IP 地址操作

```python
#!/usr/bin/env python3
"""ipaddress 模块基础用法"""

import ipaddress

# === 创建 IP 地址对象 ===
ipv4 = ipaddress.ip_address('192.168.1.100')
ipv6 = ipaddress.ip_address('2001:db8::1')

print(f"IPv4: {ipv4}")
print(f"IPv6: {ipv6}")
print(f"版本: {ipv4.version}")  # 4
print(f"是否私有: {ipv4.is_private}")  # True
print(f"是否环回: {ipv4.is_loopback}")  # False
print(f"压缩格式: {ipv6.compressed}")
print(f"展开格式: {ipv6.exploded}")

# === IP 地址属性判断 ===
test_ips = [
    '127.0.0.1',      # 环回
    '192.168.1.1',     # 私有（C类）
    '10.0.0.1',        # 私有（A类）
    '172.16.0.1',      # 私有（B类）
    '8.8.8.8',         # 公网
    '169.254.1.1',     # 链路本地
    '224.0.0.1',       # 多播
    '255.255.255.255', # 广播
]

print(f"\n{'IP 地址':<20s} {'版本':<5s} {'私有':<5s} {'环回':<5s} {'保留':<5s} {'多播':<5s}")
print("-" * 60)
for ip_str in test_ips:
    ip = ipaddress.ip_address(ip_str)
    print(f"{str(ip):<20s} {ip.version:<5d} {str(ip.is_private):<5s} "
          f"{str(ip.is_loopback):<5s} {str(ip.is_reserved):<5s} {str(ip.is_multicast):<5s}")

# === IP 地址运算 ===
ip = ipaddress.ip_address('192.168.1.100')
print(f"\n{ip} + 10 = {ip + 10}")        # 192.168.1.110
print(f"{ip} - 50 = {ip - 50}")          # 192.168.1.50
print(f"整数: {int(ip)}")                 # 3232235876
print(f"从整数: {ipaddress.ip_address(3232235876)}")

# === IP 比较 ===
ip1 = ipaddress.ip_address('10.0.0.1')
ip2 = ipaddress.ip_address('10.0.0.2')
print(f"\n{ip1} < {ip2}: {ip1 < ip2}")    # True
print(f"{ip1} == {ip2}: {ip1 == ip2}")    # False
```

### 4.2 网络与子网操作

```python
#!/usr/bin/env python3
"""子网/网络操作"""

import ipaddress

# === 创建网络对象 ===
network = ipaddress.ip_network('192.168.1.0/24')

print(f"网络: {network}")
print(f"网络地址: {network.network_address}")      # 192.168.1.0
print(f"广播地址: {network.broadcast_address}")     # 192.168.1.255
print(f"子网掩码: {network.netmask}")               # 255.255.255.0
print(f"反掩码: {network.hostmask}")                # 0.0.0.255
print(f"前缀长度: {network.prefixlen}")             # 24
print(f"总地址数: {network.num_addresses}")          # 256
print(f"可用主机数: {network.num_addresses - 2}")   # 254

# === 遍历网络中的 IP ===
print(f"\n网络中的前 5 个主机 IP:")
for i, host in enumerate(network.hosts()):
    if i >= 5:
        break
    print(f"  {host}")
# 192.168.1.1 到 192.168.1.254

# === 判断 IP 是否在网络中 ===
ip1 = ipaddress.ip_address('192.168.1.100')
ip2 = ipaddress.ip_address('10.0.0.1')

print(f"\n{ip1} 在 {network} 中: {ip1 in network}")   # True
print(f"{ip2} 在 {network} 中: {ip2 in network}")    # False

# === 子网划分 ===
big_network = ipaddress.ip_network('10.0.0.0/16')
print(f"\n将 {big_network} 划分为 /24 子网:")
subnets_24 = list(big_network.subnets(new_prefix=24))
print(f"  共 {len(subnets_24)} 个 /24 子网")
print(f"  前 5 个: {subnets_24[:5]}")
print(f"  最后 5 个: {subnets_24[-5:]}")

# === 子网合并（超网） ===
nets = [
    ipaddress.ip_network('192.168.0.0/24'),
    ipaddress.ip_network('192.168.1.0/24'),
    ipaddress.ip_network('192.168.2.0/24'),
    ipaddress.ip_network('192.168.3.0/24'),
]
collapsed = list(ipaddress.collapse_addresses(nets))
print(f"\n子网合并:")
print(f"  原始: {nets}")
print(f"  合并: {collapsed}")  # [192.168.0.0/22]

# === 网络重叠检测 ===
net1 = ipaddress.ip_network('192.168.1.0/24')
net2 = ipaddress.ip_network('192.168.1.128/25')
net3 = ipaddress.ip_network('10.0.0.0/8')

print(f"\n{net1} 与 {net2} 重叠: {net1.overlaps(net2)}")  # True
print(f"{net1} 与 {net3} 重叠: {net1.overlaps(net3)}")   # False

# === 接口对象（IP + 网络） ===
interface = ipaddress.ip_interface('192.168.1.100/24')
print(f"\n接口: {interface}")
print(f"  IP: {interface.ip}")
print(f"  网络: {interface.network}")
print(f"  网关（通常）: {interface.network.network_address + 1}")
```

### 4.3 netaddr 高级操作

```python
#!/usr/bin/env python3
"""netaddr 高级网络地址操作"""

from netaddr import IPAddress, IPNetwork, IPSet, IPRange, cidr_merge

# === 基础使用 ===
ip = IPAddress('192.168.1.100')
print(f"IP: {ip}")
print(f"二进制: {ip.bin}")
print(f"十六进制: {ip.bits()}")
print(f"OUI 供应商: {ip.info}")  # 更丰富的信息

# === IPSet 集合运算 ===
# 白名单
whitelist = IPSet([
    '10.0.0.0/8',       # 内网
    '192.168.0.0/16',   # 内网
    '172.16.0.0/12',    # 内网
])

# 黑名单（从白名单中排除）
blacklist = IPSet([
    '10.0.99.0/24',      # 测试网段
    '192.168.100.0/24',  # 隔离网段
])

# 集合运算
allowed = whitelist - blacklist   # 差集
print(f"\n允许的网段: {allowed}")

# 判断 IP 是否在集合中
test_ip = IPAddress('10.0.0.1')
print(f"{test_ip} 在白名单中: {test_ip in whitelist}")   # True
print(f"{test_ip} 在黑名单中: {test_ip in blacklist}")   # False
print(f"{test_ip} 被允许: {test_ip in allowed}")          # True

test_ip2 = IPAddress('10.0.99.5')
print(f"{test_ip2} 被允许: {test_ip2 in allowed}")       # False（被黑名单排除）

# === IP 范围 ===
ip_range = IPRange('192.168.1.100', '192.168.1.200')
print(f"\nIP 范围: {ip_range}")
print(f"包含地址数: {ip_range.size}")

# 范围转 CIDR
cidrs = ip_range.cidrs()
print(f"转换为 CIDR: {cidrs}")

# === CIDR 合并 ===
networks = [
    IPNetwork('192.168.1.0/24'),
    IPNetwork('192.168.2.0/24'),
    IPNetwork('192.168.3.0/24'),
    IPNetwork('192.168.4.0/24'),
    IPNetwork('10.0.0.0/24'),
    IPNetwork('10.0.1.0/24'),
]
merged = cidr_merge(networks)
print(f"\n合并前: {[str(n) for n in networks]}")
print(f"合并后: {[str(n) for n in merged]}")
```

## 5. 进阶用法

### 5.1 IP 地址校验工具

```python
#!/usr/bin/env python3
"""IP 地址校验工具集"""

import ipaddress
from typing import Optional, Tuple

def validate_ip(ip_str: str) -> Tuple[bool, str]:
    """校验 IP 地址合法性"""
    try:
        ip = ipaddress.ip_address(ip_str.strip())
        return True, f"合法的 IPv{ip.version} 地址"
    except ValueError as e:
        return False, f"无效: {e}"

def validate_cidr(cidr_str: str) -> Tuple[bool, str]:
    """校验 CIDR 合法性"""
    try:
        network = ipaddress.ip_network(cidr_str.strip(), strict=True)
        return True, f"合法的网段: {network.num_addresses} 个地址"
    except ValueError as e:
        return False, f"无效: {e}"

def classify_ip(ip_str: str) -> dict:
    """IP 地址分类"""
    try:
        ip = ipaddress.ip_address(ip_str)
        return {
            'ip': str(ip),
            'version': ip.version,
            'is_private': ip.is_private,
            'is_global': ip.is_global,
            'is_loopback': ip.is_loopback,
            'is_multicast': ip.is_multicast,
            'is_reserved': ip.is_reserved,
            'is_link_local': ip.is_link_local,
            'integer': int(ip),
        }
    except ValueError:
        return {'ip': ip_str, 'error': '无效地址'}

# 测试
test_cases = [
    '192.168.1.1',
    '10.0.0.1',
    '8.8.8.8',
    '256.1.1.1',      # 无效
    '2001:db8::1',
    '192.168.1.0/24',
    '192.168.1.100/24',  # strict=True 时无效（主机位不为0）
]

print("=== IP/CIDR 校验 ===")
for tc in test_cases:
    if '/' in tc:
        valid, msg = validate_cidr(tc)
    else:
        valid, msg = validate_ip(tc)
    status = "PASS" if valid else "FAIL"
    print(f"  [{status}] {tc:25s} → {msg}")
```

### 5.2 子网计算器

```python
#!/usr/bin/env python3
"""子网计算器"""

import ipaddress

def subnet_calculator(network_str: str):
    """子网计算器：类似 ipcalc"""
    try:
        network = ipaddress.ip_network(network_str, strict=False)
    except ValueError as e:
        print(f"无效网络: {e}")
        return

    print(f"{'='*50}")
    print(f"子网计算: {network_str}")
    print(f"{'='*50}")
    print(f"  网络地址:    {network.network_address}")
    print(f"  广播地址:    {network.broadcast_address}")
    print(f"  子网掩码:    {network.netmask}")
    print(f"  反掩码:      {network.hostmask}")
    print(f"  前缀长度:    /{network.prefixlen}")
    print(f"  总地址数:    {network.num_addresses:,}")
    print(f"  可用主机数:  {max(network.num_addresses - 2, 0):,}")

    if network.num_addresses <= 256:
        hosts = list(network.hosts())
        if hosts:
            print(f"  第一个主机:  {hosts[0]}")
            print(f"  最后一个主机: {hosts[-1]}")

    # 网络类型
    first_octet = int(str(network.network_address).split('.')[0]) if network.version == 4 else 0
    if network.version == 4:
        if first_octet < 128:
            net_class = 'A'
        elif first_octet < 192:
            net_class = 'B'
        elif first_octet < 224:
            net_class = 'C'
        elif first_octet < 240:
            net_class = 'D (多播)'
        else:
            net_class = 'E (保留)'
        print(f"  网络类型:    Class {net_class}")

    # 二进制表示
    if network.version == 4:
        net_int = int(network.network_address)
        mask_int = int(network.netmask)
        net_bin = format(net_int, '032b')
        mask_bin = format(mask_int, '032b')
        print(f"\n  二进制表示:")
        print(f"  网络: {net_bin[:8]}.{net_bin[8:16]}.{net_bin[16:24]}.{net_bin[24:32]}")
        print(f"  掩码: {mask_bin[:8]}.{mask_bin[8:16]}.{mask_bin[16:24]}.{mask_bin[24:32]}")

# 测试
subnet_calculator("192.168.1.0/24")
print()
subnet_calculator("10.0.0.0/8")
print()
subnet_calculator("172.16.0.0/20")
```

## 6. SRE 实战案例：IP 白名单管理工具

```python
#!/usr/bin/env python3
"""
SRE 实战：IP 白名单管理工具
功能：
1. 白名单 CRUD 操作
2. IP 地址/CIDR 校验
3. IP 归属判断
4. 白名单合并优化
5. 冲突检测
6. 导入导出
7. 审计日志
"""

import ipaddress
import json
import time
from pathlib import Path
from datetime import datetime
from typing import List, Dict, Optional, Set, Tuple
from dataclasses import dataclass, field, asdict

# ============================================================
# 数据模型
# ============================================================
@dataclass
class WhitelistEntry:
    """白名单条目"""
    network: str            # CIDR 格式
    description: str = ""   # 描述
    owner: str = ""        # 负责人
    created_at: str = ""
    expires_at: str = ""   # 过期时间（空表示永不过期）
    tags: List[str] = field(default_factory=list)

    def __post_init__(self):
        if not self.created_at:
            self.created_at = datetime.now().isoformat()

    @property
    def ip_network(self) -> ipaddress.IPv4Network:
        return ipaddress.ip_network(self.network, strict=False)

    @property
    def is_expired(self) -> bool:
        if not self.expires_at:
            return False
        return datetime.now().isoformat() > self.expires_at

@dataclass
class AuditLog:
    """审计日志"""
    action: str         # add / remove / modify / check
    target: str
    operator: str
    result: str
    timestamp: str = ""
    details: str = ""

    def __post_init__(self):
        if not self.timestamp:
            self.timestamp = datetime.now().isoformat()

# ============================================================
# IP 白名单管理器
# ============================================================
class IPWhitelistManager:
    """IP 白名单管理工具"""

    def __init__(self, data_file: str = "/tmp/ip_whitelist.json"):
        self.data_file = Path(data_file)
        self.entries: List[WhitelistEntry] = []
        self.audit_logs: List[AuditLog] = []
        self._load()

    def _load(self):
        """从文件加载白名单"""
        if self.data_file.exists():
            data = json.loads(self.data_file.read_text())
            self.entries = [WhitelistEntry(**e) for e in data.get('entries', [])]
            self.audit_logs = [AuditLog(**a) for a in data.get('audit_logs', [])]

    def _save(self):
        """保存白名单到文件"""
        self.data_file.parent.mkdir(parents=True, exist_ok=True)
        data = {
            'entries': [asdict(e) for e in self.entries],
            'audit_logs': [asdict(a) for a in self.audit_logs[-1000:]],  # 保留最近1000条
        }
        self.data_file.write_text(json.dumps(data, ensure_ascii=False, indent=2))

    def _log_audit(self, action: str, target: str, operator: str, result: str, details: str = ""):
        """记录审计日志"""
        log = AuditLog(action=action, target=target, operator=operator,
                      result=result, details=details)
        self.audit_logs.append(log)

    def _validate_network(self, network_str: str) -> Tuple[bool, str]:
        """校验网络地址"""
        try:
            net = ipaddress.ip_network(network_str, strict=False)
            # 安全检查
            if net.prefixlen < 8:
                return False, f"前缀过短（/{net.prefixlen}），可能影响范围过大"
            return True, str(net)
        except ValueError as e:
            return False, f"无效地址: {e}"

    # === CRUD 操作 ===

    def add(self, network: str, description: str = "", owner: str = "system",
            tags: List[str] = None, expires_at: str = "") -> Tuple[bool, str]:
        """添加白名单条目"""
        valid, normalized = self._validate_network(network)
        if not valid:
            self._log_audit("add", network, owner, "failed", normalized)
            return False, normalized

        # 检查是否已存在
        for entry in self.entries:
            if entry.network == normalized:
                msg = f"已存在: {normalized}"
                self._log_audit("add", network, owner, "duplicate", msg)
                return False, msg

        # 检查冲突（新条目是否被已有条目包含，或包含已有条目）
        new_net = ipaddress.ip_network(normalized)
        conflicts = []
        for entry in self.entries:
            existing = entry.ip_network
            if new_net.overlaps(existing):
                conflicts.append(f"{entry.network} ({entry.description})")

        entry = WhitelistEntry(
            network=normalized,
            description=description,
            owner=owner,
            tags=tags or [],
            expires_at=expires_at,
        )
        self.entries.append(entry)
        self._save()

        conflict_msg = f" (与 {', '.join(conflicts)} 有重叠)" if conflicts else ""
        msg = f"已添加: {normalized}{conflict_msg}"
        self._log_audit("add", normalized, owner, "success", msg)
        return True, msg

    def remove(self, network: str, operator: str = "system") -> Tuple[bool, str]:
        """删除白名单条目"""
        try:
            normalized = str(ipaddress.ip_network(network, strict=False))
        except ValueError:
            return False, f"无效地址: {network}"

        for i, entry in enumerate(self.entries):
            if entry.network == normalized:
                removed = self.entries.pop(i)
                self._save()
                self._log_audit("remove", normalized, operator, "success",
                              f"描述: {removed.description}")
                return True, f"已删除: {normalized}"

        self._log_audit("remove", network, operator, "not_found")
        return False, f"未找到: {network}"

    def check_ip(self, ip_str: str) -> Dict:
        """检查 IP 是否在白名单中"""
        try:
            ip = ipaddress.ip_address(ip_str)
        except ValueError:
            return {'ip': ip_str, 'allowed': False, 'reason': '无效 IP'}

        matching_entries = []
        for entry in self.entries:
            if not entry.is_expired and ip in entry.ip_network:
                matching_entries.append({
                    'network': entry.network,
                    'description': entry.description,
                    'owner': entry.owner,
                })

        allowed = len(matching_entries) > 0
        self._log_audit("check", ip_str, "system",
                        "allowed" if allowed else "denied")

        return {
            'ip': ip_str,
            'allowed': allowed,
            'matching_rules': matching_entries,
        }

    def list_entries(self, tag: str = None, include_expired: bool = False) -> List[Dict]:
        """列出白名单条目"""
        results = []
        for entry in self.entries:
            if not include_expired and entry.is_expired:
                continue
            if tag and tag not in entry.tags:
                continue
            results.append(asdict(entry))
        return results

    # === 高级操作 ===

    def optimize(self) -> Tuple[int, int]:
        """优化白名单：合并可聚合的网段"""
        original_count = len(self.entries)

        # 提取所有网络
        networks = [entry.ip_network for entry in self.entries if not entry.is_expired]

        # 合并网络
        collapsed = list(ipaddress.collapse_addresses(networks))

        # 更新条目（保留第一个匹配的描述）
        new_entries = []
        for net in collapsed:
            # 找到原始条目
            desc = ""
            owner = "system"
            tags = set()
            for entry in self.entries:
                if entry.ip_network.overlaps(net) or net.overlaps(entry.ip_network):
                    if not desc:
                        desc = entry.description
                        owner = entry.owner
                    tags.update(entry.tags)
                    break

            new_entries.append(WhitelistEntry(
                network=str(net),
                description=desc or "合并后的网段",
                owner=owner,
                tags=list(tags),
            ))

        self.entries = new_entries
        self._save()

        optimized_count = len(self.entries)
        self._log_audit("optimize", "all", "system", "success",
                        f"从 {original_count} 条优化为 {optimized_count} 条")
        return original_count, optimized_count

    def find_conflicts(self) -> List[Dict]:
        """查找重叠的网段"""
        conflicts = []
        for i, e1 in enumerate(self.entries):
            for j, e2 in enumerate(self.entries):
                if i >= j:
                    continue
                if e1.ip_network.overlaps(e2.ip_network):
                    conflicts.append({
                        'network1': e1.network,
                        'desc1': e1.description,
                        'network2': e2.network,
                        'desc2': e2.description,
                    })
        return conflicts

    def cleanup_expired(self) -> int:
        """清理过期条目"""
        before = len(self.entries)
        self.entries = [e for e in self.entries if not e.is_expired]
        after = len(self.entries)
        removed = before - after
        if removed:
            self._save()
            self._log_audit("cleanup", "expired", "system", "success",
                          f"清理 {removed} 条过期规则")
        return removed

    def export_iptables(self) -> str:
        """导出为 iptables 规则"""
        rules = ["# 自动生成的 iptables 白名单规则",
                 f"# 生成时间: {datetime.now().isoformat()}",
                 ""]
        for entry in self.entries:
            if entry.is_expired:
                continue
            comment = entry.description.replace(' ', '_') if entry.description else 'whitelist'
            rules.append(
                f"iptables -A INPUT -s {entry.network} "
                f"-m comment --comment \"{comment}\" -j ACCEPT"
            )
        return '\n'.join(rules)

    def report(self) -> str:
        """生成白名单报告"""
        lines = [
            f"{'='*60}",
            f"IP 白名单报告",
            f"{'='*60}",
            f"总条目: {len(self.entries)}",
            f"过期条目: {sum(1 for e in self.entries if e.is_expired)}",
            "",
        ]

        # 按标签统计
        tag_counts = {}
        for entry in self.entries:
            for tag in entry.tags:
                tag_counts[tag] = tag_counts.get(tag, 0) + 1

        if tag_counts:
            lines.append("--- 按标签统计 ---")
            for tag, count in sorted(tag_counts.items()):
                lines.append(f"  {tag}: {count} 条")

        # 列出条目
        lines.append(f"\n--- 白名单条目 ---")
        lines.append(f"{'网段':<22s} {'描述':<20s} {'负责人':<10s} {'标签'}")
        lines.append("-" * 70)
        for entry in self.entries:
            expired = " [已过期]" if entry.is_expired else ""
            lines.append(
                f"{entry.network:<22s} {entry.description:<20s} "
                f"{entry.owner:<10s} {','.join(entry.tags)}{expired}"
            )

        # 冲突检测
        conflicts = self.find_conflicts()
        if conflicts:
            lines.append(f"\n--- 重叠警告 ({len(conflicts)} 处) ---")
            for c in conflicts:
                lines.append(f"  {c['network1']} ↔ {c['network2']}")

        return '\n'.join(lines)
```

## 7. 常见坑点与最佳实践

### 坑点

1. **strict 参数**：`ip_network('192.168.1.100/24', strict=True)` 会报错（主机位不为0）
2. **IPv4 vs IPv6**：不能直接比较 IPv4 和 IPv6 地址
3. **性能问题**：遍历大网段（/8）会非常慢，用 `in` 判断代替遍历
4. **子网掩码格式**：`ip_network` 只接受 CIDR 格式，不直接接受子网掩码

### 最佳实践

```python
# 1. 用 strict=False 容错
ipaddress.ip_network('192.168.1.100/24', strict=False)

# 2. 用 in 运算符判断归属（O(1)），不要遍历
ip in network  # 正确，高效

# 3. 使用 collapse_addresses 合并网段
# 4. 存储用字符串，计算用对象
# 5. 处理用户输入时先 strip() 再解析
```
