# 装饰器与上下文管理器 — 给你的运维函数加上超能力

## 1. 是什么 & 为什么用它

### Shell 的做法

在 Shell 中，如果你想给每个函数加上计时功能：

```bash
#!/bin/bash
# Shell 方式：每个函数都要手动加计时代码
deploy_app() {
    local start=$(date +%s)
    echo "开始部署..."
    sleep 2
    local end=$(date +%s)
    echo "部署完成，耗时: $((end - start))秒"
}

restart_service() {
    local start=$(date +%s)
    echo "重启服务..."
    sleep 1
    local end=$(date +%s)
    echo "重启完成，耗时: $((end - start))秒"
}
```

**问题**：每个函数都要重复写计时代码，违反 DRY（Don't Repeat Yourself）原则。

### Python 的做法

```python
import time
import functools

# 装饰器：一次定义，处处复用
def timer(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        elapsed = time.time() - start
        print(f"[{func.__name__}] 耗时: {elapsed:.2f}秒")
        return result
    return wrapper

@timer
def deploy_app():
    print("开始部署...")
    time.sleep(1)

@timer
def restart_service():
    print("重启服务...")
    time.sleep(0.5)

deploy_app()        # 自动打印耗时
restart_service()   # 自动打印耗时
```

**核心优势**：装饰器让你能在不修改原函数代码的情况下，给函数"增强"功能。

## 2. 核心概念与原理

### 2.1 装饰器的本质

装饰器就是一个接收函数、返回函数的函数。`@decorator` 是语法糖。

```python
# 这两种写法完全等价：
@timer
def my_func():
    pass

# 等价于：
def my_func():
    pass
my_func = timer(my_func)
```

### 2.2 执行流程图

```
调用 deploy_app()
       │
       ▼
  wrapper(*args, **kwargs)     ← 装饰器包装的函数
       │
       ├─ 1. 记录开始时间
       ├─ 2. 调用原始 deploy_app()
       ├─ 3. 记录结束时间
       ├─ 4. 打印耗时
       └─ 5. 返回原函数的结果
```

### 2.3 上下文管理器的本质

上下文管理器确保资源的"获取-使用-释放"模式。类比 Shell 的 `trap` 命令。

```bash
# Shell: 用 trap 确保清理
tmpfile=$(mktemp)
trap "rm -f $tmpfile" EXIT
echo "data" > $tmpfile
# 脚本退出时自动删除临时文件
```

```python
# Python: 用 with 确保清理
import tempfile
import os

with tempfile.NamedTemporaryFile(delete=True) as tmp:
    tmp.write(b"data")
    # with 块结束时自动删除临时文件
```

## 3. 基础用法

### 3.1 最简单的装饰器

```python
#!/usr/bin/env python3
"""
装饰器基础：逐步理解
"""
import functools


# === 第一步：理解函数是对象 ===
def greet():
    return "你好"

# 函数可以赋值给变量
say_hi = greet
print(say_hi())  # 输出: 你好

# 函数可以作为参数传递
def call_func(func):
    return func()

print(call_func(greet))  # 输出: 你好


# === 第二步：最简单的装饰器 ===
def simple_log(func):
    """给函数加上日志的装饰器"""
    @functools.wraps(func)  # 保留原函数的名字和文档
    def wrapper(*args, **kwargs):
        print(f">>> 调用 {func.__name__}()")
        result = func(*args, **kwargs)
        print(f"<<< {func.__name__}() 返回: {result}")
        return result
    return wrapper


@simple_log
def add(a, b):
    """两数相加"""
    return a + b


@simple_log
def get_hostname():
    """获取主机名"""
    import socket
    return socket.gethostname()


# --- 使用示例 ---
if __name__ == "__main__":
    result = add(3, 5)
    # 输出:
    # >>> 调用 add()
    # <<< add() 返回: 8

    hostname = get_hostname()
    # 输出:
    # >>> 调用 get_hostname()
    # <<< get_hostname() 返回: xxx

    # @functools.wraps 的作用：保留原函数信息
    print(f"函数名: {add.__name__}")      # add（不是 wrapper）
    print(f"文档: {add.__doc__}")          # 两数相加
```

### 3.2 @wraps 为什么重要

```python
#!/usr/bin/env python3
"""
没有 @wraps 的后果
"""
import functools


# 不用 @wraps 的装饰器
def bad_decorator(func):
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

# 用 @wraps 的装饰器
def good_decorator(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@bad_decorator
def func_a():
    """我是函数A"""
    pass

@good_decorator
def func_b():
    """我是函数B"""
    pass

print(f"bad:  name={func_a.__name__}, doc={func_a.__doc__}")
# bad:  name=wrapper, doc=None   ← 原函数信息丢失了！

print(f"good: name={func_b.__name__}, doc={func_b.__doc__}")
# good: name=func_b, doc=我是函数B  ← 原函数信息保留
```

## 4. 进阶用法

### 4.1 带参数的装饰器

```python
#!/usr/bin/env python3
"""
带参数的装饰器：三层嵌套
"""
import functools
import time
import logging

logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")


def retry(max_retries: int = 3, delay: float = 1.0, exceptions: tuple = (Exception,)):
    """
    重试装饰器（带参数）

    参数:
        max_retries: 最大重试次数
        delay: 重试间隔（秒）
        exceptions: 需要捕获的异常类型

    对比 Shell:
        Shell 中的重试通常这样写:
        for i in $(seq 1 3); do
            command && break
            sleep 1
        done
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(1, max_retries + 1):
                try:
                    return func(*args, **kwargs)
                except exceptions as e:
                    last_exception = e
                    logging.warning(
                        f"[{func.__name__}] 第 {attempt}/{max_retries} 次失败: {e}"
                    )
                    if attempt < max_retries:
                        time.sleep(delay)
            logging.error(f"[{func.__name__}] 全部 {max_retries} 次重试失败")
            raise last_exception
        return wrapper
    return decorator


@retry(max_retries=3, delay=0.5, exceptions=(ConnectionError, TimeoutError))
def connect_to_server(host: str) -> str:
    """模拟连接服务器（随机失败）"""
    import random
    if random.random() < 0.7:  # 70% 概率失败
        raise ConnectionError(f"无法连接 {host}")
    return f"已连接 {host}"


# --- 使用示例 ---
if __name__ == "__main__":
    try:
        result = connect_to_server("192.168.1.100")
        print(f"结果: {result}")
    except ConnectionError as e:
        print(f"最终失败: {e}")
```

### 4.2 类装饰器

```python
#!/usr/bin/env python3
"""
类装饰器：用类实现装饰器（适合需要维护状态的场景）
"""
import functools
import time


class RateLimiter:
    """
    限流装饰器（类实现）
    限制函数在指定时间窗口内的调用次数

    对比 Shell: Shell 中很难实现精确限流，通常靠 sleep 粗略控制
    """

    def __init__(self, max_calls: int = 5, period: float = 60.0):
        """
        参数:
            max_calls: 时间窗口内最大调用次数
            period: 时间窗口（秒）
        """
        self.max_calls = max_calls
        self.period = period
        self.calls = []  # 记录调用时间戳

    def __call__(self, func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            now = time.time()
            # 清理过期的调用记录
            self.calls = [t for t in self.calls if now - t < self.period]

            if len(self.calls) >= self.max_calls:
                wait_time = self.period - (now - self.calls[0])
                raise RuntimeError(
                    f"[限流] {func.__name__} 已达上限 "
                    f"({self.max_calls}次/{self.period}秒)，"
                    f"请等待 {wait_time:.1f} 秒"
                )

            self.calls.append(now)
            return func(*args, **kwargs)
        return wrapper


@RateLimiter(max_calls=3, period=10.0)
def call_api(endpoint: str) -> str:
    """调用API（限制10秒内最多3次）"""
    print(f"调用 API: {endpoint}")
    return f"响应来自 {endpoint}"


# --- 使用示例 ---
if __name__ == "__main__":
    for i in range(5):
        try:
            result = call_api(f"/api/v1/servers?page={i}")
            print(f"  结果: {result}")
        except RuntimeError as e:
            print(f"  {e}")
```

### 4.3 多个装饰器叠加

```python
#!/usr/bin/env python3
"""
装饰器叠加：从下往上执行
"""
import functools
import time
import logging

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(message)s")


def timer(func):
    """计时装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        elapsed = time.time() - start
        logging.info(f"[计时] {func.__name__} 耗时: {elapsed:.3f}秒")
        return result
    return wrapper


def log_call(func):
    """日志装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        logging.info(f"[日志] 调用 {func.__name__}({args}, {kwargs})")
        result = func(*args, **kwargs)
        logging.info(f"[日志] {func.__name__} 返回: {result}")
        return result
    return wrapper


# 装饰器从下往上包装：先 log_call，再 timer
# 执行顺序：timer → log_call → check_disk
@timer
@log_call
def check_disk(path: str = "/") -> float:
    """检查磁盘使用率"""
    import os
    stat = os.statvfs(path)
    total = stat.f_blocks * stat.f_frsize
    free = stat.f_bfree * stat.f_frsize
    usage = (1 - free / total) * 100
    return round(usage, 1)


if __name__ == "__main__":
    usage = check_disk("/")
    print(f"磁盘使用率: {usage}%")
```

## 5. 上下文管理器

### 5.1 用类实现上下文管理器

```python
#!/usr/bin/env python3
"""
上下文管理器：确保资源的安全获取和释放
"""
import time
import os
import tempfile


class TempWorkDir:
    """
    临时工作目录上下文管理器
    进入时创建临时目录并切换，退出时切回原目录并清理

    对比 Shell:
        tmpdir=$(mktemp -d)
        pushd $tmpdir
        # 做一些操作
        popd
        rm -rf $tmpdir

    Shell 的问题：如果中间出错，popd 和 rm 不会执行
    Python 的 with 语句无论是否出错都会执行 __exit__
    """

    def __init__(self, prefix: str = "sre_"):
        self.prefix = prefix
        self.tmpdir = None
        self.original_dir = None

    def __enter__(self):
        """进入 with 块：创建临时目录"""
        self.original_dir = os.getcwd()
        self.tmpdir = tempfile.mkdtemp(prefix=self.prefix)
        os.chdir(self.tmpdir)
        print(f"已创建临时目录: {self.tmpdir}")
        return self.tmpdir

    def __exit__(self, exc_type, exc_val, exc_tb):
        """退出 with 块：清理临时目录"""
        os.chdir(self.original_dir)
        # 清理临时目录
        import shutil
        shutil.rmtree(self.tmpdir, ignore_errors=True)
        print(f"已清理临时目录: {self.tmpdir}")

        if exc_type is not None:
            print(f"过程中发生异常: {exc_type.__name__}: {exc_val}")
        # 返回 False 不吞异常
        return False


class Timer:
    """
    计时上下文管理器

    对比 Shell:
        start=$(date +%s%N)
        # 操作
        end=$(date +%s%N)
        echo "耗时: $(( (end - start) / 1000000 )) ms"
    """

    def __init__(self, label: str = ""):
        self.label = label
        self.start = 0
        self.elapsed = 0

    def __enter__(self):
        self.start = time.time()
        return self

    def __exit__(self, *args):
        self.elapsed = time.time() - self.start
        label = f"[{self.label}] " if self.label else ""
        print(f"{label}耗时: {self.elapsed:.3f}秒")
        return False


# --- 使用示例 ---
if __name__ == "__main__":
    # 临时目录
    print("=== 临时工作目录 ===")
    with TempWorkDir(prefix="deploy_") as tmpdir:
        print(f"当前目录: {os.getcwd()}")
        # 在临时目录中操作
        with open("test.txt", "w") as f:
            f.write("临时数据")
        print(f"临时文件已创建: {os.path.exists('test.txt')}")
    # 退出后临时目录已清理
    print(f"当前目录: {os.getcwd()}")

    # 计时器
    print("\n=== 计时器 ===")
    with Timer("磁盘检查"):
        import subprocess
        subprocess.run(["df", "-h"], capture_output=True)

    # 嵌套使用
    print("\n=== 嵌套使用 ===")
    with Timer("总耗时"):
        with Timer("步骤1"):
            time.sleep(0.1)
        with Timer("步骤2"):
            time.sleep(0.2)
```

### 5.2 用 contextmanager 装饰器（更简洁）

```python
#!/usr/bin/env python3
"""
contextlib.contextmanager：用生成器实现上下文管理器
比写 __enter__/__exit__ 更简洁
"""
from contextlib import contextmanager
import os
import tempfile
import shutil
import time


@contextmanager
def temp_work_dir(prefix: str = "sre_"):
    """
    临时工作目录（用 contextmanager 实现）
    yield 之前 = __enter__
    yield 的值 = as 后面的变量
    yield 之后 = __exit__
    """
    original_dir = os.getcwd()
    tmpdir = tempfile.mkdtemp(prefix=prefix)
    try:
        os.chdir(tmpdir)
        print(f"进入临时目录: {tmpdir}")
        yield tmpdir  # 这里暂停，把控制权交给 with 块
    finally:
        # finally 确保无论是否出错都会执行清理
        os.chdir(original_dir)
        shutil.rmtree(tmpdir, ignore_errors=True)
        print(f"清理临时目录: {tmpdir}")


@contextmanager
def change_dir(path: str):
    """
    切换目录的上下文管理器

    对比 Shell: pushd/popd
    """
    old_dir = os.getcwd()
    os.chdir(path)
    try:
        yield path
    finally:
        os.chdir(old_dir)


@contextmanager
def timer(label: str = ""):
    """计时上下文管理器"""
    start = time.time()
    try:
        yield
    finally:
        elapsed = time.time() - start
        prefix = f"[{label}] " if label else ""
        print(f"{prefix}耗时: {elapsed:.3f}秒")


@contextmanager
def file_lock(lock_path: str):
    """
    简易文件锁

    对比 Shell: flock 命令
        flock -x /tmp/my.lock -c "command"
    """
    import fcntl
    lock_file = open(lock_path, "w")
    try:
        fcntl.flock(lock_file, fcntl.LOCK_EX | fcntl.LOCK_NB)
        print(f"获取锁: {lock_path}")
        yield lock_file
    except IOError:
        print(f"无法获取锁: {lock_path}（可能被其他进程持有）")
        raise
    finally:
        fcntl.flock(lock_file, fcntl.LOCK_UN)
        lock_file.close()
        print(f"释放锁: {lock_path}")


# --- 使用示例 ---
if __name__ == "__main__":
    # 临时目录
    with temp_work_dir("build_"):
        print(f"  工作目录: {os.getcwd()}")

    # 计时
    with timer("网络检测"):
        time.sleep(0.3)

    # 文件锁
    lock_file = "/tmp/sre_deploy.lock"
    with file_lock(lock_file):
        print("  持有锁，执行部署操作...")
        time.sleep(0.1)
    print("锁已释放")
```

## 6. SRE 实战案例

### 6.1 完整实战：运维装饰器工具集

```python
#!/usr/bin/env python3
"""
SRE 实战：运维常用装饰器工具集
包含：计时、重试、限流、缓存、告警、超时控制
"""
import functools
import time
import logging
import signal
from typing import Callable, Any

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(message)s"
)
logger = logging.getLogger(__name__)


# ============================================================
# 装饰器 1：函数计时
# ============================================================

def timer(func: Callable = None, *, threshold: float = 0.0):
    """
    函数计时装饰器

    参数:
        threshold: 超过此阈值（秒）才记录日志，0表示总是记录

    用法:
        @timer                          # 不带参数
        @timer(threshold=1.0)           # 超过1秒才记录
    """
    def decorator(fn):
        @functools.wraps(fn)
        def wrapper(*args, **kwargs):
            start = time.time()
            result = fn(*args, **kwargs)
            elapsed = time.time() - start
            if elapsed >= threshold:
                logger.info(f"[计时] {fn.__name__} 耗时: {elapsed:.3f}秒")
            return result
        return wrapper

    if func is not None:
        # @timer 不带括号的情况
        return decorator(func)
    # @timer(threshold=1.0) 带括号的情况
    return decorator


# ============================================================
# 装饰器 2：自动重试
# ============================================================

def retry(max_retries: int = 3, delay: float = 1.0,
          backoff: float = 2.0, exceptions: tuple = (Exception,)):
    """
    自动重试装饰器（带指数退避）

    参数:
        max_retries: 最大重试次数
        delay: 初始等待时间（秒）
        backoff: 退避倍数（每次重试等待时间翻倍）
        exceptions: 需要重试的异常类型
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            current_delay = delay
            last_exception = None

            for attempt in range(1, max_retries + 1):
                try:
                    return func(*args, **kwargs)
                except exceptions as e:
                    last_exception = e
                    if attempt < max_retries:
                        logger.warning(
                            f"[重试] {func.__name__} 第{attempt}次失败: {e}，"
                            f"{current_delay:.1f}秒后重试..."
                        )
                        time.sleep(current_delay)
                        current_delay *= backoff  # 指数退避
                    else:
                        logger.error(
                            f"[重试] {func.__name__} 全部{max_retries}次重试失败"
                        )

            raise last_exception
        return wrapper
    return decorator


# ============================================================
# 装饰器 3：限流
# ============================================================

def rate_limit(calls: int = 10, period: float = 60.0):
    """
    限流装饰器（滑动窗口）

    参数:
        calls: 时间窗口内允许的最大调用次数
        period: 时间窗口（秒）
    """
    def decorator(func):
        call_times = []

        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            now = time.time()
            # 清理窗口外的记录
            while call_times and now - call_times[0] >= period:
                call_times.pop(0)

            if len(call_times) >= calls:
                wait = period - (now - call_times[0])
                logger.warning(
                    f"[限流] {func.__name__} 触发限流，需等待 {wait:.1f}秒"
                )
                time.sleep(wait)
                # 清理过期记录后重试
                now = time.time()
                while call_times and now - call_times[0] >= period:
                    call_times.pop(0)

            call_times.append(time.time())
            return func(*args, **kwargs)
        return wrapper
    return decorator


# ============================================================
# 装饰器 4：简易缓存
# ============================================================

def cache(ttl: float = 300.0):
    """
    TTL 缓存装饰器

    参数:
        ttl: 缓存有效期（秒），默认5分钟
    """
    def decorator(func):
        _cache = {}

        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            # 用参数作为缓存键
            key = (args, tuple(sorted(kwargs.items())))
            now = time.time()

            if key in _cache:
                result, timestamp = _cache[key]
                if now - timestamp < ttl:
                    logger.debug(f"[缓存] {func.__name__} 命中缓存")
                    return result

            result = func(*args, **kwargs)
            _cache[key] = (result, now)
            return result

        # 提供清理缓存的方法
        wrapper.clear_cache = lambda: _cache.clear()
        return wrapper
    return decorator


# ============================================================
# 装饰器 5：超时控制
# ============================================================

def timeout(seconds: int = 30):
    """
    超时控制装饰器（仅限 Linux/macOS）

    对比 Shell: timeout 命令
        timeout 30 command
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            def handler(signum, frame):
                raise TimeoutError(
                    f"{func.__name__} 执行超时（{seconds}秒）"
                )

            # 设置信号处理
            old_handler = signal.signal(signal.SIGALRM, handler)
            signal.alarm(seconds)
            try:
                result = func(*args, **kwargs)
            finally:
                signal.alarm(0)  # 取消闹钟
                signal.signal(signal.SIGALRM, old_handler)
            return result
        return wrapper
    return decorator

    """获取磁盘使用率（带缓存，避免频繁读取）"""
    import os
    stat = os.statvfs(path)
```

## 7. 常见坑点与最佳实践

### 坑点 1: 忘记 @functools.wraps

```python
# 不加 @wraps，函数的 __name__、__doc__ 都会丢失
# 对日志和调试造成困扰
def bad_decorator(func):
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper  # wrapper.__name__ == "wrapper"

# 永远记得加 @functools.wraps(func)
```

### 坑点 2: 装饰器执行时机

```python
# 装饰器在模块导入时就执行（不是调用函数时）
def register(func):
    print(f"注册函数: {func.__name__}")  # 导入时就会打印！
    return func

@register
def my_func():
    pass
# "注册函数: my_func" 在导入时就打印了
```

### 坑点 3: 装饰带有不同签名的函数

```python
# 用 *args, **kwargs 来兼容任何函数签名
def universal_decorator(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):  # 接受任意参数
        return func(*args, **kwargs)
    return wrapper
```

### 坑点 4: 上下文管理器中 yield 后的代码不执行

```python
from contextlib import contextmanager

@contextmanager
def broken_context():
    print("进入")
    yield
    # 如果 with 块中抛出异常，这里不会执行！
    print("退出")

@contextmanager
def correct_context():
    print("进入")
    try:
        yield
    finally:
        # finally 确保一定执行
        print("退出")
```

### 最佳实践

| 实践 | 说明 |
|------|------|
| 总是使用 `@functools.wraps` | 保留原函数的元信息 |
| 用 `*args, **kwargs` | 让装饰器兼容任何函数签名 |
| 上下文管理器用 `try/finally` | 确保清理代码一定执行 |
| 简单场景用 `@contextmanager` | 比写类更简洁 |
| 装饰器要返回原函数的返回值 | `return func(*args, **kwargs)` |
| 复杂装饰器写成类 | 需要维护状态时用类装饰器 |

## 8. 速查表

```python
# === 基础装饰器 ===
import functools

def my_decorator(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        # 前置操作
        result = func(*args, **kwargs)
        # 后置操作
        return result
    return wrapper

@my_decorator
def my_func():
    pass

# === 带参数的装饰器 ===
def my_decorator(param1, param2):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            # 可以使用 param1, param2
            return func(*args, **kwargs)
        return wrapper
    return decorator

@my_decorator("a", "b")
def my_func():
    pass

# === 类装饰器 ===
class MyDecorator:
    def __init__(self, param):
        self.param = param
    def __call__(self, func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            return func(*args, **kwargs)
        return wrapper

@MyDecorator("参数")
def my_func():
    pass

# === 上下文管理器（类） ===
class MyContext:
    def __enter__(self):
        return self            # 获取资源
    def __exit__(self, exc_type, exc_val, exc_tb):
        return False           # 释放资源，False 不吞异常

with MyContext() as ctx:
    pass

# === 上下文管理器（生成器） ===
from contextlib import contextmanager

@contextmanager
def my_context():
    # __enter__ 逻辑
    resource = acquire()
    try:
        yield resource         # 交出控制权
    finally:
        release(resource)      # __exit__ 逻辑

with my_context() as res:
    pass

# === 常用内置装饰器 ===
@property                      # 属性访问器
@staticmethod                  # 静态方法
@classmethod                   # 类方法
@functools.lru_cache(maxsize=128)  # 内置LRU缓存
@functools.wraps(func)         # 保留函数元信息

# === Shell 对比 ===
# Shell trap → Python with 语句 / 上下文管理器
# Shell retry循环 → Python @retry 装饰器
# Shell timeout → Python @timeout 装饰器
# Shell 函数包装 → Python 装饰器
```
