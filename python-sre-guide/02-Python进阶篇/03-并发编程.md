# 并发编程 — 让你的运维脚本同时干多件事

## 1. 是什么 & 为什么用它

### Shell 的做法

```bash
#!/bin/bash
# Shell 方式：后台任务实现并发
check_host() {
    ping -c 1 -W 2 "$1" &>/dev/null
    if [ $? -eq 0 ]; then
        echo "$1 在线"
    else
        echo "$1 离线"
    fi
}

# 后台并发执行
for host in 192.168.1.{1..10}; do
    check_host "$host" &
done
wait  # 等待所有后台任务完成
```

**Shell 的问题**：
- 无法限制并发数（100台服务器同时 ping 会把带宽打满）
- 后台任务的返回值难以收集
- 错误处理几乎不可能
- 无法共享数据

### Python 的做法

```python
from concurrent.futures import ThreadPoolExecutor
import subprocess

def check_host(ip: str) -> dict:
    result = subprocess.run(
        ["ping", "-c", "1", "-W", "2", ip],
        capture_output=True
    )
    return {"ip": ip, "alive": result.returncode == 0}

# 用线程池并发，最多20个并发
with ThreadPoolExecutor(max_workers=20) as pool:
    hosts = [f"192.168.1.{i}" for i in range(1, 11)]
    results = pool.map(check_host, hosts)
    for r in results:
        status = "在线" if r["alive"] else "离线"
        print(f"{r['ip']} {status}")
```

**核心优势**：可控的并发数、方便收集结果、完善的错误处理。

## 2. 核心概念与原理

### 2.1 并发 vs 并行

```
并发（Concurrency）：一个时间段内处理多个任务（可以交替执行）
  CPU: ──A──B──A──B──A──B──
  适合：I/O密集型（网络请求、文件读写、ping）

并行（Parallelism）：同一时刻同时执行多个任务
  CPU1: ──A──A──A──A──
  CPU2: ──B──B──B──B──
  适合：CPU密集型（数据计算、日志分析）
```

### 2.2 GIL（全局解释器锁）

GIL 是 CPython 的一个锁，同一时刻只有一个线程能执行 Python 字节码。

| 场景 | 影响 | 解决方案 |
|------|------|---------|
| I/O操作（网络/磁盘） | 无影响，I/O时GIL会释放 | `threading` / `asyncio` |
| CPU密集计算 | 多线程无法利用多核 | `multiprocessing` |

**SRE 场景绝大多数是 I/O 密集型**（ping、HTTP请求、SSH命令、文件操作），所以多线程就够用了。

### 2.3 Python 并发方案对比

| 方案 | 适用场景 | 复杂度 | GIL影响 |
|------|---------|--------|---------|
| `threading` | I/O密集，简单并发 | 低 | 不受影响 |
| `multiprocessing` | CPU密集，需要多核 | 中 | 不受影响（独立进程） |
| `concurrent.futures` | 通用并发（推荐） | 低 | 取决于选择的Executor |
| `asyncio` | 大量I/O并发 | 高 | 不受影响 |

## 3. 基础用法

### 3.1 threading 多线程

```python
#!/usr/bin/env python3
"""
threading 基础：多线程并发
"""
import threading
import time
import subprocess


def ping_host(ip: str, results: dict):
    """
    ping 一台主机并记录结果

    参数:
        ip: 目标IP
        results: 共享字典，用于存储结果
    """
    ret = subprocess.run(
        ["ping", "-c", "1", "-W", "2", ip],
        capture_output=True
    )
    results[ip] = ret.returncode == 0
    status = "在线" if results[ip] else "离线"
    print(f"  [{threading.current_thread().name}] {ip} {status}")


if __name__ == "__main__":
    hosts = ["127.0.0.1", "8.8.8.8", "192.168.1.254", "10.0.0.1"]
    results = {}  # 线程共享的结果字典
    threads = []

    print("开始并发 ping...")
    start = time.time()

    # 创建并启动线程
    for ip in hosts:
        t = threading.Thread(
            target=ping_host,
            args=(ip, results),
            name=f"ping-{ip}"  # 给线程命名，方便调试
        )
        threads.append(t)
        t.start()

    # 等待所有线程完成
    for t in threads:
        t.join(timeout=10)  # 最多等10秒

    elapsed = time.time() - start
    print(f"\n完成！耗时: {elapsed:.2f}秒")
    print(f"在线: {sum(1 for v in results.values() if v)}/{len(results)}")
```

### 3.2 线程安全与锁

```python
#!/usr/bin/env python3
"""
线程安全：当多个线程修改共享数据时需要加锁
"""
import threading
import time


class SafeCounter:
    """线程安全的计数器"""

    def __init__(self):
        self.count = 0
        self._lock = threading.Lock()

    def increment(self):
        """线程安全的自增"""
        with self._lock:  # 自动获取和释放锁
            self.count += 1

    def get(self) -> int:
        with self._lock:
            return self.count


class UnsafeCounter:
    """非线程安全的计数器（用于对比）"""

    def __init__(self):
        self.count = 0

    def increment(self):
        # 这里有竞态条件：读取-修改-写入不是原子操作
        current = self.count
        time.sleep(0.0001)  # 模拟延迟，放大竞态问题
        self.count = current + 1


def test_counter(counter, num_threads=100):
    """测试计数器"""
    threads = []
    for _ in range(num_threads):
        t = threading.Thread(target=counter.increment)
        threads.append(t)
        t.start()
    for t in threads:
        t.join()
    return counter.count if hasattr(counter, 'count') else counter.get()


if __name__ == "__main__":
    # 测试非安全计数器
    unsafe = UnsafeCounter()
    result = test_counter(unsafe, 100)
    print(f"非安全计数器: 期望100, 实际{result}")  # 通常小于100

    # 测试安全计数器
    safe = SafeCounter()
    result = test_counter(safe, 100)
    print(f"安全计数器:   期望100, 实际{result}")  # 总是100
```

### 3.3 multiprocessing 多进程

```python
#!/usr/bin/env python3
"""
multiprocessing：真正的并行（绕过GIL）
适合 CPU 密集型任务
"""
import multiprocessing
import time
import hashlib
import os


def cpu_intensive_task(data: str) -> dict:
    """
    CPU密集任务：计算大量哈希（模拟日志分析等场景）
    """
    result = data
    for _ in range(100000):
        result = hashlib.md5(result.encode()).hexdigest()
    return {
        "input": data[:20],
        "hash": result,
        "pid": os.getpid()
    }


if __name__ == "__main__":
    test_data = [f"server-log-{i}-" + "x" * 100 for i in range(8)]

    # 串行执行
    print("=== 串行执行 ===")
    start = time.time()
    serial_results = [cpu_intensive_task(d) for d in test_data]
    serial_time = time.time() - start
    print(f"耗时: {serial_time:.2f}秒")

    # 多进程并行
    print("\n=== 多进程并行 ===")
    start = time.time()
    with multiprocessing.Pool(processes=4) as pool:
        parallel_results = pool.map(cpu_intensive_task, test_data)
    parallel_time = time.time() - start
    print(f"耗时: {parallel_time:.2f}秒")
    print(f"加速比: {serial_time / parallel_time:.1f}x")

    # 显示进程分布
    pids = set(r["pid"] for r in parallel_results)
    print(f"使用了 {len(pids)} 个进程: {pids}")
```

## 4. concurrent.futures（推荐方案）

### 4.1 ThreadPoolExecutor

```python
#!/usr/bin/env python3
"""
concurrent.futures.ThreadPoolExecutor
SRE 最推荐的并发方案：接口简洁、功能强大
"""
from concurrent.futures import ThreadPoolExecutor, as_completed
import subprocess
import time
import socket


def check_port(host: str, port: int, timeout: float = 3.0) -> dict:
    """
    检测单个端口是否开放

    对比 Shell:
        nc -z -w 3 host port 2>/dev/null && echo "open" || echo "closed"
    """
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(timeout)
        result = sock.connect_ex((host, port))
        sock.close()
        return {
            "host": host,
            "port": port,
            "open": result == 0,
            "error": None
        }
    except Exception as e:
        return {
            "host": host,
            "port": port,
            "open": False,
            "error": str(e)
        }


if __name__ == "__main__":
    # 要检测的目标
    targets = [
        ("127.0.0.1", 22),
        ("127.0.0.1", 80),
        ("127.0.0.1", 443),
        ("127.0.0.1", 3306),
        ("127.0.0.1", 6379),
        ("127.0.0.1", 8080),
        ("127.0.0.1", 9090),
    ]

    print("=== 方式1: pool.map（保持顺序） ===")
    start = time.time()
    with ThreadPoolExecutor(max_workers=10) as pool:
        # map 方式：结果按提交顺序返回
        results = pool.map(lambda t: check_port(t[0], t[1]), targets)
        for r in results:
            status = "开放" if r["open"] else "关闭"
            print(f"  {r['host']}:{r['port']} -> {status}")
    print(f"耗时: {time.time() - start:.2f}秒")

    print("\n=== 方式2: submit + as_completed（先完成先返回） ===")
    start = time.time()
    with ThreadPoolExecutor(max_workers=10) as pool:
        # submit 方式：可以获取 Future 对象
        futures = {
            pool.submit(check_port, host, port): (host, port)
            for host, port in targets
        }

        # as_completed：哪个先完成就先处理哪个
        for future in as_completed(futures):
            host, port = futures[future]
            try:
                r = future.result(timeout=10)  # 获取结果
                status = "开放" if r["open"] else "关闭"
                print(f"  {host}:{port} -> {status}")
            except Exception as e:
                print(f"  {host}:{port} -> 错误: {e}")
    print(f"耗时: {time.time() - start:.2f}秒")
```

### 4.2 ProcessPoolExecutor

```python
#!/usr/bin/env python3
"""
ProcessPoolExecutor：多进程版本，接口与 ThreadPoolExecutor 完全一致
只需把 ThreadPoolExecutor 换成 ProcessPoolExecutor
"""
from concurrent.futures import ProcessPoolExecutor
import hashlib
import time


def analyze_log_chunk(chunk: str) -> dict:
    """
    分析日志块（CPU密集型任务模拟）
    实际场景：分析大量日志文件、计算统计数据
    """
    lines = chunk.split("\n")
    error_count = sum(1 for line in lines if "ERROR" in line)
    warn_count = sum(1 for line in lines if "WARN" in line)
    # 模拟CPU计算
    checksum = hashlib.sha256(chunk.encode()).hexdigest()
    return {
        "lines": len(lines),
        "errors": error_count,
        "warnings": warn_count,
        "checksum": checksum[:16]
    }


if __name__ == "__main__":
    # 模拟大量日志数据
    import random
    levels = ["INFO", "WARN", "ERROR", "DEBUG"]
    log_chunks = []
    for i in range(8):
        lines = []
        for j in range(10000):
            level = random.choice(levels)
            lines.append(f"2024-01-01 12:00:{j:02d} [{level}] 消息 {j}")
        log_chunks.append("\n".join(lines))

    # 多进程分析
    print("=== 多进程日志分析 ===")
    start = time.time()
    with ProcessPoolExecutor(max_workers=4) as pool:
        results = list(pool.map(analyze_log_chunk, log_chunks))

    total_lines = sum(r["lines"] for r in results)
    total_errors = sum(r["errors"] for r in results)
    total_warns = sum(r["warnings"] for r in results)

    print(f"分析完成！耗时: {time.time() - start:.2f}秒")
    print(f"总行数: {total_lines}")
    print(f"错误数: {total_errors}")
    print(f"警告数: {total_warns}")
```

## 5. asyncio 异步编程

### 5.1 基础概念

```python
#!/usr/bin/env python3
"""
asyncio 基础：协程与事件循环
适合大量 I/O 并发（数百上千个并发连接）
"""
import asyncio
import time


async def check_port_async(host: str, port: int, timeout: float = 3.0) -> dict:
    """
    异步端口检测

    对比同步版本：socket.connect_ex 会阻塞线程
    异步版本：等待I/O时可以处理其他任务
    """
    try:
        # asyncio.open_connection 是非阻塞的
        reader, writer = await asyncio.wait_for(
            asyncio.open_connection(host, port),
            timeout=timeout
        )
        writer.close()
        await writer.wait_closed()
        return {"host": host, "port": port, "open": True}
    except (asyncio.TimeoutError, ConnectionRefusedError, OSError):
        return {"host": host, "port": port, "open": False}


async def main():
    """主协程"""
    targets = [
        ("127.0.0.1", port)
        for port in [22, 80, 443, 3306, 5432, 6379, 8080, 8443, 9090, 9200]
    ]

    print("=== asyncio 异步端口扫描 ===")
    start = time.time()

    # 创建所有任务
    tasks = [
        check_port_async(host, port)
        for host, port in targets
    ]

    # 并发执行所有任务
    results = await asyncio.gather(*tasks)

    for r in results:
        status = "开放" if r["open"] else "关闭"
        print(f"  {r['host']}:{r['port']} -> {status}")

    print(f"耗时: {time.time() - start:.2f}秒")


if __name__ == "__main__":
    asyncio.run(main())
```

### 5.2 asyncio 信号量（控制并发数）

```python
#!/usr/bin/env python3
"""
asyncio.Semaphore：控制异步并发数
当你有1000个任务但只想同时跑50个时使用
"""
import asyncio
import time
import random


async def fetch_url(semaphore: asyncio.Semaphore, url: str, session_id: int) -> dict:
    """模拟异步HTTP请求"""
    async with semaphore:  # 获取信号量（超过上限则等待）
        # 模拟网络延迟
        delay = random.uniform(0.1, 0.5)
        await asyncio.sleep(delay)
        return {
            "url": url,
            "status": 200,
            "latency": round(delay * 1000),  # 毫秒
        }


async def main():
    # 信号量：限制最多20个并发
    semaphore = asyncio.Semaphore(20)

    # 模拟100个请求
    urls = [f"http://api.example.com/server/{i}/health" for i in range(100)]

    print(f"开始检测 {len(urls)} 个接口（最大并发: 20）")
    start = time.time()

    tasks = [
        fetch_url(semaphore, url, i)
        for i, url in enumerate(urls)
    ]
    results = await asyncio.gather(*tasks)

    # 统计结果
    success = sum(1 for r in results if r["status"] == 200)
    avg_latency = sum(r["latency"] for r in results) / len(results)
    elapsed = time.time() - start

    print(f"完成！耗时: {elapsed:.2f}秒")
    print(f"成功: {success}/{len(results)}")
    print(f"平均延迟: {avg_latency:.0f}ms")
    print(f"如果串行执行，预计需要: {sum(r['latency'] for r in results) / 1000:.1f}秒")


if __name__ == "__main__":
    asyncio.run(main())
```

## 6. SRE 实战案例：并发批量检测 100 台服务器端口存活

```python
#!/usr/bin/env python3
"""
SRE 实战：并发批量检测服务器端口存活
支持多种并发模式对比，带进度显示和结果报告

对比 Shell 版本:
    for host in $(cat hosts.txt); do
        nc -z -w 3 $host 80 2>/dev/null && echo "$host:80 open" || echo "$host:80 closed" &
    done
    wait

Shell 的问题：无法控制并发数、无法统计结果、无进度显示
"""
import socket
import time
import threading
import sys
from concurrent.futures import ThreadPoolExecutor, as_completed
from dataclasses import dataclass, field
from typing import List, Dict, Optional


# ============================================================
# 数据模型
# ============================================================

@dataclass
class PortCheckResult:
    """端口检测结果"""
    host: str
    port: int
    is_open: bool
    latency_ms: float = 0.0  # 连接延迟（毫秒）
    error: Optional[str] = None


@dataclass
class HostCheckResult:
    """单台主机的检测结果"""
    host: str
    ports: List[PortCheckResult] = field(default_factory=list)

    @property
    def open_ports(self) -> List[int]:
        return [p.port for p in self.ports if p.is_open]

    @property
    def closed_ports(self) -> List[int]:
        return [p.port for p in self.ports if not p.is_open]

    @property
    def status(self) -> str:
        if not self.ports:
            return "未检测"
        if all(p.is_open for p in self.ports):
            return "全部正常"
        elif any(p.is_open for p in self.ports):
            return "部分正常"
        else:
            return "全部异常"


# ============================================================
# 端口检测器
# ============================================================

class PortScanner:
    """
    端口扫描器
    支持并发检测大量主机的端口状态
    """

    def __init__(self, timeout: float = 3.0, max_workers: int = 50):
        """
        参数:
            timeout: 单次连接超时（秒）
            max_workers: 最大并发线程数
        """
        self.timeout = timeout
        self.max_workers = max_workers
        self._progress_lock = threading.Lock()
        self._completed = 0
        self._total = 0

    def check_port(self, host: str, port: int) -> PortCheckResult:
        """
        检测单个端口

        对比 Shell:
            nc -z -w 3 host port 2>/dev/null
            echo $?
        """
        start = time.time()
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(self.timeout)
            result = sock.connect_ex((host, port))
            latency = (time.time() - start) * 1000  # 转为毫秒
            sock.close()

            return PortCheckResult(
                host=host,
                port=port,
                is_open=(result == 0),
                latency_ms=round(latency, 1)
            )
        except socket.gaierror as e:
            return PortCheckResult(
                host=host, port=port, is_open=False,
                error=f"DNS解析失败: {e}"
            )
        except Exception as e:
            return PortCheckResult(
                host=host, port=port, is_open=False,
                error=str(e)
            )
        finally:
            # 更新进度
            with self._progress_lock:
                self._completed += 1
                self._print_progress()

    def _print_progress(self):
        """打印进度条"""
        if self._total == 0:
            return
        pct = self._completed / self._total * 100
        bar_len = 40
        filled = int(bar_len * self._completed / self._total)
        bar = "=" * filled + "-" * (bar_len - filled)
        sys.stdout.write(f"\r  进度: [{bar}] {pct:.0f}% ({self._completed}/{self._total})")
        sys.stdout.flush()
        if self._completed == self._total:
            print()  # 完成后换行

    def scan(self, hosts: List[str], ports: List[int]) -> Dict[str, HostCheckResult]:
        """
        并发扫描多台主机的多个端口

        参数:
            hosts: 主机列表
            ports: 端口列表
        返回:
            {host: HostCheckResult} 字典
        """
        # 构建任务列表
        tasks = [(host, port) for host in hosts for port in ports]
        self._total = len(tasks)
        self._completed = 0

        print(f"开始扫描: {len(hosts)}台主机 x {len(ports)}个端口 = {len(tasks)}个检测任务")
        print(f"并发数: {self.max_workers}, 超时: {self.timeout}秒")
        start = time.time()

        # 存储结果
        results: Dict[str, HostCheckResult] = {}
        for host in hosts:
            results[host] = HostCheckResult(host=host)

        # 使用线程池并发执行
        with ThreadPoolExecutor(max_workers=self.max_workers) as pool:
            # 提交所有任务
            future_to_task = {
                pool.submit(self.check_port, host, port): (host, port)
                for host, port in tasks
            }

            # 收集结果（先完成先处理）
            for future in as_completed(future_to_task):
                host, port = future_to_task[future]
                try:
                    result = future.result(timeout=self.timeout + 5)
                    results[host].ports.append(result)
                except Exception as e:
                    results[host].ports.append(
                        PortCheckResult(host=host, port=port, is_open=False, error=str(e))
                    )

        elapsed = time.time() - start
        print(f"扫描完成！总耗时: {elapsed:.2f}秒")

        return results


# ============================================================
# 报告生成器
# ============================================================

class ScanReporter:
    """扫描报告生成器"""

    @staticmethod
    def print_summary(results: Dict[str, HostCheckResult], ports: List[int]):
        """打印汇总报告"""
        separator = "=" * 70
        print(f"\n{separator}")
        print("  端口扫描报告")
        print(separator)

        # 统计信息
        total_hosts = len(results)
        all_ok = sum(1 for r in results.values() if r.status == "全部正常")
        partial = sum(1 for r in results.values() if r.status == "部分正常")
        all_down = sum(1 for r in results.values() if r.status == "全部异常")

        print(f"\n  总主机数: {total_hosts}")
        print(f"  全部正常: {all_ok}")
        print(f"  部分正常: {partial}")
        print(f"  全部异常: {all_down}")

        # 表头
        port_headers = "  ".join(f"{p:>5}" for p in ports)
        print(f"\n  {'主机':<20} {port_headers}   状态")
        print(f"  {'-'*20} {'-----  ' * len(ports)}  ------")

        # 每台主机的结果
        for host, result in sorted(results.items()):
            port_status = []
            for port in ports:
                pr = next((p for p in result.ports if p.port == port), None)
                if pr is None:
                    port_status.append("  ?  ")
                elif pr.is_open:
                    port_status.append(f"  OK ")
                else:
                    port_status.append(f"  -- ")
            ports_str = "  ".join(port_status)
            print(f"  {host:<20} {ports_str}   {result.status}")

        # 异常详情
        problem_hosts = {
            h: r for h, r in results.items()
            if r.status in ("部分正常", "全部异常")
        }
        if problem_hosts:
            print(f"\n  --- 异常主机详情 ---")
            for host, result in problem_hosts.items():
                closed = result.closed_ports
                print(f"  {host}: 端口 {closed} 不可达")

        print(f"\n{separator}")

    @staticmethod
    def export_csv(results: Dict[str, HostCheckResult], filename: str):
```

## 7. 常见坑点与最佳实践

### 坑点 1: 线程数不是越多越好

```python
# 错误：1000 个线程会消耗大量内存和系统资源
with ThreadPoolExecutor(max_workers=1000) as pool:  # 太多了！
    pool.map(check_host, hosts)

# 推荐：根据任务类型设置合理的线程数
# I/O 密集型：线程数 = CPU核数 * 5~10（通常 50-200）
# CPU 密集型：进程数 = CPU核数
import os
cpu_count = os.cpu_count()
io_workers = min(200, cpu_count * 10)   # I/O 密集型
cpu_workers = cpu_count                   # CPU 密集型
```

### 坑点 2: multiprocessing 中函数必须可序列化

```python
# 错误：lambda 和闭包无法被 pickle 序列化
from concurrent.futures import ProcessPoolExecutor

with ProcessPoolExecutor() as pool:
    results = pool.map(lambda x: x * 2, range(10))  # 报错！

# 正确：使用普通函数
def double(x):
    return x * 2

with ProcessPoolExecutor() as pool:
    results = pool.map(double, range(10))  # 正常
```

### 坑点 3: 忘记处理异常

```python
from concurrent.futures import ThreadPoolExecutor, as_completed

def risky_task(x):
    if x == 5:
        raise ValueError("出错了！")
    return x * 2

# 错误：pool.map 遇到异常会直接抛出，后续结果丢失
# 正确：用 submit + as_completed 逐个处理
with ThreadPoolExecutor(max_workers=5) as pool:
    futures = {pool.submit(risky_task, i): i for i in range(10)}
    for future in as_completed(futures):
        i = futures[future]
        try:
            result = future.result()
            print(f"任务 {i}: {result}")
        except Exception as e:
            print(f"任务 {i} 失败: {e}")
```

### 坑点 4: asyncio 中使用同步阻塞函数

```python
import asyncio
import time

# 错误：time.sleep 是同步阻塞，会阻塞整个事件循环！
async def bad_async():
    time.sleep(5)   # 阻塞！其他协程也被卡住了

# 正确：使用异步版本
async def good_async():
    await asyncio.sleep(5)  # 非阻塞，其他协程可以继续执行

# 如果必须调用同步函数，放到线程池中
async def wrap_sync():
    loop = asyncio.get_event_loop()
    await loop.run_in_executor(None, time.sleep, 5)  # 在线程中执行
```

### 最佳实践

| 实践 | 说明 |
|------|------|
| I/O 密集用 `ThreadPoolExecutor` | ping、HTTP、SSH 等场景 |
| CPU 密集用 `ProcessPoolExecutor` | 日志分析、数据计算等 |
| 控制并发数 | 避免打满带宽/连接数 |
| 用 `as_completed` 收集结果 | 比 `map` 更灵活，能处理异常 |
| 设置超时 | `future.result(timeout=30)` |
| 大量 I/O 并发考虑 `asyncio` | 上千并发时比线程更高效 |

## 8. 速查表

```python
# === threading 基础 ===
import threading

t = threading.Thread(target=func, args=(arg1,))
t.start()                         # 启动线程
t.join(timeout=10)                # 等待线程结束
t.is_alive()                      # 是否还在运行
threading.current_thread().name   # 当前线程名

# 锁
lock = threading.Lock()
with lock:                        # 自动获取/释放锁
    shared_data += 1

# === concurrent.futures（推荐） ===
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor, as_completed

# 线程池（I/O密集）
with ThreadPoolExecutor(max_workers=20) as pool:
    results = pool.map(func, iterable)          # 保持顺序
    futures = [pool.submit(func, arg) for arg in args]  # 异步提交
    for f in as_completed(futures):              # 先完成先返回
        result = f.result(timeout=30)

# 进程池（CPU密集）
with ProcessPoolExecutor(max_workers=4) as pool:
    results = pool.map(func, iterable)  # 用法完全一致

# === asyncio ===
import asyncio

async def my_coroutine():
    await asyncio.sleep(1)
    return "done"

# 并发执行多个协程
results = await asyncio.gather(coro1(), coro2(), coro3())

# 带超时
result = await asyncio.wait_for(coro(), timeout=5.0)

# 信号量控制并发
sem = asyncio.Semaphore(20)
async with sem:
    await do_something()

# 运行入口
asyncio.run(main())

# === 方案选择速查 ===
# ping 100台服务器    → ThreadPoolExecutor(max_workers=50)
# HTTP健康检查1000个  → asyncio + aiohttp
# 分析100个日志文件   → ProcessPoolExecutor(max_workers=cpu_count)
# 简单的3个后台任务   → threading.Thread

# === Shell 对比 ===
# Shell: command &        → Python: pool.submit(func)
# Shell: wait             → Python: pool.shutdown() / join()
# Shell: xargs -P 10     → Python: ThreadPoolExecutor(max_workers=10)
# Shell: parallel         → Python: concurrent.futures
```
