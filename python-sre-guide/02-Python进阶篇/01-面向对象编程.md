# 面向对象编程 — SRE 够用就好，用类封装你的运维逻辑

## 1. 是什么 & 为什么用它

### Shell 的做法

在 Shell 中，我们通常用函数 + 全局变量来组织代码：

```bash
#!/bin/bash
# Shell 方式：用函数 + 全局变量管理服务器信息
SERVER_IP=""
SERVER_NAME=""
SERVER_CPU=""

get_server_info() {
    SERVER_IP=$(hostname -I | awk '{print $1}')
    SERVER_NAME=$(hostname)
    SERVER_CPU=$(nproc)
}

print_server_info() {
    echo "主机名: $SERVER_NAME"
    echo "IP: $SERVER_IP"
    echo "CPU核数: $SERVER_CPU"
}

get_server_info
print_server_info
```

**问题**：当需要同时管理多台服务器的信息时，全局变量就乱套了。你不得不用数组或者临时文件来存储，代码迅速变得难以维护。

### Python 的做法

```python
# Python 方式：用类把数据和操作绑在一起
class Server:
    def __init__(self, ip, name, cpu_cores):
        self.ip = ip
        self.name = name
        self.cpu_cores = cpu_cores

    def show_info(self):
        print(f"主机名: {self.name}, IP: {self.ip}, CPU: {self.cpu_cores}核")

# 可以轻松管理多台服务器
servers = [
    Server("192.168.1.1", "web-01", 4),
    Server("192.168.1.2", "web-02", 8),
    Server("192.168.1.3", "db-01", 16),
]

for s in servers:
    s.show_info()
```

**核心优势**：类把数据（属性）和操作（方法）封装在一起，天然适合管理多个同类对象。

## 2. 核心概念与原理

### 2.1 类与对象

- **类（Class）**：模板/蓝图，定义了一类事物的属性和行为
- **对象（Object/Instance）**：类的具体实例

```
类 Server  ──实例化──>  对象 server1
                        对象 server2
                        对象 server3
```

### 2.2 关键术语速查

| 术语 | 含义 | Shell 类比 |
|------|------|-----------|
| `class` | 定义类 | 类似一组相关函数的集合 |
| `__init__` | 构造函数，创建对象时自动调用 | 类似函数开头的变量初始化 |
| `self` | 指向当前对象自身 | 类似 `$0` 但指的是当前对象 |
| 实例属性 | 每个对象独有的数据 | 类似局部变量 |
| 类属性 | 所有对象共享的数据 | 类似全局变量 |
| 方法 | 定义在类中的函数 | 类似普通函数但绑定到对象 |

## 3. 基础用法

### 3.1 最小可运行示例：定义一个类

```python
#!/usr/bin/env python3
"""
最小类示例：服务器对象
"""

class Server:
    """服务器类 - 用于封装单台服务器的信息"""

    # 类属性：所有实例共享
    server_count = 0

    def __init__(self, ip: str, hostname: str, role: str = "unknown"):
        """
        构造函数：创建 Server 对象时自动调用

        参数:
            ip: 服务器IP地址
            hostname: 主机名
            role: 角色（web/db/cache等）
        """
        # 实例属性：每个对象独有
        self.ip = ip
        self.hostname = hostname
        self.role = role
        self.is_alive = True  # 默认存活

        # 每创建一个实例，计数器加1
        Server.server_count += 1

    def ping(self) -> bool:
        """检测服务器是否存活（示例简化版）"""
        import subprocess
        result = subprocess.run(
            ["ping", "-c", "1", "-W", "2", self.ip],
            capture_output=True
        )
        self.is_alive = (result.returncode == 0)
        return self.is_alive

    def show(self):
        """显示服务器信息"""
        status = "在线" if self.is_alive else "离线"
        print(f"[{status}] {self.hostname} ({self.ip}) - 角色: {self.role}")


# --- 使用示例 ---
if __name__ == "__main__":
    # 创建对象（实例化）
    web1 = Server("127.0.0.1", "web-01", "web")
    web2 = Server("192.168.1.2", "web-02", "web")
    db1 = Server("192.168.1.10", "db-01", "database")

    # 调用方法
    web1.show()   # 输出: [在线] web-01 (127.0.0.1) - 角色: web
    web2.show()   # 输出: [在线] web-02 (192.168.1.2) - 角色: web

    # 访问类属性
    print(f"服务器总数: {Server.server_count}")  # 输出: 3

    # 修改实例属性
    web2.is_alive = False
    web2.show()   # 输出: [离线] web-02 (192.168.1.2) - 角色: web
```

### 3.2 实例方法、类方法、静态方法

```python
#!/usr/bin/env python3
"""
三种方法的区别与使用场景
"""
import time


class AlertManager:
    """告警管理器"""

    # 类属性
    alert_count = 0
    severity_levels = ["info", "warning", "critical"]

    def __init__(self, name: str):
        self.name = name
        self.alerts = []

    # 实例方法：操作实例数据，第一个参数是 self
    def send_alert(self, message: str, severity: str = "info"):
        """发送告警（实例方法）- 需要访问实例属性"""
        alert = {
            "manager": self.name,
            "message": message,
            "severity": severity,
            "timestamp": time.time()
        }
        self.alerts.append(alert)
        AlertManager.alert_count += 1
        print(f"[{severity.upper()}] {self.name}: {message}")

    # 类方法：操作类级别的数据，第一个参数是 cls
    @classmethod
    def get_total_alerts(cls) -> int:
        """获取所有管理器发出的告警总数（类方法）"""
        return cls.alert_count

    @classmethod
    def from_config(cls, config_dict: dict):
        """
        从配置字典创建实例（类方法的经典用法：工厂方法）
        类似于提供另一种创建对象的方式
        """
        return cls(name=config_dict.get("name", "default"))

    # 静态方法：不需要访问实例或类数据，只是逻辑上属于这个类
    @staticmethod
    def validate_severity(severity: str) -> bool:
        """验证告警级别是否合法（静态方法）"""
        return severity in ["info", "warning", "critical"]


# --- 使用示例 ---
if __name__ == "__main__":
    # 创建实例
    mgr1 = AlertManager("主机监控")
    mgr2 = AlertManager("网络监控")

    # 实例方法：通过实例调用
    mgr1.send_alert("CPU 使用率超过 90%", "critical")
    mgr2.send_alert("丢包率上升", "warning")

    # 类方法：通过类名或实例调用
    print(f"总告警数: {AlertManager.get_total_alerts()}")

    # 类方法作为工厂方法
    config = {"name": "日志监控"}
    mgr3 = AlertManager.from_config(config)
    mgr3.send_alert("日志量激增")

    # 静态方法：通过类名或实例调用
    print(f"'error' 是否合法: {AlertManager.validate_severity('error')}")    # False
    print(f"'critical' 是否合法: {AlertManager.validate_severity('critical')}")  # True
```

**什么时候用哪种方法？**

| 方法类型 | 装饰器 | 第一参数 | 使用场景 |
|---------|--------|---------|---------|
| 实例方法 | 无 | `self` | 需要访问/修改实例数据 |
| 类方法 | `@classmethod` | `cls` | 工厂方法、操作类属性 |
| 静态方法 | `@staticmethod` | 无 | 工具函数，逻辑上属于这个类 |

## 4. 继承

### 4.1 基础继承

```python
#!/usr/bin/env python3
"""
继承示例：不同类型的服务器
"""


class BaseServer:
    """基础服务器类"""

    def __init__(self, ip: str, hostname: str):
        self.ip = ip
        self.hostname = hostname

    def get_info(self) -> dict:
        """获取基本信息"""
        return {"ip": self.ip, "hostname": self.hostname}

    def health_check(self) -> bool:
        """健康检查（子类可以覆盖）"""
        print(f"基础健康检查: ping {self.ip}")
        return True


class WebServer(BaseServer):
    """Web服务器 - 继承自 BaseServer"""

    def __init__(self, ip: str, hostname: str, port: int = 80):
        # 调用父类的 __init__
        super().__init__(ip, hostname)
        self.port = port  # 子类特有的属性

    def get_info(self) -> dict:
        """覆盖父类方法，增加端口信息"""
        info = super().get_info()  # 先调用父类方法
        info["port"] = self.port
        info["type"] = "web"
        return info

    def health_check(self) -> bool:
        """覆盖健康检查：检测HTTP端口"""
        print(f"Web服务器健康检查: 检测 {self.ip}:{self.port}")
        import socket
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(3)
            result = sock.connect_ex((self.ip, self.port))
            sock.close()
            return result == 0
        except Exception:
            return False


class DBServer(BaseServer):
    """数据库服务器 - 继承自 BaseServer"""

    def __init__(self, ip: str, hostname: str, db_type: str = "mysql", port: int = 3306):
        super().__init__(ip, hostname)
        self.db_type = db_type
        self.port = port

    def get_info(self) -> dict:
        info = super().get_info()
        info["db_type"] = self.db_type
        info["port"] = self.port
        info["type"] = "database"
        return info

    def health_check(self) -> bool:
        print(f"数据库健康检查: 检测 {self.ip}:{self.port} ({self.db_type})")
        return True


# --- 使用示例 ---
if __name__ == "__main__":
    # 创建不同类型的服务器
    servers = [
        WebServer("127.0.0.1", "nginx-01", 80),
        WebServer("127.0.0.1", "nginx-02", 8080),
        DBServer("127.0.0.1", "mysql-01", "mysql", 3306),
    ]

    # 多态：相同的方法调用，不同的行为
    for server in servers:
        print(f"\n--- {server.hostname} ---")
        print(f"信息: {server.get_info()}")
        result = server.health_check()
        print(f"健康状态: {'正常' if result else '异常'}")
```

## 5. 魔术方法（Magic Methods）

魔术方法让你的类更"Pythonic"，能像内置类型一样使用。

```python
#!/usr/bin/env python3
"""
SRE 常用魔术方法示例
"""


class DiskPartition:
    """磁盘分区类 - 演示常用魔术方法"""

    def __init__(self, mount_point: str, total_gb: float, used_gb: float):
        self.mount_point = mount_point
        self.total_gb = total_gb
        self.used_gb = used_gb

    def __str__(self) -> str:
        """
        面向用户的字符串表示（print 时调用）
        类比 Shell: echo 输出的格式
        """
        usage_pct = (self.used_gb / self.total_gb) * 100
        return f"{self.mount_point}: {self.used_gb:.1f}G/{self.total_gb:.1f}G ({usage_pct:.1f}%)"

    def __repr__(self) -> str:
        """
        面向开发者的字符串表示（调试时调用）
        应该能通过它重建对象
        """
        return f"DiskPartition('{self.mount_point}', {self.total_gb}, {self.used_gb})"

    def __lt__(self, other) -> bool:
        """小于比较 - 按使用率排序"""
        return self.usage_percent < other.usage_percent

    @property
    def usage_percent(self) -> float:
        """使用率属性"""
        return (self.used_gb / self.total_gb) * 100

    @property
    def free_gb(self) -> float:
        """剩余空间"""
        return self.total_gb - self.used_gb


class SSHConnection:
    """
    SSH 连接类 - 演示上下文管理器魔术方法
    __enter__ 和 __exit__ 让对象可以用 with 语句
    """

    def __init__(self, host: str, user: str = "root"):
        self.host = host
        self.user = user
        self.connected = False

    def __enter__(self):
        """进入 with 块时调用：建立连接"""
        print(f"正在连接 {self.user}@{self.host} ...")
        self.connected = True
        print("连接成功！")
        return self  # 返回自身，赋值给 as 后面的变量

    def __exit__(self, exc_type, exc_val, exc_tb):
        """
        退出 with 块时调用：清理资源
        无论是否发生异常都会执行（类似 Shell 的 trap）

        参数:
            exc_type: 异常类型（没有异常则为 None）
            exc_val: 异常值
            exc_tb: 异常追踪信息
        """
        print(f"正在断开 {self.host} 的连接...")
        self.connected = False
        print("已断开连接。")
        # 返回 False 表示不吞掉异常，让异常继续抛出
        return False

    def execute(self, command: str) -> str:
        """执行远程命令（模拟）"""
        if not self.connected:
            raise RuntimeError("未连接！")
        print(f"在 {self.host} 执行: {command}")
        return f"命令 '{command}' 的输出结果"


# --- 使用示例 ---
if __name__ == "__main__":
    # 1. __str__ 和 __repr__ 演示
    print("=== 磁盘分区信息 ===")
    partitions = [
        DiskPartition("/", 100, 75),
        DiskPartition("/data", 500, 420),
        DiskPartition("/var/log", 50, 10),
    ]

    for p in partitions:
        print(p)  # 调用 __str__

    # 按使用率排序（利用 __lt__）
    print("\n=== 按使用率排序 ===")
    for p in sorted(partitions, reverse=True):
        print(f"  {p}")

    # repr 用于调试
    print(f"\nrepr 输出: {repr(partitions[0])}")

    # 2. 上下文管理器演示
    print("\n=== SSH 连接测试 ===")
    with SSHConnection("192.168.1.100") as conn:
        result = conn.execute("uptime")
        print(f"结果: {result}")
    # 退出 with 块后自动断开连接
    print(f"连接状态: {conn.connected}")  # False
```

## 6. dataclass — 简化数据类

Python 3.7+ 的 `dataclass` 可以大大减少样板代码，非常适合 SRE 的数据模型。

```python
#!/usr/bin/env python3
"""
dataclass 示例：用更少的代码定义数据类
"""
from dataclasses import dataclass, field
from typing import List, Optional
import time


@dataclass
class Alert:
    """告警数据类 - 自动生成 __init__、__repr__、__eq__ 等"""
    host: str
    message: str
    severity: str = "info"               # 默认值
    timestamp: float = field(default_factory=time.time)  # 动态默认值
    tags: List[str] = field(default_factory=list)        # 可变默认值必须用 field

    def is_critical(self) -> bool:
        """dataclass 也可以有自定义方法"""
        return self.severity == "critical"


@dataclass
class ServerMetrics:
    """服务器指标数据类"""
    hostname: str
    ip: str
    cpu_percent: float
    memory_percent: float
    disk_percent: float
    load_avg: float = 0.0
    uptime_hours: float = 0.0
    services: List[str] = field(default_factory=list)

    @property
    def health_score(self) -> float:
        """健康评分（0-100）"""
        score = 100.0
        if self.cpu_percent > 80:
            score -= (self.cpu_percent - 80)
        if self.memory_percent > 80:
            score -= (self.memory_percent - 80)
        if self.disk_percent > 90:
            score -= (self.disk_percent - 90) * 2
        return max(0.0, score)

    @property
    def status(self) -> str:
        """状态判定"""
        score = self.health_score
        if score >= 80:
            return "健康"
        elif score >= 60:
            return "警告"
        else:
            return "危险"


@dataclass(frozen=True)
class ServiceConfig:
    """
    不可变的服务配置（frozen=True 表示创建后不能修改）
    适合用于配置信息，防止被意外修改
    """
    name: str
    port: int
    protocol: str = "tcp"
    max_connections: int = 1000


# --- 使用示例 ---
if __name__ == "__main__":
    # 创建告警
    alert1 = Alert("web-01", "CPU过高", "critical", tags=["cpu", "performance"])
    alert2 = Alert("web-01", "CPU过高", "critical", tags=["cpu", "performance"])

    print(f"告警: {alert1}")  # 自动生成的 __repr__
    print(f"是否严重: {alert1.is_critical()}")
    print(f"两个告警相等: {alert1 == alert2}")  # 自动生成的 __eq__（除了timestamp）

    # 服务器指标
    print("\n=== 服务器指标 ===")
    metrics = ServerMetrics(
        hostname="web-01",
        ip="192.168.1.1",
        cpu_percent=85.5,
        memory_percent=72.0,
        disk_percent=65.0,
        load_avg=3.2,
        services=["nginx", "php-fpm"]
    )
    print(f"{metrics.hostname}: 健康评分={metrics.health_score:.1f}, 状态={metrics.status}")

    # 不可变配置
    config = ServiceConfig("nginx", 80)
    print(f"\n服务配置: {config}")
    try:
        config.port = 8080  # 会报错！
    except Exception as e:
        print(f"尝试修改frozen对象: {e}")
```

## 7. SRE 实战案例：服务器信息采集器

```python
#!/usr/bin/env python3
"""
SRE 实战：用类封装服务器信息采集器
功能：采集本机CPU/内存/磁盘/网络信息，生成报告

对比 Shell 版本：
    Shell 中你需要写一堆 awk/sed/grep 来解析 /proc 文件
    Python 用类封装后，代码结构清晰，数据复用方便
"""
import os
import time
import json
from dataclasses import dataclass, field, asdict
from typing import List, Dict, Optional
from pathlib import Path


# ============================================================
# 数据模型
# ============================================================

@dataclass
class CPUInfo:
    """CPU 信息"""
    model: str = ""
    cores: int = 0
    usage_percent: float = 0.0


@dataclass
class MemoryInfo:
    """内存信息"""
    total_mb: float = 0.0
    used_mb: float = 0.0
    free_mb: float = 0.0
    usage_percent: float = 0.0


@dataclass
class DiskInfo:
    """单个磁盘分区信息"""
    mount_point: str = ""
    total_gb: float = 0.0
    used_gb: float = 0.0
    free_gb: float = 0.0
    usage_percent: float = 0.0


@dataclass
class ServerReport:
    """服务器采集报告"""
    hostname: str = ""
    ip: str = ""
    os_info: str = ""
    uptime: str = ""
    cpu: CPUInfo = field(default_factory=CPUInfo)
    memory: MemoryInfo = field(default_factory=MemoryInfo)
    disks: List[DiskInfo] = field(default_factory=list)
    collect_time: str = ""

    def to_dict(self) -> dict:
        """转换为字典"""
        return asdict(self)

    def to_json(self) -> str:
        """转换为 JSON 字符串"""
        return json.dumps(self.to_dict(), ensure_ascii=False, indent=2)


# ============================================================
# 采集器
# ============================================================

class ServerCollector:
    """
    服务器信息采集器

    Shell 对比:
        Shell 中你得写这样的代码：
            cpu_model=$(cat /proc/cpuinfo | grep 'model name' | head -1 | awk -F: '{print $2}')
            mem_total=$(free -m | awk '/Mem:/{print $2}')
            disk_usage=$(df -BG | grep '^/dev/')

        Python 用类封装后，采集逻辑清晰分离，数据有类型约束
    """

    def __init__(self):
        self.report = ServerReport()

    def collect_all(self) -> ServerReport:
        """执行全量采集"""
        self.report.collect_time = time.strftime("%Y-%m-%d %H:%M:%S")
        self._collect_hostname()
        self._collect_os_info()
        self._collect_uptime()
        self._collect_cpu()
        self._collect_memory()
        self._collect_disk()
        return self.report

    def _collect_hostname(self):
        """采集主机名和IP"""
        import socket
        self.report.hostname = socket.gethostname()
        try:
            self.report.ip = socket.gethostbyname(socket.gethostname())
        except socket.gaierror:
            self.report.ip = "127.0.0.1"

    def _collect_os_info(self):
        """采集操作系统信息"""
        # 对比 Shell: cat /etc/os-release | grep PRETTY_NAME
        try:
            with open("/etc/os-release") as f:
                for line in f:
                    if line.startswith("PRETTY_NAME"):
                        self.report.os_info = line.split("=")[1].strip().strip('"')
                        break
        except FileNotFoundError:
            self.report.os_info = "未知"

    def _collect_uptime(self):
        """采集运行时间"""
        # 对比 Shell: uptime -p
        try:
            with open("/proc/uptime") as f:
                uptime_seconds = float(f.read().split()[0])
                days = int(uptime_seconds // 86400)
                hours = int((uptime_seconds % 86400) // 3600)
                minutes = int((uptime_seconds % 3600) // 60)
                self.report.uptime = f"{days}天 {hours}小时 {minutes}分钟"
        except FileNotFoundError:
            self.report.uptime = "未知"

    def _collect_cpu(self):
        """采集CPU信息"""
        cpu = CPUInfo()

        # 采集CPU型号和核数
        # 对比 Shell: grep 'model name' /proc/cpuinfo | head -1
        try:
            with open("/proc/cpuinfo") as f:
                for line in f:
                    if line.startswith("model name"):
                        cpu.model = line.split(":")[1].strip()
                        break
            cpu.cores = os.cpu_count() or 0
        except FileNotFoundError:
            cpu.model = "未知"

        # 采集CPU使用率（简化版：读取 /proc/stat 快照）
        try:
            with open("/proc/stat") as f:
                line = f.readline()
                parts = line.split()
                # user + nice + system + idle + iowait + irq + softirq
                total = sum(int(x) for x in parts[1:8])
                idle = int(parts[4])
                cpu.usage_percent = round((1 - idle / total) * 100, 1) if total > 0 else 0.0
        except (FileNotFoundError, ValueError):
            cpu.usage_percent = 0.0

        self.report.cpu = cpu

    def _collect_memory(self):
        """采集内存信息"""
        # 对比 Shell: free -m | awk '/Mem:/{print $2, $3}'
        mem = MemoryInfo()
        try:
            with open("/proc/meminfo") as f:
                meminfo = {}
                for line in f:
                    parts = line.split(":")
                    if len(parts) == 2:
                        key = parts[0].strip()
                        # 值通常是 "12345 kB"
                        value_str = parts[1].strip().split()[0]
                        meminfo[key] = int(value_str)

                total_kb = meminfo.get("MemTotal", 0)
                free_kb = meminfo.get("MemFree", 0)
                buffers_kb = meminfo.get("Buffers", 0)
                cached_kb = meminfo.get("Cached", 0)

                mem.total_mb = round(total_kb / 1024, 1)
                # 可用内存 = Free + Buffers + Cached
                available_kb = free_kb + buffers_kb + cached_kb
                mem.free_mb = round(available_kb / 1024, 1)
                mem.used_mb = round(mem.total_mb - mem.free_mb, 1)
                mem.usage_percent = round((mem.used_mb / mem.total_mb) * 100, 1) if mem.total_mb > 0 else 0.0
        except (FileNotFoundError, ValueError):
            pass

        self.report.memory = mem

    def _collect_disk(self):
        """采集磁盘信息"""
        # 对比 Shell: df -BG --output=target,size,used,avail,pcent | grep '^/'
        disks = []
        try:
            with open("/proc/mounts") as f:
                seen_devices = set()
                for line in f:
                    parts = line.split()
                    device = parts[0]
                    mount_point = parts[1]

                    # 只关注真实的块设备分区
                    if not device.startswith("/dev/"):
                        continue
                    if device in seen_devices:
                        continue
                    seen_devices.add(device)

                    try:
                        stat = os.statvfs(mount_point)
                        total = stat.f_blocks * stat.f_frsize
                        free = stat.f_bfree * stat.f_frsize
                        used = total - free

                        disk = DiskInfo(
                            mount_point=mount_point,
                            total_gb=round(total / (1024**3), 1),
                            used_gb=round(used / (1024**3), 1),
                            free_gb=round(free / (1024**3), 1),
                            usage_percent=round((used / total) * 100, 1) if total > 0 else 0.0
                        )
                        disks.append(disk)
                    except OSError:
                        continue
        except FileNotFoundError:
            pass

        self.report.disks = disks
```

## 8. 常见坑点与最佳实践

### 坑点 1: 可变默认参数

```python
# 错误！列表是可变对象，所有实例会共享同一个列表
class BadServer:
    def __init__(self, name, tags=[]):
        self.name = name
        self.tags = tags

s1 = BadServer("web-01")
s1.tags.append("production")
s2 = BadServer("web-02")
print(s2.tags)  # ['production'] — 被污染了！

# 正确写法
class GoodServer:
    def __init__(self, name, tags=None):
        self.name = name
        self.tags = tags if tags is not None else []
```

### 坑点 2: 忘记调用 super().__init__

```python
class Parent:
    def __init__(self):
        self.parent_attr = "我是父类属性"

# 错误：忘记调用 super().__init__()
class BadChild(Parent):
    def __init__(self):
        self.child_attr = "我是子类属性"

# c = BadChild()
# print(c.parent_attr)  # AttributeError!

# 正确写法
class GoodChild(Parent):
    def __init__(self):
        super().__init__()  # 先调用父类构造
        self.child_attr = "我是子类属性"
```

### 坑点 3: 类属性 vs 实例属性混淆

```python
class Counter:
    count = 0  # 类属性

    def __init__(self):
        # self.count = 0  # 如果加这行，就变成了实例属性，不共享
        pass

    def increment(self):
        Counter.count += 1  # 正确：修改类属性
        # self.count += 1  # 容易出错：创建了实例属性，不影响其他实例
```

### 最佳实践

| 实践 | 说明 |
|------|------|
| 使用 `dataclass` | 纯数据类首选，减少样板代码 |
| 实现 `__str__` 和 `__repr__` | 方便调试和日志输出 |
| 用 `with` 管理资源 | 实现 `__enter__/__exit__`，确保资源释放 |
| 优先组合而非继承 | 继承层次不要超过 2-3 层 |
| 类型提示 | 加上类型提示提高可读性 |
| 私有属性用单下划线 `_` | `_internal_attr` 表示内部使用 |

## 9. 速查表

```python
# === 定义类 ===
class MyClass:
    class_attr = "共享"           # 类属性

    def __init__(self, x):        # 构造函数
        self.x = x                # 实例属性

    def method(self):             # 实例方法
        return self.x

    @classmethod
    def cls_method(cls):          # 类方法
        return cls.class_attr

    @staticmethod
    def static_method():          # 静态方法
        return "工具函数"

# === 继承 ===
class Child(Parent):
    def __init__(self):
        super().__init__()        # 调用父类构造

# === 常用魔术方法 ===
__init__(self)                    # 构造函数
__str__(self)                     # print() 输出
__repr__(self)                    # 调试输出
__eq__(self, other)               # == 比较
__lt__(self, other)               # < 比较（排序用）
__len__(self)                     # len() 支持
__enter__(self)                   # with 进入
__exit__(self, *args)             # with 退出
__getitem__(self, key)            # [] 取值
__iter__(self)                    # for 循环支持

# === dataclass ===
from dataclasses import dataclass, field

@dataclass
class Config:
    name: str                     # 必填
    port: int = 80                # 有默认值
    tags: list = field(default_factory=list)  # 可变默认值

@dataclass(frozen=True)           # 不可变
class ImmutableConfig:
    name: str
    port: int

# === 上下文管理器 ===
class MyContext:
    def __enter__(self):
        # 获取资源
        return self
    def __exit__(self, exc_type, exc_val, exc_tb):
        # 释放资源
        return False

# === 属性装饰器 ===
class Server:
    @property
    def usage(self):              # 像属性一样访问: server.usage
        return self._calc_usage()

# === isinstance 类型检查 ===
isinstance(obj, MyClass)          # True/False
isinstance(obj, (ClassA, ClassB)) # 多类型检查

# === Shell 对比 ===
# Shell: 全局变量 + 函数 → Python: 类属性 + 方法
# Shell: source 文件    → Python: import 模块 / 继承
# Shell: trap 清理      → Python: __exit__ / with 语句
# Shell: 数组           → Python: 对象列表
```
