## 一、字符串优化

### 1.1 三种拼接方式 Benchmark

```go
// string_bench_test.go
// 对比三种字符串拼接方式的性能
package main

import (
	"fmt"
	"strings"
	"testing"
)

// === 方法 1：+ 运算符拼接 ===
// 每次拼接都会创建新的字符串（字符串不可变）
func concatPlus(n int) string {
	result := ""
	for i := 0; i < n; i++ {
		result += "item-" + fmt.Sprintf("%d", i) + ","
	}
	return result
}

// === 方法 2：fmt.Sprintf ===
// 使用反射解析格式字符串，有额外开销
func concatSprintf(n int) string {
	result := ""
	for i := 0; i < n; i++ {
		result = fmt.Sprintf("%s%s%d,", result, "item-", i)
	}
	return result
}

// === 方法 3：strings.Builder（推荐） ===
// 内部维护 []byte，避免重复分配
func concatBuilder(n int) string {
	var b strings.Builder
	b.Grow(n * 10) // 预估总长度，减少扩容次数
	for i := 0; i < n; i++ {
		b.WriteString("item-")
		fmt.Fprintf(&b, "%d", i)
		b.WriteByte(',')
	}
	return b.String()
}

// === 方法 4：strings.Join（适合已有切片） ===
func concatJoin(n int) string {
	parts := make([]string, 0, n)
	for i := 0; i < n; i++ {
		parts = append(parts, fmt.Sprintf("item-%d", i))
	}
	return strings.Join(parts, ",")
}

// === 方法 5：[]byte 直接操作 ===
func concatBytes(n int) string {
	buf := make([]byte, 0, n*10)
	for i := 0; i < n; i++ {
		buf = append(buf, "item-"...)
		buf = fmt.Appendf(buf, "%d", i) // Go 1.19+ fmt.Appendf
		buf = append(buf, ',')
	}
	return string(buf)
}

func BenchmarkConcatPlus(b *testing.B) {
	for i := 0; i < b.N; i++ {
		_ = concatPlus(100)
	}
}

func BenchmarkConcatSprintf(b *testing.B) {
	for i := 0; i < b.N; i++ {
		_ = concatSprintf(100)
	}
}

func BenchmarkConcatBuilder(b *testing.B) {
	for i := 0; i < b.N; i++ {
		_ = concatBuilder(100)
	}
}

func BenchmarkConcatJoin(b *testing.B) {
	for i := 0; i < b.N; i++ {
		_ = concatJoin(100)
	}
}

func BenchmarkConcatBytes(b *testing.B) {
	for i := 0; i < b.N; i++ {
		_ = concatBytes(100)
	}
}

/*
运行 benchmark:
  go test -bench=BenchmarkConcat -benchmem -count=3

典型结果（相对性能）:
  BenchmarkConcatPlus-8      最慢，O(n^2) 复杂度
  BenchmarkConcatSprintf-8   很慢，反射 + 字符串拼接双重开销
  BenchmarkConcatBuilder-8   最快（或接近最快）
  BenchmarkConcatJoin-8      较快，但有额外的切片分配
  BenchmarkConcatBytes-8     最快（fmt.Appendf 零分配）

结论:
  - 少量拼接（2-3 个）：用 + 没问题
  - 循环拼接：用 strings.Builder + Grow
  - 已有字符串切片：用 strings.Join
  - 极致性能：用 []byte + fmt.Appendf
*/
```

### 1.2 字符串转换优化

```go
// string_conversion.go
// 字符串与字节切片转换的性能优化
package main

import (
	"fmt"
	"strconv"
	"unsafe"
)

// 安全的转换（有拷贝）
func safeConversion() {
	s := "hello world"
	b := []byte(s)   // 拷贝：string -> []byte
	s2 := string(b)  // 拷贝：[]byte -> string
	fmt.Println(s2)
}

// 零拷贝转换（危险！仅在确保不修改时使用）
// Go 1.22+ unsafe.String 和 unsafe.StringData 是推荐的方式
func unsafeStringToBytes(s string) []byte {
	// 警告：返回的 []byte 绝对不能被修改！
	// 因为 string 是不可变的，修改会导致未定义行为
	return unsafe.Slice(unsafe.StringData(s), len(s))
}

func unsafeBytesToString(b []byte) string {
	// 警告：b 在转换后不能被修改！
	return unsafe.String(unsafe.SliceData(b), len(b))
}

// 数字转字符串的多种方式
func numberToStringBenchmark() {
	n := 12345

	// 方法 1：fmt.Sprintf（最慢，使用反射）
	_ = fmt.Sprintf("%d", n)

	// 方法 2：strconv.Itoa（推荐）
	_ = strconv.Itoa(n)

	// 方法 3：strconv.FormatInt（性能与 Itoa 相同）
	_ = strconv.FormatInt(int64(n), 10)

	// 方法 4：fmt.Appendf 到预分配 buffer（最快）
	buf := make([]byte, 0, 20)
	buf = fmt.Appendf(buf, "%d", n)
	_ = string(buf)
}

func main() {
	fmt.Println("=== 字符串转换优化 ===\n")

	safeConversion()

	fmt.Println("字符串转数字性能排名:")
	fmt.Println("  1. strconv.Atoi / strconv.ParseInt（最快）")
	fmt.Println("  2. fmt.Sscanf（最慢，使用反射）")
	fmt.Println()
	fmt.Println("数字转字符串性能排名:")
	fmt.Println("  1. strconv.Itoa / strconv.FormatInt（最快）")
	fmt.Println("  2. fmt.Appendf 到 []byte（相当快）")
	fmt.Println("  3. fmt.Sprintf（最慢）")
	fmt.Println()
	fmt.Println("零拷贝转换注意事项:")
	fmt.Println("  - 仅在性能关键路径且确保安全时使用")
	fmt.Println("  - 转换后的 []byte 绝对不能修改")
	fmt.Println("  - Go 1.22+ 推荐使用 unsafe.String/unsafe.StringData")
}
```

## 二、切片与 Map 预分配

### 2.1 切片预分配

```go
// slice_prealloc_test.go
// 演示切片预分配的性能差异
package main

import (
	"testing"
)

// 不预分配：从 nil 切片开始 append
func appendWithoutPrealloc(n int) []int {
	var s []int // len=0, cap=0
	for i := 0; i < n; i++ {
		s = append(s, i) // 触发多次扩容
	}
	return s
}

// 预分配容量：使用 make 指定 cap
func appendWithPrealloc(n int) []int {
	s := make([]int, 0, n) // len=0, cap=n
	for i := 0; i < n; i++ {
		s = append(s, i) // 不会触发扩容
	}
	return s
}

// 直接使用下标赋值（最快）
func directAssign(n int) []int {
	s := make([]int, n) // len=n, cap=n
	for i := 0; i < n; i++ {
		s[i] = i
	}
	return s
}

func BenchmarkAppendNoPrealloc_100(b *testing.B) {
	for i := 0; i < b.N; i++ {
		_ = appendWithoutPrealloc(100)
	}
}

func BenchmarkAppendPrealloc_100(b *testing.B) {
	for i := 0; i < b.N; i++ {
		_ = appendWithPrealloc(100)
	}
}

func BenchmarkDirectAssign_100(b *testing.B) {
	for i := 0; i < b.N; i++ {
		_ = directAssign(100)
	}
}

func BenchmarkAppendNoPrealloc_10000(b *testing.B) {
	for i := 0; i < b.N; i++ {
		_ = appendWithoutPrealloc(10000)
	}
}

func BenchmarkAppendPrealloc_10000(b *testing.B) {
	for i := 0; i < b.N; i++ {
		_ = appendWithPrealloc(10000)
	}
}

func BenchmarkDirectAssign_10000(b *testing.B) {
	for i := 0; i < b.N; i++ {
		_ = directAssign(10000)
	}
}

/*
运行: go test -bench=Benchmark -benchmem

切片扩容规则（Go 1.21+）:
  - cap < 256: 新容量 = 旧容量 * 2
  - cap >= 256: 新容量 = 旧容量 + 旧容量/4 + 192
  - 每次扩容都需要: 分配新内存 + 拷贝旧数据 + 旧内存等待 GC

预分配建议:
  - 如果知道确切大小: make([]T, n)
  - 如果知道大致大小: make([]T, 0, n)
  - 如果完全未知: 至少给一个合理的初始值
*/
```

```go
package main

	"fmt"

		_ = mapWithoutHint(10000)
```

## 三、避免不必要的拷贝

### 3.1 大结构体传递

```go
// copy_optimization_test.go
// 演示值传递 vs 指针传递的性能差异
package main

import (
	"testing"
)

// 大结构体（280 字节）
type LargeStruct struct {
	Field1  [32]byte
	Field2  [32]byte
	Field3  [32]byte
	Field4  [32]byte
	Field5  [32]byte
	Field6  [32]byte
	Field7  [32]byte
	Field8  [32]byte
	Counter int64
}

// 小结构体（24 字节）
type SmallStruct struct {
	X, Y, Z float64
}

// 值传递大结构体（每次调用都拷贝 280 字节）
func processLargeByValue(s LargeStruct) int64 {
	return s.Counter + 1
}

// 指针传递大结构体（只拷贝 8 字节的指针）
func processLargeByPointer(s *LargeStruct) int64 {
	return s.Counter + 1
}

// 值传递小结构体（拷贝 24 字节，很快）
func processSmallByValue(s SmallStruct) float64 {
	return s.X + s.Y + s.Z
}

// 指针传递小结构体（可能导致逃逸到堆）
func processSmallByPointer(s *SmallStruct) float64 {
	return s.X + s.Y + s.Z
}

func BenchmarkLargeByValue(b *testing.B) {
	s := LargeStruct{Counter: 42}
	for i := 0; i < b.N; i++ {
		_ = processLargeByValue(s)
	}
}

func BenchmarkLargeByPointer(b *testing.B) {
	s := &LargeStruct{Counter: 42}
	for i := 0; i < b.N; i++ {
		_ = processLargeByPointer(s)
	}
}

func BenchmarkSmallByValue(b *testing.B) {
	s := SmallStruct{X: 1.0, Y: 2.0, Z: 3.0}
	for i := 0; i < b.N; i++ {
		_ = processSmallByValue(s)
	}
}

func BenchmarkSmallByPointer(b *testing.B) {
	s := &SmallStruct{X: 1.0, Y: 2.0, Z: 3.0}
	for i := 0; i < b.N; i++ {
		_ = processSmallByPointer(s)
	}
}

/*
经验法则:
  - 结构体 > 64 字节: 优先用指针传递
  - 结构体 <= 64 字节: 值传递通常更快（避免间接访问和可能的逃逸）
  - 需要修改结构体: 必须用指针
  - 并发共享: 必须用指针 + 同步机制

注意:
  指针传递可能导致对象逃逸到堆上，增加 GC 压力。
  对于小结构体，值传递让编译器有更多优化空间。
*/
```

### 3.2 Range 循环避免拷贝

```go
// range_copy.go
// range 循环中的隐式拷贝问题
package main

import (
	"fmt"
	"testing"
)

type BigItem struct {
	Data [4096]byte
	ID   int
}

// 低效: range 会拷贝每个元素
func sumBigItemsSlow(items []BigItem) int {
	sum := 0
	for _, item := range items { // item 是 BigItem 的拷贝（4104 字节）
		sum += item.ID
	}
	return sum
}

// 高效: 使用下标访问
func sumBigItemsFast(items []BigItem) int {
	sum := 0
	for i := range items { // 不拷贝
		sum += items[i].ID
	}
	return sum
}

// 高效: 使用指针切片
func sumBigItemsPointer(items []*BigItem) int {
	sum := 0
	for _, item := range items { // item 是指针的拷贝（8 字节）
		sum += item.ID
	}
	return sum
}

func BenchmarkRangeCopy(b *testing.B) {
	items := make([]BigItem, 1000)
	for i := range items {
		items[i].ID = i
	}
	b.ResetTimer()
	for i := 0; i < b.N; i++ {
		_ = sumBigItemsSlow(items)
	}
}

func BenchmarkRangeIndex(b *testing.B) {
	items := make([]BigItem, 1000)
	for i := range items {
		items[i].ID = i
	}
	b.ResetTimer()
	for i := 0; i < b.N; i++ {
		_ = sumBigItemsFast(items)
	}
}

func main() {
	fmt.Println("运行: go test -bench=BenchmarkRange -benchmem")
	fmt.Println()
	fmt.Println("建议:")
	fmt.Println("  - 小元素（<= 64B）: for _, v := range 没问题")
	fmt.Println("  - 大元素（> 64B）: 用 for i := range 加下标访问")
	fmt.Println("  - 或者使用指针切片 []*BigItem")
}
```

## 四、sync.Pool 复用对象

```go
// sync_pool_advanced.go
// sync.Pool 高级用法与实际应用场景
package main

import (
	"bytes"
	"compress/gzip"
	"fmt"
	"io"
	"sync"
	"testing"
)

// === 场景 1：gzip Writer 复用 ===
// gzip.NewWriter 内部分配了大量缓冲区，创建成本很高

var gzipWriterPool = sync.Pool{
	New: func() interface{} {
		// 初始级别为 BestSpeed，后续可以通过 Reset 更改
		w, _ := gzip.NewWriterLevel(io.Discard, gzip.BestSpeed)
		return w
	},
}

// 不使用 Pool: 每次都创建新的 gzip Writer
func compressWithoutPool(data []byte) ([]byte, error) {
	var buf bytes.Buffer
	w := gzip.NewWriter(&buf)
	if _, err := w.Write(data); err != nil {
		return nil, err
	}
	if err := w.Close(); err != nil {
		return nil, err
	}
	return buf.Bytes(), nil
}

// 使用 Pool: 复用 gzip Writer
func compressWithPool(data []byte) ([]byte, error) {
	var buf bytes.Buffer
	w := gzipWriterPool.Get().(*gzip.Writer)
	w.Reset(&buf)

	if _, err := w.Write(data); err != nil {
		gzipWriterPool.Put(w)
		return nil, err
	}
	if err := w.Close(); err != nil {
		gzipWriterPool.Put(w)
		return nil, err
	}

	gzipWriterPool.Put(w)
	return buf.Bytes(), nil
}

// === 场景 2：带最大容量限制的 Buffer Pool ===

type BoundedBufferPool struct {
	pool   sync.Pool
	maxCap int
}

func NewBoundedBufferPool(initCap, maxCap int) *BoundedBufferPool {
	return &BoundedBufferPool{
		pool: sync.Pool{
			New: func() interface{} {
				return bytes.NewBuffer(make([]byte, 0, initCap))
			},
		},
		maxCap: maxCap,
	}
}

func (p *BoundedBufferPool) Get() *bytes.Buffer {
	return p.pool.Get().(*bytes.Buffer)
}

func (p *BoundedBufferPool) Put(buf *bytes.Buffer) {
	// 如果 buffer 增长过大，丢弃而不是归还
	// 防止少量大请求导致池中都是大 buffer
	if buf.Cap() > p.maxCap {
		return
	}
	buf.Reset()
	p.pool.Put(buf)
}

// === 场景 3：结构体对象池 ===

type Request struct {
	Method  string
	URL     string
	Headers map[string]string
	Body    []byte
}

var requestPool = sync.Pool{
	New: func() interface{} {
		return &Request{
			Headers: make(map[string]string, 8),
		}
	},
}

func acquireRequest() *Request {
	return requestPool.Get().(*Request)
}

func releaseRequest(req *Request) {
	// 清理所有字段
	req.Method = ""
	req.URL = ""
	req.Body = req.Body[:0] // 保留底层数组
	// 清理 map（注意：Go 的 map 不能 "重置"，需要逐个删除）
	for k := range req.Headers {
		delete(req.Headers, k)
	}
	requestPool.Put(req)
}

func BenchmarkCompressWithoutPool(b *testing.B) {
	data := bytes.Repeat([]byte("hello world "), 100)
	b.ResetTimer()
	for i := 0; i < b.N; i++ {
		_, _ = compressWithoutPool(data)
	}
}

func BenchmarkCompressWithPool(b *testing.B) {
	data := bytes.Repeat([]byte("hello world "), 100)
	b.ResetTimer()
	for i := 0; i < b.N; i++ {
		_, _ = compressWithPool(data)
	}
}

func main() {
	// 演示使用
	data := bytes.Repeat([]byte("test data for compression "), 50)

	compressed1, _ := compressWithoutPool(data)
	compressed2, _ := compressWithPool(data)

	fmt.Printf("原始大小: %d bytes\n", len(data))
	fmt.Printf("压缩后（无 Pool）: %d bytes\n", len(compressed1))
	fmt.Printf("压缩后（有 Pool）: %d bytes\n", len(compressed2))

	// 对象池使用
	req := acquireRequest()
	req.Method = "GET"
	req.URL = "/api/data"
	req.Headers["Content-Type"] = "application/json"
	fmt.Printf("\n请求: %s %s\n", req.Method, req.URL)
	releaseRequest(req)

	fmt.Println("\n运行 benchmark: go test -bench=BenchmarkCompress -benchmem")
}
```

## 五、零拷贝技术

```go
// zero_copy.go
// 演示零拷贝技术减少内存分配和数据拷贝
package main

import (
	"bytes"
	"fmt"
	"io"
	"net/http"
	"os"
	"strings"
	"testing"
)

// === 场景 1：文件读取 ===

// 低效：ReadAll 一次性读入内存
func readFileBad(filename string) error {
	f, err := os.Open(filename)
	if err != nil {
		return err
	}
	defer f.Close()

	// 一次性读入内存，大文件会 OOM
	data, err := io.ReadAll(f)
	if err != nil {
		return err
	}
	// 处理 data...
	_ = len(data)
	return nil
}

// 高效：流式处理，使用固定大小 buffer
func readFileGood(filename string) error {
	f, err := os.Open(filename)
	if err != nil {
		return err
	}
	defer f.Close()

	buf := make([]byte, 32*1024) // 32KB 固定 buffer
	for {
		n, err := f.Read(buf)
		if n > 0 {
			// 处理 buf[:n]，不需要一次加载全部内容
			_ = buf[:n]
		}
		if err == io.EOF {
			break
		}
		if err != nil {
			return err
		}
	}
	return nil
}

// === 场景 2：io.Copy vs ReadAll ===

// 低效：ReadAll + Write（两次拷贝）
func copyDataBad(dst io.Writer, src io.Reader) error {
	data, err := io.ReadAll(src) // 第一次拷贝：src -> data
	if err != nil {
		return err
	}
	_, err = dst.Write(data) // 第二次拷贝：data -> dst
	return err
}

// 高效：io.Copy（使用 32KB 缓冲区流式拷贝）
func copyDataGood(dst io.Writer, src io.Reader) error {
	_, err := io.Copy(dst, src) // 内部使用 32KB buffer 流式拷贝
	return err
}

// 更高效：指定缓冲区大小
func copyDataBest(dst io.Writer, src io.Reader) error {
	buf := make([]byte, 64*1024) // 64KB buffer
	_, err := io.CopyBuffer(dst, src, buf)
	return err
}

	return io.ReadAll(resp.Body)

	limitedReader := io.LimitReader(resp.Body, maxSize)
```

```go
const shardCount = 32

type ShardedMap struct {
	shards [shardCount]struct {
		sync.RWMutex
		items map[string]interface{}
	}
}

func NewShardedMap() *ShardedMap {
	sm := &ShardedMap{}
	for i := range sm.shards {
		sm.shards[i].items = make(map[string]interface{})
	}
	return sm
}

// getShard 根据 key 计算分片索引
func (sm *ShardedMap) getShard(key string) uint32 {
	h := fnv.New32a()
	h.Write([]byte(key))
	return h.Sum32() % shardCount
}

func (sm *ShardedMap) Set(key string, value interface{}) {
	idx := sm.getShard(key)
	sm.shards[idx].Lock()
	sm.shards[idx].items[key] = value
	sm.shards[idx].Unlock()
}

func (sm *ShardedMap) Get(key string) (interface{}, bool) {
	idx := sm.getShard(key)
	sm.shards[idx].RLock()
	val, ok := sm.shards[idx].items[key]
	sm.shards[idx].RUnlock()
	return val, ok
}

func (sm *ShardedMap) Delete(key string) {
	idx := sm.getShard(key)
	sm.shards[idx].Lock()
	delete(sm.shards[idx].items, key)
	sm.shards[idx].Unlock()
}

}

func (lm *LockedMap) Get(key string) (interface{}, bool) {
	lm.mu.RLock()
	val, ok := lm.items[key]
	lm.mu.RUnlock()
	return val, ok
}

	}

}
```

```go
	return ac
}

func (ac *AtomicConfig) Load() *Config {
	return ac.value.Load()
}

func (ac *AtomicConfig) Store(cfg *Config) {
	ac.value.Store(cfg)
}

// === Compare-And-Swap 实现无锁栈 ===
type node struct {
	value interface{}
	next  unsafe.Pointer
}

type LockFreeStack struct {
	top unsafe.Pointer
	len atomic.Int64
}

func (s *LockFreeStack) Push(value interface{}) {
	n := &node{value: value}
	for {
		old := atomic.LoadPointer(&s.top)
		n.next = old
		if atomic.CompareAndSwapPointer(&s.top, old, unsafe.Pointer(n)) {
			s.len.Add(1)
			return
		}
		// CAS 失败，重试
	}
}

func (s *LockFreeStack) Pop() (interface{}, bool) {
	for {
		old := atomic.LoadPointer(&s.top)
		if old == nil {
			return nil, false
		}
		n := (*node)(old)
		if atomic.CompareAndSwapPointer(&s.top, old, n.next) {
			s.len.Add(-1)
			return n.value, true
		}
		// CAS 失败，重试
	}
}

func (s *LockFreeStack) Len() int64 {
	return s.len.Load()
}
```

```go
	var b strings.Builder
	b.Grow(256)

	b.WriteString(`{"id":`)
	b.WriteString(strconv.FormatInt(user.ID, 10))

// === 优化 3：使用 json.Decoder 流式解析 ===

// 流式解析大型 JSON 数组（不需要一次性加载到内存）
func streamDecode(data []byte) ([]*User, error) {
	decoder := json.NewDecoder(bytes.NewReader(data))

	// 读取开始的 [
	_, err := decoder.Token()
	if err != nil {
		return nil, err
	}

	var users []*User
	for decoder.More() {
		var user User
		if err := decoder.Decode(&user); err != nil {
			return nil, err
		}
		users = append(users, &user)
	}

	return users, nil
}
```

```go
// 创建优化的 HTTP 客户端
func newOptimizedClient() *http.Client {
	transport := &http.Transport{
		// 连接池配置
		MaxIdleConns:        100,              // 全局最大空闲连接数
		MaxIdleConnsPerHost: 20,               // 每个 host 最大空闲连接数（默认只有 2！）
		MaxConnsPerHost:     50,               // 每个 host 最大连接数
		IdleConnTimeout:     90 * time.Second, // 空闲连接超时

		// TCP 优化
		DialContext: (&net.Dialer{
			Timeout:   10 * time.Second, // 连接超时
			KeepAlive: 30 * time.Second, // TCP Keep-Alive
		}).DialContext,

		// TLS 优化
		TLSClientConfig: &tls.Config{
			InsecureSkipVerify: false,
		},
		TLSHandshakeTimeout: 10 * time.Second,

		// HTTP/2
		ForceAttemptHTTP2: true,

		// 响应头超时
		ResponseHeaderTimeout: 10 * time.Second,

		// 期望 100-continue 超时
		ExpectContinueTimeout: 1 * time.Second,
	}

	return &http.Client{
		Transport: transport,
		Timeout:   30 * time.Second, // 整体请求超时
	}
}

// 错误示例：每次请求都创建新客户端
func fetchBad(url string) ([]byte, error) {
	// 每次创建新客户端 = 每次新建 TCP 连接
	client := &http.Client{Timeout: 10 * time.Second}
	resp, err := client.Get(url)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	return io.ReadAll(resp.Body)
}

// 正确示例：复用全局客户端
var httpClient = newOptimizedClient()

func fetchGood(ctx context.Context, url string) ([]byte, error) {
	req, err := http.NewRequestWithContext(ctx, http.MethodGet, url, nil)
	if err != nil {
		return nil, err
	}

	resp, err := httpClient.Do(req)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()

	// 限制响应体大小
	limitedBody := io.LimitReader(resp.Body, 10*1024*1024) // 最大 10MB
	return io.ReadAll(limitedBody)
}

// 正确示例：必须读取并关闭 Body（即使不需要内容）
func discardResponseBody(resp *http.Response) {
	if resp != nil && resp.Body != nil {
		// 必须读取完 body 才能复用连接
		io.Copy(io.Discard, resp.Body)
		resp.Body.Close()
	}
}

// === HTTP 服务端优化 ===

func newOptimizedServer(handler http.Handler) *http.Server {
	return &http.Server{
		Addr:    ":8080",
		Handler: handler,

		// 超时配置（防止 slowloris 攻击）
		ReadTimeout:       15 * time.Second,  // 读取请求头和请求体的超时
		ReadHeaderTimeout: 5 * time.Second,   // 读取请求头的超时
		WriteTimeout:      15 * time.Second,  // 写响应的超时
		IdleTimeout:       60 * time.Second,  // Keep-Alive 空闲连接超时

		// 限制请求大小
		MaxHeaderBytes: 1 << 20, // 1MB
	}
}

// 中间件：限制请求体大小
func maxBytesMiddleware(next http.Handler, maxBytes int64) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		r.Body = http.MaxBytesReader(w, r.Body, maxBytes)
		next.ServeHTTP(w, r)
	})
}

	fmt.Println("3. 设置合理的超时（连接/读/写/整体）")
	fmt.Println("4. 使用 context 控制超时和取消")
	fmt.Println("5. 必须读取并关闭 Response.Body（否则连接不会复用）")
	fmt.Println("6. 使用 io.LimitReader 限制响应大小")
	fmt.Println("7. 开启 HTTP/2（ForceAttemptHTTP2: true）")
	fmt.Println()

	fmt.Println("--- 服务端优化 ---")
	fmt.Println("1. 配置合理的超时（ReadTimeout/WriteTimeout/IdleTimeout）")
	fmt.Println("2. 限制请求体大小（http.MaxBytesReader）")
	fmt.Println("3. 使用 gzip 压缩响应")
	fmt.Println("4. 启用 Keep-Alive（默认开启）")
	fmt.Println("5. 使用连接限制中间件防止过载")
	fmt.Println()

	fmt.Println("--- 常见坑点 ---")
	fmt.Println("1. http.Client 默认没有超时（Timeout: 0 = 无限等待）")
	fmt.Println("2. DefaultTransport 的 MaxIdleConnsPerHost 只有 2")
	fmt.Println("3. 不关闭 Response.Body 会导致连接泄漏")
	fmt.Println("4. 不读取 Response.Body 就关闭，连接不能复用")
```

## 十、常见坑点

```
坑点 1：strings.Builder 未调用 Grow
  Builder 内部使用 []byte，不 Grow 则多次扩容。
  如果能预估大小，一定要先 Grow。

坑点 2：切片 append 不预分配
  每次扩容复制数据+分配新内存，O(n) 次扩容。
  知道大小就 make([]T, 0, n)。

坑点 3：for range 大结构体
  range 会拷贝元素，大结构体性能差。
  用下标 for i := range 或指针切片。

坑点 4：sync.Pool 中对象未重置
  归还 Pool 前必须 Reset/清零。
  否则下次取出的对象包含脏数据。

坑点 5：http.Client 每次请求新建
  每次新建 = 每次新建 TCP 连接。
  应该全局复用 http.Client。

坑点 6：Response.Body 未正确处理
  不读取 Body -> 连接不可复用。
  不关闭 Body -> 连接泄漏。
  正确: defer resp.Body.Close() 且读取完。

坑点 7：map 删除不缩容
  频繁 put/delete 的 map 内存只增不减。
  定期重建 map 或使用其他数据结构。

坑点 8：PGO profile 来源不对
  PGO profile 必须来自真实生产负载。
  来自 benchmark 或非典型场景的 profile 可能适得其反。
```
