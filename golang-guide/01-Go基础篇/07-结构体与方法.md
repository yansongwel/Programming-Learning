# 07 - Go 结构体与方法

> **适用读者**：有 Python / Shell 经验的 SRE / DevOps 工程师

## 导读

Go 没有 class，用 struct（结构体）+ method（方法）来实现面向对象。Go 的设计哲学是"组合优于继承"——没有继承，只有嵌入（embedding）。理解值接收者和指针接收者的区别是写出正确 Go 代码的关键。本篇还会介绍 struct tag、构造函数模式、函数选项模式等 Go 社区广泛使用的最佳实践。

## 1. struct 定义与初始化

### 1.1 基本定义

```go
package main

import (
    "fmt"
    "time"
)

// Server 表示一个服务器实例
// 字段大写开头：导出（公开）
// 字段小写开头：未导出（私有）
type Server struct {
    Name      string
    Host      string
    Port      int
    IsHealthy bool
    StartTime time.Time
    tags      map[string]string // 未导出字段，包外不可访问
}

// 空结构体（大小为 0，不占内存）
type Empty struct{}

// 嵌套结构体
type Cluster struct {
    Name    string
    Region  string
    Servers []Server
}

func main() {
    // 空结构体的用途
    // 1. 作为 set 的值类型（map[string]struct{}）
    // 2. 作为 channel 的信号（chan struct{}）
    set := map[string]struct{}{
        "web-01": {},
        "web-02": {},
    }
    fmt.Println("set 大小:", len(set))

    // 空结构体不占内存
    fmt.Printf("Empty 大小: %d 字节\n", unsafe_Sizeof_simulation())
}

func unsafe_Sizeof_simulation() int {
    return 0 // struct{} 的大小确实是 0
}
```

### 1.2 多种初始化方式

```go
package main

import (
    "fmt"
    "time"
)

type Server struct {
    Name      string
    Host      string
    Port      int
    IsHealthy bool
    StartTime time.Time
}

func main() {
    // ============ 方式一：字段名初始化（推荐） ============
    s1 := Server{
        Name:      "web-01",
        Host:      "192.168.1.10",
        Port:      8080,
        IsHealthy: true,
        StartTime: time.Now(),
    }
    fmt.Printf("方式一: %+v\n", s1)

    // 可以只初始化部分字段（其余为零值）
    s2 := Server{
        Name: "web-02",
        Port: 8081,
    }
    fmt.Printf("部分初始化: %+v\n", s2)

    // ============ 方式二：顺序初始化（不推荐） ============
    // 必须提供所有字段，顺序必须与定义一致
    // 添加新字段会导致编译错误
    s3 := Server{"web-03", "192.168.1.12", 8082, true, time.Now()}
    fmt.Printf("方式二: %+v\n", s3)

    // ============ 方式三：零值初始化 ============
    var s4 Server
    fmt.Printf("零值: %+v\n", s4)
    // {Name: Host: Port:0 IsHealthy:false StartTime:0001-01-01 00:00:00 +0000 UTC}

    // ============ 方式四：new 函数（返回指针） ============
    s5 := new(Server) // 等价于 &Server{}
    s5.Name = "web-05"
    s5.Port = 8085
    fmt.Printf("new: %+v\n", *s5)

    // ============ 方式五：取地址（最常用） ============
    s6 := &Server{
        Name: "web-06",
        Host: "192.168.1.16",
        Port: 8086,
    }
    fmt.Printf("取地址: %+v (类型: %T)\n", *s6, s6) // *main.Server

    // ============ 结构体是值类型 ============
    original := Server{Name: "original", Port: 8080}
    copied := original   // 完整复制
    copied.Name = "copy"
    fmt.Printf("\n原始: %s, 副本: %s\n", original.Name, copied.Name)
    // 互不影响

    // 如果需要共享，使用指针
    ptr := &original
    ptr.Name = "modified"
    fmt.Printf("通过指针修改: %s\n", original.Name) // "modified"
}
```

### 1.3 匿名结构体

```go
package main

import (
    "encoding/json"
    "fmt"
)

func main() {
    // 匿名结构体：不需要定义类型名
    // 适合一次性使用的场景

    // 场景一：临时数据结构
    config := struct {
        Host string
        Port int
    }{
        Host: "localhost",
        Port: 8080,
    }
    fmt.Printf("配置: %+v\n", config)

    // 场景二：JSON 解析（不想定义完整结构体）
    jsonData := `{"name": "web-01", "status": "running", "cpu": 85.5}`
    var result struct {
        Name   string  `json:"name"`
        Status string  `json:"status"`
        CPU    float64 `json:"cpu"`
    }
    json.Unmarshal([]byte(jsonData), &result)
    fmt.Printf("JSON: %+v\n", result)

    // 场景三：测试用例
    tests := []struct {
        name     string
        input    int
        expected int
    }{
        {"正数", 5, 25},
        {"零", 0, 0},
        {"负数", -3, 9},
    }
    for _, tc := range tests {
        result := tc.input * tc.input
        status := "PASS"
        if result != tc.expected {
            status = "FAIL"
        }
        fmt.Printf("  %s: %s (input=%d, got=%d, want=%d)\n",
            status, tc.name, tc.input, result, tc.expected)
    }
}
```

## 2. 匿名字段 / 嵌入结构体

### 2.1 基本嵌入

```go
package main

import "fmt"

// 基础类型
type Address struct {
    Street  string
    City    string
    Country string
}

type ContactInfo struct {
    Email string
    Phone string
}

// 嵌入（组合）：Go 实现"继承"的方式
type Employee struct {
    Name string
    Age  int
    Address              // 匿名字段：嵌入 Address
    ContactInfo          // 匿名字段：嵌入 ContactInfo
    DepartmentID int
}

func main() {
    emp := Employee{
        Name: "张三",
        Age:  30,
        Address: Address{
            Street:  "科技路100号",
            City:    "北京",
            Country: "中国",
        },
        ContactInfo: ContactInfo{
            Email: "zhangsan@example.com",
            Phone: "138xxxx1234",
        },
        DepartmentID: 1,
    }

    // 可以直接访问嵌入结构体的字段（提升）
    fmt.Println("城市:", emp.City)        // 等同于 emp.Address.City
    fmt.Println("邮箱:", emp.Email)       // 等同于 emp.ContactInfo.Email

    // 也可以通过嵌入类型名访问
    fmt.Println("地址:", emp.Address)

    // 完整打印
    fmt.Printf("员工: %+v\n", emp)
}
```

### 2.2 组合优于继承

```go
package main

import (
    "fmt"
    "time"
)

// ============ SRE 实战：服务基础信息 + 特化 ============

// BaseService 基础服务信息
type BaseService struct {
    Name      string
    Version   string
    StartTime time.Time
}

func (b BaseService) Uptime() time.Duration {
    return time.Since(b.StartTime)
}

func (b BaseService) Info() string {
    return fmt.Sprintf("%s (v%s)", b.Name, b.Version)
}

// HTTPService 嵌入 BaseService + HTTP 特有字段
type HTTPService struct {
    BaseService          // 嵌入基础服务
    Host        string
    Port        int
    TLS         bool
}

func (h HTTPService) URL() string {
    scheme := "http"
    if h.TLS {
        scheme = "https"
    }
    return fmt.Sprintf("%s://%s:%d", scheme, h.Host, h.Port)
}

// GRPCService 嵌入 BaseService + gRPC 特有字段
type GRPCService struct {
    BaseService
    Host       string
    Port       int
    Reflection bool
}

func (g GRPCService) Endpoint() string {
    return fmt.Sprintf("%s:%d", g.Host, g.Port)
}

// 注意：嵌入不是继承！
// - HTTPService 不是 BaseService 的子类型
// - 但 HTTPService 拥有 BaseService 的所有字段和方法

func main() {
    web := HTTPService{
        BaseService: BaseService{
            Name:      "web-api",
            Version:   "2.1.0",
            StartTime: time.Now().Add(-24 * time.Hour),
        },
        Host: "0.0.0.0",
        Port: 8080,
        TLS:  true,
    }

    grpc := GRPCService{
        BaseService: BaseService{
            Name:      "user-service",
            Version:   "1.5.0",
            StartTime: time.Now().Add(-48 * time.Hour),
        },
        Host:       "0.0.0.0",
        Port:       9090,
        Reflection: true,
    }

    // 可以直接调用 BaseService 的方法
    fmt.Println("=== HTTP 服务 ===")
    fmt.Println("信息:", web.Info())      // BaseService.Info()
    fmt.Println("运行时间:", web.Uptime())  // BaseService.Uptime()
    fmt.Println("URL:", web.URL())         // HTTPService.URL()

    fmt.Println("\n=== gRPC 服务 ===")
    fmt.Println("信息:", grpc.Info())
    fmt.Println("运行时间:", grpc.Uptime())
    fmt.Println("端点:", grpc.Endpoint())
}
```

### 2.3 字段冲突

```go
package main

import "fmt"

type A struct {
    Name string
}

type B struct {
    Name string
}

type C struct {
    A
    B
    Name string // C 自己也有 Name
}

func main() {
    c := C{
        A:    A{Name: "A的Name"},
        B:    B{Name: "B的Name"},
        Name: "C的Name",
    }

    // C 自己的 Name 优先
    fmt.Println("c.Name:", c.Name)      // "C的Name"

    // 访问嵌入类型的 Name 需要显式指定
    fmt.Println("c.A.Name:", c.A.Name)  // "A的Name"
    fmt.Println("c.B.Name:", c.B.Name)  // "B的Name"

    // 如果 C 没有 Name 字段，且 A 和 B 都有 Name：
    // c.Name 会编译错误（ambiguous selector）
    // 必须显式指定 c.A.Name 或 c.B.Name
}
```

## 3. struct tag

### 3.1 JSON tag

```go
package main

import (
    "encoding/json"
    "fmt"
)

// ServerConfig 服务器配置
type ServerConfig struct {
    // json tag 控制 JSON 序列化/反序列化的字段名
    Name        string `json:"name"`
    Host        string `json:"host"`
    Port        int    `json:"port"`
    Debug       bool   `json:"debug,omitempty"`       // omitempty: 零值时省略
    MaxConns    int    `json:"max_connections"`
    Secret      string `json:"-"`                      // -: 完全忽略此字段
    Description string `json:"description,omitempty"`
}

func main() {
    // 序列化
    cfg := ServerConfig{
        Name:     "web-01",
        Host:     "0.0.0.0",
        Port:     8080,
        Debug:    false,           // omitempty: 不会出现在 JSON 中
        MaxConns: 100,
        Secret:   "s3cret",       // -: 不会出现在 JSON 中
    }

    data, _ := json.MarshalIndent(cfg, "", "  ")
    fmt.Println("=== 序列化 ===")
    fmt.Println(string(data))

    // 反序列化
    jsonStr := `{
        "name": "api-01",
        "host": "10.0.0.1",
        "port": 9090,
        "max_connections": 200,
        "extra_field": "会被忽略"
    }`

    var cfg2 ServerConfig
    json.Unmarshal([]byte(jsonStr), &cfg2)
    fmt.Println("\n=== 反序列化 ===")
    fmt.Printf("%+v\n", cfg2)
}
```

### 3.2 多种 tag

```go
package main

import (
    "encoding/json"
    "fmt"
    "reflect"
)

// User 演示多种 tag 的使用
type User struct {
    ID        int    `json:"id"         db:"id"         yaml:"id"         validate:"required"`
    Username  string `json:"username"   db:"username"    yaml:"username"   validate:"required,min=3,max=50"`
    Email     string `json:"email"      db:"email"       yaml:"email"      validate:"required,email"`
    Password  string `json:"-"          db:"password"    yaml:"-"`
    Age       int    `json:"age"        db:"age"         yaml:"age"        validate:"min=0,max=150"`
    IsActive  bool   `json:"is_active"  db:"is_active"   yaml:"is_active"`
    CreatedAt string `json:"created_at" db:"created_at"  yaml:"created_at"`
}

// AlertRule 告警规则配置
type AlertRule struct {
    Name       string   `json:"name"       yaml:"name"`
    Metric     string   `json:"metric"     yaml:"metric"`
    Threshold  float64  `json:"threshold"  yaml:"threshold"`
    Duration   string   `json:"duration"   yaml:"duration"`
    Severity   string   `json:"severity"   yaml:"severity"`
    Recipients []string `json:"recipients" yaml:"recipients"`
}

func main() {
    user := User{
        ID:       1,
        Username: "devops",
        Email:    "devops@example.com",
        Password: "secret123",
        Age:      30,
        IsActive: true,
    }

    // JSON 序列化
    data, _ := json.MarshalIndent(user, "", "  ")
    fmt.Println("JSON:")
    fmt.Println(string(data))

    // 通过反射读取 tag
    fmt.Println("\n=== Tag 信息 ===")
    t := reflect.TypeOf(user)
    for i := 0; i < t.NumField(); i++ {
        field := t.Field(i)
        fmt.Printf("  字段: %-10s json:%-12s db:%-12s validate:%s\n",
            field.Name,
            field.Tag.Get("json"),
            field.Tag.Get("db"),
            field.Tag.Get("validate"),
        )
    }

    // 告警规则
    rule := AlertRule{
        Name:       "高CPU告警",
        Metric:     "cpu_usage_percent",
        Threshold:  90.0,
        Duration:   "5m",
        Severity:   "critical",
        Recipients: []string{"oncall@example.com", "team-sre@example.com"},
    }

    ruleJSON, _ := json.MarshalIndent(rule, "", "  ")
    fmt.Println("\n告警规则 JSON:")
    fmt.Println(string(ruleJSON))
}
```

## 4. 方法定义

### 4.1 值接收者 vs 指针接收者

```go
package main

import "fmt"

type Rectangle struct {
    Width  float64
    Height float64
}

// 值接收者：方法操作的是接收者的副本
// 不会修改原始结构体
func (r Rectangle) Area() float64 {
    return r.Width * r.Height
}

func (r Rectangle) Perimeter() float64 {
    return 2 * (r.Width + r.Height)
}

func (r Rectangle) String() string {
    return fmt.Sprintf("Rectangle(%.1f x %.1f)", r.Width, r.Height)
}

// 指针接收者：方法操作的是接收者的指针
// 可以修改原始结构体
func (r *Rectangle) Scale(factor float64) {
    r.Width *= factor  // 修改原始结构体
    r.Height *= factor
}

func (r *Rectangle) SetWidth(w float64) {
    r.Width = w
}

func main() {
    rect := Rectangle{Width: 10, Height: 5}

    // 值接收者方法：值和指针都可以调用
    fmt.Println("面积:", rect.Area())
    fmt.Println("周长:", rect.Perimeter())

    // 指针接收者方法：值和指针也都可以调用
    // Go 自动取地址或解引用
    rect.Scale(2.0) // Go 自动转为 (&rect).Scale(2.0)
    fmt.Println("放大后:", rect) // Rectangle(20.0 x 10.0)

    // 通过指针调用
    p := &rect
    fmt.Println("面积:", p.Area()) // Go 自动转为 (*p).Area()
    p.Scale(0.5)
    fmt.Println("缩小后:", rect)
}
```

### 4.2 何时使用指针接收者

```go
package main

import (
    "fmt"
    "sync"
    "time"
)

// ============ 必须用指针接收者的场景 ============

// 场景一：需要修改接收者的状态
type Counter struct {
    mu    sync.Mutex
    count int
}

func (c *Counter) Increment() {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.count++
}

func (c *Counter) Value() int {
    c.mu.Lock()
    defer c.mu.Unlock()
    return c.count
}

// 场景二：结构体很大，避免复制
type LargeStruct struct {
    Data [1024 * 1024]byte // 1MB 的数组
    Name string
}

func (l *LargeStruct) Process() {
    // 如果用值接收者，每次调用都会复制 1MB
    fmt.Println("处理:", l.Name)
}

// 场景三：包含不可复制的字段（sync.Mutex, sync.WaitGroup 等）
type SafeMap struct {
    mu   sync.Mutex       // Mutex 不能复制！
    data map[string]string
}

func (m *SafeMap) Set(key, value string) {
    m.mu.Lock()
    defer m.mu.Unlock()
    m.data[key] = value
}

// ============ 可以用值接收者的场景 ============

// 场景一：小的不可变结构体
type Point struct {
    X, Y float64
}

func (p Point) Distance(other Point) float64 {
    dx := p.X - other.X
    dy := p.Y - other.Y
    return dx*dx + dy*dy // 简化，未开方
}

// 场景二：只读方法
type ServerInfo struct {
    Name    string
    Version string
}

func (s ServerInfo) String() string {
    return fmt.Sprintf("%s v%s", s.Name, s.Version)
}

// ============ 最佳实践 ============
// 如果一个类型的任何方法使用了指针接收者，那么所有方法都应该用指针接收者
// 保持一致性！

type Service struct {
    Name      string
    Port      int
    isRunning bool
    startTime time.Time
}

func (s *Service) Start() {
    s.isRunning = true
    s.startTime = time.Now()
    fmt.Printf("服务 %s 启动在端口 %d\n", s.Name, s.Port)
}

func (s *Service) Stop() {
    s.isRunning = false
    fmt.Printf("服务 %s 已停止\n", s.Name)
}

// 即使这个方法不修改状态，也用指针接收者保持一致
func (s *Service) IsRunning() bool {
    return s.isRunning
}

func (s *Service) Uptime() time.Duration {
    if !s.isRunning {
        return 0
    }
    return time.Since(s.startTime)
}

func main() {
    // Counter 示例
    c := &Counter{}
    c.Increment()
    c.Increment()
    c.Increment()
    fmt.Println("计数:", c.Value())

    // Service 示例
    svc := &Service{Name: "web-api", Port: 8080}
    svc.Start()
    fmt.Println("运行中:", svc.IsRunning())
    time.Sleep(100 * time.Millisecond)
    fmt.Println("运行时间:", svc.Uptime())
    svc.Stop()
}
```

## 5. 方法集与接口实现的关系

```go
package main

import "fmt"

// Stringer 接口
type Stringer interface {
    String() string
}

// Starter 接口
type Starter interface {
    Start() error
}

type MyService struct {
    Name string
}

// String 是值接收者方法
func (s MyService) String() string {
    return fmt.Sprintf("Service(%s)", s.Name)
}

// Start 是指针接收者方法
func (s *MyService) Start() error {
    fmt.Printf("启动 %s\n", s.Name)
    return nil
}

func main() {
    svc := MyService{Name: "web"}

    // ============ 方法集规则 ============
    // 值类型 T 的方法集：所有值接收者方法
    // 指针类型 *T 的方法集：所有值接收者方法 + 所有指针接收者方法

    // 值类型可以赋值给值接收者方法的接口
    var str Stringer = svc  // OK：MyService 有 String() 值接收者方法
    fmt.Println(str)

    // 值类型不能赋值给指针接收者方法的接口
    // var st Starter = svc // 编译错误：MyService 没有实现 Starter
    //                       // 因为 Start() 是指针接收者

    // 指针类型可以赋值给任何接口
    var st Starter = &svc  // OK：*MyService 有 Start() 方法
    st.Start()

    var str2 Stringer = &svc // OK：*MyService 也有 String() 方法（继承自值接收者）
    fmt.Println(str2)

    // ============ 实际影响 ============
    // 这就是为什么建议一个类型的所有方法都用指针接收者
    // 这样无论值还是指针都能满足接口

    fmt.Println("\n=== 规则总结 ===")
    fmt.Println("值类型 T 的方法集:   值接收者方法")
    fmt.Println("指针类型 *T 的方法集: 值接收者方法 + 指针接收者方法")
    fmt.Println("实现接口时注意: 如果方法是指针接收者,")
    fmt.Println("  那么只有 *T 实现了接口, T 没有实现")
}
```

## 6. 构造函数模式（NewXxx）

```go
package main

import (
    "fmt"
    "time"
)

// ============ 简单构造函数 ============

type Server struct {
    name      string
    host      string
    port      int
    startTime time.Time
    isRunning bool
}

// NewServer 创建服务器实例
// Go 的惯例：构造函数命名为 NewXxx
func NewServer(name, host string, port int) *Server {
    return &Server{
        name: name,
        host: host,
        port: port,
    }
}

func (s *Server) Start() {
    s.isRunning = true
    s.startTime = time.Now()
    fmt.Printf("服务器 %s 启动在 %s:%d\n", s.name, s.host, s.port)
}

func (s *Server) String() string {
    status := "stopped"
    if s.isRunning {
        status = "running"
    }
    return fmt.Sprintf("%s (%s:%d) [%s]", s.name, s.host, s.port, status)
}

// ============ 带验证的构造函数 ============

type DatabaseConfig struct {
    Host     string
    Port     int
    User     string
    Password string
    DBName   string
    MaxConns int
}

// NewDatabaseConfig 创建数据库配置（带验证）
func NewDatabaseConfig(host string, port int, user, password, dbName string) (*DatabaseConfig, error) {
    if host == "" {
        return nil, fmt.Errorf("host 不能为空")
    }
    if port < 1 || port > 65535 {
        return nil, fmt.Errorf("无效的端口号: %d", port)
    }
    if user == "" {
        return nil, fmt.Errorf("user 不能为空")
    }
    if dbName == "" {
        return nil, fmt.Errorf("数据库名不能为空")
    }

    return &DatabaseConfig{
        Host:     host,
        Port:     port,
        User:     user,
```

## 7. 函数选项模式（Functional Options）

```go
package main

import (
    "fmt"
    "time"

// ============ 问题：参数太多怎么办？ ============
// 不好的方式：
// func NewServer(name, host string, port int, tls bool, timeout time.Duration,
//     maxConns int, debug bool, logger Logger, ...) *Server
//
// 不好的方式二：Config 结构体
// 虽然可以，但无法区分"用户没设置"和"用户设置为零值"

// ============ 函数选项模式（Go 社区最佳实践） ============

type HTTPServer struct {
    host         string
    port         int
    readTimeout  time.Duration
    writeTimeout time.Duration
    maxConns     int
    tls          bool
    certFile     string
    keyFile      string
    debug        bool
}

// Option 是一个函数类型，用于配置 HTTPServer
type Option func(*HTTPServer)

// WithPort 设置端口
func WithPort(port int) Option {
    return func(s *HTTPServer) {
        s.port = port
    }
}

// WithTimeout 设置读写超时
func WithTimeout(read, write time.Duration) Option {
    return func(s *HTTPServer) {
        s.readTimeout = read
        s.writeTimeout = write
    }
}

// WithMaxConns 设置最大连接数
func WithMaxConns(n int) Option {
    return func(s *HTTPServer) {
        s.maxConns = n
    }
}

// WithTLS 启用 TLS
func WithTLS(certFile, keyFile string) Option {
    return func(s *HTTPServer) {
        s.tls = true
        s.certFile = certFile
        s.keyFile = keyFile
    }
}

// WithDebug 启用调试模式
func WithDebug(debug bool) Option {
    return func(s *HTTPServer) {
        s.debug = debug
    }
}

// NewHTTPServer 使用函数选项模式创建服务器
func NewHTTPServer(host string, opts ...Option) *HTTPServer {
    // 设置默认值
    s := &HTTPServer{
        host:         host,
        port:         8080,
        readTimeout:  30 * time.Second,
        writeTimeout: 30 * time.Second,
        maxConns:     1000,
        tls:          false,
        debug:        false,
    }

    // 应用选项
    for _, opt := range opts {
        opt(s)
    }

    return s
}

    fmt.Printf("  最大连接: %d, TLS: %v, Debug: %v\n", s.maxConns, s.tls, s.debug)
}
```

## 8. 结构体比较与拷贝

### 8.1 结构体比较

```go
package main

import "fmt"

// 可比较的结构体（所有字段都是可比较的类型）
type Point struct {
    X, Y int
}

// 不可比较的结构体（包含 slice、map 或 func）
type Config struct {
    Name  string
    Tags  []string          // slice 不可比较
    Extra map[string]string // map 不可比较
}

func main() {
    // ============ 可比较的结构体 ============
    p1 := Point{1, 2}
    p2 := Point{1, 2}
    p3 := Point{3, 4}

    fmt.Println("p1 == p2:", p1 == p2) // true
    fmt.Println("p1 == p3:", p1 == p3) // false

    // 可以作为 map 的 key
    distances := map[Point]float64{
        {0, 0}: 0,
        {1, 1}: 1.414,
        {3, 4}: 5.0,
    }
    fmt.Println("距离:", distances[Point{3, 4}])

    // ============ 不可比较的结构体 ============
    c1 := Config{Name: "web", Tags: []string{"prod"}}
    c2 := Config{Name: "web", Tags: []string{"prod"}}
    _ = c1
    _ = c2
    // fmt.Println(c1 == c2) // 编译错误：invalid operation

    // 使用 reflect.DeepEqual 比较
    // fmt.Println(reflect.DeepEqual(c1, c2)) // true
    // 但 reflect.DeepEqual 性能差，生产中建议自定义 Equal 方法

    // ============ 自定义 Equal 方法 ============
    type Server struct {
        Name string
        Port int
    }

    s1 := Server{"web", 8080}
    s2 := Server{"web", 8080}
    fmt.Println("\nServer 相等:", s1 == s2)
}
```

### 8.2 结构体拷贝

```go
package main

import (
    "fmt"
)

type ServerConfig struct {
    Name     string
    Host     string
    Port     int
    Tags     []string          // 切片：浅拷贝！
    Metadata map[string]string // map：浅拷贝！
}

// DeepCopy 深拷贝方法
func (c ServerConfig) DeepCopy() ServerConfig {
    // 值类型字段自动复制，只需处理引用类型
    newConfig := c

    // 深拷贝 slice
    if c.Tags != nil {
        newConfig.Tags = make([]string, len(c.Tags))
        copy(newConfig.Tags, c.Tags)
    }

    // 深拷贝 map
    if c.Metadata != nil {
        newConfig.Metadata = make(map[string]string, len(c.Metadata))
        for k, v := range c.Metadata {
            newConfig.Metadata[k] = v
        }
    }

    return newConfig
}

func main() {
    original := ServerConfig{
        Name: "web-01",
        Host: "192.168.1.10",
        Port: 8080,
        Tags: []string{"production", "frontend"},
        Metadata: map[string]string{
            "region": "us-east",
            "zone":   "a",
        },
    }

    // ============ 浅拷贝（直接赋值） ============
    shallow := original

    // 修改值类型字段：互不影响
    shallow.Name = "web-02"
    fmt.Println("原始 Name:", original.Name)   // "web-01"
    fmt.Println("浅拷贝 Name:", shallow.Name)   // "web-02"

    // 修改引用类型字段：互相影响！
    shallow.Tags[0] = "staging"
    fmt.Println("原始 Tags:", original.Tags)     // ["staging" "frontend"] 被影响了！
    fmt.Println("浅拷贝 Tags:", shallow.Tags)     // ["staging" "frontend"]

    shallow.Metadata["region"] = "eu-west"
    fmt.Println("原始 Metadata:", original.Metadata) // region 也被改了！

    // ============ 深拷贝 ============
    fmt.Println("\n=== 深拷贝 ===")
    original2 := ServerConfig{
        Name:     "api-01",
        Host:     "10.0.0.1",
        Port:     9090,
        Tags:     []string{"production", "backend"},
        Metadata: map[string]string{"team": "platform"},
    }

    deep := original2.DeepCopy()
    deep.Tags[0] = "development"
    deep.Metadata["team"] = "devops"

    fmt.Println("原始 Tags:", original2.Tags)         // ["production" "backend"] 不受影响
    fmt.Println("深拷贝 Tags:", deep.Tags)             // ["development" "backend"]
    fmt.Println("原始 Metadata:", original2.Metadata)   // {"team": "platform"} 不受影响
    fmt.Println("深拷贝 Metadata:", deep.Metadata)       // {"team": "devops"}
}
```

```go
    return ok && v == value
}

    ready := 0
    for _, c := range p.Containers {
        if c.Ready {
            ready++

        p.Namespace+"/"+p.Name,

}
```

## 常见坑点

### 坑点 1：值接收者方法不能修改原始数据

```go
package main

import "fmt"

type Config struct {
    Debug bool
}

// 值接收者：修改的是副本！
func (c Config) EnableDebugBad() {
    c.Debug = true // 修改的是副本，原始 Config 不变
}

// 指针接收者：修改原始数据
func (c *Config) EnableDebugGood() {
    c.Debug = true
}

func main() {
    cfg := Config{}

    cfg.EnableDebugBad()
    fmt.Println("值接收者:", cfg.Debug) // false！没生效

    cfg.EnableDebugGood()
    fmt.Println("指针接收者:", cfg.Debug) // true
}
```

### 坑点 2：嵌入结构体的方法提升不是继承

```go
package main

import "fmt"

type Base struct {
    ID int
}

func (b Base) Describe() string {
    return fmt.Sprintf("Base(ID=%d)", b.ID)
}

type Derived struct {
    Base
    Name string
}

// Derived 有自己的 Describe 方法
func (d Derived) Describe() string {
    // 可以调用 Base 的方法
    return fmt.Sprintf("Derived(Name=%s, %s)", d.Name, d.Base.Describe())
}

func main() {
    d := Derived{Base: Base{ID: 1}, Name: "test"}

    // 调用的是 Derived 的 Describe
    fmt.Println(d.Describe())

    // 可以显式调用 Base 的
    fmt.Println(d.Base.Describe())

    // 注意：这不是多态！Base 的方法中的 b 永远是 Base 类型
    // Go 没有虚方法表
}
```

### 坑点 3：结构体包含引用类型时的浅拷贝问题

```go
package main

import "fmt"

type Service struct {
    Name string
    Tags []string
}

func main() {
    s1 := Service{Name: "web", Tags: []string{"prod", "v2"}}
    s2 := s1 // 浅拷贝！Tags 切片共享底层数组

    s2.Tags[0] = "staging"
    fmt.Println("s1.Tags:", s1.Tags) // [staging v2] 被影响了！
    fmt.Println("s2.Tags:", s2.Tags) // [staging v2]

    // 解决：手动深拷贝
    s3 := Service{Name: s1.Name}
    s3.Tags = make([]string, len(s1.Tags))
    copy(s3.Tags, s1.Tags)
}
```

### 坑点 4：nil 指针接收者调用方法

```go
package main

import "fmt"

type MyStruct struct {
    Name string
}

// 指针接收者方法可以被 nil 指针调用
// 但如果方法内访问了字段，会 panic
func (m *MyStruct) SafeMethod() string {
    if m == nil {
        return "<nil>"
    }
    return m.Name
}

func main() {
    var p *MyStruct // nil 指针

    // 不会 panic！因为方法内做了 nil 检查
    fmt.Println(p.SafeMethod()) // "<nil>"

    // 但如果直接访问字段会 panic
    // fmt.Println(p.Name) // panic: nil pointer dereference
}
```

## 速查表

### 结构体操作速查

| 操作 | 代码 |
|------|------|
| 定义 | `type Name struct { Field Type }` |
| 字面量初始化 | `s := Name{Field: value}` |
| 取地址 | `p := &Name{Field: value}` |
| new | `p := new(Name)` |
| 零值 | `var s Name` |
| 访问字段 | `s.Field` |
| 嵌入 | `type B struct { A }` |
| 值方法 | `func (s T) Method()` |
| 指针方法 | `func (s *T) Method()` |
| 构造函数 | `func NewT(...) *T` |

### 接收者选择指南

| 场景 | 接收者 | 原因 |
|------|--------|------|
| 需要修改状态 | `*T` | 值接收者只修改副本 |
| 结构体很大 | `*T` | 避免复制 |
| 包含 Mutex/WaitGroup | `*T` | 这些类型不能复制 |
| 只读小结构体 | `T` | 更安全，无副作用 |
| 一致性 | `*T` | 如果任何方法用 `*T`，全部用 `*T` |

### 与 Python 对比

| 概念 | Go | Python |
|------|-----|--------|
| 类定义 | `type T struct {}` | `class T:` |
| 构造函数 | `func NewT() *T` | `def __init__(self):` |
| 方法 | `func (t *T) M()` | `def m(self):` |
| 继承 | 嵌入（组合） | `class B(A):` |
| 私有 | 小写开头 | `_` 前缀（约定） |
| 公开 | 大写开头 | 无前缀 |
| 序列化 tag | `json:"name"` | 无（或 dataclass） |
| 多态 | 接口 | 鸭子类型 |

## 小结

1. **Go 用组合替代继承**：嵌入结构体获得字段和方法提升，但这不是面向对象的继承。
2. **值接收者 vs 指针接收者**：需要修改状态或结构体很大时用指针接收者。保持一致性——一个类型的所有方法用同一种接收者。
3. **struct tag 是序列化的核心**：`json`、`yaml`、`db` 等 tag 控制字段的映射关系。
4. **NewXxx 是 Go 的构造函数惯例**：返回 `*T`，可以包含验证逻辑。
5. **函数选项模式（Functional Options）**是 Go 社区处理多参数的最佳实践：类型安全、可扩展、有默认值。
6. **浅拷贝陷阱**：结构体直接赋值是浅拷贝，包含 slice/map 的字段会共享底层数据。需要深拷贝时手动处理。
7. **方法集决定接口实现**：值类型只有值接收者方法集，指针类型有全部方法集。
