# 02 - Go 基础语法

> **适用读者**：有 Python / Shell 经验的 SRE / DevOps 工程师
> **Go 版本**：Go 1.24

## 导读

Go 的语法设计哲学是"少即是多"——关键字只有 25 个（Python 有 35 个），没有三元运算符、没有类继承、没有异常机制。这种简洁并非功能缺失，而是刻意为之，目的是让代码一目了然、减少团队中的风格分歧。

本篇覆盖 Go 的基本语法元素：变量、常量、数据类型、运算符和格式化输出。如果你熟悉 Python，会发现 Go 在很多地方"更严格但更安全"。

## 1. 变量声明

### 1.1 var 关键字声明

```go
package main

import "fmt"

func main() {
    // ============ 显式类型声明 ============
    var name string = "DevOps"
    var age int = 30
    var salary float64 = 15000.50
    var isActive bool = true

    fmt.Println(name, age, salary, isActive)

    // ============ 类型推断（省略类型） ============
    // 编译器根据右侧值自动推断类型
    var city = "Beijing"    // 推断为 string
    var count = 42          // 推断为 int
    var rate = 3.14         // 推断为 float64

    fmt.Println(city, count, rate)

    // ============ 只声明不赋值（零值初始化） ============
    var s string    // 零值: ""
    var n int       // 零值: 0
    var f float64   // 零值: 0.0
    var b bool      // 零值: false
    var p *int      // 零值: nil

    fmt.Printf("string零值: %q\n", s)   // ""
    fmt.Printf("int零值: %d\n", n)       // 0
    fmt.Printf("float64零值: %f\n", f)   // 0.000000
    fmt.Printf("bool零值: %t\n", b)      // false
    fmt.Printf("pointer零值: %v\n", p)   // <nil>
}
```

### 1.2 短变量声明 :=（最常用）

```go
package main

import "fmt"

func main() {
    // := 只能在函数内部使用
    name := "DevOps"           // 等价于 var name string = "DevOps"
    age := 30                  // 等价于 var age int = 30
    x, y := 10, 20            // 同时声明多个变量
    host, port := "localhost", 8080

    fmt.Println(name, age, x, y, host, port)

    // ============ := 的特殊规则：左侧至少有一个新变量 ============
    a := 1
    // a := 2  // 编译错误：no new variables on left side of :=

    // 但如果左侧有新变量，已存在的变量会被重新赋值
    a, b := 2, 3  // a 被重新赋值，b 是新变量
    fmt.Println(a, b)

    // ============ 常见场景：错误处理 ============
    // Go 中大量使用 := 配合 if 进行错误处理
    // result, err := someFunction()
    // if err != nil {
    //     log.Fatal(err)
    // }
}
```

> **与 Python 对比**：Python 直接 `name = "DevOps"` 即可，无需声明。Go 的 `:=` 是最接近 Python 风格的写法，但 Go 在编译期就确定了类型，之后不能改变。

### 1.3 批量声明

```go
package main

import "fmt"

// ============ 包级别的批量声明（函数外部） ============
var (
    // 数据库配置
    dbHost     = "localhost"
    dbPort     = 5432
    dbUser     = "admin"
    dbPassword = "secret"
    dbName     = "myapp"

    // 服务器配置
    serverHost = "0.0.0.0"
    serverPort = 8080
    debugMode  = false
)

func main() {
    fmt.Printf("数据库: %s@%s:%d/%s\n", dbUser, dbHost, dbPort, dbName)
    fmt.Printf("服务器: %s:%d (debug=%v)\n", serverHost, serverPort, debugMode)

    // ============ 函数内部的批量声明 ============
    var (
        width  = 100
        height = 200
        depth  = 50
    )
    volume := width * height * depth
    fmt.Printf("体积: %d\n", volume)
}
```

### 1.4 变量作用域

```go
package main

import "fmt"

// 包级变量：整个包可见
var packageVar = "我是包级变量"

func main() {
    // 函数级变量：整个函数可见
    funcVar := "我是函数级变量"

    {
        // 块级变量：只在 {} 内可见
        blockVar := "我是块级变量"
        fmt.Println(packageVar) // 可以访问包级变量
        fmt.Println(funcVar)    // 可以访问函数级变量
        fmt.Println(blockVar)   // 可以访问块级变量
    }

    // fmt.Println(blockVar) // 编译错误：blockVar 未定义

    // if/for 中声明的变量也是块级作用域
    if x := 10; x > 5 {
        fmt.Println(x) // x 只在 if 块内可见
    }
    // fmt.Println(x) // 编译错误：x 未定义

    fmt.Println(funcVar)
}
```

### 1.5 未使用的变量

```go
package main

func main() {
    // Go 不允许声明了变量却不使用——编译直接报错！
    // x := 10  // 编译错误：x declared but not used

    // 如果确实需要忽略某个值，使用下划线 _
    _, err := doSomething()
    if err != nil {
        // 处理错误
    }
}

func doSomething() (int, error) {
    return 42, nil
}
```

> **与 Python 对比**：Python 中未使用的变量只是一个 lint 警告，不影响运行。Go 直接编译失败，强制你写更干净的代码。

## 2. 常量与 iota

### 2.1 常量声明

```go
package main

import "fmt"

// 常量在编译时确定，不能修改
const Pi = 3.14159265358979323846

// 批量声明常量
const (
    StatusOK    = 200
    StatusError = 500
    AppName     = "health-checker"
    MaxRetry    = 3
)

// 常量可以是无类型的（untyped constant）
// 无类型常量在使用时会根据上下文自动转换
const Big = 1 << 100  // 这个值超过了 int64，但作为常量可以存在

func main() {
    fmt.Println("Pi =", Pi)
    fmt.Println("Status:", StatusOK)

    // 无类型常量可以赋值给兼容的类型
    var f32 float32 = Pi  // Pi 自动转为 float32
    var f64 float64 = Pi  // Pi 自动转为 float64
    fmt.Println(f32, f64) // 精度不同

    // 有类型常量不能赋值给不同类型
    const typedPi float64 = 3.14
    // var f32b float32 = typedPi // 编译错误：cannot use typedPi (float64) as float32
}
```

### 2.2 iota 枚举生成器

```go
package main

import "fmt"

// iota 是一个在 const 块中自动递增的计数器
// 每个 const 块从 0 开始

// ============ 基础用法 ============
const (
    Sunday    = iota // 0
    Monday           // 1（自动使用上一行的表达式 = iota）
    Tuesday          // 2
    Wednesday        // 3
    Thursday         // 4
    Friday           // 5
    Saturday         // 6
)

// ============ 跳过某个值 ============
const (
    _     = iota // 0，使用 _ 跳过
    One          // 1
    Two          // 2
    Three        // 3
)

// ============ 位运算：定义权限标志 ============
const (
    ReadPermission   = 1 << iota // 1 (001)
    WritePermission              // 2 (010)
    ExecutePermission            // 4 (100)
)

// ============ 字节单位 ============
const (
    _  = iota             // 跳过 0
    KB = 1 << (10 * iota) // 1 << 10 = 1024
    MB                    // 1 << 20 = 1048576
    GB                    // 1 << 30 = 1073741824
    TB                    // 1 << 40
    PB                    // 1 << 50
)

// ============ 自定义类型 + iota（推荐做法） ============
type LogLevel int

const (
    DEBUG LogLevel = iota
    INFO
    WARN
    ERROR
    FATAL
)

// 为自定义类型实现 String() 方法，方便打印
func (l LogLevel) String() string {
    switch l {
    case DEBUG:
        return "DEBUG"
    case INFO:
        return "INFO"
    case WARN:
        return "WARN"
    case ERROR:
        return "ERROR"
    case FATAL:
        return "FATAL"
    default:
        return "UNKNOWN"
    }
}

// ============ HTTP 方法枚举 ============
type HTTPMethod string

const (
    GET    HTTPMethod = "GET"
    POST   HTTPMethod = "POST"
    PUT    HTTPMethod = "PUT"
    DELETE HTTPMethod = "DELETE"
    PATCH  HTTPMethod = "PATCH"
)

func main() {
    fmt.Println("=== 星期 ===")
    fmt.Println("Sunday:", Sunday, "Saturday:", Saturday)

    fmt.Println("\n=== 权限 ===")
    // 组合权限：读 + 写
    myPerm := ReadPermission | WritePermission
    fmt.Printf("读写权限: %03b (%d)\n", myPerm, myPerm)

    // 检查是否有某个权限
    hasRead := myPerm&ReadPermission != 0
    hasExec := myPerm&ExecutePermission != 0
    fmt.Printf("有读权限: %v, 有执行权限: %v\n", hasRead, hasExec)

    fmt.Println("\n=== 字节单位 ===")
    fileSize := 2.5 * float64(GB)
    fmt.Printf("文件大小: %.2f GB\n", fileSize/float64(GB))

    fmt.Println("\n=== 日志级别 ===")
    level := WARN
    fmt.Println("当前日志级别:", level)             // 输出: WARN
    fmt.Printf("级别值: %d, 类型: %T\n", level, level) // 输出: 级别值: 2, 类型: main.LogLevel

    fmt.Println("\n=== HTTP 方法 ===")
    method := GET
    fmt.Printf("方法: %s, 类型: %T\n", method, method)
}
```

> **与 Python 对比**：Python 3.4+ 有 `enum.Enum`，但使用频率远不如 Go 的 iota。Go 中 iota 配合 `1 << iota` 定义位标志是非常常见的模式。

## 3. 基本数据类型

### 3.1 数值类型

```go
package main

import (
    "fmt"
    "math"
    "unsafe"
)

func main() {
    // ============ 整数类型 ============

    // 有符号整数
    var i8 int8 = 127                   // -128 ~ 127
    var i16 int16 = 32767               // -32768 ~ 32767
    var i32 int32 = 2147483647          // -2^31 ~ 2^31-1
    var i64 int64 = 9223372036854775807 // -2^63 ~ 2^63-1
    var i int = 42                      // 平台相关：32 位系统是 int32，64 位系统是 int64

    fmt.Printf("int8: %d (大小: %d 字节)\n", i8, unsafe.Sizeof(i8))
    fmt.Printf("int16: %d (大小: %d 字节)\n", i16, unsafe.Sizeof(i16))
    fmt.Printf("int32: %d (大小: %d 字节)\n", i32, unsafe.Sizeof(i32))
    fmt.Printf("int64: %d (大小: %d 字节)\n", i64, unsafe.Sizeof(i64))
    fmt.Printf("int: %d (大小: %d 字节)\n", i, unsafe.Sizeof(i))

    // 无符号整数
    var u8 uint8 = 255                    // 0 ~ 255（等同于 byte）
    var u16 uint16 = 65535                // 0 ~ 65535
    var u32 uint32 = 4294967295           // 0 ~ 2^32-1
    var u64 uint64 = 18446744073709551615 // 0 ~ 2^64-1
    var u uint = 42                       // 平台相关

    fmt.Printf("\nuint8: %d\n", u8)
    fmt.Printf("uint16: %d\n", u16)
    fmt.Printf("uint32: %d\n", u32)
    fmt.Printf("uint64: %d\n", u64)
    fmt.Printf("uint: %d\n", u)

    // 特殊整数类型
    var b byte = 'A'         // byte 是 uint8 的别名
    var r rune = '中'        // rune 是 int32 的别名，表示 Unicode 码点
    var ptr uintptr = 0xDEAD // uintptr 用于存放指针值（不常用）

    fmt.Printf("\nbyte: %c (%d)\n", b, b)
    fmt.Printf("rune: %c (U+%04X)\n", r, r)
    fmt.Printf("uintptr: %#x\n", ptr)

    // 数字字面量（Go 1.13+ 支持下划线分隔）
    million := 1_000_000           // 下划线分隔，增强可读性
    hex := 0xFF                    // 十六进制
    oct := 0o77                    // 八进制（Go 1.13+ 推荐 0o 前缀）
    bin := 0b1010_1010             // 二进制（Go 1.13+）

    fmt.Printf("\n一百万: %d\n", million)
    fmt.Printf("十六进制 0xFF = %d\n", hex)
    fmt.Printf("八进制 0o77 = %d\n", oct)
    fmt.Printf("二进制 0b10101010 = %d\n", bin)

    // ============ 浮点数类型 ============

    var f32 float32 = 3.14
    var f64 float64 = 3.141592653589793

    fmt.Printf("\nfloat32: %.10f (精度约 7 位)\n", f32)
    fmt.Printf("float64: %.15f (精度约 15 位)\n", f64)
    fmt.Printf("float32 最大值: %e\n", math.MaxFloat32)
    fmt.Printf("float64 最大值: %e\n", math.MaxFloat64)

    // 特殊浮点值
    inf := math.Inf(1)
    negInf := math.Inf(-1)
    nan := math.NaN()
    fmt.Printf("\n正无穷: %f\n", inf)
    fmt.Printf("负无穷: %f\n", negInf)
    fmt.Printf("NaN: %f, IsNaN: %v\n", nan, math.IsNaN(nan))

    // ============ 复数类型（不常用） ============

    var c64 complex64 = 3 + 4i
    var c128 complex128 = complex(3.0, 4.0) // 等同于 3+4i
    fmt.Printf("\n复数: %v, 实部: %f, 虚部: %f\n", c128, real(c128), imag(c128))
    _ = c64
}
```

### 3.2 布尔类型

```go
package main

import "fmt"

func main() {
    var isReady bool = true
    var isFailed bool // 零值: false

    fmt.Println("isReady:", isReady)
    fmt.Println("isFailed:", isFailed)

    // 布尔运算
    fmt.Println("\n=== 逻辑运算 ===")
    fmt.Println("true && false =", true && false)   // 逻辑与
    fmt.Println("true || false =", true || false)   // 逻辑或
    fmt.Println("!true =", !true)                   // 逻辑非

    // 注意：Go 中布尔值不能与整数互转！
    // if 1 { }           // 编译错误
    // num := int(true)   // 编译错误
    // 这与 Python 不同，Python 中 bool 是 int 的子类
}
```

> **与 Python 对比**：Python 中 `True == 1`，空字符串、0、None 等都是 falsy。Go 中只有 `true`/`false`，没有 truthy/falsy 的概念，`if` 条件必须是 `bool` 类型。

### 3.3 字符串类型

```go
package main

import (
    "fmt"
    "strings"
)

func main() {
    // ============ 字符串基础 ============

    // 双引号：解释型字符串（处理转义）
    s1 := "Hello\tWorld\n"
    fmt.Print(s1)

    // 反引号：原始字符串（不处理转义）
    s2 := `Hello\tWorld\n`    // 原样输出，包括 \t 和 \n
    fmt.Println(s2)

    // 反引号支持多行（类似 Python 的三引号）
    jsonTemplate := `{
    "name": "DevOps",
    "role": "SRE",
    "active": true
}`
    fmt.Println(jsonTemplate)

    // ============ 字符串是不可变的 ============
    s := "hello"
    // s[0] = 'H'  // 编译错误：cannot assign to s[0]

    // 要修改字符串，需要转换为 []byte 或 []rune
    bs := []byte(s)
    bs[0] = 'H'
    s = string(bs)
    fmt.Println(s) // "Hello"

    // ============ 字符串长度 ============
    ascii := "hello"
    chinese := "你好世界"

    fmt.Printf("\n%q 的 len: %d (字节数)\n", ascii, len(ascii))       // 5
    fmt.Printf("%q 的 len: %d (字节数)\n", chinese, len(chinese))     // 12（每个中文字 3 字节 UTF-8）

    // 获取字符数（rune 数量）
    fmt.Printf("%q 的字符数: %d\n", chinese, len([]rune(chinese)))     // 4
    fmt.Printf("%q 的字符数: %d\n", chinese, strings.Count(chinese, "")-1) // 4

    // ============ 遍历字符串 ============
    fmt.Println("\n按字节遍历:")
    for i := 0; i < len(chinese); i++ {
        fmt.Printf("  索引 %d: %#x\n", i, chinese[i])
    }

    fmt.Println("按字符(rune)遍历:")
    for i, ch := range chinese {
        fmt.Printf("  索引 %d: %c (U+%04X)\n", i, ch, ch)
    }

    // ============ 字符串拼接 ============
    // 方式一：+ 运算符（简单场景）
    greeting := "Hello" + ", " + "World"
    fmt.Println(greeting)

    // 方式二：fmt.Sprintf（格式化拼接）
    name := "DevOps"
    age := 30
    info := fmt.Sprintf("Name: %s, Age: %d", name, age)
    fmt.Println(info)

    // 方式三：strings.Builder（高性能场景，推荐循环拼接时使用）
    var builder strings.Builder
    for i := 0; i < 5; i++ {
        fmt.Fprintf(&builder, "item-%d ", i)
    }
    fmt.Println(builder.String())
}
```

### 3.4 byte 与 rune

```go
package main

import "fmt"

func main() {
    // byte = uint8，表示一个 ASCII 字符或一个 UTF-8 字节
    var b byte = 'A'
    fmt.Printf("byte: %c, 值: %d, 类型: %T\n", b, b, b)

    // rune = int32，表示一个 Unicode 码点
    var r rune = '中'
    fmt.Printf("rune: %c, 值: %d, Unicode: U+%04X, 类型: %T\n", r, r, r, r)

    // 字符串 -> []byte（UTF-8 编码的字节序列）
    s := "Go语言"
    bytes := []byte(s)
    fmt.Printf("\n%q -> []byte: %v\n", s, bytes)
    fmt.Printf("字节数: %d\n", len(bytes)) // 8 = 2(Go) + 3(语) + 3(言)

    // 字符串 -> []rune（Unicode 码点序列）
    runes := []rune(s)
    fmt.Printf("%q -> []rune: %v\n", s, runes)
    fmt.Printf("字符数: %d\n", len(runes)) // 4

    // 单个字符的类型取决于声明方式
    c1 := 'A'    // 类型是 rune (int32)，不是 byte！
    c2 := byte('A') // 显式声明为 byte
    fmt.Printf("\n'A' 默认类型: %T\n", c1) // int32
    fmt.Printf("byte('A') 类型: %T\n", c2) // uint8
}
```

## 4. 类型转换

```go
package main

import (
    "fmt"
    "strconv"
)

func main() {
    // ============ Go 没有隐式类型转换！ ============
    // 所有类型转换都必须显式进行

    // 整数之间的转换
    var i32 int32 = 100
    var i64 int64 = int64(i32) // 必须显式转换
    var i int = int(i64)

    fmt.Println(i32, i64, i)

    // 注意溢出问题！
    var big int64 = 256
    var small int8 = int8(big)   // 溢出：256 -> 0
    fmt.Printf("int64(%d) -> int8(%d) 溢出了！\n", big, small)

    // 整数 <-> 浮点数
    n := 42
    f := float64(n)            // int -> float64
    n2 := int(3.14)            // float64 -> int（直接截断，不是四舍五入！）
    fmt.Printf("int(3.14) = %d (截断，不是四舍五入)\n", n2)
    fmt.Printf("int(3.99) = %d (截断，不是四舍五入)\n", int(3.99))
    _ = f

    // ============ 字符串 <-> 数字（使用 strconv 包） ============

    // 字符串 -> 整数
    s1 := "42"
    num, err := strconv.Atoi(s1) // Atoi = ASCII to Integer
    if err != nil {
        fmt.Println("转换失败:", err)
    }
    fmt.Printf("Atoi(%q) = %d\n", s1, num)

    // 字符串 -> 整数（更灵活）
    num64, err := strconv.ParseInt("FF", 16, 64) // 十六进制字符串转 int64
    fmt.Printf("ParseInt(\"FF\", 16) = %d\n", num64)

    // 字符串 -> 浮点数
    fStr := "3.14"
    fVal, err := strconv.ParseFloat(fStr, 64)
    fmt.Printf("ParseFloat(%q) = %f\n", fStr, fVal)

    // 字符串 -> 布尔
    bStr := "true"
    bVal, err := strconv.ParseBool(bStr)
    fmt.Printf("ParseBool(%q) = %v\n", bStr, bVal)

    // 整数 -> 字符串
    n3 := 42
    s2 := strconv.Itoa(n3) // Itoa = Integer to ASCII
    fmt.Printf("Itoa(%d) = %q\n", n3, s2)

    // 注意：string(42) 不是 "42"！
    s3 := string(42)                                           // Unicode 码点 42 = '*'
    fmt.Printf("string(42) = %q (这是 Unicode 码点，不是数字转字符串！)\n", s3)

    // 浮点数 -> 字符串
    s4 := strconv.FormatFloat(3.14159, 'f', 2, 64) // 'f'=小数格式, 2=精度, 64=float64
    fmt.Printf("FormatFloat(3.14159) = %q\n", s4)

    // 布尔 -> 字符串
    s5 := strconv.FormatBool(true)
    fmt.Printf("FormatBool(true) = %q\n", s5)

    // ============ []byte <-> string ============
    bs := []byte("hello")
    str := string(bs)
    fmt.Printf("[]byte -> string: %q\n", str)

    // 注意：这个转换会拷贝数据，频繁转换有性能开销
    _ = err
}
```

> **与 Python 对比**：Python 会自动做很多隐式转换（如 `int + float`），Go 完全不会。Go 的哲学是"显式优于隐式"，这避免了很多难以发现的 bug。

## 5. 运算符

### 5.1 算术运算符

```go
package main

import "fmt"

func main() {

    fmt.Println("a + b =", a+b)  // 13

    fmt.Println("\nfloat: 7.0 / 2.0 =", 7.0/2.0)       // 3.5

    // 自增/自减（只有后置，不能作为表达式）
    x := 5
    x++           // x = 6
    x--           // x = 5
```

### 5.3 位运算符

```go
package main

import "fmt"

func main() {
    a, b := 0b1100, 0b1010 // 12, 10

    fmt.Printf("a     = %04b (%d)\n", a, a)
    fmt.Printf("b     = %04b (%d)\n", b, b)
    fmt.Printf("a & b = %04b (%d) 按位与\n", a&b, a&b)     // 1000 (8)
    fmt.Printf("a | b = %04b (%d) 按位或\n", a|b, a|b)     // 1110 (14)
    fmt.Printf("a ^ b = %04b (%d) 按位异或\n", a^b, a^b)   // 0110 (6)
    fmt.Printf("a &^ b = %04b (%d) 按位清除\n", a&^b, a&^b) // 0100 (4) Go 特有！
    fmt.Printf("a << 2 = %04b (%d) 左移\n", a<<2, a<<2)     // 110000 (48)
    fmt.Printf("a >> 1 = %04b (%d) 右移\n", a>>1, a>>1)     // 0110 (6)

    // 实际应用：权限管理
    const (
        Read    = 1 << iota // 001
        Write               // 010
        Execute             // 100
    )

    // 设置权限
    perm := Read | Write     // 011
    fmt.Printf("\n权限: %03b\n", perm)

    // 检查权限
    fmt.Printf("有读权限: %v\n", perm&Read != 0)
    fmt.Printf("有执行权限: %v\n", perm&Execute != 0)

    // 添加权限
    perm |= Execute          // 111
    fmt.Printf("添加执行后: %03b\n", perm)

    // 移除权限
    perm &^= Write           // 101
    fmt.Printf("移除写后: %03b\n", perm)
}
```

## 6. fmt 格式化输出（Printf 占位符详解）

### 6.1 通用占位符

```go
package main

import "fmt"

type Server struct {
    Host string
    Port int
}

    s := Server{Host: "localhost", Port: 8080}
    n := 42
    f := 3.14159
    str := "hello"
    bs := []byte{72, 101, 108, 108, 111}

    fmt.Printf("%%v:  %v\n", s)      // {localhost 8080}

    // %+v 带字段名的结构体
    fmt.Printf("%%+v: %+v\n", s)     // {Host:localhost Port:8080}

    // %#v Go 语法表示（调试利器）
    fmt.Printf("%%#v: %#v\n", s)     // main.Server{Host:"localhost", Port:8080}

    // %T  类型
    fmt.Printf("%%T:  %T\n", s)      // main.Server
    fmt.Printf("%%T:  %T\n", n)      // int
    fmt.Printf("%%T:  %T\n", str)    // string

    // %% 输出百分号
    fmt.Printf("CPU 使用率: 85%%\n")

    _ = f
    _ = bs
}
```

## 7. 零值概念

```go
package main

import "fmt"

func main() {
    // Go 的每种类型都有零值，声明时不赋值会自动初始化为零值
    // 这是 Go 的重要特性：没有"未初始化的变量"

    // ============ 各类型的零值 ============
    var b bool       // false
    var i int        // 0
    var f float64    // 0.0
    var s string     // "" (空字符串)
    var p *int       // nil
    var sl []int     // nil
    var m map[string]int  // nil
    var ch chan int   // nil
    var fn func()    // nil
    var iface error  // nil

    fmt.Printf("bool:      %v\n", b)
    fmt.Printf("int:       %v\n", i)
    fmt.Printf("float64:   %v\n", f)
    fmt.Printf("string:    %q\n", s)
    fmt.Printf("*int:      %v\n", p)
    fmt.Printf("[]int:     %v (nil=%v)\n", sl, sl == nil)
    fmt.Printf("map:       %v (nil=%v)\n", m, m == nil)
    fmt.Printf("chan:       %v\n", ch)
    fmt.Printf("func:      %v\n", fn)
    fmt.Printf("interface: %v\n", iface)

    // ============ 结构体的零值 ============
    type Config struct {
        Host    string
        Port    int
        Debug   bool
        Timeout float64
    }

    var cfg Config
    fmt.Printf("\nConfig 零值: %+v\n", cfg)
    // {Host: Port:0 Debug:false Timeout:0}

    // 零值可用性：很多类型的零值是可以直接使用的
    // 例如：
    var builder []byte
    builder = append(builder, 'H', 'i')
    fmt.Println("nil slice append:", string(builder))

    // 但 nil map 不能写入！
    // m["key"] = 1  // panic: assignment to entry in nil map
    // 必须先初始化
    m = make(map[string]int)
    m["key"] = 1
    fmt.Println("map:", m)
}
```

> **与 Python 对比**：Python 中不赋值就没有变量，引用会抛 `NameError`。Go 中声明了就一定有值（零值），不存在未定义的变量。这使得 Go 代码更加可预测。

## 8. 类型别名与自定义类型

```go
package main

import "fmt"

// ============ 类型定义（创建新类型） ============
// 新类型与原类型是不同的类型，不能直接互相赋值
type UserID int64
type Email string
type Celsius float64
type Fahrenheit float64

// 为新类型添加方法
func (c Celsius) ToFahrenheit() Fahrenheit {
    return Fahrenheit(c*9/5 + 32)
}

func (f Fahrenheit) ToCelsius() Celsius {
    return Celsius((f - 32) * 5 / 9)
}

func (c Celsius) String() string {
    return fmt.Sprintf("%.1f°C", float64(c))
}

// ============ 类型别名（Go 1.9+） ============
// 类型别名和原类型完全相同，可以互相赋值
type Byte = uint8 // byte 就是这样定义的
type Rune = int32 // rune 就是这样定义的

// 实际项目中的类型别名用法（通常用于渐进式重构）
type OldConfig = NewConfig
type NewConfig struct {
    Host string
    Port int
}

func main() {
    // ============ 类型定义的使用 ============
    var uid UserID = 12345
    var email Email = "admin@example.com"

    fmt.Printf("UserID: %d (类型: %T)\n", uid, uid)
    fmt.Printf("Email: %s (类型: %T)\n", email, email)

    // 不同类型不能直接运算或赋值
    // var n int64 = uid          // 编译错误
    var n int64 = int64(uid)     // 必须显式转换
    fmt.Println("转换后:", n)

    // 温度转换示例
    bodyTemp := Celsius(37.0)
    fmt.Printf("\n体温: %s = %.1f°F\n", bodyTemp, bodyTemp.ToFahrenheit())

    boiling := Fahrenheit(212)
    fmt.Printf("沸点: %.1f°F = %s\n", boiling, boiling.ToCelsius())

    // ============ 类型别名的使用 ============
    var b Byte = 255
    var u uint8 = b // 类型别名可以直接赋值，因为它们是同一个类型
    fmt.Printf("\nByte: %d, uint8: %d\n", b, u)
    fmt.Printf("Byte 类型: %T\n", b) // uint8（别名不会显示，显示原类型）

    // ============ 区分类型定义和类型别名 ============
    type MyInt int     // 类型定义：MyInt 是新类型
    type YourInt = int // 类型别名：YourInt 就是 int

    var mi MyInt = 42
    var yi YourInt = 42

    fmt.Printf("\nMyInt 类型: %T\n", mi)   // main.MyInt（新类型）
    fmt.Printf("YourInt 类型: %T\n", yi)    // int（就是 int）

    // var x int = mi   // 编译错误：MyInt != int
    var y int = yi      // 正确：YourInt == int
    _ = y
}
```

## 实战场景：配置解析

```go
package main

import (
    "fmt"
    "strconv"
    "strings"
)

// 模拟从环境变量或配置文件解析配置

// ServerConfig 服务器配置
type ServerConfig struct {
    Host         string
    Port         int
    ReadTimeout  int // 秒
    WriteTimeout int // 秒
    Debug        bool
}

// ParseConfig 从 key=value 格式的字符串切片解析配置
func ParseConfig(lines []string) (ServerConfig, error) {
    cfg := ServerConfig{
        // 默认值
        Host:         "0.0.0.0",
        Port:         8080,
        ReadTimeout:  30,
        WriteTimeout: 30,
        Debug:        false,
    }

    for _, line := range lines {
        // 跳过空行和注释
        line = strings.TrimSpace(line)
        if line == "" || strings.HasPrefix(line, "#") {
            continue
        }

        // 分割 key=value
        parts := strings.SplitN(line, "=", 2)
        if len(parts) != 2 {
            return cfg, fmt.Errorf("无效的配置行: %q", line)
        }

        key := strings.TrimSpace(parts[0])
        value := strings.TrimSpace(parts[1])

        // 根据 key 设置对应字段
        switch strings.ToLower(key) {
        case "host":
            cfg.Host = value
        case "port":
            port, err := strconv.Atoi(value)
            if err != nil {
                return cfg, fmt.Errorf("无效的端口号: %q", value)
            }
            if port < 1 || port > 65535 {
                return cfg, fmt.Errorf("端口号超出范围: %d", port)
            }
            cfg.Port = port
        case "read_timeout":
            timeout, err := strconv.Atoi(value)
            if err != nil {
                return cfg, fmt.Errorf("无效的读超时: %q", value)
            }
            cfg.ReadTimeout = timeout
        case "write_timeout":
            timeout, err := strconv.Atoi(value)
            if err != nil {
                return cfg, fmt.Errorf("无效的写超时: %q", value)
            }
            cfg.WriteTimeout = timeout
        case "debug":
            debug, err := strconv.ParseBool(value)
            if err != nil {
                return cfg, fmt.Errorf("无效的 debug 值: %q", value)
            }
            cfg.Debug = debug
        default:
            fmt.Printf("警告: 未知的配置项 %q\n", key)
        }
    }

    return cfg, nil
}

func main() {
    // 模拟配置文件内容
    configLines := []string{
        "# 服务器配置",
        "host = 192.168.1.100",
        "port = 9090",
        "read_timeout = 60",
        "write_timeout = 60",
        "debug = true",
        "",
        "# 这是注释行",
    }

    cfg, err := ParseConfig(configLines)
    if err != nil {
        fmt.Printf("解析配置失败: %v\n", err)
        return
    }

    fmt.Printf("配置解析结果:\n")
    fmt.Printf("  Host:         %s\n", cfg.Host)
    fmt.Printf("  Port:         %d\n", cfg.Port)
    fmt.Printf("  ReadTimeout:  %d秒\n", cfg.ReadTimeout)
    fmt.Printf("  WriteTimeout: %d秒\n", cfg.WriteTimeout)
    fmt.Printf("  Debug:        %v\n", cfg.Debug)

    // 生成监听地址
    addr := fmt.Sprintf("%s:%d", cfg.Host, cfg.Port)
    fmt.Printf("  监听地址:      %s\n", addr)
}
```

## 常见坑点

### 坑点 1：:= 在不同作用域中的变量遮蔽（Variable Shadowing）

```go
package main

import "fmt"

func main() {
    x := 1
    fmt.Println("外层 x:", x) // 1

    {
        // 这里的 := 创建了一个新的 x，遮蔽了外层的 x
        x := 2
        fmt.Println("内层 x:", x) // 2
    }

    fmt.Println("外层 x:", x) // 还是 1！不是 2！

    // 如果想修改外层的 x，用 = 而不是 :=
    {
        x = 3
        fmt.Println("内层修改后 x:", x) // 3
    }
    fmt.Println("外层 x:", x) // 3

    // 最坑的场景：if err 中的变量遮蔽
    var err error
    if true {
        // 如果左侧有其他新变量，err 会被遮蔽而不是赋值
        // result, err := doSomething()  // 这里的 err 是新变量！
        // 外层的 err 不会被修改
        _ = err
    }
}
```

### 坑点 2：string(int) 不是数字转字符串

```go
package main

import (
    "fmt"
    "strconv"
)

func main() {
    n := 65
    s1 := string(n)           // "A"（Unicode 码点 65 = 'A'）
    s2 := strconv.Itoa(n)     // "65"（正确的数字转字符串）
    s3 := fmt.Sprintf("%d", n) // "65"（也可以）

    fmt.Printf("string(%d) = %q (Unicode 字符！)\n", n, s1)
    fmt.Printf("Itoa(%d) = %q (数字字符串)\n", n, s2)
    fmt.Printf("Sprintf(%d) = %q\n", n, s3)
}
```

### 坑点 3：浮点数精度问题

```go
package main

import (
    "fmt"
    "math"
)

func main() {
    // 浮点数不精确
    a := 0.1 + 0.2
    fmt.Printf("0.1 + 0.2 = %.20f\n", a)      // 0.30000000000000004441
    fmt.Println("0.1 + 0.2 == 0.3?", a == 0.3) // false!

    // 正确的浮点数比较方式
    epsilon := 1e-9
    fmt.Println("近似相等?", math.Abs(a-0.3) < epsilon) // true

    // 涉及金钱计算时，使用整数（分）或第三方 decimal 库
    // 不要用 float 来处理金额！
    priceCents := 199 // 1.99 元 = 199 分
    quantity := 3
    totalCents := priceCents * quantity
    fmt.Printf("总价: %.2f 元\n", float64(totalCents)/100) // 5.97 元
}
```

## 小结

1. **变量声明**：函数内优先使用 `:=`，包级变量用 `var`，批量声明用 `var()` 块。
2. **Go 没有隐式类型转换**：所有转换必须显式进行，这比 Python 严格但更安全。
3. **零值是 Go 的重要特性**：每个类型都有零值，没有"未初始化"的变量。
4. **iota 是优雅的枚举方式**：配合类型定义和 `String()` 方法，比 Python 的 Enum 更轻量。
5. **fmt.Printf 是调试利器**：`%+v` 和 `%#v` 是你最好的朋友。
6. **strconv 包是类型转换的核心**：字符串和数字之间的转换全靠它。
