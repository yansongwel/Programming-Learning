# Go 进阶篇 04 — 指针与内存（Pointer & Memory）

> **适用版本**：Go 1.24 | **面向读者**：有编程经验的 SRE / DevOps 工程师

---

## 导读

指针是 Go 语言中一个核心概念——虽然 Go 有垃圾回收，但理解指针和内存管理仍然至关重要。与 C/C++ 不同，Go 的指针更加安全：没有指针运算、没有悬挂指针、没有手动内存管理。但你需要理解：
- 何时使用指针、何时使用值
- Go 的参数传递机制（全部是值传递！）
- `new` 和 `make` 的区别
- 内存逃逸分析——编译器如何决定变量分配在栈还是堆
- GC 的基本原理

对于 SRE/DevOps 工程师来说，理解内存模型有助于编写高性能的运维工具、排查内存泄漏问题。

## 1. 指针基础

### 1.1 取地址 & 和解引用 *

```go
package main

import "fmt"

func main() {
	// 声明一个普通变量
	x := 42

	// & 取地址：获取变量的内存地址
	p := &x // p 是一个指向 int 的指针，类型为 *int

	fmt.Println("x 的值:", x)          // 42
	fmt.Println("x 的地址:", &x)       // 0xc000018090（每次运行不同）
	fmt.Println("p 的值（地址）:", p)   // 0xc000018090
	fmt.Println("p 指向的值:", *p)     // 42（解引用：通过指针获取值）

	// * 解引用：通过指针修改原始值
	*p = 100
	fmt.Println("修改后 x 的值:", x) // 100（x 被通过指针修改了）

	// 指针的类型
	fmt.Printf("x 的类型: %T\n", x)  // int
	fmt.Printf("p 的类型: %T\n", p)  // *int
}
```

### 1.2 指针的零值是 nil

```go
package main

import "fmt"

func main() {
	// 未初始化的指针是 nil
	var p *int
	fmt.Println("p 是 nil?", p == nil) // true
	fmt.Println("p 的值:", p)          // <nil>

	// 解引用 nil 指针会 panic
	// fmt.Println(*p) // panic: runtime error: invalid memory address or nil pointer dereference

	// 安全使用指针：先检查 nil
	if p != nil {
		fmt.Println("值:", *p)
	} else {
		fmt.Println("指针为 nil，跳过")
	}

	// 初始化指针
	x := 42
	p = &x
	fmt.Println("初始化后:", *p) // 42
}
```

### 1.3 指针与结构体

```go
package main

import "fmt"

type Server struct {
	Name   string
	Host   string
	Port   int
	Online bool
}

func main() {
	// 方式1：先创建值，再取地址
	srv1 := Server{Name: "web-01", Host: "10.0.0.1", Port: 80}
	p1 := &srv1

	// 方式2：直接创建指针（更常见）
	p2 := &Server{
		Name: "web-02",
		Host: "10.0.0.2",
		Port: 443,
	}

	// 通过指针访问字段（Go 自动解引用，不需要 -> 操作符）
	fmt.Println(p1.Name) // web-01（等价于 (*p1).Name）
	fmt.Println(p2.Name) // web-02

	// 通过指针修改字段
	p2.Online = true
	fmt.Printf("服务器: %+v\n", *p2)

	// 方式3：使用 new 创建指针（不常用，因为无法同时初始化字段）
	p3 := new(Server)
	p3.Name = "web-03"
	p3.Host = "10.0.0.3"
	p3.Port = 8080
	fmt.Printf("服务器: %+v\n", *p3)
}
```

## 2. Go 没有指针运算

### 2.1 与 C 的对比

```go
package main

import "fmt"

func main() {
	/*
	C 语言中的指针运算：
	  int arr[] = {1, 2, 3, 4, 5};
	  int *p = arr;
	  p++;        // 指针移动到下一个元素
	  *(p + 2);   // 访问第3个后续元素

	Go 不允许指针运算！以下代码都会编译报错：
	  arr := [5]int{1, 2, 3, 4, 5}
	  p := &arr[0]
	  p++            // 编译错误
	  p = p + 1      // 编译错误
	  *(p + 2)       // 编译错误

	这是 Go 的设计哲学：安全 > 灵活
	*/

	// Go 中遍历数组/切片用 range，不需要指针运算
	nums := []int{10, 20, 30, 40, 50}
	for i, v := range nums {
		fmt.Printf("索引 %d: 值 %d, 地址 %p\n", i, v, &nums[i])
	}

	// 如果确实需要底层指针运算（极少数情况），使用 unsafe 包
	// 但这几乎从不推荐（见本文第9节）
	fmt.Println("\nGo 不支持指针运算，这让代码更安全")
}
```

## 3. 值类型 vs 引用类型

### 3.1 值类型

```go
package main

import "fmt"

func main() {
	fmt.Println("=== 值类型 ===")
	fmt.Println("赋值和传参会复制整个值")
	fmt.Println()

	// 基本类型都是值类型
	a := 42
	b := a  // b 是 a 的拷贝
	b = 100
	fmt.Printf("a = %d, b = %d\n", a, b) // a=42, b=100（互不影响）

	// 数组是值类型（！注意：切片不是）
	arr1 := [3]int{1, 2, 3}
	arr2 := arr1 // arr2 是 arr1 的完整拷贝
	arr2[0] = 999
	fmt.Printf("arr1 = %v, arr2 = %v\n", arr1, arr2) // arr1=[1,2,3], arr2=[999,2,3]

	// 结构体是值类型
	type Point struct{ X, Y int }
	p1 := Point{1, 2}
	p2 := p1 // p2 是 p1 的拷贝
	p2.X = 100
	fmt.Printf("p1 = %v, p2 = %v\n", p1, p2) // p1={1,2}, p2={100,2}

	// 值类型列表：
	// int, float64, bool, string, array, struct
}
```

### 3.2 引用类型

```go
package main

import "fmt"

func main() {
	fmt.Println("=== 引用类型 ===")
	fmt.Println("赋值和传参会共享底层数据")
	fmt.Println()

	// 切片是引用类型（底层共享数组）
	s1 := []int{1, 2, 3}
	s2 := s1 // s2 和 s1 指向同一个底层数组
	s2[0] = 999
	fmt.Printf("s1 = %v, s2 = %v\n", s1, s2) // 都变成 [999,2,3]

	// map 是引用类型
	m1 := map[string]int{"a": 1, "b": 2}
	m2 := m1 // m2 和 m1 指向同一个底层哈希表
	m2["a"] = 999
	fmt.Printf("m1 = %v, m2 = %v\n", m1, m2) // m1 也被修改了

	// channel 是引用类型
	ch1 := make(chan int, 1)
	ch2 := ch1 // ch2 和 ch1 是同一个 channel
	ch1 <- 42
	fmt.Println("从 ch2 接收:", <-ch2) // 42

	// 指针本身也是引用语义
	x := 42
	p1 := &x
	p2 := p1 // p2 指向同一个变量 x
	*p2 = 100
	fmt.Println("通过 p1 读取:", *p1) // 100

	fmt.Println()
	fmt.Println("引用类型列表: slice, map, channel, pointer, function")

	// 注意：interface 也有引用语义
	// 但 string 虽然底层是指针+长度，但行为上是值类型（不可变）
}
```

### 3.3 完整对比

```go
package main

import "fmt"

type Config struct {
	Name    string
	Tags    []string          // 切片是引用类型
	Options map[string]string // map 是引用类型
}

func main() {
	// 复制结构体时的陷阱
	cfg1 := Config{
		Name:    "原始配置",
		Tags:    []string{"prod", "critical"},
		Options: map[string]string{"timeout": "30s"},
	}

	cfg2 := cfg1 // 浅拷贝！
	cfg2.Name = "拷贝配置" // 字符串是值类型，不影响 cfg1

	cfg2.Tags[0] = "staging" // 切片是引用类型，cfg1 也被修改！
	cfg2.Options["timeout"] = "60s" // map 是引用类型，cfg1 也被修改！

	fmt.Println("cfg1:", cfg1.Name, cfg1.Tags, cfg1.Options)
	fmt.Println("cfg2:", cfg2.Name, cfg2.Tags, cfg2.Options)
	// cfg1: 原始配置 [staging critical] map[timeout:60s]  ← Tags 和 Options 被修改了！
	// cfg2: 拷贝配置 [staging critical] map[timeout:60s]

	fmt.Println()

	// 正确的深拷贝方式
	cfg3 := Config{
		Name: cfg1.Name,
		Tags: make([]string, len(cfg1.Tags)),
		Options: make(map[string]string),
	}
	copy(cfg3.Tags, cfg1.Tags) // 复制切片内容
	for k, v := range cfg1.Options { // 复制 map 内容
		cfg3.Options[k] = v
	}

	cfg3.Tags[0] = "dev"
	cfg3.Options["timeout"] = "10s"

	fmt.Println("深拷贝后:")
	fmt.Println("cfg1:", cfg1.Tags, cfg1.Options) // 不受影响
	fmt.Println("cfg3:", cfg3.Tags, cfg3.Options) // 独立修改
}
```

## 4. 函数参数传递：Go 全部是值传递！

### 4.1 值传递的证明

```go
package main

import "fmt"

// 传值：函数接收的是拷贝
func tryModifyInt(n int) {
	n = 999
	fmt.Println("函数内 n:", n) // 999
}

func tryModifyStruct(s struct{ Name string }) {
	s.Name = "被修改"
	fmt.Println("函数内 s:", s.Name) // 被修改
}

// 传指针：函数接收的是指针的拷贝（但指向同一个值）
func modifyViaPointer(p *int) {
	*p = 999
	fmt.Println("函数内 *p:", *p) // 999
}

func main() {
	// 基本类型：值传递
	x := 42
	tryModifyInt(x)
	fmt.Println("函数外 x:", x) // 42（未被修改）

	// 结构体：值传递
	s := struct{ Name string }{"原始值"}
	tryModifyStruct(s)
	fmt.Println("函数外 s:", s.Name) // 原始值（未被修改）

	// 指针：传递的是指针的拷贝（指向同一地址）
	y := 42
	modifyViaPointer(&y)
	fmt.Println("函数外 y:", y) // 999（被通过指针修改了）

	fmt.Println()
	fmt.Println("关键理解：")
	fmt.Println("  Go 中所有参数都是值传递（传递的是值的拷贝）")
	fmt.Println("  传指针时，传的是指针本身的拷贝")
	fmt.Println("  但拷贝的指针指向相同的内存地址")
	fmt.Println("  所以通过指针可以修改原始值")
}
```

### 4.2 切片和 map 的传递

```go
package main

import "fmt"

// 切片传递：切片头（指针+长度+容量）被拷贝，但底层数组共享
func modifySlice(s []int) {
	s[0] = 999 // 修改底层数组——外部可见
	fmt.Println("函数内:", s)
}

func appendToSlice(s []int) {
	s = append(s, 100) // append 可能创建新底层数组——外部不可见
	fmt.Println("函数内 append 后:", s)
}

// map 传递：map 本身是指针（指向 runtime.hmap），拷贝后指向同一个哈希表
func modifyMap(m map[string]int) {
	m["new_key"] = 42 // 外部可见
	fmt.Println("函数内:", m)
}

func main() {
	// 切片：修改元素外部可见
	s := []int{1, 2, 3}
	modifySlice(s)
	fmt.Println("函数外:", s) // [999 2 3]（被修改了）

	fmt.Println()

	// 切片：append 可能外部不可见
	s2 := []int{1, 2, 3}
	appendToSlice(s2)
	fmt.Println("函数外:", s2) // [1 2 3]（没变！因为 append 在函数内创建了新切片头）

	fmt.Println()

	// 如果想让 append 的结果外部可见，有两种方式：
	// 方式1：返回新切片
	s3 := appendSafe(s2)
	fmt.Println("返回新切片:", s3) // [1 2 3 100]

	// 方式2：传切片指针
	s4 := []int{1, 2, 3}
	appendViaPointer(&s4)
	fmt.Println("指针方式:", s4) // [1 2 3 100]

	fmt.Println()

	// map：修改外部可见
	m := map[string]int{"a": 1}
	modifyMap(m)
	fmt.Println("函数外:", m) // map[a:1 new_key:42]
}

func appendSafe(s []int) []int {
	return append(s, 100)
}

func appendViaPointer(s *[]int) {
	*s = append(*s, 100)
}
```

### 4.3 何时使用指针，何时使用值

```go
package main

import "fmt"

/*
选择指针 vs 值的指导原则：

使用指针 (*T) 的场景：
1. 需要修改接收者/参数的值
2. 结构体很大（避免拷贝开销）
3. 需要表达 "可能为空" 的语义（nil）
4. 实现接口时用了指针接收者，就要一直用指针
5. sync.Mutex 等同步原语必须使用指针（不能拷贝）

使用值 (T) 的场景：
1. 小的结构体（几个字段）
2. 基本类型（int, string, bool）
3. 不需要修改的只读数据
4. 想要值语义（拷贝安全）
5. 线程安全考虑（每个 goroutine 有自己的拷贝）
*/

// 小结构体：使用值接收者
type Point struct {
	X, Y float64
}

func (p Point) Distance() float64 {
	return p.X*p.X + p.Y*p.Y // 简化的距离计算
}

// 大结构体：使用指针接收者
type ClusterConfig struct {
	Name      string
	Nodes     []string
	Labels    map[string]string
	Replicas  int
	Namespace string
	Image     string
	// ... 很多字段
}

func (c *ClusterConfig) AddNode(node string) {
	c.Nodes = append(c.Nodes, node) // 需要修改，用指针
}

func (c *ClusterConfig) NodeCount() int {
	return len(c.Nodes) // 虽然不修改，但为了一致性也用指针
}

// 需要表达 nil 语义
type OptionalConfig struct {
	Timeout *int    // nil 表示未设置，0 表示零超时
	Debug   *bool   // nil 表示未设置
	Name    *string // nil 表示未设置
}

func main() {
	// 值类型示例
	p := Point{3, 4}
	fmt.Println("距离:", p.Distance())

	// 指针类型示例
	cfg := &ClusterConfig{
		Name:      "prod-cluster",
		Nodes:     []string{"node-01"},
		Namespace: "default",
	}
	cfg.AddNode("node-02")
	fmt.Printf("集群: %s, 节点数: %d\n", cfg.Name, cfg.NodeCount())

	// 可选配置示例
	timeout := 30
	debug := true
	opt := OptionalConfig{
		Timeout: &timeout, // 设置了超时
		Debug:   &debug,   // 设置了调试
		Name:    nil,       // 未设置名称
	}

	if opt.Timeout != nil {
		fmt.Println("超时:", *opt.Timeout)
	}
	if opt.Name != nil {
		fmt.Println("名称:", *opt.Name)
	} else {
		fmt.Println("名称: 未设置")
	}
}
```

## 5. new vs make

### 5.1 new — 分配内存并返回指针

```go
package main

import "fmt"

func main() {
	// new(T) 分配 T 类型的零值内存，返回 *T 指针
	// new 可以用于任何类型

	// new 一个 int
	p := new(int) // *int，指向零值 0
	fmt.Printf("new(int): 值=%d, 类型=%T, 地址=%p\n", *p, p, p)

	// new 一个 string
	s := new(string) // *string，指向零值 ""
	fmt.Printf("new(string): 值=%q, 类型=%T\n", *s, s)

	// new 一个结构体
	type Server struct {
		Name string
		Port int
	}
	srv := new(Server) // *Server，所有字段为零值
	fmt.Printf("new(Server): %+v\n", *srv)

	// new 很少直接使用，因为以下写法更常见且更灵活：
	srv2 := &Server{} // 等价于 new(Server)
	fmt.Printf("&Server{}: %+v\n", *srv2)

	srv3 := &Server{Name: "api-01", Port: 8080} // 可以同时初始化
	fmt.Printf("&Server{...}: %+v\n", *srv3)

	fmt.Println()
	fmt.Println("new 的使用场景：")
	fmt.Println("  1. 需要一个指向零值的指针")
	fmt.Println("  2. 泛型代码中创建任意类型的指针")
	fmt.Println("  3. 其实很少直接使用 new")
}
```

### 5.2 make — 初始化内置引用类型

```go
package main

import "fmt"

func main() {
	// make 只能用于 slice、map、channel 这三种类型
	// make 返回的是类型本身（不是指针），且已经初始化好（可以直接使用）

	// === Slice ===
	// make([]T, length, capacity)
	s1 := make([]int, 5)    // 长度5，容量5
	s2 := make([]int, 3, 10) // 长度3，容量10
	fmt.Printf("s1: len=%d, cap=%d, 值=%v\n", len(s1), cap(s1), s1)
	fmt.Printf("s2: len=%d, cap=%d, 值=%v\n", len(s2), cap(s2), s2)

	// === Map ===
	// make(map[K]V, initialCapacity) — capacity 是可选的
	m1 := make(map[string]int)     // 空 map，默认容量
	m2 := make(map[string]int, 100) // 空 map，预分配约100个元素的空间
	m1["a"] = 1
	m2["b"] = 2
	fmt.Printf("m1: %v\n", m1)
	fmt.Printf("m2: %v\n", m2)

	// === Channel ===
	// make(chan T, bufferSize) — bufferSize 是可选的
	ch1 := make(chan int)    // 无缓冲 channel
	ch2 := make(chan int, 5) // 带5个缓冲的 channel
	fmt.Printf("ch1: %T\n", ch1)
	fmt.Printf("ch2: 容量=%d\n", cap(ch2))

	// ❌ 不能对 slice/map/channel 使用 new（不会正确初始化）
	// 以下代码虽然编译通过，但使用时会 panic
	// mp := new(map[string]int) // *map[string]int，但 *mp 是 nil
	// (*mp)["key"] = 1          // panic: assignment to entry in nil map

	// ✅ 必须使用 make
	mp := make(map[string]int) // 正确初始化的 map
	mp["key"] = 1
	fmt.Println("map:", mp)
}
```

### 5.3 new vs make 总结对比

```go
package main

import "fmt"

func main() {
	fmt.Println("new vs make 对比")
	fmt.Println()

	comparison := []struct {
		aspect, newFunc, makeFunc string
	}{
		{"适用类型", "任何类型", "只有 slice, map, channel"},
		{"返回类型", "*T（指针）", "T（值本身）"},
		{"初始化", "只分配零值内存", "完整初始化内部数据结构"},
		{"使用频率", "很少直接使用", "非常常用"},
	}

	for _, c := range comparison {
		fmt.Printf("  %-10s | new: %-25s | make: %s\n", c.aspect, c.newFunc, c.makeFunc)
	}

	fmt.Println()
	fmt.Println("示例对比:")
	fmt.Println("  new([]int)             → *[]int（指向 nil 切片的指针，几乎没用）")
	fmt.Println("  make([]int, 5)         → []int（已初始化的切片，长度5）")
	fmt.Println()
	fmt.Println("  new(map[string]int)    → *map[string]int（指向 nil map 的指针，赋值会 panic）")
	fmt.Println("  make(map[string]int)   → map[string]int（已初始化的 map，可以直接使用）")
	fmt.Println()
	fmt.Println("  &Server{}              → *Server（等价于 new(Server)，但可以同时初始化字段）")
}
```

## 6. 内存逃逸分析

### 6.1 什么是逃逸分析

```go
package main

import "fmt"

// Go 编译器通过逃逸分析决定变量分配在栈还是堆上。
// 如果变量在函数返回后仍然被引用，它就"逃逸"到堆上。

// 不逃逸：x 分配在栈上
func noEscape() int {
	x := 42
	return x // 返回的是值的拷贝，x 本身可以在栈上
}

// 逃逸：x 分配在堆上
func escape() *int {
	x := 42
	return &x // 返回了 x 的地址，x 在函数返回后仍然需要存在
}

// 逃逸：传给接口类型
func escapeToInterface() {
	x := 42
	fmt.Println(x) // fmt.Println 接受 any 类型，x 被装箱到接口，逃逸到堆
}

// 大对象可能逃逸
func bigArray() [10000]int {
	var arr [10000]int
	return arr // 大数组即使是值返回，编译器也可能选择堆分配
}

func main() {
	_ = noEscape()
	p := escape()
	fmt.Println("逃逸的值:", *p)
	escapeToInterface()
}
```

### 6.2 使用 go build -gcflags 查看逃逸分析

```bash
# 查看逃逸分析结果
go build -gcflags="-m" main.go

# 更详细的输出
```

## 7. 栈 vs 堆分配

### 7.1 栈和堆的特点

```go
package main

import "fmt"

func main() {
	fmt.Println("栈 (Stack) vs 堆 (Heap) 分配")
	fmt.Println()

	/*
	栈 (Stack)：
	- 分配/释放极快（只需移动栈指针）
	- 函数返回时自动回收（零 GC 开销）
	- 大小有限（Go 的 goroutine 栈初始很小，按需增长）
	- 每个 goroutine 有自己的栈
	- 局部变量默认在栈上

	堆 (Heap)：
	- 分配较慢（需要找到合适的内存块）
	- 需要 GC 回收（有性能开销）
	- 大小几乎不受限制
	- 所有 goroutine 共享
	- 逃逸的变量在堆上

	对 SRE 的启示：
	- 减少堆分配 = 减少 GC 压力 = 降低延迟毛刺
	- 高频调用的热路径上尤其要注意
	*/

	comparison := []struct {
		aspect, stack, heap string
	}{
		{"分配速度", "极快（ns 级）", "较慢（可能 us 级）"},
		{"回收方式", "自动（函数返回即回收）", "GC 回收"},
		{"GC 压力", "零", "有"},
		{"大小限制", "有（goroutine 栈大小）", "几乎无限（受物理内存限制）"},
		{"访问速度", "快（缓存友好）", "可能较慢（缓存不友好）"},
		{"共享性", "goroutine 私有", "所有 goroutine 共享"},
	}

	for _, c := range comparison {
		fmt.Printf("  %-10s | 栈: %-25s | 堆: %s\n", c.aspect, c.stack, c.heap)
	}
}
```

```go
func buildStringBad() string {

	for i := 0; i < 100; i++ {

	}
	return s

func buildStringGood() string {
	var b strings.Builder
	b.Grow(400) // 预估大小

		b.WriteString(strconv.Itoa(i))
		b.WriteByte(',')
	}
```

## 8. 垃圾回收（GC）基础

### 8.1 Go GC 概述

```go
package main

import (
	"fmt"
	"runtime"
)

func main() {
	fmt.Println("Go 垃圾回收器 (GC) 基础")
	fmt.Println()

	/*
	Go 使用三色标记-清除（Tri-color Mark-and-Sweep）并发垃圾回收器

	三种颜色：
	- 白色：未被扫描的对象（GC 结束后白色对象会被回收）
	- 灰色：已被发现但子引用未处理的对象
	- 黑色：已被扫描且所有子引用已处理的对象

	GC 特点：
	1. 并发：GC 与应用代码并发运行（STW 时间极短，通常 < 1ms）
	2. 非分代：不像 Java 分新生代/老生代
	3. 非压缩：不移动对象（简化了指针管理）
	4. 低延迟：设计目标是最小化停顿时间

	触发条件：
	1. 堆大小增长到 GOGC 设定的阈值（默认 100%，即堆翻倍）
	2. 距离上次 GC 超过 2 分钟
	3. 手动调用 runtime.GC()
	*/

	// 查看 GC 统计信息
	var stats runtime.MemStats
	runtime.ReadMemStats(&stats)

	fmt.Printf("堆分配总量: %d MB\n", stats.TotalAlloc/1024/1024)
	fmt.Printf("当前堆使用: %d KB\n", stats.HeapAlloc/1024)
	fmt.Printf("堆对象数: %d\n", stats.HeapObjects)
	fmt.Printf("GC 次数: %d\n", stats.NumGC)
	fmt.Printf("系统分配: %d MB\n", stats.Sys/1024/1024)

	// 分配一些内存
	data := make([]byte, 10*1024*1024) // 10 MB
	_ = data
	runtime.ReadMemStats(&stats)
	fmt.Printf("\n分配 10MB 后:\n")
	fmt.Printf("当前堆使用: %d KB\n", stats.HeapAlloc/1024)

	// 手动触发 GC（正常不需要，仅用于演示）
	runtime.GC()
	runtime.ReadMemStats(&stats)
	fmt.Printf("\nGC 后:\n")
	fmt.Printf("当前堆使用: %d KB\n", stats.HeapAlloc/1024)
	fmt.Printf("GC 次数: %d\n", stats.NumGC)
}
```

## 10. 实战场景

### 10.1 使用指针实现链表（理解指针的经典例子）

```go
package main

import "fmt"

// Node 链表节点
type Node struct {
	Value string
	Next  *Node // 指向下一个节点的指针
}

// LinkedList 简单链表
type LinkedList struct {
	Head *Node
	Size int
}

// Push 在头部添加节点
func (l *LinkedList) Push(value string) {
	l.Head = &Node{
		Value: value,
		Next:  l.Head,
	}
	l.Size++
}

// Pop 从头部移除节点
func (l *LinkedList) Pop() (string, bool) {
	if l.Head == nil {
		return "", false
	}
	value := l.Head.Value
	l.Head = l.Head.Next
	l.Size--
	return value, true
}

// Print 打印链表
func (l *LinkedList) Print() {
	current := l.Head
	for current != nil {
		fmt.Printf("[%s]", current.Value)
		if current.Next != nil {
			fmt.Print(" -> ")
		}
		current = current.Next
	}
	fmt.Println()
}

func main() {
	list := &LinkedList{}

	// 添加运维任务
	list.Push("检查磁盘")
	list.Push("重启服务")
	list.Push("更新证书")

	fmt.Printf("任务队列 (共 %d 个):\n", list.Size)
	list.Print() // [更新证书] -> [重启服务] -> [检查磁盘]

	// 执行任务
	if task, ok := list.Pop(); ok {
		fmt.Printf("\n执行任务: %s\n", task)
	}

	fmt.Printf("剩余任务 (共 %d 个):\n", list.Size)
	list.Print() // [重启服务] -> [检查磁盘]
}
```

### 10.2 SRE 场景：连接池的指针使用

```go
package main

import (
	"fmt"
	"sync"
)

// Connection 模拟数据库连接
type Connection struct {
	ID     int
	DSN    string
	InUse  bool
}

// Pool 连接池
type Pool struct {
	mu    sync.Mutex
	conns []*Connection // 使用指针切片——共享连接对象
	max   int
	next  int
}

// NewPool 创建连接池
func NewPool(dsn string, maxSize int) *Pool {
	pool := &Pool{
		conns: make([]*Connection, maxSize),
		max:   maxSize,
	}
	for i := 0; i < maxSize; i++ {
		pool.conns[i] = &Connection{
			ID:  i + 1,
			DSN: dsn,
		}
	}
	return pool
}

// Get 获取一个空闲连接
func (p *Pool) Get() (*Connection, error) {
	p.mu.Lock()
	defer p.mu.Unlock()

	for _, conn := range p.conns {
		if !conn.InUse {
			conn.InUse = true
			return conn, nil // 返回指针——调用方和池共享同一个对象
		}
	}
	return nil, fmt.Errorf("连接池已满（最大 %d）", p.max)
}

// Release 释放连接回池
func (p *Pool) Release(conn *Connection) {
	p.mu.Lock()
	defer p.mu.Unlock()
	conn.InUse = false // 通过指针修改——池中的对象也被修改
}

// Stats 连接池统计
func (p *Pool) Stats() (total, inUse, free int) {
	p.mu.Lock()
	defer p.mu.Unlock()
	total = len(p.conns)
	for _, conn := range p.conns {
		if conn.InUse {
			inUse++
		}
	}
	free = total - inUse
	return
}

func main() {
	pool := NewPool("postgres://localhost:5432/mydb", 3)

	// 获取连接
	conn1, err := pool.Get()
	if err != nil {
		fmt.Println("错误:", err)
		return
	}
	fmt.Printf("获取连接 #%d\n", conn1.ID)

	conn2, _ := pool.Get()
	fmt.Printf("获取连接 #%d\n", conn2.ID)

	// 查看统计
	total, inUse, free := pool.Stats()
	fmt.Printf("总连接: %d, 使用中: %d, 空闲: %d\n", total, inUse, free)

	// 释放连接
	pool.Release(conn1)
	fmt.Printf("释放连接 #%d\n", conn1.ID)

	total, inUse, free = pool.Stats()
	fmt.Printf("总连接: %d, 使用中: %d, 空闲: %d\n", total, inUse, free)

	// 再次获取——会复用刚释放的连接
	conn3, _ := pool.Get()
	fmt.Printf("获取连接 #%d（复用的）\n", conn3.ID)
}
```

## 常见坑点总结

| 坑点 | 说明 | 解决方案 |
|------|------|----------|
| nil 指针解引用 | 对 nil 指针执行 `*p` 会 panic | 使用前检查 `if p != nil` |
| 结构体浅拷贝 | 拷贝包含 slice/map 的结构体会共享底层数据 | 手动深拷贝 slice 和 map |
| 切片 append 不可见 | 函数内 append 不影响外部切片 | 返回新切片或传切片指针 |
| sync.Mutex 拷贝 | 拷贝含 Mutex 的结构体会导致死锁 | 始终使用指针传递含 Mutex 的类型 |
| 循环变量取地址 | 循环变量取地址指向同一个变量 | 在循环内创建副本 |
| 大结构体值传递 | 大结构体值传递性能差 | 使用指针传递大结构体 |
| 结构体对齐浪费 | 字段排列不当导致内存浪费 | 按大小从大到小排列字段 |

## 速查表 / 小结

### 指针操作速查

| 操作 | 语法 | 说明 |
|------|------|------|
| 取地址 | `p := &x` | 获取变量 x 的地址 |
| 解引用 | `v := *p` | 获取指针指向的值 |
| 声明指针 | `var p *int` | 声明一个 int 指针（零值为 nil） |
| 创建指针 | `p := &T{}` 或 `p := new(T)` | 创建并返回指针 |
| 判空 | `if p != nil` | 检查指针是否为 nil |

### new vs make 速查

| 函数 | 适用类型 | 返回值 | 说明 |
|------|----------|--------|------|
| `new(T)` | 任何类型 | `*T` | 分配零值内存，返回指针 |
| `make(T, ...)` | slice/map/chan | `T` | 初始化内部结构，返回值本身 |

### 值类型 vs 引用类型速查

| 类别 | 类型 | 赋值行为 |
|------|------|----------|
| 值类型 | int, float, bool, string, array, struct | 完整拷贝 |
| 引用类型 | slice, map, channel, pointer, function | 共享底层数据 |

### 逃逸分析触发条件

| 条件 | 示例 | 分配位置 |
|------|------|----------|
| 返回局部变量指针 | `return &x` | 堆 |
| 传给接口类型 | `fmt.Println(x)` | 堆 |
| 闭包捕获 | `func() { return x }` | 堆 |
| 发送到 channel | `ch <- &x` | 堆 |
| 切片扩容 | `append(s, ...)` | 可能堆 |
| 局部变量不外逃 | `x := 42; _ = x` | 栈 |

### GC 调优速查

| 参数 | 默认值 | 说明 |
|------|--------|------|
| `GOGC` | 100 | 堆增长百分比触发 GC |
| `GOMEMLIMIT` | 无限制 | Go 运行时软内存限制 |
| `GODEBUG=gctrace=1` | - | 打印 GC 跟踪信息 |

```
# 查看逃逸分析
go build -gcflags="-m" ./...

# 查看 GC 行为
GODEBUG=gctrace=1 ./myapp

# 查看内存使用
go tool pprof http://localhost:6060/debug/pprof/heap
```
