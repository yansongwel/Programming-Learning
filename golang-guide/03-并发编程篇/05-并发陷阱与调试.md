# 05 - 并发陷阱与调试

## 导读

并发编程是 Go 的强项，但也是最容易出错的领域。数据竞争、死锁、goroutine 泄漏等问题往往难以复现，在生产环境中造成严重后果。幸运的是，Go 提供了一系列强大的工具来检测和调试这些问题。

作为 SRE/DevOps 工程师，你需要掌握：
- **数据竞争**（Data Race）的识别与修复
- **Race Detector** 的使用
- **死锁**的检测与避免
- **goroutine 泄漏**的排查
- **channel 使用陷阱**的规避
- **pprof** 查看 goroutine 状态
- **goleak** 工具检测泄漏

这些技能将帮助你构建可靠的并发系统，并在生产环境中快速定位问题。

## 1. 数据竞争（Data Race）

### 1.1 什么是数据竞争

数据竞争发生在两个或多个 goroutine 同时访问同一个变量，且至少有一个是写操作时。

```go
package main

import (
	"fmt"
	"sync"
)

func main() {
	// 经典的数据竞争示例
	fmt.Println("=== 数据竞争演示 ===")
	fmt.Println("请使用 go run -race main.go 运行以检测数据竞争")
	fmt.Println()

	var counter int
	var wg sync.WaitGroup

	for i := 0; i < 1000; i++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			counter++ // 数据竞争！读取-修改-写入不是原子的
		}()
	}

	wg.Wait()
	// 结果是不确定的，可能不是 1000
	fmt.Printf("counter = %d（期望 1000，实际可能不同）\n", counter)
}
```

### 1.2 数据竞争的危害

```go
package main

import (
	"fmt"
	"sync"
)

// 危害1: 读到不一致的数据
type Config struct {
	Host string
	Port int
}

var (
	globalConfig Config
	configMu     sync.Mutex // 正确做法：用锁保护
)

// 危害2: map 并发读写导致 panic
func mapCrash() {
	defer func() {
		if r := recover(); r != nil {
			fmt.Println("捕获到 panic:", r)
		}
	}()

	m := make(map[int]int)
	var wg sync.WaitGroup

	// 并发写
	for i := 0; i < 100; i++ {
		wg.Add(1)
		go func(v int) {
			defer wg.Done()
			m[v] = v // 并发写 map 会导致 fatal error (不是 panic，无法 recover)
		}(i)
	}
	wg.Wait()
}

// 危害3: 接口（interface）的数据竞争
type Animal interface {
	Speak() string
}

type Dog struct{}
func (d Dog) Speak() string { return "Woof" }

type Cat struct{}
func (c Cat) Speak() string { return "Meow" }

func main() {
	fmt.Println("=== 数据竞争的危害 ===")
	fmt.Println()

	// 危害清单
	dangers := []string{
		"1. 变量值不确定：counter++ 不是原子操作",
		"2. 内存损坏：map 并发读写导致 fatal error（程序直接崩溃）",
		"3. 接口竞争：interface 内部有两个 word，竞争可能读到不匹配的类型和值",
		"4. 切片竞争：修改 slice 的 len/cap/ptr 不是原子的",
		"5. 结构体竞争：大于 8 字节的结构体赋值不是原子的",
		"6. 字符串竞争：string 内部有 ptr 和 len，竞争可能读到不匹配的组合",
	}

	for _, d := range dangers {
		fmt.Println(d)
	}

	fmt.Println("\n重要：Go 中 map 的并发读写不是 panic，而是 fatal error")
	fmt.Println("fatal error 无法被 recover 捕获，程序会直接崩溃！")

	// 不要运行 mapCrash()，会导致程序崩溃
	// mapCrash()
}
```

### 1.3 常见数据竞争场景

```go
package main

import (
	"fmt"
	"sync"
	"sync/atomic"
)

func main() {
	fmt.Println("=== 常见数据竞争场景及修复 ===")

	// 场景1: 循环变量捕获（Go < 1.22）
	fmt.Println("\n--- 场景1: 循环变量捕获 ---")
	var wg sync.WaitGroup

	// 有问题的写法（Go < 1.22）
	for i := 0; i < 5; i++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			// 在 Go < 1.22 中，这里的 i 是外部循环变量的引用
			// 当 goroutine 执行时，循环可能已经结束，i 可能是 5
			fmt.Printf("(不安全) i = %d\n", i)
		}()
	}
	wg.Wait()

	// 修复写法
	for i := 0; i < 5; i++ {
		wg.Add(1)
		go func(val int) { // 通过参数传递值的拷贝
			defer wg.Done()
			fmt.Printf("(安全) val = %d\n", val)
		}(i)
	}
	wg.Wait()

	// 场景2: 共享变量读写
	fmt.Println("\n--- 场景2: 共享变量读写 ---")
	// 错误：没有同步
	var balance int64 = 100

	// 修复方案1：使用 Mutex
	var mu sync.Mutex
	safeDeposit := func(amount int64) {
		mu.Lock()
		balance += amount
		mu.Unlock()
	}

	// 修复方案2：使用 atomic
	var atomicBalance atomic.Int64
	atomicBalance.Store(100)
	atomicDeposit := func(amount int64) {
		atomicBalance.Add(amount)
	}

	// 修复方案3：使用 channel
	depositCh := make(chan int64)
	go func() {
		bal := int64(100)
		for amount := range depositCh {
			bal += amount
			fmt.Printf("余额: %d\n", bal)
		}
	}()

	_ = safeDeposit
	_ = atomicDeposit
	depositCh <- 50
	depositCh <- -20
	close(depositCh)

	// 场景3: slice append 竞争
	fmt.Println("\n--- 场景3: slice append 竞争 ---")
	// 错误：并发 append 到同一个 slice
	// var results []int
	// for i := 0; i < 10; i++ {
	//     go func(v int) {
	//         results = append(results, v) // 数据竞争！
	//     }(i)
	// }

	// 修复：使用锁
	var results []int
	var resultMu sync.Mutex
	for i := 0; i < 10; i++ {
		wg.Add(1)
		go func(v int) {
			defer wg.Done()
			resultMu.Lock()
			results = append(results, v)
			resultMu.Unlock()
		}(i)
	}
	wg.Wait()
	fmt.Printf("安全 append 结果: %v\n", results)

	// 或者使用预分配的 slice（每个 goroutine 写不同索引）
	results2 := make([]int, 10)
	for i := 0; i < 10; i++ {
		wg.Add(1)
		go func(idx int) {
			defer wg.Done()
			results2[idx] = idx * idx // 不同索引，无竞争
		}(i)
	}
	wg.Wait()
	fmt.Printf("预分配结果: %v\n", results2)
}
```

## 2. Race Detector 使用

### 2.1 启用 Race Detector

```go
package main

import (
	"fmt"
	"sync"
)

// 运行方式：
//   go run -race main.go
//   go test -race ./...
//   go build -race -o myapp && ./myapp

var shared int

func increment(wg *sync.WaitGroup) {
	defer wg.Done()
	shared++ // 数据竞争
}

func main() {
	var wg sync.WaitGroup

	for i := 0; i < 10; i++ {
		wg.Add(1)
		go increment(&wg)
	}

	wg.Wait()
	fmt.Println("shared =", shared)
}

// Race Detector 输出示例：
// ==================
// WARNING: DATA RACE
// Read at 0x00000064c3d8 by goroutine 8:

// Previous write at 0x00000064c3d8 by goroutine 7:
//   main.increment()

//
// Goroutine 8 (running) created at:
```

### 2.3 常见的 Race Detector 报告解读

```go
package main

import (
	"fmt"
	"sync"
)

// 示例1: 简单的读写竞争
func example1() {
	var data int
	var wg sync.WaitGroup

	wg.Add(2)
	go func() {
		defer wg.Done()
		data = 42 // 写
	}()
	go func() {
		defer wg.Done()
		_ = data // 读 - 与写竞争
	}()
	wg.Wait()
}

// 示例2: map 竞争
func example2() {
	m := make(map[string]int)
	var wg sync.WaitGroup

	wg.Add(2)
	go func() {
		defer wg.Done()
		m["a"] = 1 // 写
	}()
	go func() {
		defer wg.Done()
		m["b"] = 2 // 写 - 即使是不同的 key，map 的写也不是并发安全的
	}()
	wg.Wait()
}

// 示例3: 修复后的版本
func example3() {
	m := make(map[string]int)
	var mu sync.Mutex
	var wg sync.WaitGroup

	wg.Add(2)
	go func() {
		defer wg.Done()
		mu.Lock()
		m["a"] = 1
		mu.Unlock()
	}()
	go func() {
		defer wg.Done()
		mu.Lock()
		m["b"] = 2
		mu.Unlock()
	}()
	wg.Wait()
	fmt.Printf("安全的 map: %v\n", m)
}

func main() {
	fmt.Println("运行 go run -race main.go 查看竞争报告")
	fmt.Println()

	// example1() // 有竞争
	// example2() // 有竞争

	example3() // 安全

	fmt.Println("\n如何解读 Race Detector 报告:")
	fmt.Println("  1. 'Read at ...' - 竞争的读操作位置")
	fmt.Println("  2. 'Previous write at ...' - 竞争的写操作位置")
	fmt.Println("  3. 'Goroutine N created at' - 创建 goroutine 的位置")
	fmt.Println("  4. 每条信息都包含文件名和行号")
	fmt.Println("  5. 根据行号找到具体代码，添加适当的同步")
}
```

## 3. 死锁检测

### 3.1 Go 运行时自动检测

```go
package main

import "fmt"

func main() {
	// Go 运行时能检测到所有 goroutine 都被阻塞的情况
	// 此时会输出 "fatal error: all goroutines are asleep - deadlock!"

	// 死锁场景1: 无缓冲 channel 自发自收
	// ch := make(chan int)
	// ch <- 1   // 阻塞：没有其他 goroutine 接收
	// <-ch      // 永远执行不到

	// 死锁场景2: 互相等待
	// ch1 := make(chan int)
	// ch2 := make(chan int)
	// go func() {
	//     <-ch1    // 等 ch1
	//     ch2 <- 1 // 永远执行不到
	// }()
	// <-ch2        // 等 ch2（但 ch2 的发送依赖 ch1，没人往 ch1 发）
	// ch1 <- 1     // 永远执行不到

	// 死锁场景3: Mutex 重入
	// var mu sync.Mutex
	// mu.Lock()
	// mu.Lock() // 死锁：同一 goroutine 不能重复加锁

	fmt.Println("=== Go 运行时死锁检测 ===")
	fmt.Println()
	fmt.Println("Go 运行时能检测到的死锁条件:")
	fmt.Println("  - 所有 goroutine 都处于等待状态（睡眠/阻塞）")
	fmt.Println("  - 没有任何 goroutine 可以继续执行")
	fmt.Println()
	fmt.Println("Go 运行时不能检测到的情况:")
	fmt.Println("  - 部分 goroutine 死锁（只要有一个 goroutine 在运行就不报错）")
	fmt.Println("  - goroutine 泄漏（goroutine 一直阻塞但不影响其他 goroutine）")
	fmt.Println("  - 活锁（goroutine 在忙循环但没有进展）")
}
```

### 3.2 常见死锁模式

```go
package main

import (
	"fmt"
	"sync"
	"time"
)

// 死锁模式1: 锁顺序不一致
func lockOrderDeadlock() {
	var mu1, mu2 sync.Mutex

	// goroutine 1: 先锁 mu1 再锁 mu2
	go func() {
		mu1.Lock()
		time.Sleep(100 * time.Millisecond) // 增加死锁概率
		mu2.Lock()
		fmt.Println("G1 完成")
		mu2.Unlock()
		mu1.Unlock()
	}()

	// goroutine 2: 先锁 mu2 再锁 mu1（顺序相反 = 死锁风险）
	go func() {
		mu2.Lock()
		time.Sleep(100 * time.Millisecond)
		mu1.Lock()
		fmt.Println("G2 完成")
		mu1.Unlock()
		mu2.Unlock()
	}()
}

// 修复：保持一致的加锁顺序
func lockOrderFixed() {
	var mu1, mu2 sync.Mutex

	// 两个 goroutine 都按 mu1 -> mu2 的顺序加锁
	var wg sync.WaitGroup
	wg.Add(2)

	go func() {
		defer wg.Done()
		mu1.Lock()
		defer mu1.Unlock()
		time.Sleep(100 * time.Millisecond)
		mu2.Lock()
		defer mu2.Unlock()
		fmt.Println("G1 完成（修复后）")
	}()

	go func() {
		defer wg.Done()
		mu1.Lock() // 与 G1 相同的加锁顺序
		defer mu1.Unlock()
		time.Sleep(100 * time.Millisecond)
		mu2.Lock()
		defer mu2.Unlock()
		fmt.Println("G2 完成（修复后）")
	}()

	wg.Wait()
}

// 死锁模式2: channel + 锁 交叉
func channelLockDeadlock() {
	// 不运行，仅展示模式
	/*
		var mu sync.Mutex
		ch := make(chan int)

		go func() {
			mu.Lock()
			ch <- 1    // 等待 channel 被读取
			mu.Unlock()
		}()

		mu.Lock()      // 等待锁（但锁被上面的 goroutine 持有）
		<-ch           // 永远执行不到
		mu.Unlock()
	*/
}

func main() {
	fmt.Println("=== 死锁模式与修复 ===")

	// 不运行 lockOrderDeadlock()，它可能会死锁
	// lockOrderDeadlock()

	lockOrderFixed()

	_ = channelLockDeadlock

	fmt.Println("\n死锁预防规则:")
	fmt.Println("  1. 保持一致的加锁顺序")
	fmt.Println("  2. 避免持有锁时进行 channel 操作")
	fmt.Println("  3. 使用 defer 确保解锁")
	fmt.Println("  4. 优先使用 channel 而非锁来同步")
	fmt.Println("  5. 使用 context 超时避免无限等待")
	fmt.Println("  6. 使用 TryLock（Go 1.18+）避免阻塞")
}
```

### 3.3 使用 TryLock 避免死锁（Go 1.18+）

```go
package main

import (
	"fmt"
	"sync"
	"time"
)

func main() {
	var mu sync.Mutex

	// TryLock: 尝试加锁，如果锁已被持有则立即返回 false
	fmt.Println("=== TryLock (Go 1.18+) ===")

	mu.Lock()
	fmt.Println("锁已被持有")

	// 在另一个 goroutine 中尝试获取锁
	go func() {
		if mu.TryLock() {
			fmt.Println("成功获取锁")
			mu.Unlock()
		} else {
			fmt.Println("锁不可用，跳过（不阻塞）")
		}
	}()

	time.Sleep(100 * time.Millisecond)
	mu.Unlock()
	fmt.Println("锁已释放")

	// RWMutex 也有 TryLock 和 TryRLock
	var rw sync.RWMutex
	rw.Lock()

	go func() {
		if rw.TryRLock() {
			fmt.Println("获取读锁成功")
			rw.RUnlock()
		} else {
			fmt.Println("读锁不可用（有写锁）")
		}
	}()

	time.Sleep(100 * time.Millisecond)
	rw.Unlock()

	time.Sleep(100 * time.Millisecond)

	fmt.Println("\nTryLock 使用场景:")
	fmt.Println("  - 乐观锁定：尝试获取锁，失败则走其他路径")
	fmt.Println("  - 避免死锁：不会无限阻塞")
	fmt.Println("  - 注意：不要用 TryLock 替代正确的锁设计")
}
```

## 4. Goroutine 泄漏

### 4.1 什么是 Goroutine 泄漏

goroutine 泄漏是指 goroutine 启动后，由于某种原因永远无法退出，导致其占用的资源（内存、goroutine 数量）无法被回收。

```go
package main

import (
	"fmt"
	"runtime"
	"time"
)

// 泄漏原因1: channel 操作永远阻塞
func leakByChannel() {
	ch := make(chan int) // 无缓冲 channel
	go func() {
		val := <-ch // 永远等待，因为没有人发送
		fmt.Println(val)
	}()
	// ch 不可达了，但 goroutine 还在等待
}

// 泄漏原因2: 无限循环没有退出条件
func leakByInfiniteLoop() {
	go func() {
		for {
			time.Sleep(time.Second) // 永远运行
		}
	}()
}

// 泄漏原因3: 阻塞在锁上
func leakByLock() {
	var mu = make(chan struct{}, 1) // 用 channel 模拟锁
	mu <- struct{}{}               // 锁被持有
	go func() {
		mu <- struct{}{} // 永远阻塞，因为锁永远不释放
	}()
}

// 泄漏原因4: select 没有退出 case
func leakBySelect() {
	ch := make(chan int)
	go func() {
		select {
		case v := <-ch:
			fmt.Println(v)
			// 没有 default 或 timeout case
			// 如果 ch 永远没有数据，goroutine 永远阻塞
		}
	}()
}

func main() {
	fmt.Printf("初始 goroutine 数: %d\n", runtime.NumGoroutine())

	leakByChannel()
	leakByInfiniteLoop()
	leakByLock()
	leakBySelect()

	time.Sleep(100 * time.Millisecond)
	fmt.Printf("泄漏后 goroutine 数: %d（泄漏了 4 个）\n", runtime.NumGoroutine())

	fmt.Println("\n常见泄漏原因:")
	fmt.Println("  1. 向无缓冲 channel 发送/接收，但没有对应的接收/发送方")
	fmt.Println("  2. 向满的缓冲 channel 发送，没人消费")
	fmt.Println("  3. 无限循环没有退出条件")
	fmt.Println("  4. 等待永远不会释放的锁")
	fmt.Println("  5. select 没有超时或取消 case")
}
```

### 4.2 防止 Goroutine 泄漏

```go
package main

import (
	"context"
	"fmt"
	"runtime"
	"time"
)

// 修复1: 使用 context 控制生命周期
func safeWorker(ctx context.Context, name string) {
	go func() {
		ticker := time.NewTicker(500 * time.Millisecond)
		defer ticker.Stop()

		for {
			select {
			case <-ctx.Done():
				fmt.Printf("[%s] 退出: %v\n", name, ctx.Err())
				return
			case <-ticker.C:
				fmt.Printf("[%s] 工作中\n", name)
			}
		}
	}()
}

// 修复2: 使用 done channel
func safeProducer(done <-chan struct{}) <-chan int {
	ch := make(chan int)
	go func() {
		defer close(ch)
		i := 0
		for {
			select {
			case <-done:
				fmt.Println("Producer 退出")
				return
			case ch <- i:
				i++
			}
		}
	}()
	return ch
}

// 修复3: 使用缓冲 channel 避免 goroutine 阻塞
func safeAsyncTask() <-chan string {
	// 使用缓冲为 1 的 channel，即使没人读，goroutine 也能发完就退出
	ch := make(chan string, 1)
	go func() {
		time.Sleep(100 * time.Millisecond)
		ch <- "任务完成"
		// 即使没人读 ch，goroutine 也不会泄漏（缓冲区可以容纳这个结果）
	}()
	return ch
}

func main() {
	fmt.Printf("初始 goroutine 数: %d\n", runtime.NumGoroutine())

	// 使用 context
	ctx, cancel := context.WithTimeout(context.Background(), 2*time.Second)
	safeWorker(ctx, "Worker-1")
	safeWorker(ctx, "Worker-2")

	// 使用 done channel
	done := make(chan struct{})
	producer := safeProducer(done)

	// 消费一些数据
	for i := 0; i < 5; i++ {
		fmt.Printf("收到: %d\n", <-producer)
	}

	time.Sleep(time.Second)
	fmt.Printf("\n运行中 goroutine 数: %d\n", runtime.NumGoroutine())

	// 清理

	cancel()

	time.Sleep(200 * time.Millisecond)
	fmt.Printf("清理后 goroutine 数: %d\n", runtime.NumGoroutine())
```

## 5. Channel 使用陷阱

### 5.1 向已关闭 channel 发送

```go
package main

import "fmt"

func main() {
	fmt.Println("=== Channel 陷阱1: 向已关闭 channel 发送 ===")

	ch := make(chan int, 1)
	close(ch)

	// 向已关闭 channel 发送会 panic
	func() {
		defer func() {
			if r := recover(); r != nil {
				fmt.Printf("panic: %v\n", r)
			}
		}()
		ch <- 1 // panic: send on closed channel
	}()

	// 安全的发送方式：使用 select + recover
	safeSend := func(ch chan int, value int) (closed bool) {
		defer func() {
			if r := recover(); r != nil {
				closed = true
			}
		}()
		ch <- value
		return false
	}

	ch2 := make(chan int, 1)
	close(ch2)
	if safeSend(ch2, 42) {
		fmt.Println("channel 已关闭，发送被捕获")
	}

	// 更好的方式：使用 sync.Once 确保只关闭一次
	fmt.Println("\n推荐: 只有发送方关闭 channel，接收方不应关闭")
}
```

### 5.2 重复关闭 channel

```go
package main

import (
	"fmt"
	"sync"
)

// SafeChannel 安全的 channel 包装
type SafeChannel struct {
	ch     chan int
	once   sync.Once
	closed bool
	mu     sync.RWMutex
}

func NewSafeChannel(size int) *SafeChannel {
	return &SafeChannel{
		ch: make(chan int, size),
	}
}

// Close 安全关闭（可以调用多次，只生效一次）
func (sc *SafeChannel) Close() {
	sc.once.Do(func() {
		sc.mu.Lock()
		sc.closed = true
		close(sc.ch)
		sc.mu.Unlock()
		fmt.Println("channel 已关闭")
	})
}

// Send 安全发送
func (sc *SafeChannel) Send(value int) bool {
	sc.mu.RLock()
	defer sc.mu.RUnlock()
	if sc.closed {
		return false
	}
	sc.ch <- value
	return true
}

// Receive 获取底层 channel 用于接收
func (sc *SafeChannel) Receive() <-chan int {
	return sc.ch
}

func main() {
	fmt.Println("=== 安全的 Channel 操作 ===")

	sc := NewSafeChannel(5)

	// 安全发送
	sc.Send(1)
	sc.Send(2)
	sc.Send(3)

	// 安全关闭（多次调用不 panic）
	sc.Close()
	sc.Close() // 不会 panic
	sc.Close() // 不会 panic

	// 关闭后安全发送
	ok := sc.Send(4)
	fmt.Printf("关闭后发送: ok=%v\n", ok) // false

	// 读取已有数据
	for v := range sc.Receive() {
		fmt.Printf("收到: %d\n", v)
	}
}
```

### 6.2 编程方式获取 Goroutine 信息

```go
package main

import (
	"fmt"
	"runtime"
	"strings"
	"time"
)

// getGoroutineProfile 获取 goroutine 概要信息
func getGoroutineProfile() {
	// 方式1: runtime.NumGoroutine()
	fmt.Printf("当前 goroutine 数量: %d\n", runtime.NumGoroutine())

	// 方式2: runtime.Stack() 获取所有 goroutine 堆栈
	buf := make([]byte, 64*1024) // 64KB 缓冲区
	n := runtime.Stack(buf, true) // true = 所有 goroutine
	stackStr := string(buf[:n])

	// 解析堆栈
	goroutines := strings.Split(stackStr, "\n\n")
	fmt.Printf("堆栈中的 goroutine 段数: %d\n\n", len(goroutines))

	// 统计各种状态的 goroutine
	states := make(map[string]int)
	for _, g := range goroutines {
		lines := strings.Split(g, "\n")
		if len(lines) > 0 {
			// 第一行格式: "goroutine N [state]:"
			line := lines[0]
			start := strings.Index(line, "[")
			end := strings.Index(line, "]")
			if start != -1 && end != -1 {
				state := line[start+1 : end]
				// 去掉时间部分（如 "chan receive, 5 minutes"）
				if idx := strings.Index(state, ","); idx != -1 {
					state = state[:idx]
				}
				states[state]++
			}
		}
	}

	fmt.Println("Goroutine 状态分布:")
	for state, count := range states {
		fmt.Printf("  %-20s %d\n", state, count)
	}
}

// GoroutineMonitor 周期性监控 goroutine 数量
func GoroutineMonitor(interval time.Duration, threshold int) <-chan string {
	alerts := make(chan string, 10)

	go func() {
		ticker := time.NewTicker(interval)
		defer ticker.Stop()
		defer close(alerts)

		prevCount := runtime.NumGoroutine()

		for range ticker.C {
			current := runtime.NumGoroutine()
			delta := current - prevCount

			if current > threshold {
				alerts <- fmt.Sprintf("警告: goroutine 数量 %d 超过阈值 %d (增长 %+d)",
					current, threshold, delta)
			}

			if delta > 100 { // 短时间内增长超过 100
				alerts <- fmt.Sprintf("警告: goroutine 快速增长 %+d (当前 %d)",
					delta, current)
			}

			prevCount = current
		}
	}()

	return alerts
}

func main() {
	fmt.Println("=== Goroutine 监控 ===\n")

	getGoroutineProfile()

	// 启动监控
	alerts := GoroutineMonitor(time.Second, 20)

	// 制造一些 goroutine
	for i := 0; i < 30; i++ {
		go func() {
			time.Sleep(10 * time.Second)
		}()
	}

	// 读取告警
	timeout := time.After(3 * time.Second)
	for {
		select {
		case alert, ok := <-alerts:
			if !ok {
				return
			}
			fmt.Println(alert)
		case <-timeout:
			fmt.Println("\n最终状态:")
			getGoroutineProfile()
			return
		}
	}
}
```

```go
package main

import (
	"fmt"
	"math/rand"
	"runtime"
	"sync"
	"time"
)

// GoroutineMetrics goroutine 指标收集
type GoroutineMetrics struct {
	mu           sync.Mutex
	history      []int
	maxHistory   int
	alertFunc    func(msg string)
	threshold    int
	growthRate   int // 每秒最大增长数
}

func NewGoroutineMetrics(threshold, growthRate, maxHistory int) *GoroutineMetrics {
	return &GoroutineMetrics{
		maxHistory: maxHistory,
		threshold:  threshold,
		growthRate: growthRate,
		alertFunc: func(msg string) {
			fmt.Printf("[ALERT] %s\n", msg)
		},
	}
}

// Collect 收集当前 goroutine 数量
func (gm *GoroutineMetrics) Collect() {
	gm.mu.Lock()
	defer gm.mu.Unlock()

	current := runtime.NumGoroutine()
	gm.history = append(gm.history, current)

	// 保持历史记录不超过上限
	if len(gm.history) > gm.maxHistory {
		gm.history = gm.history[1:]
	}

	// 检查阈值
	if current > gm.threshold {
		gm.alertFunc(fmt.Sprintf("goroutine 数量 %d 超过阈值 %d", current, gm.threshold))
	}

	// 检查增长率
	if len(gm.history) >= 2 {
		prev := gm.history[len(gm.history)-2]
		growth := current - prev
		if growth > gm.growthRate {
			gm.alertFunc(fmt.Sprintf("goroutine 快速增长: %d -> %d (+%d/s)",
				prev, current, growth))
		}
	}
}

// Stats 获取统计信息
func (gm *GoroutineMetrics) Stats() map[string]int {
	gm.mu.Lock()
	defer gm.mu.Unlock()

	if len(gm.history) == 0 {
		return nil
	}

	min, max, sum := gm.history[0], gm.history[0], 0
	for _, v := range gm.history {
		if v < min {
			min = v
		}
		if v > max {
			max = v
		}
		sum += v
	}

	return map[string]int{
		"current": gm.history[len(gm.history)-1],
		"min":     min,
		"max":     max,
		"avg":     sum / len(gm.history),
		"samples": len(gm.history),
	}
}

func main() {
	metrics := NewGoroutineMetrics(50, 20, 60) // 阈值50, 增长率20/s, 保留60个采样

	// 周期性采集
	go func() {
		ticker := time.NewTicker(time.Second)
		defer ticker.Stop()
		for range ticker.C {
			metrics.Collect()
		}
	}()

	// 模拟正常工作负载
	var wg sync.WaitGroup
	for i := 0; i < 20; i++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			time.Sleep(time.Duration(3+rand.Intn(5)) * time.Second)
		}()
	}

	// 模拟突发负载
	time.Sleep(2 * time.Second)
	fmt.Println("\n模拟突发负载...")
	for i := 0; i < 50; i++ {
		go func() {
			time.Sleep(5 * time.Second)
		}()
	}

	time.Sleep(3 * time.Second)

	// 打印统计
	stats := metrics.Stats()
	fmt.Println("\n=== Goroutine 统计 ===")
	for k, v := range stats {
		fmt.Printf("  %s: %d\n", k, v)
	}

	wg.Wait()
}
```

## 9. 综合实战：并发安全的服务注册中心

```go
package main

import (
	"context"
	"fmt"
	"math/rand"
	"sync"
	"sync/atomic"
	"time"
)

// ServiceStatus 服务状态
type ServiceStatus int

const (
	StatusUnknown ServiceStatus = iota
	StatusHealthy
	StatusUnhealthy
	StatusDraining
)

func (s ServiceStatus) String() string {
	switch s {
	case StatusHealthy:
		return "HEALTHY"
	case StatusUnhealthy:
		return "UNHEALTHY"
	case StatusDraining:
		return "DRAINING"
	default:
		return "UNKNOWN"
	}
}

// ServiceInstance 服务实例
type ServiceInstance struct {
	ID        string
	Name      string
	Address   string
	Port      int
	Status    ServiceStatus
	LastCheck time.Time
}

// Registry 服务注册中心（并发安全）
type Registry struct {
	mu        sync.RWMutex
	services  map[string]map[string]*ServiceInstance // name -> id -> instance
	listeners []chan<- RegistryEvent
	closed    atomic.Bool
}

// RegistryEvent 注册中心事件
type RegistryEvent struct {
	Type     string // "register", "deregister", "status_change"
	Instance ServiceInstance
}

func NewRegistry() *Registry {
	return &Registry{
		services: make(map[string]map[string]*ServiceInstance),
	}
}

// Register 注册服务实例
func (r *Registry) Register(instance ServiceInstance) error {
	if r.closed.Load() {
		return fmt.Errorf("注册中心已关闭")
	}

	r.mu.Lock()
	defer r.mu.Unlock()

	if r.services[instance.Name] == nil {
		r.services[instance.Name] = make(map[string]*ServiceInstance)
	}
	instance.LastCheck = time.Now()
	r.services[instance.Name][instance.ID] = &instance

	// 通知监听者
	r.notifyListeners(RegistryEvent{
		Type:     "register",
		Instance: instance,
	})

	return nil
}

// Deregister 注销服务实例
func (r *Registry) Deregister(name, id string) {
	r.mu.Lock()
	defer r.mu.Unlock()

	if instances, ok := r.services[name]; ok {
		if inst, ok := instances[id]; ok {
			r.notifyListeners(RegistryEvent{
				Type:     "deregister",
				Instance: *inst,
			})
			delete(instances, id)
			if len(instances) == 0 {
				delete(r.services, name)
			}
		}
	}
}

// GetHealthy 获取健康的服务实例
func (r *Registry) GetHealthy(name string) []ServiceInstance {
	r.mu.RLock()
	defer r.mu.RUnlock()

	var healthy []ServiceInstance
	if instances, ok := r.services[name]; ok {
		for _, inst := range instances {
			if inst.Status == StatusHealthy {
				healthy = append(healthy, *inst)
			}
		}
	}
	return healthy
}

// UpdateStatus 更新服务状态
func (r *Registry) UpdateStatus(name, id string, status ServiceStatus) {
	r.mu.Lock()
	defer r.mu.Unlock()

	if instances, ok := r.services[name]; ok {
		if inst, ok := instances[id]; ok {
			oldStatus := inst.Status
			inst.Status = status
			inst.LastCheck = time.Now()

			if oldStatus != status {
				r.notifyListeners(RegistryEvent{
					Type:     "status_change",
					Instance: *inst,
				})
			}
		}
	}
}

// Subscribe 订阅事件
func (r *Registry) Subscribe() <-chan RegistryEvent {
	r.mu.Lock()
	defer r.mu.Unlock()

	ch := make(chan RegistryEvent, 100)
	r.listeners = append(r.listeners, ch)
	return ch
}

// notifyListeners 通知所有监听者（必须在锁内调用）
func (r *Registry) notifyListeners(event RegistryEvent) {
	for _, ch := range r.listeners {
		select {
		case ch <- event:
		default:
			// channel 满了，丢弃事件（避免阻塞）
		}
	}
}

// HealthCheck 启动定期健康检查
func (r *Registry) HealthCheck(ctx context.Context, interval time.Duration) {
	go func() {
		ticker := time.NewTicker(interval)
		defer ticker.Stop()

		for {
			select {
			case <-ctx.Done():
				return
			case <-ticker.C:
				r.mu.RLock()
				var toCheck []struct {
					name string
					id   string
				}
				for name, instances := range r.services {
					for id := range instances {
						toCheck = append(toCheck, struct {
							name string
							id   string
						}{name, id})
					}
				}
				r.mu.RUnlock()

				// 并发检查所有实例
				var wg sync.WaitGroup
				sem := make(chan struct{}, 10) // 限制并发数

				for _, tc := range toCheck {
					wg.Add(1)
					go func(name, id string) {
						defer wg.Done()
						sem <- struct{}{}
						defer func() { <-sem }()

						// 模拟健康检查
						healthy := rand.Float64() > 0.1
						if healthy {
							r.UpdateStatus(name, id, StatusHealthy)
						} else {
							r.UpdateStatus(name, id, StatusUnhealthy)
						}
					}(tc.name, tc.id)
				}
				wg.Wait()
			}
		}
	}()
}

// Stats 获取注册中心统计
func (r *Registry) Stats() map[string]int {
	r.mu.RLock()
	defer r.mu.RUnlock()

	stats := map[string]int{
		"total_services":  len(r.services),
		"total_instances": 0,
		"healthy":         0,
		"unhealthy":       0,
	}

	for _, instances := range r.services {
		stats["total_instances"] += len(instances)
		for _, inst := range instances {
			switch inst.Status {
			case StatusHealthy:
				stats["healthy"]++
			case StatusUnhealthy:
				stats["unhealthy"]++
			}
		}
	}

	return stats
}

func main() {
	registry := NewRegistry()

	// 订阅事件
	events := registry.Subscribe()
	go func() {
		for event := range events {
			if event.Type == "status_change" {
				fmt.Printf("[Event] %s: %s/%s -> %s\n",
					event.Type, event.Instance.Name, event.Instance.ID, event.Instance.Status)
			}
		}
	}()

	// 启动健康检查
	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()
	registry.HealthCheck(ctx, time.Second)

	// 注册一批服务
	services := []struct {
		name    string
		count   int
		portBase int
	}{
		{"web-frontend", 3, 8080},
		{"api-gateway", 2, 9090},
		{"user-service", 3, 9100},
		{"order-service", 2, 9200},
	}

	var wg sync.WaitGroup
	for _, svc := range services {
		for i := 0; i < svc.count; i++ {
			inst := ServiceInstance{
				ID:      fmt.Sprintf("%s-%d", svc.name, i),
				Name:    svc.name,
				Address: fmt.Sprintf("10.0.1.%d", rand.Intn(255)),
				Port:    svc.portBase + i,
				Status:  StatusHealthy,
			}
			wg.Add(1)
			go func(inst ServiceInstance) {
				defer wg.Done()
				if err := registry.Register(inst); err != nil {
					fmt.Printf("注册失败: %v\n", err)
				}
			}(inst)
		}
	}
	wg.Wait()

	// 打印统计
	stats := registry.Stats()
	fmt.Println("\n=== 注册中心统计 ===")
	for k, v := range stats {
		fmt.Printf("  %s: %d\n", k, v)
	}

	// 查询健康实例
	fmt.Println("\n=== 健康的 web-frontend 实例 ===")
	for _, inst := range registry.GetHealthy("web-frontend") {
		fmt.Printf("  %s (%s:%d) - %s\n", inst.ID, inst.Address, inst.Port, inst.Status)
	}

	// 等待一段时间看健康检查效果
	time.Sleep(3 * time.Second)

	fmt.Println("\n=== 健康检查后的统计 ===")
	stats = registry.Stats()
	for k, v := range stats {
		fmt.Printf("  %s: %d\n", k, v)
	}
}
```

### 并发安全检查清单

```
[ ] 共享变量是否有适当的同步保护（Mutex/atomic/channel）
[ ] map 是否有并发读写保护
[ ] slice 是否有并发 append 保护
[ ] 所有 goroutine 是否都有退出机制
[ ] channel 是否由发送方关闭
[ ] context 的 cancel 是否总是被调用
[ ] 锁的加解锁是否配对（使用 defer）
[ ] 是否在锁内做了可能阻塞的操作
[ ] CI 是否运行了 -race 测试
[ ] 是否有 goroutine 数量的监控和告警
```

### 核心原则

| 原则 | 说明 |
|------|------|
| 使用 -race | 开发和测试阶段始终启用 race detector |
| 最小化共享 | 优先使用 channel 通信，减少共享变量 |
| 控制生命周期 | 每个 goroutine 都必须有明确的退出机制 |
| context 传播 | 使用 context 实现取消和超时的级联传播 |
| 监控 goroutine | 在生产环境监控 goroutine 数量和增长趋势 |
| 防御性编程 | 假设并发总会出错，使用工具验证 |
| 先正确后优化 | 先用 Mutex 保证正确性，再考虑用 atomic 优化 |
