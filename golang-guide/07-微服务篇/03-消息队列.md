# 03-消息队列

## 导读

在微服务架构中，并非所有服务间通信都需要同步的请求-响应模式。很多场景下，异步消息传递更加合适：用户注册后发送欢迎邮件、订单创建后扣减库存、监控指标采集后异步处理告警。消息队列（Message Queue）正是解决这些问题的核心组件。

对于 SRE/DevOps 工程师而言，消息队列是日常运维中不可或缺的基础设施。理解其原理和 Go 客户端的使用，能够帮助你构建告警通知系统、日志收集管道、事件驱动的自动化运维工具。

## 一、消息队列核心概念

### 1.1 为什么需要消息队列

```
同步调用的问题：

  用户服务 → 订单服务 → 库存服务 → 物流服务 → 通知服务
  总耗时 = 所有服务耗时之和，任一环节失败整个调用失败

使用消息队列后：

  用户服务 → 订单服务 → [消息队列]
                            ├→ 库存服务（异步消费）
                            ├→ 物流服务（异步消费）
                            └→ 通知服务（异步消费）
  订单服务只需将消息写入队列即可返回，其余服务异步处理
```

三大核心价值：
1. **解耦**：生产者和消费者互不感知，独立演进
2. **削峰**：流量高峰时消息缓存在队列中，消费者按自身能力处理
3. **异步**：耗时操作异步执行，提升系统响应速度

### 1.2 核心术语

| 术语 | 说明 |
|------|------|
| Producer | 消息生产者，发送消息到队列 |
| Consumer | 消息消费者，从队列接收并处理消息 |
| Broker | 消息代理服务器，存储和转发消息 |
| Topic | 主题/话题，消息的逻辑分类（Kafka/NATS） |
| Queue | 队列，消息的存储容器（RabbitMQ） |
| Partition | 分区，Topic 的物理分片（Kafka） |
| Offset | 偏移量，消费者在分区中的位置（Kafka） |
| Exchange | 交换机，消息路由器（RabbitMQ） |
| Consumer Group | 消费者组，组内消费者分摊消费 |

```go
// kafka_demo/producer/main.go
package main

import (
	"context"
	"encoding/json"
	"fmt"
	"log"
	"time"

	"github.com/segmentio/kafka-go"
)

// AlertEvent 告警事件
type AlertEvent struct {
	ID        string    `json:"id"`
	Service   string    `json:"service"`
	Level     string    `json:"level"`     // info, warning, critical
	Message   string    `json:"message"`
	Timestamp time.Time `json:"timestamp"`
}

func main() {
	// ===== 方式一：使用 Writer（推荐，带批量发送和重试） =====
	writer := &kafka.Writer{
		Addr:         kafka.TCP("localhost:9092"),
		Topic:        "alerts",            // 发送到 alerts 主题
		Balancer:     &kafka.LeastBytes{}, // 负载均衡策略：最少字节数
		BatchSize:    100,                 // 批量大小
		BatchTimeout: 10 * time.Millisecond, // 批量超时
		RequiredAcks: kafka.RequireAll,    // 等待所有副本确认（最安全）
		Async:        false,               // 同步发送（Async=true 时为异步）
		// 自动创建 Topic（需要 broker 配置允许）
		AllowAutoTopicCreation: true,
	}
	defer writer.Close()

	ctx := context.Background()

	// 发送单条消息
	alert := AlertEvent{
		ID:        "alert-001",
		Service:   "api-gateway",
		Level:     "critical",
		Message:   "CPU 使用率超过 95%",
		Timestamp: time.Now(),
	}

	data, err := json.Marshal(alert)
	if err != nil {
		log.Fatalf("序列化失败: %v", err)
	}

	err = writer.WriteMessages(ctx, kafka.Message{
		Key:   []byte(alert.Service), // 相同 key 的消息发送到同一个分区（保序）
		Value: data,
		Headers: []kafka.Header{
			{Key: "source", Value: []byte("monitoring-system")},
		},
	})
	if err != nil {
		log.Fatalf("发送失败: %v", err)
	}
	fmt.Println("消息发送成功")

	// 批量发送消息
	messages := make([]kafka.Message, 0, 10)
	for i := 0; i < 10; i++ {
		alert := AlertEvent{
			ID:        fmt.Sprintf("alert-%03d", i),
			Service:   fmt.Sprintf("service-%d", i%3),
			Level:     "warning",
			Message:   fmt.Sprintf("第 %d 条告警消息", i),
			Timestamp: time.Now(),
		}
		data, _ := json.Marshal(alert)
		messages = append(messages, kafka.Message{
			Key:   []byte(alert.Service),
			Value: data,
		})
	}

	err = writer.WriteMessages(ctx, messages...)
	if err != nil {
		log.Fatalf("批量发送失败: %v", err)
	}
	fmt.Printf("批量发送 %d 条消息成功\n", len(messages))

	// 打印 Writer 统计信息
	stats := writer.Stats()
	fmt.Printf("发送统计: 消息数=%d, 字节数=%d, 错误数=%d\n",
		stats.Messages, stats.Bytes, stats.Errors)
}
```

### 2.3 Kafka Consumer

```go
// kafka_demo/consumer/main.go
package main

import (
	"context"
	"encoding/json"
	"fmt"
	"log"
	"os"
	"os/signal"
	"syscall"
	"time"

	"github.com/segmentio/kafka-go"
)

// AlertEvent 告警事件（与 producer 保持一致）
type AlertEvent struct {
	ID        string    `json:"id"`
	Service   string    `json:"service"`
	Level     string    `json:"level"`
	Message   string    `json:"message"`
	Timestamp time.Time `json:"timestamp"`
}

func main() {
	// ===== 方式一：使用 Reader（单消费者） =====
	fmt.Println("===== 单消费者模式 =====")
	singleConsumer()

	// ===== 方式二：使用 Reader + ConsumerGroup =====
	// fmt.Println("===== 消费者组模式 =====")
	// consumerGroup()
}

// singleConsumer 单消费者模式
func singleConsumer() {
	reader := kafka.NewReader(kafka.ReaderConfig{
		Brokers:        []string{"localhost:9092"},
		Topic:          "alerts",
		GroupID:        "alert-processor-group", // 消费者组（同组内的消费者分摊消费）
		MinBytes:       1,                       // 最少拉取字节数
		MaxBytes:       10e6,                    // 最大拉取字节数（10MB）
		MaxWait:        1 * time.Second,         // 最大等待时间
		StartOffset:    kafka.FirstOffset,       // 从最早的消息开始消费
		CommitInterval: time.Second,             // 自动提交偏移量间隔
	})
	defer reader.Close()

	ctx, cancel := context.WithCancel(context.Background())
	defer cancel()

	// 处理退出信号
	go func() {
		quit := make(chan os.Signal, 1)
		signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
		<-quit
		cancel()
	}()

	fmt.Println("开始消费消息...")

	for {
		// FetchMessage 拉取消息但不提交偏移量
		msg, err := reader.FetchMessage(ctx)
		if err != nil {
			if ctx.Err() != nil {
				fmt.Println("消费者退出")
				return
			}
			log.Printf("拉取消息失败: %v", err)
			continue
		}

		// 解析消息
		var alert AlertEvent
		if err := json.Unmarshal(msg.Value, &alert); err != nil {
			log.Printf("反序列化失败: %v", err)
			// 仍然需要提交偏移量，否则会重复消费
			reader.CommitMessages(ctx, msg)
			continue
		}

		// 处理消息
		fmt.Printf("收到告警: [%s] %s - %s (分区=%d, 偏移=%d)\n",
			alert.Level, alert.Service, alert.Message,
			msg.Partition, msg.Offset)

		// 处理成功后手动提交偏移量（确保消息不丢失）
		if err := reader.CommitMessages(ctx, msg); err != nil {
			log.Printf("提交偏移量失败: %v", err)
		}
	}
}

// consumerGroup 消费者组模式
// 多个消费者实例共同消费一个 Topic，每个分区只分配给组内一个消费者
func consumerGroup() {
	reader := kafka.NewReader(kafka.ReaderConfig{
		Brokers: []string{"localhost:9092"},
		Topic:   "alerts",
		GroupID: "alert-processor-group",
		// 消费者组内的多个实例自动分配分区
		// 如果 Topic 有 3 个分区，启动 3 个消费者实例可以做到每个实例消费一个分区
		MinBytes: 1,
		MaxBytes: 10e6,
	})
	defer reader.Close()

	ctx, cancel := context.WithCancel(context.Background())
	defer cancel()

	go func() {
		quit := make(chan os.Signal, 1)
		signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
		<-quit
		cancel()
	}()

	for {
		msg, err := reader.FetchMessage(ctx)
		if err != nil {
			if ctx.Err() != nil {
				return
			}
			log.Printf("拉取失败: %v", err)
			continue
		}

		// 并发处理消息（注意：同一分区的消息需要保持有序处理）
		processMessage(msg)

		// 手动提交
		reader.CommitMessages(ctx, msg)
	}
}

func processMessage(msg kafka.Message) {
	var alert AlertEvent
	json.Unmarshal(msg.Value, &alert)
	fmt.Printf("处理告警: [%s] %s (P%d:O%d)\n",
		alert.Level, alert.Message, msg.Partition, msg.Offset)
}
```

### 2.4 Kafka Topic 管理

```go
// kafka_demo/admin/main.go
package main

import (
	"fmt"
	"log"
	"net"
	"strconv"

	"github.com/segmentio/kafka-go"
)

func main() {
	// 连接到 Kafka broker
	conn, err := kafka.Dial("tcp", "localhost:9092")
	if err != nil {
		log.Fatalf("连接失败: %v", err)
	}
	defer conn.Close()

	// 获取 Controller 节点
	controller, err := conn.Controller()
	if err != nil {
		log.Fatalf("获取 Controller 失败: %v", err)
	}
	controllerConn, err := kafka.Dial("tcp", net.JoinHostPort(controller.Host, strconv.Itoa(controller.Port)))
	if err != nil {
		log.Fatalf("连接 Controller 失败: %v", err)
	}
	defer controllerConn.Close()

	// 创建 Topic
	err = controllerConn.CreateTopics(kafka.TopicConfig{
		Topic:             "alerts",
		NumPartitions:     3,  // 3 个分区（影响并行消费能力）
		ReplicationFactor: 1,  // 副本数（生产环境建议 3）
	})
	if err != nil {
		log.Printf("创建 Topic 失败: %v", err)
	} else {
		fmt.Println("Topic 'alerts' 创建成功")
	}

	// 列出所有 Topic
	partitions, err := conn.ReadPartitions()
	if err != nil {
		log.Fatalf("读取分区信息失败: %v", err)
	}

	topics := make(map[string]int)
	for _, p := range partitions {
		topics[p.Topic]++
	}

	fmt.Println("\n现有 Topic:")
	for topic, partCount := range topics {
		fmt.Printf("  %s (分区数: %d)\n", topic, partCount)
	}
}
```

## 三、RabbitMQ Go 客户端

### 3.1 RabbitMQ 概述

RabbitMQ 是基于 AMQP 协议的消息代理，以灵活的路由机制、可靠的消息传递和丰富的功能著称。适用于任务队列、请求分发、微服务通信等场景。

```bash
# Docker 启动 RabbitMQ（含管理界面）
docker run -d \
  --name rabbitmq \
  -p 5672:5672 \
  -p 15672:15672 \
  rabbitmq:3-management

# 管理界面：http://localhost:15672
# 默认账号：guest / guest
```

```bash
# 安装 Go RabbitMQ 客户端
go get github.com/rabbitmq/amqp091-go
```

### 3.2 RabbitMQ 核心模型

```
Producer → Exchange → Binding → Queue → Consumer

Exchange 类型：
  - direct  : 精确路由（routing key 完全匹配）
  - fanout  : 广播（忽略 routing key，发送到所有绑定队列）
  - topic   : 模式匹配路由（routing key 支持通配符 * 和 #）
  - headers : 基于消息头匹配
```

### 3.3 RabbitMQ Producer

```go
// rabbitmq_demo/producer/main.go
package main

import (
	"context"
	"encoding/json"
	"fmt"
	"log"
	"time"

	amqp "github.com/rabbitmq/amqp091-go"
)

// AlertEvent 告警事件
type AlertEvent struct {
	ID        string    `json:"id"`
	Service   string    `json:"service"`
	Level     string    `json:"level"`
	Message   string    `json:"message"`
	Timestamp time.Time `json:"timestamp"`
}

// RabbitMQProducer RabbitMQ 生产者
type RabbitMQProducer struct {
	conn    *amqp.Connection
	channel *amqp.Channel
}

// NewRabbitMQProducer 创建生产者
func NewRabbitMQProducer(url string) (*RabbitMQProducer, error) {
	// 建立连接
	conn, err := amqp.Dial(url)
	if err != nil {
		return nil, fmt.Errorf("连接 RabbitMQ 失败: %w", err)
	}

	// 创建 Channel（轻量级连接，实际操作都在 Channel 上）
	ch, err := conn.Channel()
	if err != nil {
		conn.Close()
		return nil, fmt.Errorf("创建 Channel 失败: %w", err)
	}

	// 开启发布确认模式（Publisher Confirms）
	// 确保消息成功到达 broker
	if err := ch.Confirm(false); err != nil {
		ch.Close()
		conn.Close()
		return nil, fmt.Errorf("开启发布确认失败: %w", err)
	}

	return &RabbitMQProducer{
		conn:    conn,
		channel: ch,
	}, nil
}

// SetupExchangeAndQueue 声明交换机和队列
func (p *RabbitMQProducer) SetupExchangeAndQueue() error {
	// 声明 Topic 类型的交换机
	err := p.channel.ExchangeDeclare(
		"alerts.topic", // 交换机名称
		"topic",        // 类型：topic（支持通配符路由）
		true,           // durable：交换机持久化
		false,          // autoDelete
		false,          // internal
		false,          // noWait
		nil,            // 额外参数
	)
	if err != nil {
		return fmt.Errorf("声明交换机失败: %w", err)
	}

	// 声明队列
	queues := []struct {
		name       string
		routingKey string
	}{
		{"alerts.critical", "alert.critical.*"}, // 匹配所有 critical 告警
		{"alerts.all", "alert.#"},               // 匹配所有告警（# 匹配零个或多个词）
		{"alerts.api", "alert.*.api-gateway"},   // 匹配 api-gateway 的所有级别告警
	}

	for _, q := range queues {
		// 声明队列
		_, err := p.channel.QueueDeclare(
			q.name, // 队列名称
			true,   // durable：队列持久化（broker 重启后队列仍存在）
			false,  // autoDelete
			false,  // exclusive
			false,  // noWait
			amqp.Table{
				"x-message-ttl": int32(86400000), // 消息 TTL：24小时（毫秒）
				"x-max-length":  int32(100000),   // 队列最大消息数
			},
		)
		if err != nil {
			return fmt.Errorf("声明队列 %s 失败: %w", q.name, err)
		}

		// 绑定队列到交换机
		err = p.channel.QueueBind(
			q.name,         // 队列名
			q.routingKey,   // 路由键
			"alerts.topic", // 交换机名
			false,
			nil,
		)
		if err != nil {
			return fmt.Errorf("绑定队列 %s 失败: %w", q.name, err)
		}
	}

	return nil
}

// Publish 发布告警消息
func (p *RabbitMQProducer) Publish(ctx context.Context, alert *AlertEvent) error {
	data, err := json.Marshal(alert)
	if err != nil {
		return fmt.Errorf("序列化失败: %w", err)
	}

	// 构建 routing key: alert.{level}.{service}
	routingKey := fmt.Sprintf("alert.%s.%s", alert.Level, alert.Service)

	// 获取确认通道
	confirms := p.channel.NotifyPublish(make(chan amqp.Confirmation, 1))

	// 发布消息
	err = p.channel.PublishWithContext(ctx,
		"alerts.topic", // 交换机
		routingKey,     // 路由键
		true,           // mandatory：true 表示如果消息无法路由到队列则返回给生产者
		false,          // immediate（RabbitMQ 3.x 已废弃）
		amqp.Publishing{
			DeliveryMode: amqp.Persistent,    // 消息持久化
			ContentType:  "application/json",
			Body:         data,
			MessageId:    alert.ID,
			Timestamp:    alert.Timestamp,
			Headers: amqp.Table{
				"source": "monitoring-system",
				"level":  alert.Level,
			},
		},
	)
	if err != nil {
		return fmt.Errorf("发布失败: %w", err)
	}

	// 等待 broker 确认
	select {
	case confirm := <-confirms:
		if !confirm.Ack {
			return fmt.Errorf("消息被 broker 拒绝: deliveryTag=%d", confirm.DeliveryTag)
		}
		return nil
	case <-ctx.Done():
		return ctx.Err()
	}
}

// Close 关闭连接
func (p *RabbitMQProducer) Close() {
	if p.channel != nil {
		p.channel.Close()
	}
	if p.conn != nil {
		p.conn.Close()
	}
}

func main() {
	producer, err := NewRabbitMQProducer("amqp://guest:guest@localhost:5672/")
	if err != nil {
		log.Fatal(err)
	}
	defer producer.Close()

	// 设置交换机和队列
	if err := producer.SetupExchangeAndQueue(); err != nil {
		log.Fatal(err)
	}
	fmt.Println("交换机和队列设置完成")

	ctx := context.Background()

	// 发送告警
	alerts := []AlertEvent{
		{ID: "a001", Service: "api-gateway", Level: "critical", Message: "服务不可用", Timestamp: time.Now()},
		{ID: "a002", Service: "user-service", Level: "warning", Message: "响应时间增加", Timestamp: time.Now()},
		{ID: "a003", Service: "api-gateway", Level: "info", Message: "部署完成", Timestamp: time.Now()},
		{ID: "a004", Service: "db-service", Level: "critical", Message: "连接池耗尽", Timestamp: time.Now()},
	}

	for _, alert := range alerts {
		if err := producer.Publish(ctx, &alert); err != nil {
			log.Printf("发送失败: %v", err)
			continue
		}
		fmt.Printf("已发送: [%s] %s - %s (routing: alert.%s.%s)\n",
			alert.Level, alert.Service, alert.Message, alert.Level, alert.Service)
	}
}
```

### 3.4 RabbitMQ Consumer

```go
// rabbitmq_demo/consumer/main.go
package main

import (
	"encoding/json"
	"fmt"
	"log"
	"os"
	"os/signal"
	"syscall"
	"time"

	amqp "github.com/rabbitmq/amqp091-go"
)

// AlertEvent 告警事件
type AlertEvent struct {
	ID        string    `json:"id"`
	Service   string    `json:"service"`
	Level     string    `json:"level"`
	Message   string    `json:"message"`
	Timestamp time.Time `json:"timestamp"`
}

// RabbitMQConsumer RabbitMQ 消费者
type RabbitMQConsumer struct {
	conn    *amqp.Connection
	channel *amqp.Channel
}

// NewRabbitMQConsumer 创建消费者
func NewRabbitMQConsumer(url string) (*RabbitMQConsumer, error) {
	conn, err := amqp.Dial(url)
	if err != nil {
		return nil, fmt.Errorf("连接失败: %w", err)
	}

	ch, err := conn.Channel()
	if err != nil {
		conn.Close()
		return nil, fmt.Errorf("创建 Channel 失败: %w", err)
	}

	// 设置 QoS：预取数量
	// 控制消费者同时处理的消息数，防止消费者被压垮
	err = ch.Qos(
		10,    // prefetchCount：每次最多取 10 条未确认消息
		0,     // prefetchSize：不限制字节数
		false, // global：false 表示仅对当前 Channel 生效
	)
	if err != nil {
		ch.Close()
		conn.Close()
		return nil, fmt.Errorf("设置 QoS 失败: %w", err)
	}

	return &RabbitMQConsumer{
		conn:    conn,
		channel: ch,
	}, nil
}

// Consume 消费指定队列的消息
func (c *RabbitMQConsumer) Consume(queueName string, handler func(*AlertEvent) error) error {
	// 开始消费
	deliveries, err := c.channel.Consume(
		queueName, // 队列名
		"",        // 消费者标签（空字符串会自动生成）
		false,     // autoAck：false 表示手动确认（重要！保证消息不丢失）
		false,     // exclusive
		false,     // noLocal
		false,     // noWait
		nil,       // 额外参数
	)
	if err != nil {
		return fmt.Errorf("启动消费失败: %w", err)
	}

	fmt.Printf("开始消费队列: %s\n", queueName)

	for delivery := range deliveries {
		// 反序列化消息
		var alert AlertEvent
		if err := json.Unmarshal(delivery.Body, &alert); err != nil {
			log.Printf("反序列化失败: %v", err)
			// 消息格式错误，拒绝并不重新入队（避免无限循环）
			delivery.Nack(false, false) // multiple=false, requeue=false
			continue
		}

		// 处理消息
		if err := handler(&alert); err != nil {
			log.Printf("处理失败: %v", err)
			// 处理失败，根据策略决定是否重新入队
			// requeue=true: 重新入队（可能导致重复处理，需要幂等）
			// requeue=false: 进入死信队列（如果配置了的话）
			delivery.Nack(false, true) // requeue=true
			continue
		}

		// 处理成功，确认消息
		delivery.Ack(false) // multiple=false 只确认当前消息
	}

	return nil
}

// Close 关闭连接
func (c *RabbitMQConsumer) Close() {
	if c.channel != nil {
		c.channel.Close()
	}
	if c.conn != nil {
		c.conn.Close()
	}
}
```

```go
// rabbitmq_demo/reconnect/reconnect.go
package main

import (
	"fmt"
	"log"
	"sync"
	"time"

	amqp "github.com/rabbitmq/amqp091-go"
)

// ReconnectableConsumer 支持断线重连的消费者
type ReconnectableConsumer struct {
	url     string
	conn    *amqp.Connection
	channel *amqp.Channel
	mu      sync.Mutex
	done    chan struct{}
}

// NewReconnectableConsumer 创建可重连的消费者
func NewReconnectableConsumer(url string) *ReconnectableConsumer {
	return &ReconnectableConsumer{
		url:  url,
		done: make(chan struct{}),
	}
}

// connect 建立连接
func (c *ReconnectableConsumer) connect() error {
	c.mu.Lock()
	defer c.mu.Unlock()

	var err error
	c.conn, err = amqp.Dial(c.url)
	if err != nil {
		return fmt.Errorf("连接失败: %w", err)
	}

	c.channel, err = c.conn.Channel()
	if err != nil {
		c.conn.Close()
		return fmt.Errorf("创建 Channel 失败: %w", err)
	}

	// 监听连接关闭事件
	closeChan := c.conn.NotifyClose(make(chan *amqp.Error, 1))
	go func() {
		amqpErr := <-closeChan
		if amqpErr != nil {
			log.Printf("连接断开: %v，准备重连...", amqpErr)
			c.reconnect()
		}
	}()

	log.Println("RabbitMQ 连接建立成功")
	return nil
}

// reconnect 重连逻辑（指数退避）
func (c *ReconnectableConsumer) reconnect() {
	backoff := time.Second

	for {
		select {
		case <-c.done:
			return
		default:
		}

		log.Printf("尝试重连（等待 %v）...", backoff)
		time.Sleep(backoff)

		if err := c.connect(); err != nil {
			log.Printf("重连失败: %v", err)
			// 指数退避，最长 30 秒
			backoff *= 2
			if backoff > 30*time.Second {
				backoff = 30 * time.Second
			}
			continue
		}

		log.Println("重连成功")
		return
	}
}

// Close 关闭
func (c *ReconnectableConsumer) Close() {
	close(c.done)
	c.mu.Lock()
	defer c.mu.Unlock()
	if c.channel != nil {
		c.channel.Close()
	}
	if c.conn != nil {
		c.conn.Close()
	}
}
```

```go
	// 所有订阅者都会收到消息（扇出模式）

	sub, err := nc.Subscribe("alerts.>", func(msg *nats.Msg) {

	time.Sleep(100 * time.Millisecond) // 等待消息处理

	// ===== 队列组（Queue Group） =====
	// 同一个队列组内的订阅者只有一个会收到消息（负载均衡）

	fmt.Println("\n===== 队列组演示 =====")
	for i := 0; i < 3; i++ {
		workerID := i

		})
	}

	for i := 0; i < 6; i++ {
		nc.Publish("tasks", []byte(fmt.Sprintf("任务 #%d", i)))
	}
	nc.Flush()

	time.Sleep(100 * time.Millisecond)

	// 类似同步 RPC 调用

	fmt.Println("\n===== 请求/响应演示 =====")

	if err != nil {
		log.Printf("请求超时: %v", err)
	} else {
		fmt.Printf("[客户端] 健康检查响应: %s\n", string(resp.Data))
	}

	// * 匹配单个 token: alerts.* 匹配 alerts.critical 但不匹配 alerts.critical.api
	// > 匹配一个或多个 token: alerts.> 匹配 alerts.critical 和 alerts.critical.api
```

### 4.3 NATS JetStream（持久化）

```go
// nats_demo/jetstream/main.go
package main

import (
	"context"
	"fmt"
	"log"
	"time"

	"github.com/nats-io/nats.go"
	"github.com/nats-io/nats.go/jetstream"
)

func main() {
	// 连接 NATS
	nc, err := nats.Connect(nats.DefaultURL)
	if err != nil {
		log.Fatal(err)
	}
	defer nc.Close()

	// 创建 JetStream 上下文
	js, err := jetstream.New(nc)
	if err != nil {
		log.Fatalf("创建 JetStream 失败: %v", err)
	}

	ctx := context.Background()

	// ===== 创建 Stream（类似 Kafka 的 Topic） =====
	stream, err := js.CreateOrUpdateStream(ctx, jetstream.StreamConfig{
		Name:        "ALERTS",                     // Stream 名称
		Subjects:    []string{"alerts.>"},          // 匹配的主题
		Storage:     jetstream.FileStorage,         // 存储方式：文件（持久化）
		Retention:   jetstream.LimitsPolicy,        // 保留策略：基于限制
		MaxAge:      24 * time.Hour,                // 消息最大保留时间
		MaxMsgs:     100000,                        // 最大消息数
		MaxBytes:    1024 * 1024 * 1024,            // 最大存储空间 1GB
		Replicas:    1,                             // 副本数（生产环境建议 3）
		Duplicates:  5 * time.Minute,               // 去重窗口
	})
	if err != nil {
		log.Fatalf("创建 Stream 失败: %v", err)
	}
	fmt.Printf("Stream 创建成功: %s (主题: %v)\n", stream.CachedInfo().Config.Name, stream.CachedInfo().Config.Subjects)

	// ===== 发布消息 =====
	for i := 0; i < 5; i++ {
		ack, err := js.Publish(ctx, "alerts.critical", []byte(fmt.Sprintf(`{"id":"js-%d","message":"告警 %d"}`, i, i)),
			jetstream.WithMsgID(fmt.Sprintf("msg-%d", i)), // 消息 ID（用于去重）
		)
		if err != nil {
			log.Printf("发布失败: %v", err)
			continue
		}
		fmt.Printf("发布成功: seq=%d, stream=%s\n", ack.Sequence, ack.Stream)
	}

	// ===== 创建 Consumer =====
	consumer, err := js.CreateOrUpdateConsumer(ctx, "ALERTS", jetstream.ConsumerConfig{
		Name:          "alert-processor",             // 消费者名称
		Durable:       "alert-processor",             // 持久化消费者（重启后恢复消费位置）
		FilterSubject: "alerts.>",                    // 过滤主题
		AckPolicy:     jetstream.AckExplicitPolicy,   // 需要显式确认
		AckWait:       30 * time.Second,              // 确认超时（超时后重新投递）
		MaxDeliver:    3,                             // 最大重试次数
		DeliverPolicy: jetstream.DeliverAllPolicy,    // 从最早消息开始
	})
	if err != nil {
		log.Fatalf("创建 Consumer 失败: %v", err)
	}
	fmt.Printf("Consumer 创建成功: %s\n", consumer.CachedInfo().Name)

	// ===== 消费消息（拉取模式） =====
	fmt.Println("\n开始消费消息...")

	// 批量拉取
	msgBatch, err := consumer.Fetch(10, jetstream.FetchMaxWait(5*time.Second))
	if err != nil {
		log.Fatalf("拉取消息失败: %v", err)
	}

	for msg := range msgBatch.Messages() {
		fmt.Printf("收到消息: subject=%s, data=%s\n", msg.Subject(), string(msg.Data()))

		// 确认消息
		if err := msg.Ack(); err != nil {
			log.Printf("确认失败: %v", err)
		}
	}

	if msgBatch.Error() != nil {
		log.Printf("批量拉取错误: %v", msgBatch.Error())
	}

	// ===== 消费消息（持续消费模式） =====
	fmt.Println("\n启动持续消费...")

	consumeCtx, consumeCancel := context.WithTimeout(ctx, 10*time.Second)
	defer consumeCancel()

	cc, err := consumer.Consume(func(msg jetstream.Msg) {
		fmt.Printf("[持续消费] subject=%s, data=%s\n", msg.Subject(), string(msg.Data()))
		msg.Ack()
	})
	if err != nil {
		log.Fatalf("启动持续消费失败: %v", err)
	}
	defer cc.Stop()

	// 发布更多消息
	go func() {
		time.Sleep(time.Second)
		for i := 5; i < 8; i++ {
			js.Publish(ctx, "alerts.warning", []byte(fmt.Sprintf(`{"id":"js-%d","message":"新告警 %d"}`, i, i)))
		}
	}()

	<-consumeCtx.Done()
	fmt.Println("消费结束")

	// ===== 查看 Stream 信息 =====
	info, _ := stream.Info(ctx)
	fmt.Printf("\nStream 统计: 消息数=%d, 字节数=%d\n", info.State.Msgs, info.State.Bytes)
}
```

## 五、消息队列选型对比

| 特性 | Kafka | RabbitMQ | NATS (JetStream) |
|------|-------|----------|-------------------|
| **定位** | 分布式流平台 | 消息代理 | 轻量消息系统 |
| **协议** | 自有协议 | AMQP | 自有协议 |
| **消息模型** | 日志追加 (append-only) | 队列+交换机路由 | 主题发布/订阅 |
| **吞吐量** | 极高（百万/秒） | 高（万级/秒） | 极高（百万/秒） |
| **延迟** | 毫秒级 | 微秒~毫秒级 | 微秒级 |
| **持久化** | 默认持久化 | 可选 | JetStream 可选 |
| **消息顺序** | 分区内有序 | 队列内有序 | Stream 内有序 |
| **消费模式** | Pull（拉取） | Push（推送） | Pull + Push |
| **消息回溯** | 支持（按偏移量） | 不支持 | 支持 |
| **死信队列** | 无内置 | 内置 DLX | 有 |
| **运维复杂度** | 高 | 中 | 低 |
| **Go 客户端** | segmentio/kafka-go | amqp091-go | nats.go |
| **适用场景** | 日志/事件流/大数据 | 任务队列/路由分发 | 微服务通信/IoT |

### 选型建议

```
日志收集 / 事件溯源 / 大数据管道 → Kafka
  - 需要持久化和回溯
  - 吞吐量要求高
  - 可接受较高的运维成本

任务分发 / 复杂路由 / 传统微服务 → RabbitMQ
  - 需要灵活的路由规则
  - 需要优先级队列、延迟队列
  - 消息可靠性要求高

云原生微服务 / 轻量级事件总线 → NATS
  - 追求极简部署和低延迟
  - 云原生环境（NATS 是 CNCF 项目）
  - 不需要太复杂的路由
```

```go
// InMemoryQueue 基于 channel 的内存消息队列
type InMemoryQueue struct {
	topics map[string][]chan []byte
	mu     sync.RWMutex
}

func NewInMemoryQueue() *InMemoryQueue {
	return &InMemoryQueue{
		topics: make(map[string][]chan []byte),
	}
}

func (q *InMemoryQueue) Publish(ctx context.Context, topic string, data []byte) error {
	q.mu.RLock()
	defer q.mu.RUnlock()

	subscribers, ok := q.topics[topic]
	if !ok {
		return nil // 没有订阅者，消息丢弃
	}

	for _, ch := range subscribers {
		select {
		case ch <- data:
		default:
			log.Printf("消息队列: 订阅者处理不过来，消息丢弃")
		}
	}
	return nil
}

func (q *InMemoryQueue) Subscribe(topic string, handler func(data []byte) error) error {
	ch := make(chan []byte, 100)

	q.mu.Lock()
	q.topics[topic] = append(q.topics[topic], ch)
	q.mu.Unlock()

	go func() {
		for data := range ch {
			if err := handler(data); err != nil {
				log.Printf("消息处理失败: %v", err)
			}
		}
	}()

	return nil
}

func (q *InMemoryQueue) Close() error {
	q.mu.Lock()
	defer q.mu.Unlock()
	for _, subscribers := range q.topics {
		for _, ch := range subscribers {
			close(ch)
		}
	}
	return nil

	log.Printf("[邮件通知] [%s] %s: %s (收件人: %v)", alert.Level, alert.Service, alert.Message, n.to)
	return nil
}

	return ChannelEmail

// ==================== 告警通知系统 ====================

// AlertNotificationSystem 告警通知系统
type AlertNotificationSystem struct {
	queue     MessageQueue
	notifiers map[NotifyChannel]Notifier
	rules     []NotifyRule
	mu        sync.RWMutex

	// 告警去重/抑制
	alertCache   map[string]time.Time // alertKey -> 最后通知时间
	cacheMu      sync.Mutex
	silencePeriod time.Duration // 静默期（同一告警在此期间不重复通知）
}

// NewAlertNotificationSystem 创建告警通知系统
func NewAlertNotificationSystem(queue MessageQueue) *AlertNotificationSystem {
	return &AlertNotificationSystem{
		queue:         queue,
		notifiers:     make(map[NotifyChannel]Notifier),
		alertCache:    make(map[string]time.Time),
		silencePeriod: 5 * time.Minute, // 5 分钟静默期
	}
}

// RegisterNotifier 注册通知器
func (s *AlertNotificationSystem) RegisterNotifier(notifier Notifier) {
	s.mu.Lock()
	defer s.mu.Unlock()
	s.notifiers[notifier.Channel()] = notifier
}

// SetRules 设置通知规则
func (s *AlertNotificationSystem) SetRules(rules []NotifyRule) {
	s.mu.Lock()
	defer s.mu.Unlock()
	s.rules = rules
}

// ReceiveAlert 接收告警（HTTP 接口）
func (s *AlertNotificationSystem) ReceiveAlert(ctx context.Context, alert *Alert) error {
	// 生成告警唯一键（用于去重）
	alertKey := fmt.Sprintf("%s:%s:%s", alert.Service, alert.Level, alert.Message)

	// 检查是否在静默期内
	s.cacheMu.Lock()
	if lastTime, ok := s.alertCache[alertKey]; ok {
		if time.Since(lastTime) < s.silencePeriod {
			s.cacheMu.Unlock()
			log.Printf("告警抑制（静默期内）: %s", alertKey)
			return nil
		}
	}
	s.alertCache[alertKey] = time.Now()
	s.cacheMu.Unlock()

	// 序列化并发送到消息队列
	data, err := json.Marshal(alert)
	if err != nil {
		return fmt.Errorf("序列化告警失败: %w", err)
	}

	topic := fmt.Sprintf("alerts.%s", alert.Level)
	return s.queue.Publish(ctx, topic, data)
}

// StartProcessing 启动告警处理
func (s *AlertNotificationSystem) StartProcessing() error {
	// 订阅各级别告警
	levels := []AlertLevel{AlertInfo, AlertWarning, AlertCritical}

	for _, level := range levels {
		topic := fmt.Sprintf("alerts.%s", level)
		currentLevel := level

		err := s.queue.Subscribe(topic, func(data []byte) error {
			var alert Alert
			if err := json.Unmarshal(data, &alert); err != nil {
				return fmt.Errorf("反序列化告警失败: %w", err)
			}

			return s.processAlert(context.Background(), &alert, currentLevel)
		})
		if err != nil {
			return fmt.Errorf("订阅 %s 失败: %w", topic, err)
		}
		log.Printf("已订阅告警主题: %s", topic)
	}

	return nil
}

// processAlert 处理告警：根据规则分发到对应渠道
func (s *AlertNotificationSystem) processAlert(ctx context.Context, alert *Alert, level AlertLevel) error {
	s.mu.RLock()
	defer s.mu.RUnlock()

	// 查找匹配的规则
	for _, rule := range s.rules {
		if rule.Level == level {
			for _, channel := range rule.Channels {
				notifier, ok := s.notifiers[channel]
				if !ok {
					log.Printf("未找到通知渠道: %s", channel)
					continue
				}

				// 异步发送通知
				go func(n Notifier) {
					if err := n.Send(ctx, alert); err != nil {
						log.Printf("发送通知失败 (渠道=%s): %v", n.Channel(), err)
					}
				}(notifier)
			}
		}
	}

	return nil
}
```

## 七、常见坑点

### 坑点 1：Kafka 消费者不提交偏移量导致重复消费

```go
// 错误：使用 ReadMessage（自动提交）后又手动提交
msg, _ := reader.ReadMessage(ctx) // 已经自动提交了偏移量
reader.CommitMessages(ctx, msg)    // 多余的操作

// 正确方式一：使用 FetchMessage + 手动提交（推荐）
msg, _ := reader.FetchMessage(ctx)  // 不自动提交
// 处理消息...
reader.CommitMessages(ctx, msg)     // 处理成功后提交

// 正确方式二：使用 ReadMessage 自动提交（适用于允许少量丢失的场景）
msg, _ := reader.ReadMessage(ctx)   // 自动提交偏移量
// 处理消息（如果处理失败，消息已被跳过）
```

### 坑点 2：RabbitMQ autoAck=true 导致消息丢失

```go
// 错误：自动确认，消费者处理失败时消息直接丢失
deliveries, _ := ch.Consume(queue, "", true, ...) // autoAck=true
for d := range deliveries {
    process(d) // 如果 process 崩溃，消息已经被确认，无法重试
}

// 正确：手动确认
deliveries, _ := ch.Consume(queue, "", false, ...) // autoAck=false
for d := range deliveries {
    if err := process(d); err != nil {
        d.Nack(false, true) // 处理失败，重新入队
    } else {
        d.Ack(false) // 处理成功，确认
    }
}
```

### 坑点 3：不处理消息队列连接断开

```go
// 错误：假设连接永远不会断
conn, _ := amqp.Dial(url)
ch, _ := conn.Channel()
// 用到天荒地老...

// 正确：监听连接关闭事件并重连
closeChan := conn.NotifyClose(make(chan *amqp.Error, 1))
go func() {
    err := <-closeChan
    log.Printf("连接断开: %v", err)
    reconnect() // 实现重连逻辑
}()
```

### 坑点 4：Kafka Consumer Group 分区再平衡导致重复消费

```go
// 问题：当消费者组成员变化时，会触发分区再平衡
// 未提交的偏移量对应的消息会被重新分配，导致重复消费

// 解决方案：保证消息处理的幂等性
func processMessage(msg kafka.Message) error {
    var event Event
    json.Unmarshal(msg.Value, &event)

    // 使用消息 ID 做幂等检查
    if isProcessed(event.ID) {
        return nil // 已处理过，跳过
    }

    // 处理消息...

    markProcessed(event.ID) // 标记为已处理
    return nil
}
```

### 坑点 5：NATS Core 模式消息丢失

```go
// 错误：在 Core NATS 模式下假设消息一定会被收到
nc.Publish("important.event", data)
// Core NATS 是 at-most-once 语义：如果没有订阅者在线，消息直接丢失

// 正确：需要持久化保证时使用 JetStream
js, _ := jetstream.New(nc)
js.Publish(ctx, "important.event", data)
// JetStream 提供 at-least-once 甚至 exactly-once 语义
```

## 小结

本章完整介绍了三种主流消息队列在 Go 中的使用：

1. **Kafka** 适合高吞吐量的日志收集和事件流处理，分区机制天然支持水平扩展
2. **RabbitMQ** 适合需要灵活路由的任务队列场景，Exchange/Queue/Binding 模型功能强大
3. **NATS** 是云原生环境下的轻量选择，JetStream 模式补齐了持久化能力
4. 消息队列的核心价值是**解耦、削峰、异步**，选型需要根据具体场景权衡
5. 无论使用哪种消息队列，都要注意**消息确认、连接重连、幂等处理**三大关键点

通过 SRE 实战的告警通知系统，我们展示了如何将消息队列应用于实际运维场景，实现异步、可靠的告警分发。
