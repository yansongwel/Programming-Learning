# 系统安全加固

> 一句话说明：用 Shell 脚本实现系统安全基线配置和安全审计。

---

## 1. SSH 安全加固

```bash
# /etc/ssh/sshd_config 关键配置
ssh_harden() {
    local SSHD_CONFIG="/etc/ssh/sshd_config"
    local BACKUP="${SSHD_CONFIG}.bak.$(date +%Y%m%d)"
    cp "$SSHD_CONFIG" "$BACKUP"

    # 修改端口
    sed -i 's/^#Port 22/Port 2222/' "$SSHD_CONFIG"

    # 禁止 root 登录
    sed -i 's/^#PermitRootLogin.*/PermitRootLogin no/' "$SSHD_CONFIG"

    # 禁止密码登录（确保已配好密钥）
    sed -i 's/^#PasswordAuthentication.*/PasswordAuthentication no/' "$SSHD_CONFIG"

    # 限制最大尝试次数
    sed -i 's/^#MaxAuthTries.*/MaxAuthTries 3/' "$SSHD_CONFIG"

    # 空闲超时
    sed -i 's/^#ClientAliveInterval.*/ClientAliveInterval 300/' "$SSHD_CONFIG"
    sed -i 's/^#ClientAliveCountMax.*/ClientAliveCountMax 2/' "$SSHD_CONFIG"

    # 禁止空密码
    sed -i 's/^#PermitEmptyPasswords.*/PermitEmptyPasswords no/' "$SSHD_CONFIG"

    sshd -t && systemctl reload sshd
    echo "SSH 加固完成"
}
```

## 2. 防火墙配置

```bash
# firewalld（CentOS/RHEL）
firewall-cmd --list-all
firewall-cmd --permanent --add-port=2222/tcp
firewall-cmd --permanent --remove-service=ssh
firewall-cmd --reload

# ufw（Ubuntu）
ufw default deny incoming
ufw default allow outgoing
ufw allow 2222/tcp
ufw allow from 10.0.0.0/8 to any port 22
ufw enable

# iptables 基本规则
iptables -A INPUT -p tcp --dport 2222 -j ACCEPT
iptables -A INPUT -p tcp --dport 80 -j ACCEPT
iptables -A INPUT -p tcp --dport 443 -j ACCEPT
iptables -A INPUT -m state --state ESTABLISHED,RELATED -j ACCEPT
iptables -A INPUT -i lo -j ACCEPT
iptables -P INPUT DROP
```

## 3. 安全审计

```bash
# 检查异常用户
# UID 为 0 的用户（应该只有 root）
awk -F: '$3==0{print $1}' /etc/passwd

# 无密码的用户
awk -F: '($2==""){print $1}' /etc/shadow

# 可登录的用户
grep -v '/nologin\|/false' /etc/passwd

# 检查 SUID/SGID 文件
find / -perm -4000 -type f 2>/dev/null    # SUID
find / -perm -2000 -type f 2>/dev/null    # SGID

# 检查无属主文件
find / -nouser -o -nogroup 2>/dev/null

# 检查 world-writable 文件
find / -type f -perm -002 2>/dev/null

# 最近修改的文件
find /etc -mtime -1 -type f               # 24 小时内修改的配置

# 登录记录
last -20                                    # 最近 20 次登录
lastb -20 2>/dev/null                      # 最近 20 次失败登录
```

## 4. 实战：安全巡检脚本

```bash
#!/usr/bin/env bash
# security_audit.sh — 系统安全巡检

set -euo pipefail

WARN=0

check() {
    local desc="$1" result="$2"
    if [[ "$result" == "PASS" ]]; then
        printf "  [通过] %s\n" "$desc"
    else
        printf "  [警告] %s — %s\n" "$desc" "$result"
        ((WARN++))
    fi
}

echo "===== 系统安全巡检 ====="
echo "时间: $(date)"
echo ""

# SSH 检查
echo "--- SSH 配置 ---"
SSHD="/etc/ssh/sshd_config"
ROOT_LOGIN=$(grep -i "^PermitRootLogin" "$SSHD" 2>/dev/null | awk '{print $2}')
check "禁止root登录" "$( [[ "$ROOT_LOGIN" == "no" ]] && echo PASS || echo "当前: ${ROOT_LOGIN:-未配置}" )"

PASS_AUTH=$(grep -i "^PasswordAuthentication" "$SSHD" 2>/dev/null | awk '{print $2}')
check "禁止密码登录" "$( [[ "$PASS_AUTH" == "no" ]] && echo PASS || echo "当前: ${PASS_AUTH:-未配置}" )"

# 用户检查
echo ""
echo "--- 用户安全 ---"
ROOT_USERS=$(awk -F: '$3==0{print $1}' /etc/passwd | tr '\n' ' ')
check "UID=0 用户" "$( [[ "$ROOT_USERS" == "root " ]] && echo PASS || echo "发现: $ROOT_USERS" )"

EMPTY_PASS=$(awk -F: '($2==""){print $1}' /etc/shadow 2>/dev/null | tr '\n' ' ')
check "无密码用户" "$( [[ -z "$EMPTY_PASS" ]] && echo PASS || echo "发现: $EMPTY_PASS" )"

# 文件权限
echo ""
echo "--- 关键文件权限 ---"
for f in /etc/passwd /etc/shadow /etc/ssh/sshd_config; do
    if [[ -f "$f" ]]; then
        perm=$(stat -c %a "$f")
        case "$f" in
            */shadow) expected=640 ;;
            */sshd_config) expected=600 ;;
            *) expected=644 ;;
        esac
        check "$f 权限=$perm" "$( (( perm <= expected )) && echo PASS || echo "建议: $expected" )"
    fi
done

echo ""
echo "===== 巡检完成: $WARN 项警告 ====="
```
