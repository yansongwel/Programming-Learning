# 实战项目：一键环境搭建工具

> 一句话说明：构建一个通用的开发/生产环境一键搭建工具，支持多种技术栈的自动化安装配置。

---

## 1. 项目需求

快速搭建常见技术栈环境：
- LNMP (Linux + Nginx + MySQL + PHP)
- Docker + Docker Compose
- Node.js 开发环境
- Python 开发环境
- 监控栈 (Prometheus + Grafana)

---

## 2. 核心脚本

```bash
#!/usr/bin/env bash
# env_setup.sh — 一键环境搭建工具

set -euo pipefail

# ========================
# 基础设施
# ========================
RED='\033[0;31m'; GREEN='\033[0;32m'; YELLOW='\033[0;33m'; BLUE='\033[0;34m'; NC='\033[0m'
log_info()  { echo -e "${GREEN}[✓]${NC} $*"; }
log_step()  { echo -e "\n${BLUE}==>${NC} $*"; }
log_warn()  { echo -e "${YELLOW}[!]${NC} $*"; }
die()       { echo -e "${RED}[✗]${NC} $*" >&2; exit 1; }

# 检测操作系统
detect_os() {
    if [[ -f /etc/debian_version ]]; then
        OS="debian"
        PKG_UPDATE="apt-get update -qq"
        PKG_INSTALL="apt-get install -y -qq"
    elif [[ -f /etc/redhat-release ]]; then
        OS="rhel"
        PKG_UPDATE="yum makecache -q"
        PKG_INSTALL="yum install -y -q"
    else
        die "不支持的操作系统"
    fi
}

require_root() {
    [[ "$EUID" -eq 0 ]] || die "请使用 sudo 运行"
}

# ========================
# 安装模块
# ========================
install_docker() {
    log_step "安装 Docker"
    if command -v docker &>/dev/null; then
        log_info "Docker 已安装 ($(docker --version | awk '{print $3}'))"
        return 0
    fi

    if [[ "$OS" == "debian" ]]; then
        $PKG_INSTALL ca-certificates curl gnupg
        install -m 0755 -d /etc/apt/keyrings
        curl -fsSL https://download.docker.com/linux/ubuntu/gpg | gpg --dearmor -o /etc/apt/keyrings/docker.gpg
        chmod a+r /etc/apt/keyrings/docker.gpg
        echo "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu $(. /etc/os-release && echo $VERSION_CODENAME) stable" > /etc/apt/sources.list.d/docker.list
        apt-get update -qq
        $PKG_INSTALL docker-ce docker-ce-cli containerd.io docker-compose-plugin
    elif [[ "$OS" == "rhel" ]]; then
        $PKG_INSTALL yum-utils
        yum-config-manager --add-repo https://download.docker.com/linux/centos/docker-ce.repo
        $PKG_INSTALL docker-ce docker-ce-cli containerd.io docker-compose-plugin
    fi

    systemctl enable --now docker
    # 将当前用户加入 docker 组
    usermod -aG docker "${SUDO_USER:-$USER}" 2>/dev/null || true

    log_info "Docker 安装完成 ($(docker --version | awk '{print $3}'))"
}

install_nginx() {
    log_step "安装 Nginx"
    if command -v nginx &>/dev/null; then
        log_info "Nginx 已安装"
        return 0
    fi
    $PKG_INSTALL nginx
    systemctl enable --now nginx
    log_info "Nginx 安装完成"
}

install_mysql() {
    log_step "安装 MySQL"
    if command -v mysql &>/dev/null; then
        log_info "MySQL 已安装"
        return 0
    fi
    if [[ "$OS" == "debian" ]]; then
        $PKG_INSTALL mysql-server
    else
        $PKG_INSTALL mysql-server
    fi
    systemctl enable --now mysqld 2>/dev/null || systemctl enable --now mysql 2>/dev/null
    log_info "MySQL 安装完成"
}

install_nodejs() {
    log_step "安装 Node.js"
    if command -v node &>/dev/null; then
        log_info "Node.js 已安装 ($(node --version))"
        return 0
    fi
    # 使用 nvm
    export NVM_DIR="/opt/nvm"
    mkdir -p "$NVM_DIR"
    curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash
    source "$NVM_DIR/nvm.sh"
    nvm install --lts
    nvm use --lts
    log_info "Node.js 安装完成 ($(node --version))"
}

install_python() {
    log_step "安装 Python 开发环境"
    if command -v python3 &>/dev/null; then
        log_info "Python3 已安装 ($(python3 --version))"
    fi
    $PKG_INSTALL python3 python3-pip python3-venv python3-dev
    pip3 install --quiet virtualenv
    log_info "Python 开发环境就绪"
}

# ========================
# 套件组合
# ========================
setup_lnmp() {
    log_step "搭建 LNMP 环境"
    $PKG_UPDATE
    install_nginx
    install_mysql
    $PKG_INSTALL php-fpm php-mysql php-cli
    systemctl enable --now php*-fpm 2>/dev/null || true
    log_info "LNMP 环境搭建完成"
}

setup_docker_env() {
    log_step "搭建 Docker 环境"
    install_docker
    # 安装常用工具
    $PKG_INSTALL curl wget git vim htop
    log_info "Docker 环境搭建完成"
}

setup_monitoring() {
    log_step "搭建监控栈 (Docker 方式)"
    install_docker

    local MONITOR_DIR="/opt/monitoring"
    mkdir -p "$MONITOR_DIR"

    cat > "$MONITOR_DIR/docker-compose.yml" << 'EOF'
services:
  prometheus:
    image: prom/prometheus:latest
    ports:
      - "9090:9090"
    volumes:
      - ./prometheus.yml:/etc/prometheus/prometheus.yml
    restart: unless-stopped

  grafana:
    image: grafana/grafana:latest
    ports:
      - "3000:3000"
    environment:
      - GF_SECURITY_ADMIN_PASSWORD=admin
    restart: unless-stopped

  node-exporter:
    image: prom/node-exporter:latest
    ports:
      - "9100:9100"
    restart: unless-stopped
EOF

    cat > "$MONITOR_DIR/prometheus.yml" << 'EOF'
global:
  scrape_interval: 15s

scrape_configs:
  - job_name: 'node'
    static_configs:
      - targets: ['node-exporter:9100']
EOF

    cd "$MONITOR_DIR"
    docker compose up -d
    log_info "监控栈启动完成"
    echo "  Prometheus: http://localhost:9090"
    echo "  Grafana:    http://localhost:3000 (admin/admin)"
}

# ========================
# 交互菜单
# ========================
show_menu() {
    echo ""
    echo "╔══════════════════════════════════════╗"
    echo "║      一键环境搭建工具 v1.0           ║"
    echo "╚══════════════════════════════════════╝"
    echo ""
    echo "请选择要搭建的环境："
    echo ""

    select choice in \
        "Docker 环境" \
        "LNMP (Nginx+MySQL+PHP)" \
        "Node.js 开发环境" \
        "Python 开发环境" \
        "监控栈 (Prometheus+Grafana)" \
        "全部安装" \
        "退出"; do
        case "$choice" in
            "Docker 环境")           setup_docker_env ;;
            "LNMP (Nginx+MySQL+PHP)") setup_lnmp ;;
            "Node.js 开发环境")       install_nodejs ;;
            "Python 开发环境")        install_python ;;
            "监控栈 (Prometheus+Grafana)") setup_monitoring ;;
            "全部安装")
                setup_docker_env
                setup_lnmp
                install_nodejs
                install_python
                setup_monitoring
                ;;
            "退出") break ;;
            *) echo "无效选项" ;;
        esac
        echo ""
        echo "按回车继续..."
        read -r
        show_menu
        break
    done
}

# ========================
# 主入口
# ========================
main() {
    require_root
    detect_os
    $PKG_UPDATE

    if [[ $# -gt 0 ]]; then
        case "$1" in
            docker)     setup_docker_env ;;
            lnmp)       setup_lnmp ;;
            nodejs)     install_nodejs ;;
            python)     install_python ;;
            monitoring) setup_monitoring ;;
            all)        setup_docker_env; setup_lnmp; install_nodejs; install_python; setup_monitoring ;;
            *)          echo "用法: $0 [docker|lnmp|nodejs|python|monitoring|all]"; exit 1 ;;
        esac
    else
        show_menu
    fi
}

main "$@"
```

## 3. 使用方式

```bash
# 交互式菜单
sudo bash env_setup.sh

# 命令行直接指定
sudo bash env_setup.sh docker
sudo bash env_setup.sh lnmp
sudo bash env_setup.sh monitoring
sudo bash env_setup.sh all
```
