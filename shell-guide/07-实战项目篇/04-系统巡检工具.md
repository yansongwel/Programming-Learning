# 实战项目：系统巡检工具

> 一句话说明：构建一个全面的系统巡检工具，支持多主机巡检和报告导出。

---

## 1. 完整巡检脚本

```bash
#!/usr/bin/env bash
# system_patrol.sh — 系统巡检工具
# 用法:
#   本机巡检: bash system_patrol.sh
#   远程巡检: bash system_patrol.sh -f servers.txt
#   输出报告: bash system_patrol.sh -o report.txt

set -euo pipefail

OUTPUT=""
SERVERS_FILE=""
VERBOSE=false

while [[ $# -gt 0 ]]; do
    case "$1" in
        -o|--output)  OUTPUT="$2"; shift 2 ;;
        -f|--file)    SERVERS_FILE="$2"; shift 2 ;;
        -v|--verbose) VERBOSE=true; shift ;;
        -h|--help)    echo "Usage: $0 [-o output] [-f servers.txt] [-v]"; exit 0 ;;
        *)            shift ;;
    esac
done

# ========================
# 巡检函数
# ========================
SEP="$(printf '─%.0s' {1..60})"

section() {
    echo ""
    echo "┌${SEP}┐"
    printf "│ %-58s │\n" "$1"
    echo "└${SEP}┘"
}

check_pass() { printf "  %-40s [  OK  ]\n" "$1"; }
check_warn() { printf "  %-40s [ WARN ] %s\n" "$1" "$2"; }
check_fail() { printf "  %-40s [ FAIL ] %s\n" "$1" "$2"; }

run_inspection() {
    local WARNINGS=0 FAILURES=0

    echo "╔══════════════════════════════════════════════════════════════╗"
    echo "║              系统巡检报告                                   ║"
    echo "║  主机: $(hostname)"
    echo "║  时间: $(date '+%Y-%m-%d %H:%M:%S')"
    echo "╚══════════════════════════════════════════════════════════════╝"

    # --- 基本信息 ---
    section "基本信息"
    echo "  主机名:     $(hostname)"
    echo "  IP:         $(hostname -I 2>/dev/null | awk '{print $1}' || echo '未知')"
    echo "  系统:       $(grep PRETTY_NAME /etc/os-release 2>/dev/null | cut -d'"' -f2 || uname -s)"
    echo "  内核:       $(uname -r)"
    echo "  运行时间:   $(uptime -p 2>/dev/null || uptime)"

    # --- CPU ---
    section "CPU 检查"
    local load1 cores
    load1=$(cat /proc/loadavg | awk '{print $1}')
    cores=$(nproc)
    local load_pct=$(awk "BEGIN{printf \"%d\", $load1/$cores*100}")
    if (( load_pct > 90 )); then
        check_fail "CPU 负载" "Load: $load1 (${load_pct}%)"
        ((FAILURES++))
    elif (( load_pct > 70 )); then
        check_warn "CPU 负载" "Load: $load1 (${load_pct}%)"
        ((WARNINGS++))
    else
        check_pass "CPU 负载 ($load1, ${cores}核, ${load_pct}%)"
    fi

    # --- 内存 ---
    section "内存检查"
    local mem_pct
    mem_pct=$(free | awk '/Mem:/{printf "%d", $3/$2*100}')
    if (( mem_pct > 90 )); then
        check_fail "内存使用率" "${mem_pct}%"
        ((FAILURES++))
    elif (( mem_pct > 80 )); then
        check_warn "内存使用率" "${mem_pct}%"
        ((WARNINGS++))
    else
        check_pass "内存使用率 (${mem_pct}%)"
    fi

    # Swap
    local swap_total swap_used
    swap_total=$(free | awk '/Swap:/{print $2}')
    swap_used=$(free | awk '/Swap:/{print $3}')
    if (( swap_total > 0 && swap_used * 100 / swap_total > 50 )); then
        check_warn "Swap 使用" "$(free -h | awk '/Swap:/{print $3"/"$2}')"
        ((WARNINGS++))
    else
        check_pass "Swap 正常"
    fi

    # --- 磁盘 ---
    section "磁盘检查"
    while read -r usage mount; do
        local pct=${usage%\%}
        if (( pct > 90 )); then
            check_fail "磁盘 $mount" "$usage"
            ((FAILURES++))
        elif (( pct > 80 )); then
            check_warn "磁盘 $mount" "$usage"
            ((WARNINGS++))
        else
            check_pass "磁盘 $mount ($usage)"
        fi
    done < <(df -h | awk 'NR>1 && $1!~/tmpfs|devtmpfs/{print $5, $6}')

    # --- 服务 ---
    section "服务检查"
    local services=("sshd" "cron" "rsyslog")
    # 检测并添加常见服务
    for svc in nginx mysql redis docker; do
        systemctl list-unit-files "${svc}.service" &>/dev/null && services+=("$svc")
    done

    for svc in "${services[@]}"; do
        if systemctl is-active --quiet "$svc" 2>/dev/null; then
            check_pass "$svc 运行中"
        elif systemctl is-enabled --quiet "$svc" 2>/dev/null; then
            check_warn "$svc 已启用但未运行"
            ((WARNINGS++))
        fi
    done

    # --- 安全 ---
    section "安全检查"
    # Root SSH
    local root_login
    root_login=$(grep -i "^PermitRootLogin" /etc/ssh/sshd_config 2>/dev/null | awk '{print $2}')
    if [[ "$root_login" == "no" ]]; then
        check_pass "SSH 禁止 Root 登录"
    else
        check_warn "SSH Root 登录" "当前: ${root_login:-未配置}"
        ((WARNINGS++))
    fi

    # 失败登录
    local fail_count
    fail_count=$(grep -c "Failed password" /var/log/auth.log 2>/dev/null || journalctl -u sshd --since "24h ago" 2>/dev/null | grep -c "Failed password" || echo 0)
    if (( fail_count > 100 )); then
        check_warn "24h 登录失败" "${fail_count} 次"
        ((WARNINGS++))
    else
        check_pass "登录失败次数 (${fail_count}次/24h)"
    fi

    # --- 汇总 ---
    section "巡检结果"
    echo "  正常项: 多项"
    echo "  警告:   $WARNINGS 项"
    echo "  严重:   $FAILURES 项"

    if (( FAILURES > 0 )); then
        echo ""
        echo "  整体状态: ❌ 需要立即处理"
        return 2
    elif (( WARNINGS > 0 )); then
        echo ""
        echo "  整体状态: ⚠️  需要关注"
        return 1
    else
        echo ""
        echo "  整体状态: ✅ 健康"
        return 0
    fi
}

# ========================
# 执行
# ========================
if [[ -n "$SERVERS_FILE" ]]; then
    # 多主机巡检
    while read -r server; do
        [[ -z "$server" || "$server" =~ ^# ]] && continue
        echo ""
        echo "======== 巡检: $server ========"
        ssh -o ConnectTimeout=5 "$server" "$(cat "$0")" 2>&1 || echo "连接失败: $server"
    done < "$SERVERS_FILE"
elif [[ -n "$OUTPUT" ]]; then
    run_inspection | tee "$OUTPUT"
    echo ""
    echo "报告已保存: $OUTPUT"
else
    run_inspection
fi
```

---

## 2. 使用方式

```bash
# 本机巡检
bash system_patrol.sh

# 保存报告
bash system_patrol.sh -o /tmp/patrol_report.txt

# 多主机巡检
bash system_patrol.sh -f servers.txt
```
