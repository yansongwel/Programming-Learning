# 实战项目：服务器初始化脚本

> 一句话说明：编写一个完整的服务器初始化脚本，新服务器开箱即用。

---

## 1. 项目需求

新服务器交付后需要完成以下初始化工作：
- 系统基础配置（主机名、时区、DNS）
- 软件包安装
- 用户创建与 SSH 配置
- 安全加固
- 性能优化（内核参数）
- 监控 Agent 安装

---

## 2. 完整脚本

```bash
#!/usr/bin/env bash
#
# server_init.sh — 服务器初始化脚本
# 用法: bash server_init.sh --hostname web01 --role web
#

set -euo pipefail

# ========================
# 颜色和日志
# ========================
RED='\033[0;31m'; GREEN='\033[0;32m'; YELLOW='\033[0;33m'; NC='\033[0m'
log_info()  { echo -e "${GREEN}[INFO]${NC}  $(date '+%H:%M:%S') $*"; }
log_warn()  { echo -e "${YELLOW}[WARN]${NC}  $(date '+%H:%M:%S') $*"; }
log_error() { echo -e "${RED}[ERROR]${NC} $(date '+%H:%M:%S') $*"; }
die()       { log_error "$@"; exit 1; }

# ========================
# 配置
# ========================
NEW_HOSTNAME=""
ROLE="base"
SSH_PORT=22
TIMEZONE="Asia/Shanghai"
DNS_SERVERS=("223.5.5.5" "8.8.8.8")
NTP_SERVER="ntp.aliyun.com"
ADMIN_USERS=("ops:ssh-ed25519 AAAA...ops@company.com")
PACKAGES_BASE="vim curl wget git htop iotop lsof net-tools bash-completion"
PACKAGES_WEB="nginx"
PACKAGES_DB=""

# ========================
# 参数解析
# ========================
parse_args() {
    while [[ $# -gt 0 ]]; do
        case "$1" in
            --hostname) NEW_HOSTNAME="$2"; shift 2 ;;
            --role)     ROLE="$2"; shift 2 ;;
            --ssh-port) SSH_PORT="$2"; shift 2 ;;
            -h|--help)  usage ;;
            *)          die "未知参数: $1" ;;
        esac
    done
}

usage() {
    cat << EOF
Usage: $(basename "$0") [OPTIONS]

Options:
    --hostname NAME     设置主机名
    --role ROLE         服务器角色 (base|web|db) [默认: base]
    --ssh-port PORT     SSH 端口 [默认: 22]
    -h, --help          帮助信息
EOF
    exit 0
}

# ========================
# 前置检查
# ========================
preflight() {
    log_info "执行前置检查..."
    [[ "$EUID" -eq 0 ]] || die "请使用 root 权限运行"

    # 检测发行版
    if [[ -f /etc/redhat-release ]]; then
        OS_FAMILY="rhel"
        PKG_INSTALL="yum install -y"
    elif [[ -f /etc/debian_version ]]; then
        OS_FAMILY="debian"
        PKG_INSTALL="apt-get install -y"
    else
        die "不支持的操作系统"
    fi
    log_info "检测到: $OS_FAMILY"
}

# ========================
# 各初始化步骤
# ========================
setup_hostname() {
    if [[ -n "$NEW_HOSTNAME" ]]; then
        log_info "设置主机名: $NEW_HOSTNAME"
        hostnamectl set-hostname "$NEW_HOSTNAME"
    fi
}

setup_timezone() {
    log_info "设置时区: $TIMEZONE"
    timedatectl set-timezone "$TIMEZONE"
    timedatectl set-ntp true

    # NTP 同步
    if command -v chronyc &>/dev/null; then
        sed -i "s/^server .*/server $NTP_SERVER iburst/" /etc/chrony.conf 2>/dev/null || true
        systemctl restart chronyd
    fi
}

setup_dns() {
    log_info "配置 DNS"
    cat > /etc/resolv.conf << EOF
$(for dns in "${DNS_SERVERS[@]}"; do echo "nameserver $dns"; done)
EOF
}

install_packages() {
    log_info "安装基础软件包..."
    if [[ "$OS_FAMILY" == "debian" ]]; then
        apt-get update -qq
    fi

    local packages="$PACKAGES_BASE"
    case "$ROLE" in
        web) packages+=" $PACKAGES_WEB" ;;
        db)  packages+=" $PACKAGES_DB" ;;
    esac

    $PKG_INSTALL $packages
    log_info "软件包安装完成"
}

setup_users() {
    log_info "配置用户..."
    for entry in "${ADMIN_USERS[@]}"; do
        local user="${entry%%:*}"
        local pubkey="${entry#*:}"

        if ! id "$user" &>/dev/null; then
            useradd -m -s /bin/bash -G wheel "$user" 2>/dev/null || \
            useradd -m -s /bin/bash -G sudo "$user" 2>/dev/null || true
        fi

        local ssh_dir="/home/${user}/.ssh"
        mkdir -p "$ssh_dir"
        echo "$pubkey" >> "${ssh_dir}/authorized_keys"
        sort -u "${ssh_dir}/authorized_keys" -o "${ssh_dir}/authorized_keys"
        chmod 700 "$ssh_dir"
        chmod 600 "${ssh_dir}/authorized_keys"
        chown -R "${user}:${user}" "$ssh_dir"
        log_info "  用户 $user 已配置"
    done
}

harden_ssh() {
    log_info "SSH 安全加固..."
    local sshd="/etc/ssh/sshd_config"
    cp "$sshd" "${sshd}.bak"

    sed -i "s/^#*Port .*/Port $SSH_PORT/" "$sshd"
    sed -i 's/^#*PermitRootLogin .*/PermitRootLogin no/' "$sshd"
    sed -i 's/^#*PasswordAuthentication .*/PasswordAuthentication no/' "$sshd"
    sed -i 's/^#*MaxAuthTries .*/MaxAuthTries 3/' "$sshd"
    sed -i 's/^#*ClientAliveInterval .*/ClientAliveInterval 300/' "$sshd"

    sshd -t && systemctl reload sshd
    log_info "SSH 加固完成 (端口: $SSH_PORT)"
}

tune_kernel() {
    log_info "优化内核参数..."
    cat >> /etc/sysctl.d/99-custom.conf << 'EOF'
# 网络优化
net.core.somaxconn = 65535
net.ipv4.tcp_max_syn_backlog = 65535
net.ipv4.tcp_tw_reuse = 1
net.ipv4.tcp_fin_timeout = 30
net.ipv4.tcp_keepalive_time = 600

# 文件描述符
fs.file-max = 655350
fs.inotify.max_user_watches = 524288

# 内存
vm.swappiness = 10
vm.overcommit_memory = 1
EOF
    sysctl --system > /dev/null 2>&1

    # 文件描述符限制
    cat >> /etc/security/limits.d/99-custom.conf << 'EOF'
*  soft  nofile  655350
*  hard  nofile  655350
*  soft  nproc   65535
*  hard  nproc   65535
EOF
    log_info "内核参数优化完成"
}

setup_firewall() {
    log_info "配置防火墙..."
    if command -v firewall-cmd &>/dev/null; then
        firewall-cmd --permanent --add-port=${SSH_PORT}/tcp
        [[ "$ROLE" == "web" ]] && firewall-cmd --permanent --add-port={80,443}/tcp
        firewall-cmd --reload
    elif command -v ufw &>/dev/null; then
        ufw allow "$SSH_PORT/tcp"
        [[ "$ROLE" == "web" ]] && ufw allow 80/tcp && ufw allow 443/tcp
        ufw --force enable
    fi
    log_info "防火墙配置完成"
}

# ========================
# 主函数
# ========================
main() {
    echo "============================================"
    echo "  服务器初始化脚本"
    echo "  $(date '+%Y-%m-%d %H:%M:%S')"
    echo "============================================"

    parse_args "$@"
    preflight

    setup_hostname
    setup_timezone
    setup_dns
    install_packages
    setup_users
    harden_ssh
    tune_kernel
    setup_firewall

    echo ""
    echo "============================================"
    log_info "初始化完成!"
    echo "  主机名: $(hostname)"
    echo "  角色:   $ROLE"
    echo "  SSH:    端口 $SSH_PORT"
    echo "============================================"
}

main "$@"
```

---

## 3. 使用方法

```bash
# 基础初始化
bash server_init.sh --hostname web01

# Web 服务器初始化
bash server_init.sh --hostname web01 --role web --ssh-port 2222

# 数据库服务器
bash server_init.sh --hostname db01 --role db
```

## 4. 项目要点总结

| 设计要素 | 实现方式 |
|----------|----------|
| 参数化 | `--hostname`, `--role` 支持不同场景 |
| 幂等性 | 重复执行不会出错 |
| 可扩展 | 按角色安装不同软件 |
| 安全 | SSH 加固 + 防火墙 + 内核优化 |
| 日志 | 彩色输出 + 分步骤提示 |
| 错误处理 | `set -euo pipefail` + `die()` |
