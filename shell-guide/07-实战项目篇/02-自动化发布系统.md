# 实战项目：自动化发布系统

> 一句话说明：构建一个支持多环境、灰度发布、自动回滚的完整发布系统。

---

## 1. 项目架构

```
release-system/
├── bin/
│   └── release.sh         # 主入口
├── lib/

```

## 2. 核心代码

### lib/common.sh

```bash
#!/usr/bin/env bash
# 公共函数库

RED='\033[0;31m'; GREEN='\033[0;32m'; YELLOW='\033[0;33m'; BLUE='\033[0;34m'; NC='\033[0m'

log_info()  { echo -e "${GREEN}[INFO]${NC}  $(date '+%H:%M:%S') $*"; }
log_warn()  { echo -e "${YELLOW}[WARN]${NC}  $(date '+%H:%M:%S') $*"; }
log_error() { echo -e "${RED}[ERROR]${NC} $(date '+%H:%M:%S') $*" >&2; }
log_step()  { echo -e "\n${BLUE}==>${NC} $*"; }
die()       { log_error "$@"; exit 1; }

confirm() {
    local prompt="${1:-确认执行?}"
    read -rp "$prompt [y/N]: " answer
    [[ "$answer" =~ ^[Yy]$ ]]
}

retry() {
    local max=$1 delay=$2; shift 2
    for ((i=1; i<=max; i++)); do
        "$@" && return 0
        (( i < max )) && { log_warn "第 $i 次失败，${delay}s 后重试..."; sleep "$delay"; }
    done
    return 1
}

health_check() {
    local url="$1" timeout="${2:-30}"
    local i=0
    until curl -sf -o /dev/null "$url"; do
        ((i++))
        (( i >= timeout )) && return 1
        sleep 1
    done
    return 0
}
```

### bin/release.sh（主入口）

```bash
#!/usr/bin/env bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
ROOT_DIR="$(dirname "$SCRIPT_DIR")"

source "$ROOT_DIR/lib/common.sh"

# 参数解析
ENV=""
VERSION=""
ACTION="deploy"
CANARY=false

while [[ $# -gt 0 ]]; do
    case "$1" in
        -e|--env)     ENV="$2"; shift 2 ;;
        -v|--version) VERSION="$2"; shift 2 ;;
        -a|--action)  ACTION="$2"; shift 2 ;;
        --canary)     CANARY=true; shift ;;
        -h|--help)    echo "Usage: $0 -e <env> -v <version> [-a deploy|rollback] [--canary]"; exit 0 ;;
        *)            die "未知参数: $1" ;;
    esac
done

[[ -z "$ENV" ]] && die "请指定环境: -e dev|staging|prod"
[[ -z "$VERSION" && "$ACTION" == "deploy" ]] && die "请指定版本: -v x.y.z"

# 加载配置
source "$ROOT_DIR/conf/global.conf"
source "$ROOT_DIR/conf/${ENV}.conf"
INVENTORY="$ROOT_DIR/inventory/${ENV}.txt"
[[ -f "$INVENTORY" ]] || die "服务器列表不存在: $INVENTORY"

# 读取服务器列表
mapfile -t SERVERS < <(grep -v '^#' "$INVENTORY" | grep -v '^$')

log_step "发布系统 v1.0"
echo "  环境: $ENV"
echo "  版本: $VERSION"
echo "  操作: $ACTION"
echo "  服务器: ${#SERVERS[@]} 台"
echo "  灰度: $CANARY"

# 生产环境需要确认
if [[ "$ENV" == "prod" ]]; then
    confirm "确认在生产环境执行 $ACTION ?" || { echo "已取消"; exit 0; }
fi

case "$ACTION" in
    deploy)
        source "$ROOT_DIR/lib/build.sh"
        source "$ROOT_DIR/lib/deploy.sh"

        log_step "构建阶段"
        build "$VERSION"

        log_step "部署阶段"
        if $CANARY; then
            # 灰度发布：先部署一台
            deploy_to_host "${SERVERS[0]}" "$VERSION"
            log_info "灰度实例已部署: ${SERVERS[0]}"
            confirm "灰度验证通过，继续全量部署?" || { echo "灰度中止"; exit 0; }
            for server in "${SERVERS[@]:1}"; do
                deploy_to_host "$server" "$VERSION"
            done
        else
            for server in "${SERVERS[@]}"; do
                deploy_to_host "$server" "$VERSION"
            done
        fi

        log_step "发布完成!"
        ;;
    rollback)
        source "$ROOT_DIR/lib/rollback.sh"
        rollback_all "${SERVERS[@]}"
        ;;
esac
```

## 3. 使用示例

```bash
# 部署到开发环境
./bin/release.sh -e dev -v 1.2.0

# 预发环境灰度发布
./bin/release.sh -e staging -v 1.2.0 --canary

# 生产环境发布
./bin/release.sh -e prod -v 1.2.0 --canary

# 回滚
./bin/release.sh -e prod -a rollback
```

## 4. 项目要点

| 特性 | 实现 |
|------|------|
| 多环境 | 配置文件 + 服务器列表分环境管理 |
| 灰度发布 | `--canary` 先部署一台验证 |
| 回滚 | 符号链接切换到上一个版本 |
| 安全 | 生产环境二次确认 |
| 健康检查 | 部署后自动检测 |
| 日志 | 全过程记录 |
