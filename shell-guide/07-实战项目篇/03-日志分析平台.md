# 实战项目：日志分析平台

> 一句话说明：用 Shell 构建一个轻量级日志分析工具，支持多种日志格式的实时分析和报表生成。

---

## 1. 项目需求

- 支持 Nginx、应用日志等多种格式
- 实时监控模式和批量分析模式
- 生成 HTML 报表
- 告警通知

---

## 2. 核心脚本

```bash
#!/usr/bin/env bash
# log_platform.sh — 轻量级日志分析平台

set -euo pipefail

# ========================
# 配置
# ========================
REPORT_DIR="/tmp/log_reports"
mkdir -p "$REPORT_DIR"

# ========================
# Nginx 日志分析
# ========================
analyze_nginx() {
    local log_file="$1"
    local total=$(wc -l < "$log_file")

    echo "=== Nginx 日志分析 ==="
    echo "文件: $log_file"
    echo "总请求: $total"
    echo ""

    echo "--- 状态码分布 ---"
    awk '{print $9}' "$log_file" | sort | uniq -c | sort -rn | \
        awk -v total="$total" '{printf "  %-6s %8d (%5.1f%%)\n", $2, $1, $1*100/total}'

    echo ""
    echo "--- Top 10 IP ---"
    awk '{print $1}' "$log_file" | sort | uniq -c | sort -rn | head -10 | \
        awk '{printf "  %-18s %8d\n", $2, $1}'

    echo ""
    echo "--- Top 10 URL ---"
    awk '{print $7}' "$log_file" | sort | uniq -c | sort -rn | head -10 | \
        awk '{printf "  %-50s %8d\n", $2, $1}'

    echo ""
    echo "--- 每小时流量 ---"
    awk '{h=substr($4,14,2); req[h]++} END{for(i=0;i<24;i++) printf "  %02d:00  %8d\n", i, req[sprintf("%02d",i)]+0}' "$log_file"

    echo ""
    echo "--- 5xx 错误详情 (最近 10 条) ---"
    awk '$9>=500{print $4, $1, $7, $9}' "$log_file" | tail -10 | \
        awk '{printf "  %s %-15s %-40s [%s]\n", $1, $2, $3, $4}'
}

# ========================
# 生成 HTML 报表
# ========================
generate_html_report() {
    local log_file="$1"
    local report="${REPORT_DIR}/report_$(date +%Y%m%d_%H%M%S).html"

    cat > "$report" << 'HEADER'
<!DOCTYPE html>
<html><head>
<meta charset="utf-8">
<title>日志分析报告</title>
<style>
  body { font-family: monospace; margin: 20px; background: #1a1a2e; color: #eee; }
  h1 { color: #0f3460; background: #e94560; padding: 10px; border-radius: 5px; }
  h2 { color: #e94560; border-bottom: 1px solid #333; padding-bottom: 5px; }
  table { border-collapse: collapse; width: 100%; margin: 10px 0; }
  th { background: #16213e; padding: 8px; text-align: left; }
  td { padding: 8px; border-bottom: 1px solid #333; }
  tr:hover { background: #16213e; }
  .bar { background: #e94560; height: 20px; display: inline-block; }
  .stat { display: inline-block; background: #16213e; padding: 15px 25px; margin: 5px; border-radius: 5px; }
  .stat .num { font-size: 2em; color: #e94560; }
</style>
</head><body>
HEADER

    echo "<h1>日志分析报告</h1>" >> "$report"
    echo "<p>生成时间: $(date '+%Y-%m-%d %H:%M:%S') | 文件: $log_file</p>" >> "$report"

    local total=$(wc -l < "$log_file")
    local err_count=$(awk '$9>=400' "$log_file" | wc -l)
    local uniq_ips=$(awk '{print $1}' "$log_file" | sort -u | wc -l)

    echo "<div>" >> "$report"
    echo "<div class='stat'><div class='num'>$total</div>总请求</div>" >> "$report"
    echo "<div class='stat'><div class='num'>$err_count</div>错误数</div>" >> "$report"
    echo "<div class='stat'><div class='num'>$uniq_ips</div>独立IP</div>" >> "$report"
    echo "</div>" >> "$report"

    # 状态码表格
    echo "<h2>状态码分布</h2><table><tr><th>状态码</th><th>数量</th><th>占比</th></tr>" >> "$report"
    awk '{print $9}' "$log_file" | sort | uniq -c | sort -rn | \
        awk -v t="$total" '{printf "<tr><td>%s</td><td>%d</td><td>%.1f%%</td></tr>\n", $2, $1, $1*100/t}' >> "$report"
    echo "</table>" >> "$report"

    # Top IP
    echo "<h2>Top 20 IP</h2><table><tr><th>IP</th><th>请求数</th><th>占比</th></tr>" >> "$report"
    awk '{print $1}' "$log_file" | sort | uniq -c | sort -rn | head -20 | \
        awk -v t="$total" '{printf "<tr><td>%s</td><td>%d</td><td>%.1f%%</td></tr>\n", $2, $1, $1*100/t}' >> "$report"
    echo "</table>" >> "$report"

    echo "</body></html>" >> "$report"
    echo "HTML 报表: $report"
}

# ========================
# 实时监控模式
# ========================
realtime_monitor() {
    local log_file="$1"
    echo "实时监控: $log_file (Ctrl+C 退出)"
    echo ""

    tail -f "$log_file" | awk '
    {
        total++
        ip[$1]++
        status[$9]++
        if ($9 >= 500) errors++

        if (total % 100 == 0) {
            printf "\r请求: %d | 5xx: %d | 独立IP: %d    ", total, errors+0, length(ip)
        }

        if ($9 >= 500) {
            printf "\n[5xx] %s %s %s → %s\n", strftime("%H:%M:%S"), $1, $7, $9
        }
    }'
}

# ========================
# 主入口
# ========================
case "${1:-help}" in
    analyze)
        LOG_FILE="${2:?请指定日志文件}"
        analyze_nginx "$LOG_FILE"
        ;;
    html)
        LOG_FILE="${2:?请指定日志文件}"
        generate_html_report "$LOG_FILE"
        ;;
    monitor)
        LOG_FILE="${2:?请指定日志文件}"
        realtime_monitor "$LOG_FILE"
        ;;
    *)
        echo "用法: $0 {analyze|html|monitor} <日志文件>"
        echo ""
        echo "  analyze  — 终端分析报表"
        echo "  html     — 生成 HTML 报表"
        echo "  monitor  — 实时监控模式"
        ;;
esac
```

---

## 3. 使用示例

```bash
# 终端分析
./log_platform.sh analyze /var/log/nginx/access.log

# 生成 HTML 报告
./log_platform.sh html /var/log/nginx/access.log

# 实时监控
./log_platform.sh monitor /var/log/nginx/access.log
```
