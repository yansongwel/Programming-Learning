# 网络工具

> 一句话说明：掌握常用的网络诊断和管理命令，快速排查网络问题。

---

## 1. 网络信息查看

```bash
# IP 地址
ip addr show                     # 查看所有网卡 IP（推荐）
ip -4 addr show                  # 只看 IPv4
hostname -I                      # 快速获取 IP

# 路由表
ip route show                    # 查看路由
ip route get 8.8.8.8             # 查看到达目标的路由

# DNS
cat /etc/resolv.conf             # DNS 配置
nslookup example.com             # DNS 查询
dig example.com                  # 详细 DNS 查询
dig +short example.com           # 简洁输出
host example.com                 # 简单查询
```

## 2. 连通性测试

```bash
# ping — ICMP 连通性
ping -c 4 8.8.8.8               # 发送 4 个包
ping -c 4 -W 2 host             # 超时 2 秒

# traceroute — 路由追踪
traceroute 8.8.8.8
traceroute -n 8.8.8.8            # 不做 DNS 反解析（更快）
mtr 8.8.8.8                     # 动态追踪（推荐）

# telnet / nc — TCP 连通性
# 检查端口是否开放
nc -zv host 80                   # TCP 端口测试
nc -zv host 80-100               # 端口范围扫描
nc -zvw 3 host 443               # 设置超时 3 秒

# bash 内置 TCP 测试（不需要额外工具）
timeout 3 bash -c "echo >/dev/tcp/host/80" 2>/dev/null && echo "开放" || echo "关闭"
```

## 3. 端口与连接

```bash
# ss — 查看网络连接（替代 netstat）
ss -tlnp                         # TCP 监听端口
ss -ulnp                         # UDP 监听端口
ss -ant                          # 所有 TCP 连接
ss -s                            # 连接统计摘要
ss -ant state established         # 已建立的连接
ss -tnp | grep :80               # 查看 80 端口连接

# 统计连接状态
ss -ant | awk 'NR>1{s[$1]++} END{for(k in s) print k, s[k]}'

# 统计连接数最多的 IP
ss -ant | awk 'NR>1{print $5}' | cut -d: -f1 | sort | uniq -c | sort -rn | head -10
```

## 4. 数据传输

```bash
# curl — HTTP 请求工具
curl https://example.com              # GET 请求
curl -o file.zip URL                  # 下载到文件
curl -O URL                           # 保留原始文件名
curl -I URL                           # 只获取 HTTP 头
curl -s URL                           # 静默模式
curl -w "%{http_code}" -so /dev/null URL   # 只获取状态码

# POST 请求
curl -X POST -d '{"key":"value"}' -H "Content-Type: application/json" URL

# 带认证
curl -u user:pass URL
curl -H "Authorization: Bearer TOKEN" URL

# wget — 下载工具
wget URL                              # 下载
wget -c URL                           # 断点续传
wget -r -l 2 URL                      # 递归下载（深度 2）
wget --mirror URL                     # 镜像网站

# scp — 安全复制
scp file user@host:/path/             # 上传
scp user@host:/path/file ./           # 下载
scp -r dir/ user@host:/path/          # 传目录

# rsync — 高效同步
rsync -avz src/ user@host:/dest/      # 同步到远程
rsync -avz --delete src/ dest/        # 同步并删除目标端多余文件
rsync -avz --exclude='*.log' src/ dest/
```

## 5. 抓包与分析

```bash
# tcpdump — 命令行抓包
tcpdump -i eth0                       # 抓 eth0 的所有包
tcpdump -i eth0 port 80               # 只抓 80 端口
tcpdump -i eth0 host 192.168.1.100    # 只抓特定 IP
tcpdump -i eth0 -w capture.pcap       # 保存到文件
tcpdump -r capture.pcap               # 读取抓包文件
tcpdump -i eth0 -nn port 80 -c 100   # 抓 100 个包，不解析域名

# 常用过滤
tcpdump 'tcp port 80 and host 10.0.0.1'
tcpdump 'tcp[tcpflags] & tcp-syn != 0'   # 只看 SYN 包
```

## 6. 实战：端口扫描脚本

```bash
#!/usr/bin/env bash
# port_scan.sh — 端口扫描

set -euo pipefail

HOST="${1:?用法: $0 <主机> [起始端口] [结束端口]}"
START="${2:-1}"
END="${3:-1024}"

echo "扫描 $HOST 端口 $START-$END"
echo "=========================="

OPEN=0
for ((port=START; port<=END; port++)); do
    if timeout 1 bash -c "echo >/dev/tcp/$HOST/$port" 2>/dev/null; then
        SERVICE=$(grep -w "$port/tcp" /etc/services 2>/dev/null | awk '{print $1}' | head -1)
        printf "端口 %-6d 开放  %s\n" "$port" "${SERVICE:-未知}"
        ((OPEN++))
    fi
done

echo "=========================="
echo "开放端口: $OPEN 个"
```
