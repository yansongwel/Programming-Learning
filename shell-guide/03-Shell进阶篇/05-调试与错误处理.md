# 调试与错误处理

> 一句话说明：掌握 Shell 脚本的调试技巧和健壮的错误处理模式，写出生产级质量的脚本。

---

## 1. 调试方法

### 1.1 set 调试选项

```bash
set -x           # 打印每条命令及其展开结果（最常用）
set +x           # 关闭 -x

set -v           # 打印输入的原始命令
set -n           # 只检查语法，不执行

# 组合使用
set -xv          # 既打印原始命令又打印展开结果

# 运行时开启调试
bash -x script.sh

# 局部调试
set -x
# 这段代码会打印调试信息
problematic_function
set +x
# 这段不打印
```

### 1.2 调试输出技巧

```bash
# 自定义调试函数
DEBUG=${DEBUG:-false}

debug() {
    $DEBUG && echo "[DEBUG] $*" >&2
}

debug "变量值: VAR=$VAR"

# 运行方式
DEBUG=true bash script.sh     # 开启调试
bash script.sh                # 正常运行
```

### 1.3 ShellCheck

```bash
# 静态分析工具，发现常见错误
shellcheck script.sh

# 常见 ShellCheck 警告
# SC2086: 变量未加引号 → "$VAR"
# SC2034: 变量未使用
# SC2046: 命令替换未加引号
# SC2002: 无用的 cat → 直接用 < file
# SC2115: rm -rf "$DIR/" 中 DIR 为空时的风险
```

## 2. 错误处理模式

### 2.1 严格模式

```bash
#!/usr/bin/env bash
set -euo pipefail

# set -e: 命令失败立即退出
# set -u: 引用未定义变量报错
# set -o pipefail: 管道中任一命令失败则整体失败

# 需要允许失败的命令
grep "pattern" file || true              # 找不到也不退出
result=$(command) || result="default"    # 失败时用默认值
if ! command; then                       # if 中不受 -e 影响
    echo "命令失败"
fi
```

### 2.2 错误处理框架

```bash
#!/usr/bin/env bash
set -euo pipefail

LOG_FILE="/tmp/script_$(date +%Y%m%d).log"

log()   { echo "[$(date '+%H:%M:%S')] $*" | tee -a "$LOG_FILE"; }
error() { log "ERROR: $*" >&2; }
die()   { error "$@"; exit 1; }

on_error() {
    error "失败于行 $1: $2 (退出码: $3)"
    # 可在此发送告警
}
trap 'on_error ${LINENO} "${BASH_COMMAND}" $?' ERR

cleanup() {
    local exit_code=$?
    # 清理逻辑
    if (( exit_code != 0 )); then
        error "脚本异常退出 (退出码: $exit_code)"
    fi
}
trap cleanup EXIT

# 主逻辑
main() {
    log "开始执行"
    # 前置检查
    command -v docker &>/dev/null || die "docker 未安装"
    [[ -f config.yml ]] || die "配置文件不存在"

    log "执行完成"
}

main "$@"
```

## 3. 常见错误处理技巧

```bash
# 检查命令是否存在
require_cmd() {
    for cmd in "$@"; do
        command -v "$cmd" &>/dev/null || die "缺少命令: $cmd"
    done
}
require_cmd curl jq docker

# 检查 root 权限
[[ "$EUID" -eq 0 ]] || die "需要 root 权限"

# 安全的 cd
cd /some/path || die "无法进入目录"

# 带超时的操作
timeout 30 some_command || die "操作超时"

# 重试模式
retry() {
    local n=0 max=$1 delay=$2; shift 2
    until "$@"; do
        ((n++))
        (( n >= max )) && { error "重试 $max 次后仍失败"; return 1; }
        log "第 $n 次失败，${delay}s 后重试..."
        sleep "$delay"
    done
}
retry 3 5 curl -sf http://api.example.com/health
```

## 4. 实战：健壮的部署脚本

```bash
#!/usr/bin/env bash
set -euo pipefail

readonly SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
readonly LOG_FILE="/var/log/deploy_$(date +%Y%m%d_%H%M%S).log"

log()   { echo "[$(date '+%H:%M:%S')] $*" | tee -a "$LOG_FILE"; }
error() { log "ERROR: $*" >&2; }
die()   { error "$@"; exit 1; }
trap 'error "行 ${LINENO}: ${BASH_COMMAND} 失败"' ERR

# 前置检查
preflight() {
    log "执行前置检查..."
    command -v docker &>/dev/null || die "docker 未安装"
    [[ -f "$SCRIPT_DIR/docker-compose.yml" ]] || die "缺少 docker-compose.yml"
    docker info &>/dev/null || die "docker 服务未运行"
    log "前置检查通过"
}

# 备份
backup() {
    log "备份当前版本..."
    local backup_dir="/backup/$(date +%Y%m%d_%H%M%S)"
    mkdir -p "$backup_dir"
    docker compose config > "$backup_dir/docker-compose.yml.bak"
    log "备份完成: $backup_dir"
}

# 部署
deploy() {
    log "开始部署..."
    docker compose pull
    docker compose up -d
    log "等待服务就绪..."
    local retries=0
    until curl -sf http://localhost:8080/health &>/dev/null; do
        ((retries++))
        (( retries > 30 )) && die "服务未能在 30s 内就绪"
        sleep 1
    done
    log "部署成功！"
}

main() {
    preflight
    backup
    deploy
}

main "$@"
```

## 5. 小结

| 知识点 | 要点 |
|--------|------|
| 调试 | `set -x` 跟踪，`shellcheck` 静态检查 |
| 严格模式 | `set -euo pipefail` 标配 |
| ERR trap | 自动捕获错误并记录上下文 |
| EXIT trap | 确保清理工作执行 |
| 健壮模式 | 前置检查 + 错误处理 + 日志记录 |
