# 日志轮转与清理

> 一句话说明：掌握 logrotate 配置和自定义日志清理脚本，防止磁盘被日志撑爆。

---

## 1. logrotate 配置

```bash
# 全局配置: /etc/logrotate.conf
# 应用配置: /etc/logrotate.d/

# 示例: /etc/logrotate.d/myapp
cat > /etc/logrotate.d/myapp << 'EOF'
/var/log/myapp/*.log {
    daily                   # 每天轮转
    rotate 30               # 保留 30 份
    compress                # gzip 压缩
    delaycompress           # 延迟压缩（最近一份不压缩）
    missingok               # 文件不存在不报错
    notifempty              # 空文件不轮转
    copytruncate            # 复制后截断（不需要重启服务）
    dateext                 # 用日期做后缀（而非数字）
    dateformat -%Y%m%d      # 日期格式
    maxsize 100M            # 超过 100M 也触发轮转
    create 644 appuser appgroup   # 创建新文件的权限
    postrotate
        # 轮转后执行的命令
        systemctl reload myapp > /dev/null 2>&1 || true
    endscript
}
EOF

# 测试配置
logrotate -d /etc/logrotate.d/myapp     # 调试模式（不实际执行）
logrotate -f /etc/logrotate.d/myapp     # 强制执行
```

### Nginx 日志轮转

```bash
cat > /etc/logrotate.d/nginx << 'EOF'
/var/log/nginx/*.log {
    daily
    rotate 14
    compress
    delaycompress
    missingok
    notifempty
    sharedscripts
    postrotate
        if [ -f /var/run/nginx.pid ]; then
            kill -USR1 $(cat /var/run/nginx.pid)
        fi
    endscript
}
EOF
```

## 2. 自定义清理脚本

```bash
#!/usr/bin/env bash
# log_cleaner.sh — 日志清理脚本

set -euo pipefail

# 配置
LOG_DIRS=(
    "/var/log/myapp"
    "/var/log/nginx"
    "/opt/app/logs"
)
MAX_AGE_DAYS=30
MAX_SIZE_MB=500

log() { echo "[$(date '+%H:%M:%S')] $*"; }

for dir in "${LOG_DIRS[@]}"; do
    [[ -d "$dir" ]] || continue

    log "清理目录: $dir"

    # 按时间清理
    old_count=$(find "$dir" -name "*.log*" -mtime +$MAX_AGE_DAYS | wc -l)
    if (( old_count > 0 )); then
        find "$dir" -name "*.log*" -mtime +$MAX_AGE_DAYS -delete
        log "  删除 ${old_count} 个超过 ${MAX_AGE_DAYS} 天的文件"
    fi

    # 压缩未压缩的旧日志
    find "$dir" -name "*.log.*" ! -name "*.gz" -mtime +1 -exec gzip {} \;

    # 按大小清理（目录超过阈值时删除最旧的）
    dir_size=$(du -sm "$dir" | awk '{print $1}')
    while (( dir_size > MAX_SIZE_MB )); do
        oldest=$(find "$dir" -name "*.log*" -type f -printf '%T+ %p\n' | sort | head -1 | cut -d' ' -f2-)
        [[ -z "$oldest" ]] && break
        rm -f "$oldest"
        log "  删除(超限): $oldest"
        dir_size=$(du -sm "$dir" | awk '{print $1}')
    done

    log "  当前大小: ${dir_size}MB"
done

# 清理 journal 日志
if command -v journalctl &>/dev/null; then
    journalctl --vacuum-size=200M --vacuum-time=30d 2>/dev/null
    log "journal 日志已清理"
fi

log "清理完成"
```

## 3. 大文件检测

```bash
#!/usr/bin/env bash
# find_large_logs.sh — 查找大日志文件

set -euo pipefail

THRESHOLD="${1:-100}"   # 默认 100MB

echo "查找超过 ${THRESHOLD}MB 的日志文件..."
echo ""

printf "%-10s %-15s %s\n" "大小" "修改时间" "路径"
printf "%-10s %-15s %s\n" "----" "--------" "----"

find /var/log /opt /home -name "*.log*" -size +${THRESHOLD}M -type f 2>/dev/null | \
while read -r file; do
    size=$(du -h "$file" | cut -f1)
    mtime=$(stat -c %y "$file" | cut -d. -f1)
    printf "%-10s %-15s %s\n" "$size" "$mtime" "$file"
done | sort -rh

echo ""

# 检查已删除但未释放的文件
echo "已删除但未释放的文件:"
lsof +L1 2>/dev/null | awk '$7>100000000{printf "  %s (%.0fMB) PID:%s\n", $9, $7/1024/1024, $2}' || echo "  (无)"
```

## 4. 小结

| 方式 | 适用场景 |
|------|----------|
| logrotate | 标准日志轮转（首选） |
| 自定义脚本 | 特殊清理逻辑 |
| `find -mtime -delete` | 快速清理过期文件 |
| `journalctl --vacuum` | systemd 日志清理 |

**关键配置项**：`daily/weekly` + `rotate N` + `compress` + `maxsize` + `copytruncate`
