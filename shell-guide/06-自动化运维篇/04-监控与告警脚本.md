# 监控与告警脚本

> 一句话说明：用 Shell 实现系统资源、服务状态和业务指标的监控与告警。

---

## 1. 资源监控

```bash
#!/usr/bin/env bash
# resource_monitor.sh — 系统资源监控

set -euo pipefail

# 阈值配置
CPU_WARN=80
MEM_WARN=85
DISK_WARN=85

ALERTS=()

# CPU 检查
cpu_usage=$(top -bn1 | grep "Cpu(s)" | awk '{print int(100-$8)}')
(( cpu_usage > CPU_WARN )) && ALERTS+=("CPU 使用率 ${cpu_usage}% (阈值 ${CPU_WARN}%)")

# 内存检查
mem_usage=$(free | awk '/Mem:/{printf "%d", $3/$2*100}')
(( mem_usage > MEM_WARN )) && ALERTS+=("内存使用率 ${mem_usage}% (阈值 ${MEM_WARN}%)")

# 磁盘检查
while read -r usage mount; do
    pct=${usage%\%}
    (( pct > DISK_WARN )) && ALERTS+=("磁盘 ${mount} 使用率 ${usage} (阈值 ${DISK_WARN}%)")
done < <(df -h | awk 'NR>1 && $1!~/tmpfs/{print $5, $6}')

# 发送告警
if (( ${#ALERTS[@]} > 0 )); then
    MSG="[$(hostname)] 资源告警 $(date '+%H:%M')\n"
    for alert in "${ALERTS[@]}"; do
        MSG+="- $alert\n"
    done
    echo -e "$MSG"
    # send_alert "$MSG"  # 调用告警发送函数
fi
```

## 2. 服务监控

```bash
#!/usr/bin/env bash
# service_monitor.sh — 服务状态监控

set -euo pipefail

SERVICES=("nginx" "mysql" "redis")
URLS=("http://localhost:80" "http://localhost:8080/health")

check_services() {
    for svc in "${SERVICES[@]}"; do
        if ! systemctl is-active --quiet "$svc" 2>/dev/null; then
            echo "服务异常: $svc 未运行"
        fi
    done
}

check_urls() {
    for url in "${URLS[@]}"; do
        code=$(curl -sf -o /dev/null -w "%{http_code}" --connect-timeout 5 "$url" 2>/dev/null || echo "000")
        if [[ "$code" != "200" ]]; then
            echo "URL 异常: $url (状态码: $code)"
        fi
    done
}

check_ports() {
    local ports=(22 80 443 3306 6379)
    for port in "${ports[@]}"; do
        if ! ss -tlnp | grep -q ":${port} "; then
            echo "端口未监听: $port"
        fi
    done
}

# 执行检查
ISSUES=()
while IFS= read -r issue; do
    [[ -n "$issue" ]] && ISSUES+=("$issue")
done < <(check_services; check_urls; check_ports)

if (( ${#ISSUES[@]} > 0 )); then
    echo "[告警] $(hostname) - $(date '+%H:%M:%S')"
    printf '  %s\n' "${ISSUES[@]}"
fi
```

## 3. 告警发送

```bash
# 钉钉机器人
send_dingtalk() {
    local webhook="$1" message="$2"
    curl -sf -H 'Content-Type: application/json' -d "{
        \"msgtype\": \"text\",
        \"text\": {\"content\": \"$message\"}
    }" "$webhook" > /dev/null
}

# 企业微信机器人
send_wechat() {
    local webhook="$1" message="$2"
    curl -sf -H 'Content-Type: application/json' -d "{
        \"msgtype\": \"text\",
        \"text\": {\"content\": \"$message\"}
    }" "$webhook" > /dev/null
}

# 飞书机器人
send_feishu() {
    local webhook="$1" message="$2"
    curl -sf -H 'Content-Type: application/json' -d "{
        \"msg_type\": \"text\",
        \"content\": {\"text\": \"$message\"}
    }" "$webhook" > /dev/null
}

# 邮件
send_email() {
    local to="$1" subject="$2" body="$3"
    echo "$body" | mail -s "$subject" "$to"
}
```

## 4. 完整监控框架

```bash
#!/usr/bin/env bash
# monitor.sh — 通用监控框架

set -euo pipefail

HOSTNAME=$(hostname)
WEBHOOK="${ALERT_WEBHOOK:-}"
ALERT_FILE="/tmp/monitor_alerts"
INTERVAL=60

log() { echo "[$(date '+%H:%M:%S')] $*"; }

collect_metrics() {
    CPU=$(top -bn1 | grep "Cpu(s)" | awk '{print int(100-$8)}')
    MEM=$(free | awk '/Mem:/{printf "%d", $3/$2*100}')
    LOAD=$(cat /proc/loadavg | awk '{print $1}')
    DISK_MAX=$(df -h | awk 'NR>1 && $1!~/tmpfs/{gsub(/%/,""); if(+$5>max) max=+$5} END{print max}')
    CONN=$(ss -ant state established | wc -l)
}

check_thresholds() {
    local alerts=""
    (( CPU > 85 )) && alerts+="CPU: ${CPU}%\n"
    (( MEM > 90 )) && alerts+="内存: ${MEM}%\n"
    (( DISK_MAX > 90 )) && alerts+="磁盘最大: ${DISK_MAX}%\n"
    echo -ne "$alerts"
}

main() {
    log "监控启动 (间隔: ${INTERVAL}s)"

    while true; do
        collect_metrics
        ALERTS=$(check_thresholds)

        if [[ -n "$ALERTS" ]]; then
            # 防止重复告警（1 小时内同样的告警不重复发送）
            ALERT_HASH=$(echo "$ALERTS" | md5sum | cut -d' ' -f1)
            if [[ ! -f "$ALERT_FILE" ]] || ! grep -q "$ALERT_HASH" "$ALERT_FILE" 2>/dev/null; then
                MSG="[${HOSTNAME}] 监控告警\n${ALERTS}"
                log "发送告警: $(echo -e "$ALERTS" | tr '\n' ' ')"
                [[ -n "$WEBHOOK" ]] && send_dingtalk "$WEBHOOK" "$(echo -e "$MSG")"
                echo "$ALERT_HASH $(date +%s)" >> "$ALERT_FILE"
            fi
        fi

        # 清理 1 小时前的告警记录
        if [[ -f "$ALERT_FILE" ]]; then
            local now; now=$(date +%s)
            awk -v now="$now" '$2 > now-3600' "$ALERT_FILE" > "${ALERT_FILE}.tmp"
            mv "${ALERT_FILE}.tmp" "$ALERT_FILE"
        fi

        sleep "$INTERVAL"
    done
}

main
```

## 5. 小结

| 监控类型 | 方法 |
|----------|------|
| CPU/内存 | `top`, `free`, `/proc` |
| 磁盘 | `df -h` |
| 服务状态 | `systemctl is-active` |
| URL 检测 | `curl -sf -w "%{http_code}"` |
| 端口检测 | `ss -tlnp`, `nc -zv` |
| 告警通道 | 钉钉/企微/飞书 Webhook |
