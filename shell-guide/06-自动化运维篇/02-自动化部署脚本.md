# 自动化部署脚本

> 一句话说明：用 Shell 实现应用的自动化构建、发布和回滚，支持滚动部署和蓝绿部署。

---

## 1. 部署脚本基础模板

```bash
#!/usr/bin/env bash
# deploy.sh — 通用部署脚本模板

set -euo pipefail

# ========================
# 配置
# ========================
APP_NAME="myapp"
DEPLOY_DIR="/opt/${APP_NAME}"
RELEASE_DIR="${DEPLOY_DIR}/releases"
CURRENT_LINK="${DEPLOY_DIR}/current"
SHARED_DIR="${DEPLOY_DIR}/shared"
KEEP_RELEASES=5
LOG_FILE="/var/log/${APP_NAME}/deploy.log"

# ========================
# 函数
# ========================
log()   { echo "[$(date '+%H:%M:%S')] $*" | tee -a "$LOG_FILE"; }
error() { log "ERROR: $*" >&2; }
die()   { error "$@"; exit 1; }

# 初始化目录
init() {
    mkdir -p "$RELEASE_DIR" "$SHARED_DIR/logs" "$SHARED_DIR/config"
    log "目录初始化完成"
}

# 发布新版本
deploy() {
    local version="${1:?请指定版本号}"
    local release_path="${RELEASE_DIR}/${version}"

    [[ -d "$release_path" ]] && die "版本 $version 已存在"

    log "开始部署 v${version}..."

    # 1. 创建发布目录
    mkdir -p "$release_path"

    # 2. 下载/复制代码
    log "下载代码..."
    # tar xzf "/tmp/${APP_NAME}-${version}.tar.gz" -C "$release_path"
    # 或 git clone
    # 或 rsync

    # 3. 链接共享文件
    ln -sf "$SHARED_DIR/config" "$release_path/config"
    ln -sf "$SHARED_DIR/logs" "$release_path/logs"

    # 4. 切换 current 链接
    ln -sfn "$release_path" "$CURRENT_LINK"
    log "已切换到 v${version}"

    # 5. 重启服务
    systemctl restart "$APP_NAME" || die "服务重启失败"

    # 6. 健康检查
    health_check || { rollback; die "健康检查失败，已回滚"; }

    # 7. 清理旧版本
    cleanup_old_releases

    log "部署 v${version} 完成!"
}

# 回滚
rollback() {
    local previous
    previous=$(ls -1t "$RELEASE_DIR" | sed -n '2p')
    [[ -z "$previous" ]] && die "没有可回滚的版本"

    log "回滚到: $previous"
    ln -sfn "${RELEASE_DIR}/${previous}" "$CURRENT_LINK"
    systemctl restart "$APP_NAME"
    log "回滚完成"
}

# 健康检查
health_check() {
    local retries=0 max=30
    log "等待服务就绪..."
    until curl -sf http://localhost:8080/health &>/dev/null; do
        ((retries++))
        (( retries >= max )) && return 1
        sleep 1
    done
    log "健康检查通过 (${retries}s)"
}

# 清理旧版本
cleanup_old_releases() {
    local count
    count=$(ls -1 "$RELEASE_DIR" | wc -l)
    if (( count > KEEP_RELEASES )); then
        ls -1t "$RELEASE_DIR" | tail -n +$((KEEP_RELEASES+1)) | while read -r old; do
            rm -rf "${RELEASE_DIR}/${old}"
            log "已清理旧版本: $old"
        done
    fi
}

# ========================
# 主入口
# ========================
case "${1:-help}" in
    init)     init ;;
    deploy)   deploy "${2:-}" ;;
    rollback) rollback ;;
    status)   ls -la "$CURRENT_LINK"; ls -1t "$RELEASE_DIR" ;;
    *)        echo "用法: $0 {init|deploy <版本>|rollback|status}" ;;
esac
```

## 2. 滚动部署

```bash
#!/usr/bin/env bash
# rolling_deploy.sh — 滚动部署到多台服务器

set -euo pipefail

SERVERS=("web01" "web02" "web03")
PACKAGE="$1"
BATCH_SIZE=1    # 每次部署的服务器数

log() { echo "[$(date '+%H:%M:%S')] $*"; }

deploy_to_host() {
    local host="$1" pkg="$2"
    log "部署到 $host..."

    # 1. 从负载均衡中摘除
    # ssh lb "disable_backend $host"

    # 2. 上传并部署
    scp "$pkg" "${host}:/tmp/"
    ssh "$host" "cd /opt/myapp && tar xzf /tmp/$(basename "$pkg") && systemctl restart myapp"

    # 3. 健康检查
    local retries=0
    until ssh "$host" "curl -sf http://localhost:8080/health" &>/dev/null; do
        ((retries++))
        (( retries >= 30 )) && { log "$host 健康检查失败!"; return 1; }
        sleep 1
    done

    # 4. 重新加入负载均衡
    # ssh lb "enable_backend $host"

    log "$host 部署成功"
}

for server in "${SERVERS[@]}"; do
    if ! deploy_to_host "$server" "$PACKAGE"; then
        log "部署在 $server 失败，停止滚动部署"
        exit 1
    fi
done

log "全部部署完成!"
```

## 3. Git 自动部署

```bash
#!/usr/bin/env bash
# git_deploy.sh — 基于 Git 的部署

set -euo pipefail

REPO_URL="git@github.com:company/myapp.git"
DEPLOY_DIR="/opt/myapp"
BRANCH="${1:-main}"

log() { echo "[$(date '+%H:%M:%S')] $*"; }

cd "$DEPLOY_DIR" || { git clone "$REPO_URL" "$DEPLOY_DIR"; cd "$DEPLOY_DIR"; }

log "拉取最新代码 ($BRANCH)..."
git fetch origin
git checkout "$BRANCH"
git pull origin "$BRANCH"

COMMIT=$(git rev-parse --short HEAD)
log "当前版本: $COMMIT"

log "安装依赖..."
# npm install --production
# pip install -r requirements.txt

log "重启服务..."
systemctl restart myapp

log "部署完成: $COMMIT"
```

## 4. 小结

| 部署方式 | 适用场景 |
|----------|----------|
| 符号链接切换 | 单机部署，快速回滚 |
| 滚动部署 | 多机集群，逐台更新 |
| Git 部署 | 开发/测试环境 |
| 蓝绿部署 | 零停机，需要双倍资源 |

核心要素：版本管理 + 健康检查 + 回滚机制 + 日志记录
