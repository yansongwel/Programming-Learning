# 备份与恢复方案

> 一句话说明：用 Shell 实现自动化备份策略，包括文件备份、数据库备份和异地备份。

---

## 1. 文件备份

### 1.1 tar 全量备份

```bash
#!/usr/bin/env bash
# full_backup.sh — 全量备份

set -euo pipefail

SOURCE="${1:?用法: $0 <源目录>}"
BACKUP_DIR="/backup"
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="${BACKUP_DIR}/full_$(basename "$SOURCE")_${DATE}.tar.gz"

mkdir -p "$BACKUP_DIR"

echo "全量备份: $SOURCE"
tar czf "$BACKUP_FILE" -C "$(dirname "$SOURCE")" "$(basename "$SOURCE")"

SIZE=$(du -h "$BACKUP_FILE" | cut -f1)
echo "完成: $BACKUP_FILE ($SIZE)"
```

### 1.2 rsync 增量备份

```bash
#!/usr/bin/env bash
# incremental_backup.sh — 基于 rsync 的增量备份

set -euo pipefail

SOURCE="${1:?用法: $0 <源目录>}"
BACKUP_BASE="/backup/incremental"
DATE=$(date +%Y%m%d_%H%M%S)
CURRENT="${BACKUP_BASE}/${DATE}"
LATEST="${BACKUP_BASE}/latest"

mkdir -p "$BACKUP_BASE"

# --link-dest 利用硬链接实现增量（未变化的文件硬链接到上次备份）
rsync -avz --delete \
    ${LATEST:+--link-dest="$LATEST"} \
    "$SOURCE/" "$CURRENT/"

# 更新 latest 链接
ln -sfn "$CURRENT" "$LATEST"

echo "增量备份完成: $CURRENT"
```

## 2. 数据库备份

```bash
#!/usr/bin/env bash
# db_backup.sh — MySQL 数据库备份

set -euo pipefail

DB_HOST="${DB_HOST:-localhost}"
DB_USER="${DB_USER:-root}"
DB_PASS="${DB_PASS:?请设置 DB_PASS 环境变量}"
DATABASES="${1:-all}"
BACKUP_DIR="/backup/mysql"
DATE=$(date +%Y%m%d_%H%M%S)

mkdir -p "$BACKUP_DIR"

if [[ "$DATABASES" == "all" ]]; then
    DUMP_FILE="${BACKUP_DIR}/all_${DATE}.sql.gz"
    mysqldump -h"$DB_HOST" -u"$DB_USER" -p"$DB_PASS" \
        --all-databases --single-transaction --routines --triggers \
        | gzip > "$DUMP_FILE"
    echo "全库备份: $DUMP_FILE ($(du -h "$DUMP_FILE" | cut -f1))"
else
    for db in $DATABASES; do
        DUMP_FILE="${BACKUP_DIR}/${db}_${DATE}.sql.gz"
        mysqldump -h"$DB_HOST" -u"$DB_USER" -p"$DB_PASS" \
            --single-transaction "$db" | gzip > "$DUMP_FILE"
        echo "备份: $db → $DUMP_FILE ($(du -h "$DUMP_FILE" | cut -f1))"
    done
fi

# 清理 7 天前的备份
find "$BACKUP_DIR" -name "*.sql.gz" -mtime +7 -delete
echo "已清理过期备份"
```

## 3. 恢复验证

```bash
# 文件恢复
tar xzf /backup/full_data_20240115.tar.gz -C /restore/

# MySQL 恢复
gunzip < /backup/mysql/mydb_20240115.sql.gz | mysql -u root -p mydb

# 验证恢复完整性
verify_backup() {
    local backup="$1"
    if [[ "$backup" == *.tar.gz ]]; then
        tar tzf "$backup" > /dev/null && echo "验证通过: $backup" || echo "验证失败: $backup"
    elif [[ "$backup" == *.sql.gz ]]; then
        gunzip -t "$backup" && echo "验证通过: $backup" || echo "验证失败: $backup"
    fi
}
```

## 4. 完整备份策略脚本

```bash
#!/usr/bin/env bash
# backup_strategy.sh — 企业级备份策略

set -euo pipefail

BACKUP_ROOT="/backup"
LOG_FILE="/var/log/backup.log"
RETENTION_DAILY=7
RETENTION_WEEKLY=4
RETENTION_MONTHLY=6

log() { echo "[$(date '+%Y-%m-%d %H:%M:%S')] $*" | tee -a "$LOG_FILE"; }

# 判断备份类型
DAY=$(date +%d)
DOW=$(date +%u)   # 1=周一

if (( DAY == 1 )); then
    TYPE="monthly"
elif (( DOW == 7 )); then
    TYPE="weekly"
else
    TYPE="daily"
fi

DEST="${BACKUP_ROOT}/${TYPE}/$(date +%Y%m%d)"
mkdir -p "$DEST"

log "开始 ${TYPE} 备份..."

# 1. 文件备份
log "备份文件..."
rsync -az /opt/app/ "${DEST}/app/"
rsync -az /etc/ "${DEST}/etc/"

# 2. 数据库备份
log "备份数据库..."
mysqldump --all-databases --single-transaction 2>/dev/null | gzip > "${DEST}/mysql_all.sql.gz" || log "MySQL 备份跳过"

# 3. 验证
log "验证备份..."
[[ -d "${DEST}/app" ]] && log "文件备份: OK" || log "文件备份: FAIL"
[[ -f "${DEST}/mysql_all.sql.gz" ]] && gunzip -t "${DEST}/mysql_all.sql.gz" 2>/dev/null && log "数据库备份: OK"

# 4. 清理
log "清理过期备份..."
find "${BACKUP_ROOT}/daily" -maxdepth 1 -mtime +$RETENTION_DAILY -exec rm -rf {} \; 2>/dev/null
find "${BACKUP_ROOT}/weekly" -maxdepth 1 -mtime +$((RETENTION_WEEKLY*7)) -exec rm -rf {} \; 2>/dev/null
find "${BACKUP_ROOT}/monthly" -maxdepth 1 -mtime +$((RETENTION_MONTHLY*30)) -exec rm -rf {} \; 2>/dev/null

log "${TYPE} 备份完成: $DEST ($(du -sh "$DEST" | cut -f1))"
```

## 5. 小结

| 备份类型 | 工具 | 特点 |
|----------|------|------|
| 全量 | `tar czf` | 简单，恢复快，空间大 |
| 增量 | `rsync --link-dest` | 空间省，硬链接去重 |
| 数据库 | `mysqldump/pg_dump` | `--single-transaction` 不锁表 |
| 异地 | `rsync` + SSH | 抵御物理灾难 |

**3-2-1 原则**：3 份副本、2 种存储介质、1 份异地备份
