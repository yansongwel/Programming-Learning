# Shell 循环结构

> 一句话说明：掌握 for、while、until 循环及循环控制，让脚本自动化处理重复任务。

---

## 1. for 循环

### 1.1 列表遍历

```bash
# 遍历字符串列表
for fruit in apple banana cherry; do
    echo "水果: $fruit"
done

# 遍历变量列表
SERVERS="web01 web02 web03 db01"
for server in $SERVERS; do
    echo "检查服务器: $server"
done

# 遍历文件
for file in /etc/*.conf; do
    echo "配置文件: $file"
done

# 遍历命令输出
for user in $(cut -d: -f1 /etc/passwd); do
    echo "用户: $user"
done

# 遍历数组
PORTS=(80 443 8080 3306 6379)
for port in "${PORTS[@]}"; do
    echo "检查端口: $port"
done
```

### 1.2 C 风格 for 循环

```bash
# 类似 C/Java 的写法
for ((i=1; i<=10; i++)); do
    echo "第 $i 次"
done

# 倒序
for ((i=10; i>=1; i--)); do
    echo "倒计时: $i"
done

# 步进
for ((i=0; i<=100; i+=10)); do
    echo "进度: $i%"
done
```

### 1.3 序列生成

```bash
# {start..end}
for i in {1..5}; do
    echo "$i"
done

# {start..end..step}
for i in {0..100..10}; do
    echo "$i"
done

# seq 命令（更灵活，支持变量）
END=5
for i in $(seq 1 "$END"); do
    echo "$i"
done

# seq 指定步长和格式
for i in $(seq -w 001 010); do
    echo "server-$i"       # server-001 ... server-010
done
```

## 2. while 循环

```bash
# 基本语法
COUNT=1
while [[ $COUNT -le 5 ]]; do
    echo "第 $COUNT 次循环"
    ((COUNT++))
done

# 无限循环
while true; do
    echo "运行中... (Ctrl+C 退出)"
    sleep 1
done

# 读取文件的标准模式
while IFS= read -r line; do
    echo "行内容: $line"
done < /etc/hostname

# 读取命令输出
ps aux | while read -r user pid cpu mem _ _ _ _ _ _ cmd; do
    if (( $(echo "$mem > 5.0" | bc -l) )); then
        echo "高内存进程: $cmd (${mem}%)"
    fi
done

# ⚠️ 管道中的 while 在子 Shell 中运行，变量不会传回
COUNT=0
echo -e "a\nb\nc" | while read -r line; do
    ((COUNT++))
done
echo "$COUNT"   # 0（不是 3！）

# 解决方案：进程替换
COUNT=0
while read -r line; do
    ((COUNT++))
done < <(echo -e "a\nb\nc")
echo "$COUNT"   # 3 ✓
```

## 3. until 循环

与 while 相反：条件为 false 时继续循环，为 true 时停止。

```bash
# 等待服务启动
until curl -sf http://localhost:8080/health &>/dev/null; do
    echo "等待服务启动..."
    sleep 2
done
echo "服务已就绪！"

# 等待文件出现
until [[ -f /tmp/signal.txt ]]; do
    echo "等待信号文件..."
    sleep 1
done
echo "收到信号，继续执行"

# 带超时的等待
TIMEOUT=30
ELAPSED=0
until systemctl is-active --quiet nginx; do
    if (( ELAPSED >= TIMEOUT )); then
        echo "超时：nginx 未能在 ${TIMEOUT}s 内启动"
        exit 1
    fi
    sleep 1
    ((ELAPSED++))
done
echo "nginx 已启动 (耗时 ${ELAPSED}s)"
```

## 4. 循环控制

### 4.1 break 与 continue

```bash
# break — 跳出循环
for i in {1..100}; do
    if (( i > 5 )); then
        break
    fi
    echo "$i"
done

# continue — 跳过当前迭代
for file in /var/log/*.log; do
    # 跳过空文件
    [[ ! -s "$file" ]] && continue
    echo "$file: $(wc -l < "$file") 行"
done

# break N — 跳出 N 层循环
for i in {1..3}; do
    for j in {1..3}; do
        if (( i == 2 && j == 2 )); then
            break 2    # 跳出两层循环
        fi
        echo "i=$i, j=$j"
    done
done
```

### 4.2 循环与管道结合

```bash
# 批量检查端口
for port in 22 80 443 3306 6379 8080; do
    if timeout 2 bash -c "echo >/dev/tcp/localhost/$port" 2>/dev/null; then
        echo "端口 $port: 开放"
    else
        echo "端口 $port: 关闭"
    fi
done

# 批量 ping 检测
for ip in 192.168.1.{1..254}; do
    ping -c 1 -W 1 "$ip" &>/dev/null && echo "$ip 在线" &
done
wait    # 等待所有后台任务完成
```

## 5. select 菜单循环

```bash
#!/usr/bin/env bash
# 交互式菜单

echo "请选择操作："
select action in "查看进程" "查看磁盘" "查看内存" "退出"; do
    case "$action" in
        "查看进程") ps aux --sort=-%mem | head -10 ;;
        "查看磁盘") df -h ;;
        "查看内存") free -h ;;
        "退出") echo "再见！"; break ;;
        *) echo "无效选项，请重新选择" ;;
    esac
done
```

## 6. 实战练习

### 练习 1：批量用户创建

```bash
#!/usr/bin/env bash
# create_users.sh — 从文件批量创建用户

set -euo pipefail

USER_FILE="${1:?用法: $0 <用户列表文件>}"
# 文件格式: 用户名:密码:描述（每行一个）

[[ ! -f "$USER_FILE" ]] && { echo "文件不存在: $USER_FILE"; exit 1; }

SUCCESS=0
FAIL=0

while IFS=: read -r username password comment; do
    # 跳过空行和注释
    [[ -z "$username" || "$username" =~ ^# ]] && continue

    if id "$username" &>/dev/null; then
        echo "跳过: $username (已存在)"
        continue
    fi

    if useradd -m -c "${comment:-$username}" "$username" 2>/dev/null; then
        echo "$username:$password" | chpasswd
        echo "创建成功: $username"
        ((SUCCESS++))
    else
        echo "创建失败: $username"
        ((FAIL++))
    fi
done < "$USER_FILE"

echo ""
echo "完成: 成功 $SUCCESS 个, 失败 $FAIL 个"
```

### 练习 2：服务监控守护

```bash
#!/usr/bin/env bash
# watchdog.sh — 监控服务并自动重启

set -euo pipefail

SERVICE="${1:?用法: $0 <服务名> [检查间隔秒]}"
INTERVAL="${2:-10}"
MAX_RESTART=3
RESTART_COUNT=0

echo "开始监控 ${SERVICE}, 检查间隔 ${INTERVAL}s"

while true; do
    if ! systemctl is-active --quiet "$SERVICE"; then
        ((RESTART_COUNT++))
        echo "[$(date '+%H:%M:%S')] ${SERVICE} 已停止 (第 ${RESTART_COUNT} 次)"

        if (( RESTART_COUNT > MAX_RESTART )); then
            echo "已达最大重启次数 ${MAX_RESTART}，退出监控"
            exit 1
        fi

        echo "尝试重启..."
        systemctl start "$SERVICE" && echo "重启成功" || echo "重启失败"
    else
        RESTART_COUNT=0    # 服务正常则重置计数
    fi

    sleep "$INTERVAL"
done
```

## 7. 小结

| 知识点 | 要点 |
|--------|------|
| for-in | 遍历列表、文件、命令输出 |
| C 风格 for | `for ((i=0; i<n; i++))` |
| while | 条件为 true 时继续循环 |
| until | 条件为 false 时继续循环 |
| break/continue | 跳出/跳过，支持指定层数 |
| select | 快速创建交互式菜单 |
| 管道+while | 用进程替换 `< <(cmd)` 避免子 Shell 问题 |
