# Shell 函数

> 一句话说明：将重复逻辑封装成函数，提高脚本的可读性、可维护性和复用性。

---

## 1. 函数定义与调用

```bash
# 方式一：标准写法（推荐）
greet() {
    echo "Hello, $1!"
}

# 方式二：带 function 关键字
function greet {
    echo "Hello, $1!"
}

# 调用函数（不需要括号）
greet "World"     # Hello, World!
greet "Shell"     # Hello, Shell!
```

## 2. 参数与返回值

### 2.1 函数参数

```bash
# 函数内的 $1, $2 是函数的参数，不是脚本的参数
show_info() {
    echo "函数名: ${FUNCNAME[0]}"
    echo "参数个数: $#"
    echo "第一个参数: $1"
    echo "第二个参数: $2"
    echo "所有参数: $@"
}

show_info "hello" "world"
```

### 2.2 返回值

```bash
# return 只能返回 0-255 的整数（退出码）
is_root() {
    [[ "$EUID" -eq 0 ]]   # 最后一条命令的退出码就是返回值
}

if is_root; then
    echo "是 root"
else
    echo "不是 root"
fi

# 返回字符串/复杂数据：用 echo + 命令替换
get_ip() {
    hostname -I | awk '{print $1}'
}
MY_IP=$(get_ip)
echo "IP: $MY_IP"

# 返回多个值
get_memory() {
    local total used
    read -r total used <<< "$(free -m | awk '/Mem:/{print $2, $3}')"
    echo "$total $used"
}
read -r TOTAL USED <<< "$(get_memory)"
echo "内存: ${USED}MB / ${TOTAL}MB"
```

### 2.3 局部变量

```bash
# local 声明局部变量（强烈推荐！）
counter() {
    local count=0             # 局部变量，不影响外部
    for i in {1..5}; do
        ((count++))
    done
    echo "$count"
}

count=100                     # 外部变量
result=$(counter)
echo "函数返回: $result"     # 5
echo "外部变量: $count"      # 100（未被修改）
```

## 3. 常用函数模式

### 3.1 日志函数

```bash
# 带颜色和级别的日志
RED='\033[0;31m'
YELLOW='\033[0;33m'
GREEN='\033[0;32m'
NC='\033[0m'

log_info()  { echo -e "${GREEN}[INFO]${NC}  $(date '+%H:%M:%S') $*"; }
log_warn()  { echo -e "${YELLOW}[WARN]${NC}  $(date '+%H:%M:%S') $*" >&2; }
log_error() { echo -e "${RED}[ERROR]${NC} $(date '+%H:%M:%S') $*" >&2; }

die() {
    log_error "$@"
    exit 1
}

# 使用
log_info "开始部署..."
log_warn "磁盘空间不足 20%"
log_error "数据库连接失败"
die "无法继续，退出"
```

### 3.2 重试函数

```bash
retry() {
    local max_attempts="${1:?}"
    local delay="${2:?}"
    shift 2
    local cmd=("$@")

    for ((i=1; i<=max_attempts; i++)); do
        if "${cmd[@]}"; then
            return 0
        fi
        echo "第 $i 次尝试失败，${delay}s 后重试..."
        sleep "$delay"
    done

    echo "已达最大重试次数 $max_attempts"
    return 1
}

# 使用：最多重试 3 次，间隔 5 秒
retry 3 5 curl -sf http://api.example.com/health
```

### 3.3 确认函数

```bash
confirm() {
    local prompt="${1:-确认执行?}"
    read -rp "${prompt} [y/N]: " answer
    [[ "$answer" =~ ^[Yy]$ ]]
}

if confirm "是否重启 nginx"; then
    systemctl restart nginx
    echo "已重启"
else
    echo "已取消"
fi
```

### 3.4 参数解析

```bash
#!/usr/bin/env bash
set -euo pipefail

usage() {
    cat << EOF
Usage: $(basename "$0") [OPTIONS]

Options:
    -h, --help          显示帮助
    -v, --verbose       详细输出
    -f, --file FILE     指定文件
    -n, --count NUM     指定数量（默认: 10）
    -d, --dry-run       仅模拟，不实际执行
EOF
    exit 0
}

# 默认值
VERBOSE=false
FILE=""
COUNT=10
DRY_RUN=false

parse_args() {
    while [[ $# -gt 0 ]]; do
        case "$1" in
            -h|--help)    usage ;;
            -v|--verbose) VERBOSE=true; shift ;;
            -f|--file)    FILE="$2"; shift 2 ;;
            -n|--count)   COUNT="$2"; shift 2 ;;
            -d|--dry-run) DRY_RUN=true; shift ;;
            --)           shift; break ;;
            -*)           die "未知选项: $1" ;;
            *)            break ;;
        esac
    done
    REMAINING_ARGS=("$@")
}

parse_args "$@"

$VERBOSE && echo "详细模式已开启"
echo "文件: ${FILE:-未指定}"
echo "数量: $COUNT"
$DRY_RUN && echo "模拟模式"
```

## 4. 函数库

将常用函数放到单独文件中，在其他脚本中 source 加载。

```bash
# lib/common.sh — 公共函数库
#!/usr/bin/env bash

readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly RED='\033[0;31m'
readonly GREEN='\033[0;32m'
readonly YELLOW='\033[0;33m'
readonly NC='\033[0m'

log_info()  { echo -e "${GREEN}[INFO]${NC}  $(date '+%H:%M:%S') $*"; }
log_warn()  { echo -e "${YELLOW}[WARN]${NC}  $(date '+%H:%M:%S') $*" >&2; }
log_error() { echo -e "${RED}[ERROR]${NC} $(date '+%H:%M:%S') $*" >&2; }
die()       { log_error "$@"; exit 1; }

require_root() {
    [[ "$EUID" -eq 0 ]] || die "请使用 root 权限运行此脚本"
}

require_cmd() {
    for cmd in "$@"; do
        command -v "$cmd" &>/dev/null || die "缺少命令: $cmd"
    done
}
```

```bash
# deploy.sh — 使用函数库
#!/usr/bin/env bash
set -euo pipefail

source "$(dirname "$0")/lib/common.sh"

require_root
require_cmd docker docker-compose curl

log_info "开始部署..."
# 后续逻辑...
```

## 5. 实战练习

### 练习：工具箱脚本

```bash
#!/usr/bin/env bash
# toolbox.sh — 运维工具箱

set -euo pipefail

show_cpu()    { echo "CPU 核心: $(nproc)"; top -bn1 | head -5; }
show_memory() { free -h; }
show_disk()   { df -h | grep -v tmpfs; }
show_network(){ ip -brief addr show; echo ""; ss -tlnp; }

show_top_processes() {
    local count="${1:-10}"
    echo "=== CPU TOP ${count} ==="
    ps aux --sort=-%cpu | head -$((count+1))
    echo ""
    echo "=== 内存 TOP ${count} ==="
    ps aux --sort=-%mem | head -$((count+1))
}

main() {
    echo "===== 运维工具箱 ====="
    select tool in "CPU信息" "内存信息" "磁盘信息" "网络信息" "TOP进程" "退出"; do
        case "$tool" in
            "CPU信息")  show_cpu ;;
            "内存信息") show_memory ;;
            "磁盘信息") show_disk ;;
            "网络信息") show_network ;;
            "TOP进程")
                read -rp "显示前几个? [10]: " n
                show_top_processes "${n:-10}"
                ;;
            "退出") break ;;
            *) echo "无效选项" ;;
        esac
        echo ""
    done
}

main
```

## 6. 小结

| 知识点 | 要点 |
|--------|------|
| 定义方式 | `func_name() { }` 推荐 |
| 参数 | `$1 $2 $@ $#`，与脚本参数语法相同 |
| 返回值 | `return` 返回退出码，`echo` 返回数据 |
| 局部变量 | 始终用 `local` 声明 |
| 函数库 | `source lib/common.sh` 复用 |
| 最佳实践 | 函数做一件事、命名清晰、参数校验 |
