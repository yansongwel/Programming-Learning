# 数据统计与报表

> 一句话说明：用 Shell 命令组合实现数据统计、汇总和报表输出。

---

## 1. 统计工具链

```bash
# 经典统计管道模式
cat data | sort | uniq -c | sort -rn | head -N

# sort 常用选项
sort -n         # 数值排序
sort -r         # 逆序
sort -k2        # 按第 2 列
sort -t: -k3 -n # 以 : 分隔，按第 3 列数值排序
sort -u         # 排序并去重

# uniq 常用选项
uniq -c         # 计数
uniq -d         # 只显示重复行
uniq -u         # 只显示唯一行
# 注意：uniq 只处理相邻行，所以通常需要先 sort
```

## 2. awk 统计

```bash
# 求和
awk '{sum+=$1} END{print sum}' numbers.txt

# 平均值
awk '{sum+=$1; n++} END{printf "平均: %.2f\n", sum/n}' numbers.txt

# 最大/最小值
awk 'NR==1||$1>max{max=$1} END{print "最大:", max}' numbers.txt
awk 'NR==1||$1<min{min=$1} END{print "最小:", min}' numbers.txt

# 完整统计
awk '{
    sum+=$1; sumsq+=$1*$1; n++
    if(NR==1||$1>max) max=$1
    if(NR==1||$1<min) min=$1
}
END{
    avg=sum/n
    printf "数量: %d\n", n
    printf "总和: %.2f\n", sum
    printf "平均: %.2f\n", avg
    printf "最大: %.2f\n", max
    printf "最小: %.2f\n", min
    printf "标准差: %.2f\n", sqrt(sumsq/n - avg*avg)
}' numbers.txt

# 分组统计
awk -F, '{group[$1]+=$2; count[$1]++}
END{
    for(g in group)
        printf "%-10s 总和: %8.2f  平均: %8.2f  数量: %d\n",
            g, group[g], group[g]/count[g], count[g]
}' data.csv | sort -k3 -rn
```

## 3. 实战：系统资源报表

```bash
#!/usr/bin/env bash
# system_report.sh — 生成系统资源使用报表

set -euo pipefail

REPORT_FILE="/tmp/system_report_$(date +%Y%m%d).txt"

{
    printf "╔══════════════════════════════════════╗\n"
    printf "║       系统资源使用报表               ║\n"
    printf "║  %s                   ║\n" "$(date '+%Y-%m-%d %H:%M:%S')"
    printf "╚══════════════════════════════════════╝\n"
    echo ""

    echo "【CPU 信息】"
    echo "  核心数: $(nproc)"
    echo "  负载: $(cat /proc/loadavg | awk '{print $1, $2, $3}')"
    echo ""

    echo "【内存使用】"
    free -h | awk '
        /Mem:/{printf "  总量: %s  已用: %s  可用: %s  使用率: %.1f%%\n", $2, $3, $7, $3/$2*100}
        /Swap:/{if($2!="0B") printf "  Swap: %s  已用: %s\n", $2, $3}
    '
    echo ""

    echo "【磁盘使用】"
    printf "  %-20s %8s %8s %8s %6s\n" "挂载点" "总量" "已用" "可用" "使用率"
    printf "  %-20s %8s %8s %8s %6s\n" "----" "----" "----" "----" "----"
    df -h | awk 'NR>1 && $1!~/tmpfs/{printf "  %-20s %8s %8s %8s %6s\n", $6, $2, $3, $4, $5}'
    echo ""

    echo "【Top 10 进程 (CPU)】"
    printf "  %-8s %-10s %6s %6s  %s\n" "PID" "USER" "CPU%" "MEM%" "COMMAND"
    ps aux --sort=-%cpu | awk 'NR>1 && NR<=11{printf "  %-8s %-10s %6s %6s  %s\n", $2, $1, $3, $4, $11}'
    echo ""

    echo "【Top 10 进程 (内存)】"
    printf "  %-8s %-10s %6s %6s  %s\n" "PID" "USER" "CPU%" "MEM%" "COMMAND"
    ps aux --sort=-%mem | awk 'NR>1 && NR<=11{printf "  %-8s %-10s %6s %6s  %s\n", $2, $1, $3, $4, $11}'
    echo ""

    echo "【网络连接统计】"
    ss -ant 2>/dev/null | awk 'NR>1{s[$1]++} END{for(k in s) printf "  %-15s %d\n", k, s[k]}' | sort -k2 -rn

} | tee "$REPORT_FILE"

echo ""
echo "报告已保存: $REPORT_FILE"
```

## 4. 报表格式化技巧

```bash
# 表格线
printf '=%.0s' {1..50}; echo

# 对齐输出
printf "%-20s %10s %10s\n" "Name" "Value" "Percent"
printf "%-20s %10d %9.1f%%\n" "CPU Usage" 75 75.3
printf "%-20s %10d %9.1f%%\n" "Memory" 8192 62.5

# 进度条
show_progress() {
    local percent=$1 width=30
    local filled=$((percent * width / 100))
    local empty=$((width - filled))
    printf "\r[%s%s] %3d%%" \
        "$(printf '█%.0s' $(seq 1 $filled 2>/dev/null) )" \
        "$(printf '░%.0s' $(seq 1 $empty 2>/dev/null) )" \
        "$percent"
}

# 颜色输出
RED='\033[0;31m'; GREEN='\033[0;32m'; YELLOW='\033[0;33m'; NC='\033[0m'
colorize_percent() {
    local val=$1
    if (( val >= 90 )); then echo -e "${RED}${val}%${NC}"
    elif (( val >= 70 )); then echo -e "${YELLOW}${val}%${NC}"
    else echo -e "${GREEN}${val}%${NC}"
    fi
}
```
