# 配置文件处理

> 一句话说明：用 Shell 解析和管理 INI、YAML、key=value 等常见配置文件格式。

---

## 1. Key=Value 格式

```bash
# 读取配置
get_config() {
    local file="$1" key="$2"
    grep "^${key}=" "$file" | cut -d= -f2- | sed 's/^"//;s/"$//'
}

# 修改配置
set_config() {
    local file="$1" key="$2" value="$3"
    if grep -q "^${key}=" "$file"; then
        sed -i "s|^${key}=.*|${key}=${value}|" "$file"
    else
        echo "${key}=${value}" >> "$file"
    fi
}

# 加载整个配置文件为环境变量
load_env() {
    local file="$1"
    while IFS='=' read -r key value; do
        [[ -z "$key" || "$key" =~ ^# ]] && continue
        key=$(echo "$key" | xargs)
        value=$(echo "$value" | xargs | sed 's/^"//;s/"$//')
        export "$key=$value"
    done < "$file"
}

# 使用
load_env ".env"
echo "DB_HOST=$DB_HOST"
```

## 2. INI 格式

```bash
# 示例 INI 文件
# [database]
# host = localhost
# port = 3306
# [redis]
# host = 127.0.0.1
# port = 6379

# 读取指定 section 的 key
ini_get() {
    local file="$1" section="$2" key="$3"
    awk -F '=' -v sec="$section" -v k="$key" '
        /^\[/{current=substr($0,2,length($0)-2)}
        current==sec && $1~k{gsub(/^[ \t]+|[ \t]+$/, "", $2); print $2}
    ' "$file"
}

# 列出所有 section
ini_sections() {
    grep '^\[' "$1" | tr -d '[]'
}

# 使用
DB_HOST=$(ini_get config.ini database host)
DB_PORT=$(ini_get config.ini database port)
```

## 3. YAML 简单解析

```bash
# 对于简单的 YAML，可以用 awk/grep
# 复杂 YAML 建议用 yq

# 安装 yq
# pip install yq  或  snap install yq

# yq 用法
yq '.database.host' config.yml
yq '.servers[0].name' config.yml
yq '.database.port = 5432' config.yml      # 修改

# 没有 yq 时的简易解析（仅适用于简单平铺格式）
parse_yaml() {
    local file="$1" prefix="${2:-}"
    local s='[[:space:]]*' w='[a-zA-Z0-9_]*'
    sed -ne "s|^\($s\)\($w\)$s:$s\"\(.*\)\"$s\$|\2=\"\3\"|p" \
         -e "s|^\($s\)\($w\)$s:$s\(.*\)$s\$|\2=\"\3\"|p" "$file"
}

eval "$(parse_yaml config.yml)"
```

## 4. JSON 处理

```bash
# jq — JSON 处理神器
# 安装: apt install jq / yum install jq

# 基本用法
echo '{"name":"test","port":8080}' | jq '.name'       # "test"
echo '{"name":"test","port":8080}' | jq -r '.name'    # test（无引号）

# 数组
echo '[1,2,3]' | jq '.[]'          # 遍历
echo '[1,2,3]' | jq '.[0]'         # 第一个
echo '[1,2,3]' | jq 'length'       # 3

# 过滤
cat data.json | jq '.servers[] | select(.status == "active") | .name'

# 构造
jq -n --arg name "test" --arg port "8080" '{name: $name, port: $port}'

# 没有 jq 时用 grep/awk（仅简单场景）
grep -o '"name":"[^"]*"' data.json | cut -d'"' -f4
```

## 5. 实战：配置管理工具

```bash
#!/usr/bin/env bash
# config_tool.sh — 通用配置文件管理

set -euo pipefail

CONFIG_FILE="${1:?用法: $0 <配置文件> <get|set|list> [key] [value]}"
ACTION="${2:?请指定操作: get|set|list|diff}"

case "$ACTION" in
    get)
        KEY="${3:?请指定 key}"
        VALUE=$(grep "^${KEY}=" "$CONFIG_FILE" | cut -d= -f2-)
        echo "${VALUE:-未找到}"
        ;;
    set)
        KEY="${3:?请指定 key}"
        VALUE="${4:?请指定 value}"
        if grep -q "^${KEY}=" "$CONFIG_FILE"; then
            sed -i "s|^${KEY}=.*|${KEY}=${VALUE}|" "$CONFIG_FILE"
            echo "已更新: ${KEY}=${VALUE}"
        else
            echo "${KEY}=${VALUE}" >> "$CONFIG_FILE"
            echo "已添加: ${KEY}=${VALUE}"
        fi
        ;;
    list)
        grep -v '^#' "$CONFIG_FILE" | grep -v '^$' | sort
        ;;
    diff)
        FILE2="${3:?请指定对比文件}"
        diff --color <(sort "$CONFIG_FILE") <(sort "$FILE2") || true
        ;;
esac
```

## 6. 小结

| 格式 | 解析方式 |
|------|----------|
| key=value | `grep + cut` 或 `source` |
| INI | `awk` 按 section 提取 |
| YAML | `yq` 推荐，简单场景用 `sed` |
| JSON | `jq` 必备工具 |
